56      @Description("convert Unicode code point to a string")
57      @ScalarFunction
58      @SqlType(StandardTypes.VARCHAR)
59      public static Slice chr(@SqlType(StandardTypes.BIGINT) long codepoint)
60      {
61          try {
62              return SliceUtf8.codePointToUtf8(Ints.saturatedCast(codepoint));
63          }
64          catch (InvalidCodePointException e) {
65              throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Not a valid Unicode code point: " + codepoint, e);
66          }
67      }

85      @Description("greedily replaces occurrences of a pattern with a string")
86      @ScalarFunction
87      @SqlType(StandardTypes.VARCHAR)
88      public static Slice replace(@SqlType(StandardTypes.VARCHAR) Slice str, @SqlType(StandardTypes.VARCHAR) Slice search, @SqlType(StandardTypes.VARCHAR) Slice replace)
89      {
90          // Empty search?
91          if (search.length() == 0) {
92              // With empty `search` we insert `replace` in front of every character and and the end
93              Slice buffer = Slices.allocate((countCodePoints(str) + 1) * replace.length() + str.length());
94              // Always start with replace
95              buffer.setBytes(0, replace);
96              int indexBuffer = replace.length();
97              // After every code point insert `replace`
98              int index = 0;
99              while (index < str.length()) {
100                 int codePointLength = lengthOfCodePointSafe(str, index);
101                 // Append current code point
102                 buffer.setBytes(indexBuffer, str, index, codePointLength);
103                 indexBuffer += codePointLength;
104                 // Append `replace`
105                 buffer.setBytes(indexBuffer, replace);
106                 indexBuffer += replace.length();
107                 // Advance pointer to current code point
108                 index += codePointLength;
109             }
110 
111             return buffer;
112         }
113         // Allocate a reasonable buffer
114         Slice buffer = Slices.allocate(str.length());
115 
116         int index = 0;
117         int indexBuffer = 0;
118         while (index < str.length()) {
119             int matchIndex = str.indexOf(search, index);
120             // Found a match?
121             if (matchIndex < 0) {
122                 // No match found so copy the rest of string
123                 int bytesToCopy = str.length() - index;
124                 buffer = Slices.ensureSize(buffer, indexBuffer + bytesToCopy);
125                 buffer.setBytes(indexBuffer, str, index, bytesToCopy);
126                 indexBuffer += bytesToCopy;
127 
128                 break;
129             }
130 
131             int bytesToCopy = matchIndex - index;
132             buffer = Slices.ensureSize(buffer, indexBuffer + bytesToCopy + replace.length());
133             // Non empty match?
134             if (bytesToCopy > 0) {
135                 buffer.setBytes(indexBuffer, str, index, bytesToCopy);
136                 indexBuffer += bytesToCopy;
137             }
138             // Non empty replace?
139             if (replace.length() > 0) {
140                 buffer.setBytes(indexBuffer, replace);
141                 indexBuffer += replace.length();
142             }
143             // Continue searching after match
144             index = matchIndex + search.length();
145         }
146 
147         return buffer.slice(0, indexBuffer);
148     }

158     @Description("returns index of first occurrence of a substring (or 0 if not found)")
159     @ScalarFunction("strpos")
160     @SqlType(StandardTypes.BIGINT)
161     public static long stringPosition(@SqlType(StandardTypes.VARCHAR) Slice string, @SqlType(StandardTypes.VARCHAR) Slice substring)
162     {
163         if (substring.length() == 0) {
164             return 1;
165         }
166 
167         int index = string.indexOf(substring);
168         if (index < 0) {
169             return 0;
170         }
171         return countCodePoints(string, 0, index) + 1;
172     }

174     @Description("suffix starting at given index")
175     @ScalarFunction
176     @SqlType(StandardTypes.VARCHAR)
177     public static Slice substr(@SqlType(StandardTypes.VARCHAR) Slice utf8, @SqlType(StandardTypes.BIGINT) long start)
178     {
179         if ((start == 0) || utf8.length() == 0) {
180             return Slices.EMPTY_SLICE;
181         }
182 
183         int startCodePoint = Ints.saturatedCast(start);
184 
185         if (startCodePoint > 0) {
186             int indexStart = offsetOfCodePoint(utf8, startCodePoint - 1);
187             if (indexStart < 0) {
188                 // before beginning of string
189                 return Slices.EMPTY_SLICE;
190             }
191             int indexEnd = utf8.length();
192 
193             return utf8.slice(indexStart, indexEnd - indexStart);
194         }
195 
196         // negative start is relative to end of string
197         int codePoints = countCodePoints(utf8);
198         startCodePoint += codePoints;
199 
200         // before beginning of string
201         if (startCodePoint < 0) {
202             return Slices.EMPTY_SLICE;
203         }
204 
205         int indexStart = offsetOfCodePoint(utf8, startCodePoint);
206         int indexEnd = utf8.length();
207 
208         return utf8.slice(indexStart, indexEnd - indexStart);
209     }

211     @Description("substring of given length starting at an index")
212     @ScalarFunction
213     @SqlType(StandardTypes.VARCHAR)
214     public static Slice substr(@SqlType(StandardTypes.VARCHAR) Slice utf8, @SqlType(StandardTypes.BIGINT) long start, @SqlType(StandardTypes.BIGINT) long length)
215     {
216         if (start == 0 || (length <= 0) || (utf8.length() == 0)) {
217             return Slices.EMPTY_SLICE;
218         }
219 
220         int startCodePoint = Ints.saturatedCast(start);
221         int lengthCodePoints = Ints.saturatedCast(length);
222 
223         if (startCodePoint > 0) {
224             int indexStart = offsetOfCodePoint(utf8, startCodePoint - 1);
225             if (indexStart < 0) {
226                 // before beginning of string
227                 return Slices.EMPTY_SLICE;
228             }
229             int indexEnd = offsetOfCodePoint(utf8, indexStart, lengthCodePoints);
230             if (indexEnd < 0) {
231                 // after end of string
232                 indexEnd = utf8.length();
233             }
234 
235             return utf8.slice(indexStart, indexEnd - indexStart);
236         }
237 
238         // negative start is relative to end of string
239         int codePoints = countCodePoints(utf8);
240         startCodePoint += codePoints;
241 
242         // before beginning of string
243         if (startCodePoint < 0) {
244             return Slices.EMPTY_SLICE;
245         }
246 
247         int indexStart = offsetOfCodePoint(utf8, startCodePoint);
248         int indexEnd;
249         if (startCodePoint + lengthCodePoints < codePoints) {
250             indexEnd = offsetOfCodePoint(utf8, indexStart, lengthCodePoints);
251         }
252         else {
253             indexEnd = utf8.length();
254         }
255 
256         return utf8.slice(indexStart, indexEnd - indexStart);
257     }

266     @ScalarFunction
267     @SqlType("array<varchar>")
268     public static Block split(@SqlType(StandardTypes.VARCHAR) Slice string, @SqlType(StandardTypes.VARCHAR) Slice delimiter, @SqlType(StandardTypes.BIGINT) long limit)
269     {
270         checkCondition(limit > 0, INVALID_FUNCTION_ARGUMENT, "Limit must be positive");
271         checkCondition(limit <= Integer.MAX_VALUE, INVALID_FUNCTION_ARGUMENT, "Limit is too large");
272         checkCondition(delimiter.length() > 0, INVALID_FUNCTION_ARGUMENT, "The delimiter may not be the empty string");
273         BlockBuilder parts = VARCHAR.createBlockBuilder(new BlockBuilderStatus(), 1, string.length());
274         // If limit is one, the last and only element is the complete string
275         if (limit == 1) {
276             VARCHAR.writeSlice(parts, string);
277             return parts.build();
278         }
279 
280         int index = 0;
281         while (index < string.length()) {
282             int splitIndex = string.indexOf(delimiter, index);
283             // Found split?
284             if (splitIndex < 0) {
285                 break;
286             }
287             // Add the part from current index to found split
288             VARCHAR.writeSlice(parts, string, index, splitIndex - index);
289             // Continue searching after delimiter
290             index = splitIndex + delimiter.length();
291             // Reached limit-1 parts so we can stop
292             if (parts.getPositionCount() == limit - 1) {
293                 break;
294             }
295         }
296         // Rest of string
297         VARCHAR.writeSlice(parts, string, index, string.length() - index);
298 
299         return parts.build();
300     }

302     @Nullable
303     @Description("splits a string by a delimiter and returns the specified field (counting from one)")
304     @ScalarFunction
305     @SqlType(StandardTypes.VARCHAR)
306     public static Slice splitPart(@SqlType(StandardTypes.VARCHAR) Slice string, @SqlType(StandardTypes.VARCHAR) Slice delimiter, @SqlType(StandardTypes.BIGINT) long index)
307     {
308         checkCondition(index > 0, INVALID_FUNCTION_ARGUMENT, "Index must be greater than zero");
309         // Empty delimiter? Then every character will be a split
310         if (delimiter.length() == 0) {
311             int startCodePoint = Ints.checkedCast(index);
312 
313             int indexStart = offsetOfCodePoint(string, startCodePoint - 1);
314             if (indexStart < 0) {
315                 // index too big
316                 return null;
317             }
318             int length = lengthOfCodePoint(string, indexStart);
319             if (indexStart + length > string.length()) {
320                 throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Invalid UTF-8 encoding");
321             }
322             return string.slice(indexStart, length);
323         }
324 
325         int matchCount = 0;
326 
327         int previousIndex = 0;
328         while (previousIndex < string.length()) {
329             int matchIndex = string.indexOf(delimiter, previousIndex);
330             // No match
331             if (matchIndex < 0) {
332                 break;
333             }
334             // Reached the requested part?
335             if (++matchCount == index) {
336                 return string.slice(previousIndex, matchIndex - previousIndex);
337             }
338             // Continue searching after the delimiter
339             previousIndex = matchIndex + delimiter.length();
340         }
341 
342         if (matchCount == index - 1) {
343             // returns last section of the split
344             return string.slice(previousIndex, string.length() - previousIndex);
345         }
346 
347         // index is too big, null is returned
348         return null;
349     }

391     @Description("transforms the string to normalized form")
392     @ScalarFunction
393     @SqlType(StandardTypes.VARCHAR)
394     public static Slice normalize(@SqlType(StandardTypes.VARCHAR) Slice slice, @SqlType(StandardTypes.VARCHAR) Slice form)
395     {
396         Normalizer.Form targetForm;
397         try {
398             targetForm = Normalizer.Form.valueOf(form.toStringUtf8());
399         }
400         catch (IllegalArgumentException e) {
401             throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Normalization form must be one of [NFD, NFC, NFKD, NFKC]");
402         }
403         return utf8Slice(Normalizer.normalize(slice.toStringUtf8(), targetForm));
404     }

414     @Description("decodes the UTF-8 encoded string")
415     @ScalarFunction
416     @SqlType(StandardTypes.VARCHAR)
417     public static Slice fromUtf8(@SqlType(StandardTypes.VARBINARY) Slice slice, @SqlType(StandardTypes.VARCHAR) Slice replacementCharacter)
418     {
419         int count = countCodePoints(replacementCharacter);
420         if (count > 1) {
421             throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Replacement character string must empty or a single character");
422         }
423 
424         OptionalInt replacementCodePoint;
425         if (count == 1) {
426             try {
427                 replacementCodePoint = OptionalInt.of(SliceUtf8.getCodePointAt(replacementCharacter, 0));
428             }
429             catch (InvalidUtf8Exception e) {
430                 throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Invalid replacement character");
431             }
432         }
433         else {
434             replacementCodePoint = OptionalInt.empty();
435         }
436         return SliceUtf8.fixInvalidUtf8(slice, replacementCodePoint);
437     }

439     @Description("decodes the UTF-8 encoded string")
440     @ScalarFunction
441     @SqlType(StandardTypes.VARCHAR)
442     public static Slice fromUtf8(@SqlType(StandardTypes.VARBINARY) Slice slice, @SqlType(StandardTypes.BIGINT) long replacementCodePoint)
443     {
444         if (replacementCodePoint > MAX_CODE_POINT || Character.getType((int) replacementCodePoint) == SURROGATE) {
445             throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Invalid replacement character");
446         }
447         return SliceUtf8.fixInvalidUtf8(slice, OptionalInt.of((int) replacementCodePoint));
448     }
