55  //
56  // NOTE:  As a general strategy the methods should "stage" a change and only
57  // process the actual change before lock release (DriverLockResult.close()).
58  // The assures that only one thread will be working with the operators at a
59  // time and state changer threads are not blocked.
60  //
61  public class Driver
62          implements Closeable
63  {
64      private static final Logger log = Logger.get(Driver.class);

66      private final DriverContext driverContext;
67      private final List<Operator> operators;
68      private final Map<PlanNodeId, SourceOperator> sourceOperators;
69      private final Optional<DeleteOperator> deleteOperator;
70      private final ConcurrentMap<PlanNodeId, TaskSource> newSources = new ConcurrentHashMap<>();
71  
72      private final AtomicReference<State> state = new AtomicReference<>(State.ALIVE);
73  
74      private final ReentrantLock exclusiveLock = new ReentrantLock();
75  
76      @GuardedBy("this")
77      private Thread lockHolder;
78  
79      @GuardedBy("exclusiveLock")
80      private final Map<PlanNodeId, TaskSource> currentSources = new ConcurrentHashMap<>();

82      private enum State
83      {
84          ALIVE, NEED_DESTRUCTION, DESTROYED
85      }

87      public Driver(DriverContext driverContext, Operator firstOperator, Operator... otherOperators)
88      {
89          this(requireNonNull(driverContext, "driverContext is null"),
90                  ImmutableList.<Operator>builder()
91                          .add(requireNonNull(firstOperator, "firstOperator is null"))
92                          .add(requireNonNull(otherOperators, "otherOperators is null"))
93                          .build());
94      }

96      public Driver(DriverContext driverContext, List<Operator> operators)
97      {
98          this.driverContext = requireNonNull(driverContext, "driverContext is null");
99          this.operators = ImmutableList.copyOf(requireNonNull(operators, "operators is null"));
100         checkArgument(!operators.isEmpty(), "There must be at least one operator");
101 
102         ImmutableMap.Builder<PlanNodeId, SourceOperator> sourceOperators = ImmutableMap.builder();
103         Optional<DeleteOperator> deleteOperator = Optional.empty();
104         for (Operator operator : operators) {
105             if (operator instanceof SourceOperator) {
106                 SourceOperator sourceOperator = (SourceOperator) operator;
107                 sourceOperators.put(sourceOperator.getSourceId(), sourceOperator);
108             }
109             else if (operator instanceof DeleteOperator) {
110                 checkArgument(!deleteOperator.isPresent(), "There must be at most one DeleteOperator");
111                 deleteOperator = Optional.of((DeleteOperator) operator);
112             }
113         }
114         this.sourceOperators = sourceOperators.build();
115         this.deleteOperator = deleteOperator;
116     }

123     public Set<PlanNodeId> getSourceIds()

125         return sourceOperators.keySet();

128     @Override
129     public void close()
130     {
131         // mark the service for destruction
132         if (!state.compareAndSet(State.ALIVE, State.NEED_DESTRUCTION)) {
133             return;
134         }
135 
136         // if we can get the lock, attempt a clean shutdown; otherwise someone else will shutdown
137         try (DriverLockResult lockResult = tryLockAndProcessPendingStateChanges(0, TimeUnit.MILLISECONDS)) {
138             // if we did not get the lock, interrupt the lock holder
139             if (!lockResult.wasAcquired()) {
140                 // there is a benign race condition here were the lock holder
141                 // can be change between attempting to get lock and grabbing
142                 // the synchronized lock here, but in either case we want to
143                 // interrupt the lock holder thread
144                 synchronized (this) {
145                     if (lockHolder != null) {
146                         lockHolder.interrupt();
147                     }
148                 }
149             }
150 
151             // clean shutdown is automatically triggered during lock release
152         }
153     }

155     public boolean isFinished()
156     {
157         checkLockNotHeld("Can not check finished status while holding the driver lock");
158 
159         // if we can get the lock, attempt a clean shutdown; otherwise someone else will shutdown
160         try (DriverLockResult lockResult = tryLockAndProcessPendingStateChanges(0, TimeUnit.MILLISECONDS)) {
161             if (lockResult.wasAcquired()) {
162                 return isFinishedInternal();
163             }
164             else {
165                 // did not get the lock, so we can't check operators, or destroy
166                 return state.get() != State.ALIVE || driverContext.isDone();
167             }
168         }
169     }
170 
171     private boolean isFinishedInternal()
172     {
173         checkLockHeld("Lock must be held to call isFinishedInternal");
174 
175         boolean finished = state.get() != State.ALIVE || driverContext.isDone() || operators.get(operators.size() - 1).isFinished();
176         if (finished) {
177             state.compareAndSet(State.ALIVE, State.NEED_DESTRUCTION);
178         }
179         return finished;
180     }

182     public void updateSource(TaskSource source)
183     {
184         checkLockNotHeld("Can not update sources while holding the driver lock");
185 
186         // does this driver have an operator for the specified source?
187         if (!sourceOperators.containsKey(source.getPlanNodeId())) {
188             return;
189         }
190 
191         // stage the new updates
192         while (true) {
193             // attempt to update directly to the new source
194             TaskSource currentNewSource = newSources.putIfAbsent(source.getPlanNodeId(), source);
195 
196             // if update succeeded, just break
197             if (currentNewSource == null) {
198                 break;
199             }
200 
201             // merge source into the current new source
202             TaskSource newSource = currentNewSource.update(source);
203 
204             // if this is not a new source, just return
205             if (newSource == currentNewSource) {
206                 break;
207             }
208 
209             // attempt to replace the currentNewSource with the new source
210             if (newSources.replace(source.getPlanNodeId(), currentNewSource, newSource)) {
211                 break;
212             }
213 
214             // someone else updated while we were processing
215         }
216 
217         // attempt to get the lock and process the updates we staged above
218         // updates will be processed in close if and only if we got the lock
219         tryLockAndProcessPendingStateChanges(0, TimeUnit.MILLISECONDS).close();
220     }

222     private void processNewSources()
223     {
224         checkLockHeld("Lock must be held to call processNewSources");
225 
226         // only update if the driver is still alive
227         if (state.get() != State.ALIVE) {
228             return;
229         }
230 
231         // copy the pending sources
232         // it is ok to "miss" a source added during the copy as it will be
233         // handled on the next call to this method
234         Map<PlanNodeId, TaskSource> sources = new HashMap<>(newSources);
235         for (Entry<PlanNodeId, TaskSource> entry : sources.entrySet()) {
236             // Remove the entries we are going to process from the newSources map.
237             // It is ok if someone already updated the entry; we will catch it on
238             // the next iteration.
239             newSources.remove(entry.getKey(), entry.getValue());
240 
241             processNewSource(entry.getValue());
242         }
243     }
244 
245     private void processNewSource(TaskSource source)
246     {
247         checkLockHeld("Lock must be held to call processNewSources");
248 
249         // create new source
250         Set<ScheduledSplit> newSplits;
251         TaskSource currentSource = currentSources.get(source.getPlanNodeId());
252         if (currentSource == null) {
253             newSplits = source.getSplits();
254             currentSources.put(source.getPlanNodeId(), source);
255         }
256         else {
257             // merge the current source and the specified source
258             TaskSource newSource = currentSource.update(source);
259 
260             // if this is not a new source, just return
261             if (newSource == currentSource) {
262                 return;
263             }
264 
265             // find the new splits to add
266             newSplits = Sets.difference(newSource.getSplits(), currentSource.getSplits());
267             currentSources.put(source.getPlanNodeId(), newSource);
268         }
269 
270         // add new splits
271         for (ScheduledSplit newSplit : newSplits) {
272             Split split = newSplit.getSplit();
273 
274             SourceOperator sourceOperator = sourceOperators.get(source.getPlanNodeId());
275             if (sourceOperator != null) {
276                 Supplier<Optional<UpdatablePageSource>> pageSource = sourceOperator.addSplit(split);
277                 if (deleteOperator.isPresent()) {
278                     deleteOperator.get().setPageSource(pageSource);
279                 }
280             }
281         }
282 
283         // set no more splits
284         if (source.isNoMoreSplits()) {
285             sourceOperators.get(source.getPlanNodeId()).noMoreSplits();
286         }
287     }

289     public ListenableFuture<?> processFor(Duration duration)
290     {
291         checkLockNotHeld("Can not process for a duration while holding the driver lock");
292 
293         requireNonNull(duration, "duration is null");
294 
295         long maxRuntime = duration.roundTo(TimeUnit.NANOSECONDS);
296 
297         try (DriverLockResult lockResult = tryLockAndProcessPendingStateChanges(100, TimeUnit.MILLISECONDS)) {
298             if (lockResult.wasAcquired()) {
299                 driverContext.startProcessTimer();
300                 try {
301                     long start = System.nanoTime();
302                     do {
303                         ListenableFuture<?> future = processInternal();
304                         if (!future.isDone()) {
305                             return future;
306                         }
307                     }
308                     while (System.nanoTime() - start < maxRuntime && !isFinishedInternal());
309                 }
310                 finally {
311                     driverContext.recordProcessed();
312                 }
313             }
314         }
315         return NOT_BLOCKED;
316     }

318     public ListenableFuture<?> process()
319     {
320         checkLockNotHeld("Can not process while holding the driver lock");
321 
322         try (DriverLockResult lockResult = tryLockAndProcessPendingStateChanges(100, TimeUnit.MILLISECONDS)) {
323             if (!lockResult.wasAcquired()) {
324                 // this is unlikely to happen unless the driver is being
325                 // destroyed and in that case the caller should notice notice
326                 // this state change by calling isFinished
327                 return NOT_BLOCKED;
328             }
329             return processInternal();
330         }
331     }

333     private ListenableFuture<?> processInternal()
334     {
335         checkLockHeld("Lock must be held to call processInternal");
336 
337         try {
338             if (!newSources.isEmpty()) {
339                 processNewSources();
340             }
341 
342             // special handling for drivers with a single operator
343             if (operators.size() == 1) {
344                 if (driverContext.isDone()) {
345                     return NOT_BLOCKED;
346                 }
347 
348                 // check if operator is blocked
349                 Operator current = operators.get(0);
350                 ListenableFuture<?> blocked = isBlocked(current);
351                 if (!blocked.isDone()) {
352                     current.getOperatorContext().recordBlocked(blocked);
353                     return blocked;
354                 }
355 
356                 // there is only one operator so just finish it
357                 current.getOperatorContext().startIntervalTimer();
358                 current.finish();
359                 current.getOperatorContext().recordFinish();
360                 return NOT_BLOCKED;
361             }
362 
363             boolean movedPage = false;
364             for (int i = 0; i < operators.size() - 1 && !driverContext.isDone(); i++) {
365                 Operator current = operators.get(i);
366                 Operator next = operators.get(i + 1);
367 
368                 // skip blocked operators
369                 if (!isBlocked(current).isDone()) {
370                     continue;
371                 }
372                 if (!isBlocked(next).isDone()) {
373                     continue;
374                 }
375 
376                 // if the current operator is not finished and next operator needs input...
377                 if (!current.isFinished() && next.needsInput()) {
378                     // get an output page from current operator
379                     current.getOperatorContext().startIntervalTimer();
380                     Page page = current.getOutput();
381                     current.getOperatorContext().recordGetOutput(page);
382 
383                     // if we got an output page, add it to the next operator
384                     if (page != null) {
385                         next.getOperatorContext().startIntervalTimer();
386                         next.addInput(page);
387                         next.getOperatorContext().recordAddInput(page);
388                         movedPage = true;
389                     }
390                 }
391 
392                 // if current operator is finished...
393                 if (current.isFinished()) {
394                     // let next operator know there will be no more data
395                     next.getOperatorContext().startIntervalTimer();
396                     next.finish();
397                     next.getOperatorContext().recordFinish();
398                 }
399             }
400 
401             // if we did not move any pages, check if we are blocked
402             if (!movedPage) {
403                 List<Operator> blockedOperators = new ArrayList<>();
404                 List<ListenableFuture<?>> blockedFutures = new ArrayList<>();
405                 for (Operator operator : operators) {
406                     ListenableFuture<?> blocked = isBlocked(operator);
407                     if (!blocked.isDone()) {
408                         blockedOperators.add(operator);
409                         blockedFutures.add(blocked);
410                     }
411                 }
412 
413                 if (!blockedFutures.isEmpty()) {
414                     // unblock when the first future is complete
415                     ListenableFuture<?> blocked = firstFinishedFuture(blockedFutures);
416                     // driver records serial blocked time
417                     driverContext.recordBlocked(blocked);
418                     // each blocked operator is responsible for blocking the execution
419                     // until one of the operators can continue
420                     for (Operator operator : blockedOperators) {
421                         operator.getOperatorContext().recordBlocked(blocked);
422                     }
423                     return blocked;
424                 }
425             }
426 
427             return NOT_BLOCKED;
428         }
429         catch (Throwable t) {
430             driverContext.failed(t);
431             throw t;
432         }
433     }

435     private void destroyIfNecessary()
436     {
437         checkLockHeld("Lock must be held to call destroyIfNecessary");
438 
439         if (!state.compareAndSet(State.NEED_DESTRUCTION, State.DESTROYED)) {
440             return;
441         }
442 
443         // record the current interrupted status (and clear the flag); we'll reset it later
444         boolean wasInterrupted = Thread.interrupted();
445 
446         // if we get an error while closing a driver, record it and we will throw it at the end
447         Throwable inFlightException = null;
448         try {
449             for (Operator operator : operators) {
450                 try {
451                     operator.close();
452                 }
453                 catch (InterruptedException t) {
454                     // don't record the stack
455                     wasInterrupted = true;
456                 }
457                 catch (Throwable t) {
458                     inFlightException = addSuppressedException(
459                             inFlightException,
460                             t,
461                             "Error closing operator %s for task %s",
462                             operator.getOperatorContext().getOperatorId(),
463                             driverContext.getTaskId());
464                 }
465                 try {
466                     operator.getOperatorContext().setMemoryReservation(0);
467                 }
468                 catch (Throwable t) {
469                     inFlightException = addSuppressedException(
470                             inFlightException,
471                             t,
472                             "Error freeing memory for operator %s for task %s",
473                             operator.getOperatorContext().getOperatorId(),
474                             driverContext.getTaskId());
475                 }
476                 try {
477                     operator.getOperatorContext().closeSystemMemoryContext();
478                 }
479                 catch (Throwable t) {
480                     inFlightException = addSuppressedException(
481                             inFlightException,
482                             t,
483                             "Error freeing system memory for operator %s for task %s",
484                             operator.getOperatorContext().getOperatorId(),
485                             driverContext.getTaskId());
486                 }
487             }
488             driverContext.finished();
489         }
490         catch (Throwable t) {
491             // this shouldn't happen but be safe
492             inFlightException = addSuppressedException(
493                     inFlightException,
494                     t,
495                     "Error destroying driver for task %s",
496                     driverContext.getTaskId());
497         }
498         finally {
499             // reset the interrupted flag
500             if (wasInterrupted) {
501                 Thread.currentThread().interrupt();
502             }
503         }
504 
505         if (inFlightException != null) {
506             // this will always be an Error or Runtime
507             throw Throwables.propagate(inFlightException);
508         }
509     }

511     private static ListenableFuture<?> isBlocked(Operator operator)
512     {
513         ListenableFuture<?> blocked = operator.isBlocked();
514         if (blocked.isDone()) {
515             blocked = operator.getOperatorContext().isWaitingForMemory();
516         }
517         return blocked;
518     }

520     private static Throwable addSuppressedException(Throwable inFlightException, Throwable newException, String message, Object... args)
521     {
522         if (newException instanceof Error) {
523             if (inFlightException == null) {
524                 inFlightException = newException;
525             }
526             else {
527                 // Self-suppression not permitted
528                 if (inFlightException != newException) {
529                     inFlightException.addSuppressed(newException);
530                 }
531             }
532         }
533         else {
534             // log normal exceptions instead of rethrowing them
535             log.error(newException, message, args);
536         }
537         return inFlightException;
538     }

540     private DriverLockResult tryLockAndProcessPendingStateChanges(int timeout, TimeUnit unit)
541     {
542         checkLockNotHeld("Can not acquire the driver lock while already holding the driver lock");
543 
544         return new DriverLockResult(timeout, unit);
545     }
546 
547     private synchronized void checkLockNotHeld(String message)
548     {
549         checkState(Thread.currentThread() != lockHolder, message);
550     }
551 
552     private synchronized void checkLockHeld(String message)
553     {
554         checkState(Thread.currentThread() == lockHolder, message);
555     }

557     private static ListenableFuture<?> firstFinishedFuture(List<ListenableFuture<?>> futures)
558     {
559         SettableFuture<?> result = SettableFuture.create();
560         ExecutorService executor = MoreExecutors.newDirectExecutorService();
561 
562         for (ListenableFuture<?> future : futures) {
563             future.addListener(() -> result.set(null), executor);
564         }
565 
566         return result;
567     }
568 
569     private class DriverLockResult
570             implements AutoCloseable
571     {
572         private final boolean acquired;
573 
574         private DriverLockResult(int timeout, TimeUnit unit)
575         {
576             acquired = tryAcquire(timeout, unit);
577         }
578 
579         private boolean tryAcquire(int timeout, TimeUnit unit)
580         {
581             boolean acquired = false;
582             try {
583                 acquired = exclusiveLock.tryLock(timeout, unit);
584             }
585             catch (InterruptedException e) {
586                 Thread.currentThread().interrupt();
587             }
588 
589             if (acquired) {
590                 synchronized (Driver.this) {
591                     lockHolder = Thread.currentThread();
592                 }
593             }
594 
595             return acquired;
596         }
597 
598         public boolean wasAcquired()
599         {
600             return acquired;
601         }
602 
603         @Override
604         public void close()
605         {
606             if (!acquired) {
607                 return;
608             }
609 
610             boolean done = false;
611             while (!done) {
612                 done = true;
613                 // before releasing the lock, process any new sources and/or destroy the driver
614                 try {
615                     try {
616                         processNewSources();
617                     }
618                     finally {
619                         destroyIfNecessary();
620                     }
621                 }
622                 finally {
623                     synchronized (Driver.this) {
624                         lockHolder = null;
625                     }
626                     exclusiveLock.unlock();
627 
628                     // if new sources were added after we processed them, go around and try again
629                     // in case someone else failed to acquire the lock and as a result won't update them
630                     if (!newSources.isEmpty() && state.get() == State.ALIVE && tryAcquire(0, TimeUnit.MILLISECONDS)) {
631                         done = false;
632                     }
633                 }
634             }
635         }
636     }
