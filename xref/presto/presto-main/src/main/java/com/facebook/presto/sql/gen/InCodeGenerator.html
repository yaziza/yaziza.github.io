58  public class InCodeGenerator
59          implements BytecodeGenerator

61      private final FunctionRegistry registry;
62  
63      public InCodeGenerator(FunctionRegistry registry)
64      {
65          this.registry = requireNonNull(registry, "registry is null");
66      }
67  
68      enum SwitchGenerationCase
69      {
70          DIRECT_SWITCH,
71          HASH_SWITCH,
72          SET_CONTAINS
73      }
74  
75      @VisibleForTesting
76      static SwitchGenerationCase checkSwitchGenerationCase(Type type, List<RowExpression> values)
77      {
78          if (values.size() > 32) {
79              // 32 is chosen because
80              // * SET_CONTAINS performs worst when smaller than but close to power of 2
81              // * Benchmark shows performance of SET_CONTAINS is better at 50, but similar at 25.
82              return SwitchGenerationCase.SET_CONTAINS;
83          }
84  
85          if (!(type instanceof BigintType || type instanceof DateType)) {
86              return SwitchGenerationCase.HASH_SWITCH;
87          }
88          for (RowExpression expression : values) {
89              // For non-constant expressions, they will be added to the default case in the generated switch code. They do not affect any of
90              // the cases other than the default one. Therefore, it's okay to skip them when choosing between DIRECT_SWITCH and HASH_SWITCH.
91              // Same argument applies for nulls.
92              if (!(expression instanceof ConstantExpression)) {
93                  continue;
94              }
95              Object constant = ((ConstantExpression) expression).getValue();
96              if (constant == null) {
97                  continue;
98              }
99              long longConstant = (Long) constant;
100             if (longConstant < Integer.MIN_VALUE || longConstant > Integer.MAX_VALUE) {
101                 return SwitchGenerationCase.HASH_SWITCH;
102             }
103         }
104         return SwitchGenerationCase.DIRECT_SWITCH;
105     }

107     @Override
108     public BytecodeNode generateExpression(Signature signature, BytecodeGeneratorContext generatorContext, Type returnType, List<RowExpression> arguments)
109     {
110         BytecodeNode value = generatorContext.generate(arguments.get(0));
111 
112         List<RowExpression> values = arguments.subList(1, arguments.size());
113 
114         ImmutableList.Builder<BytecodeNode> valuesBytecode = ImmutableList.builder();
115         for (int i = 1; i < arguments.size(); i++) {
116             BytecodeNode testNode = generatorContext.generate(arguments.get(i));
117             valuesBytecode.add(testNode);
118         }
119 
120         Type type = arguments.get(0).getType();
121         Class<?> javaType = type.getJavaType();
122 
123         SwitchGenerationCase switchGenerationCase  = checkSwitchGenerationCase(type, values);
124 
125         Signature hashCodeSignature = internalOperator(HASH_CODE, BIGINT, ImmutableList.of(type));
126         MethodHandle hashCodeFunction = generatorContext.getRegistry().getScalarFunctionImplementation(hashCodeSignature).getMethodHandle();
127 
128         ImmutableListMultimap.Builder<Integer, BytecodeNode> hashBucketsBuilder = ImmutableListMultimap.builder();
129         ImmutableList.Builder<BytecodeNode> defaultBucket = ImmutableList.builder();
130         ImmutableSet.Builder<Object> constantValuesBuilder = ImmutableSet.builder();
131 
132         for (RowExpression testValue : values) {
133             BytecodeNode testBytecode = generatorContext.generate(testValue);
134 
135             if (testValue instanceof ConstantExpression && ((ConstantExpression) testValue).getValue() != null) {
136                 ConstantExpression constant = (ConstantExpression) testValue;
137                 Object object = constant.getValue();
138                 switch (switchGenerationCase) {
139                     case DIRECT_SWITCH:
140                     case SET_CONTAINS:
141                         constantValuesBuilder.add(object);
142                         break;
143                     case HASH_SWITCH:
144                         try {
145                             int hashCode = Ints.checkedCast((Long) hashCodeFunction.invoke(object));
146                             hashBucketsBuilder.put(hashCode, testBytecode);
147                         }
148                         catch (Throwable throwable) {
149                             throw new IllegalArgumentException("Error processing IN statement: error calculating hash code for " + object, throwable);
150                         }
151                         break;
152                     default:
153                         throw new IllegalArgumentException("Not supported switch generation case: " + switchGenerationCase);
154                 }
155             }
156             else {
157                 defaultBucket.add(testBytecode);
158             }
159         }
160         ImmutableListMultimap<Integer, BytecodeNode> hashBuckets = hashBucketsBuilder.build();
161         ImmutableSet<Object> constantValues = constantValuesBuilder.build();
162 
163         LabelNode end = new LabelNode("end");
164         LabelNode match = new LabelNode("match");
165         LabelNode noMatch = new LabelNode("noMatch");
166 
167         LabelNode defaultLabel = new LabelNode("default");
168 
169         Scope scope = generatorContext.getScope();
170 
171         BytecodeNode switchBlock;
172         BytecodeBlock switchCaseBlocks = new BytecodeBlock();
173         LookupSwitch.LookupSwitchBuilder switchBuilder = lookupSwitchBuilder();
174         switch (switchGenerationCase) {
175             case DIRECT_SWITCH:
176                 // A white-list is used to select types eligible for DIRECT_SWITCH.
177                 // For these types, it's safe to not use presto HASH_CODE and EQUAL operator.
178                 for (Object constantValue : constantValues) {
179                     switchBuilder.addCase(Ints.checkedCast((Long) constantValue), match);
180                 }
181                 switchBuilder.defaultCase(defaultLabel);
182                 switchBlock = new BytecodeBlock()
183                         .comment("lookupSwitch(<stackValue>))")
184                         .dup(javaType)
185                         .append(new IfStatement()
186                                 .condition(new BytecodeBlock()
187                                         .dup(javaType)
188                                         .invokeStatic(InCodeGenerator.class, "isInteger", boolean.class, long.class))
189                                 .ifFalse(new BytecodeBlock()
190                                         .pop(javaType)
191                                         .gotoLabel(defaultLabel)))
192                         .longToInt()
193                         .append(switchBuilder.build());
194                 break;
195             case HASH_SWITCH:
196                 for (Map.Entry<Integer, Collection<BytecodeNode>> bucket : hashBuckets.asMap().entrySet()) {
197                     LabelNode label = new LabelNode("inHash" + bucket.getKey());
198                     switchBuilder.addCase(bucket.getKey(), label);
199                     Collection<BytecodeNode> testValues = bucket.getValue();
200 
201                     BytecodeBlock caseBlock = buildInCase(generatorContext, scope, type, label, match, defaultLabel, testValues, false);
202                     switchCaseBlocks.append(caseBlock.setDescription("case " + bucket.getKey()));
203                 }
204                 switchBuilder.defaultCase(defaultLabel);
205                 Binding hashCodeBinding = generatorContext
206                         .getCallSiteBinder()
207                         .bind(hashCodeFunction);
208                 switchBlock = new BytecodeBlock()
209                         .comment("lookupSwitch(hashCode(<stackValue>))")
210                         .dup(javaType)
211                         .append(invoke(hashCodeBinding, hashCodeSignature))
212                         .longToInt()
213                         .append(switchBuilder.build())
214                         .append(switchCaseBlocks);
215                 break;
216             case SET_CONTAINS:
217                 Set<?> constantValuesSet = toFastutilHashSet(constantValues, type, registry);
218                 Binding constant = generatorContext.getCallSiteBinder().bind(constantValuesSet, constantValuesSet.getClass());
219 
220                 switchBlock = new BytecodeBlock()
221                         .comment("inListSet.contains(<stackValue>)")
222                         .append(new IfStatement()
223                                 .condition(new BytecodeBlock()
224                                         .comment("value")
225                                         .dup(javaType)
226                                         .comment("set")
227                                         .append(loadConstant(constant))
228                                         // TODO: use invokeVirtual on the set instead. This requires swapping the two elements in the stack
229                                         .invokeStatic(FastutilSetHelper.class, "in", boolean.class, javaType.isPrimitive() ? javaType : Object.class, constantValuesSet.getClass()))
230                                 .ifTrue(jump(match)));
231                 break;
232             default:
233                 throw new IllegalArgumentException("Not supported switch generation case: " + switchGenerationCase);
234         }
235 
236         BytecodeBlock defaultCaseBlock = buildInCase(generatorContext, scope, type, defaultLabel, match, noMatch, defaultBucket.build(), true).setDescription("default");
237 
238         BytecodeBlock block = new BytecodeBlock()
239                 .comment("IN")
240                 .append(value)
241                 .append(ifWasNullPopAndGoto(scope, end, boolean.class, javaType))
242                 .append(switchBlock)
243                 .append(defaultCaseBlock);
244 
245         BytecodeBlock matchBlock = new BytecodeBlock()
246                 .setDescription("match")
247                 .visitLabel(match)
248                 .pop(javaType)
249                 .append(generatorContext.wasNull().set(constantFalse()))
250                 .push(true)
251                 .gotoLabel(end);
252         block.append(matchBlock);
253 
254         BytecodeBlock noMatchBlock = new BytecodeBlock()
255                 .setDescription("noMatch")
256                 .visitLabel(noMatch)
257                 .pop(javaType)
258                 .push(false)
259                 .gotoLabel(end);
260         block.append(noMatchBlock);
261 
262         block.visitLabel(end);
263 
264         return block;
265     }

267     public static boolean isInteger(long value)
268     {
269         return value == (int) value;
270     }

272     private BytecodeBlock buildInCase(BytecodeGeneratorContext generatorContext,
273             Scope scope,
274             Type type,
275             LabelNode caseLabel,
276             LabelNode matchLabel,
277             LabelNode noMatchLabel,
278             Collection<BytecodeNode> testValues,
279             boolean checkForNulls)
280     {
281         Variable caseWasNull = null;
282         if (checkForNulls) {
283             caseWasNull = scope.createTempVariable(boolean.class);
284         }
285 
286         BytecodeBlock caseBlock = new BytecodeBlock()
287                 .visitLabel(caseLabel);
288 
289         if (checkForNulls) {
290             caseBlock.putVariable(caseWasNull, false);
291         }
292 
293         LabelNode elseLabel = new LabelNode("else");
294         BytecodeBlock elseBlock = new BytecodeBlock()
295                 .visitLabel(elseLabel);
296 
297         Variable wasNull = generatorContext.wasNull();
298         if (checkForNulls) {
299             elseBlock.append(wasNull.set(caseWasNull));
300         }
301 
302         elseBlock.gotoLabel(noMatchLabel);
303 
304         ScalarFunctionImplementation operator = generatorContext.getRegistry().getScalarFunctionImplementation(internalOperator(EQUAL, BOOLEAN, ImmutableList.of(type, type)));
305 
306         Binding equalsFunction = generatorContext
307                 .getCallSiteBinder()
308                 .bind(operator.getMethodHandle());
309 
310         BytecodeNode elseNode = elseBlock;
311         for (BytecodeNode testNode : testValues) {
312             LabelNode testLabel = new LabelNode("test");
313             IfStatement test = new IfStatement();
314 
315             test.condition()
316                     .visitLabel(testLabel)
317                     .dup(type.getJavaType())
318                     .append(testNode);
319 
320             if (checkForNulls) {
321                 test.condition()
322                         .append(wasNull)
323                         .putVariable(caseWasNull)
324                         .append(ifWasNullPopAndGoto(scope, elseLabel, void.class, type.getJavaType(), type.getJavaType()));
325             }
326             test.condition()
327                     .append(invoke(equalsFunction, EQUAL.name()));
328 
329             test.ifTrue().gotoLabel(matchLabel);
330             test.ifFalse(elseNode);
331 
332             elseNode = test;
333             elseLabel = testLabel;
334         }
335         caseBlock.append(elseNode);
336         return caseBlock;
337     }
