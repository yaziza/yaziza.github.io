52      @ScalarOperator(OperatorType.CAST)
53      @SqlType(RegexpType.NAME)
54      public static Regex castToRegexp(@SqlType(StandardTypes.VARCHAR) Slice pattern)
55      {
56          Regex regex;
57          try {
58              // When normal UTF8 encoding instead of non-strict UTF8) is used, joni can infinite loop when invalid UTF8 slice is supplied to it.
59              regex = new Regex(pattern.getBytes(), 0, pattern.length(), Option.DEFAULT, NonStrictUTF8Encoding.INSTANCE, Syntax.Java);
60          }
61          catch (Exception e) {
62              throw new PrestoException(INVALID_FUNCTION_ARGUMENT, e);
63          }
64          return regex;
65      }

85      @Description("replaces substrings matching a regular expression by given string")
86      @ScalarFunction
87      @SqlType(StandardTypes.VARCHAR)
88      public static Slice regexpReplace(@SqlType(StandardTypes.VARCHAR) Slice source, @SqlType(RegexpType.NAME) Regex pattern, @SqlType(StandardTypes.VARCHAR) Slice replacement)
89      {
90          Matcher matcher = pattern.matcher(source.getBytes());
91          SliceOutput sliceOutput = new DynamicSliceOutput(source.length() + replacement.length() * 5);
92  
93          int lastEnd = 0;
94          int nextStart = 0; // nextStart is the same as lastEnd, unless the last match was zero-width. In such case, nextStart is lastEnd + 1.
95          while (true) {
96              int offset = matcher.search(nextStart, source.length(), Option.DEFAULT);
97              if (offset == -1) {
98                  break;
99              }
100             if (matcher.getEnd() == matcher.getBegin()) {
101                 nextStart = matcher.getEnd() + 1;
102             }
103             else {
104                 nextStart = matcher.getEnd();
105             }
106             Slice sliceBetweenReplacements = source.slice(lastEnd, matcher.getBegin() - lastEnd);
107             lastEnd = matcher.getEnd();
108             sliceOutput.appendBytes(sliceBetweenReplacements);
109             appendReplacement(sliceOutput, source, pattern, matcher.getEagerRegion(), replacement);
110         }
111         sliceOutput.appendBytes(source.slice(lastEnd, source.length() - lastEnd));
112 
113         return sliceOutput.slice();
114     }

116     private static void appendReplacement(SliceOutput result, Slice source, Regex pattern, Region region, Slice replacement)
117     {
118         // Handle the following items:
119         // 1. ${name};
120         // 2. $0, $1, $123 (group 123, if exists; or group 12, if exists; or group 1);
121         // 3. \\, \$, \t (literal 't').
122         // 4. Anything that doesn't starts with \ or $ is considered regular bytes
123 
124         int idx = 0;
125 
126         while (idx < replacement.length()) {
127             byte nextByte = replacement.getByte(idx);
128             if (nextByte == '$') {
129                 idx++;
130                 if (idx == replacement.length()) { // not using checkArgument because `.toStringUtf8` is expensive
131                     throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Illegal replacement sequence: " + replacement.toStringUtf8());
132                 }
133                 nextByte = replacement.getByte(idx);
134                 int backref;
135                 if (nextByte == '{') { // case 1 in the above comment
136                     idx++;
137                     int startCursor = idx;
138                     while (idx < replacement.length()) {
139                         nextByte = replacement.getByte(idx);
140                         if (nextByte == '}') {
141                             break;
142                         }
143                         idx++;
144                     }
145                     byte[] groupName = replacement.getBytes(startCursor, idx - startCursor);
146                     try {
147                         backref = pattern.nameToBackrefNumber(groupName, 0, groupName.length, region);
148                     }
149                     catch (ValueException e) {
150                         throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Illegal replacement sequence: unknown group { " + new String(groupName, StandardCharsets.UTF_8) + " }");
151                     }
152                     idx++;
153                 }
154                 else { // case 2 in the above comment
155                     backref = nextByte - '0';
156                     if (backref < 0 || backref > 9) { // not using checkArgument because `.toStringUtf8` is expensive
157                         throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Illegal replacement sequence: " + replacement.toStringUtf8());
158                     }
159                     if (region.numRegs <= backref) {
160                         throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Illegal replacement sequence: unknown group " + backref);
161                     }
162                     idx++;
163                     while (idx < replacement.length()) { // Adaptive group number: find largest group num that is not greater than actual number of groups
164                         int nextDigit = replacement.getByte(idx) - '0';
165                         if (nextDigit < 0 || nextDigit > 9) {
166                             break;
167                         }
168                         int newBackref = (backref * 10) + nextDigit;
169                         if (region.numRegs <= newBackref) {
170                             break;
171                         }
172                         backref = newBackref;
173                         idx++;
174                     }
175                 }
176                 int beg = region.beg[backref];
177                 int end = region.end[backref];
178                 if (beg != -1 && end != -1) { // the specific group doesn't exist in the current match, skip
179                     result.appendBytes(source.slice(beg, end - beg));
180                 }
181             }
182             else { // case 3 and 4 in the above comment
183                 if (nextByte == '\\') {
184                     idx++;
185                     if (idx == replacement.length()) { // not using checkArgument because `.toStringUtf8` is expensive
186                         throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Illegal replacement sequence: " + replacement.toStringUtf8());
187                     }
188                     nextByte = replacement.getByte(idx);
189                 }
190                 result.appendByte(nextByte);
191                 idx++;
192             }
193         }
194     }

204     @Description("group(s) extracted using the given pattern")
205     @ScalarFunction
206     @SqlType("array<varchar>")
207     public static Block regexpExtractAll(@SqlType(StandardTypes.VARCHAR) Slice source, @SqlType(RegexpType.NAME) Regex pattern, @SqlType(StandardTypes.BIGINT) long groupIndex)
208     {
209         Matcher matcher = pattern.matcher(source.getBytes());
210         validateGroup(groupIndex, matcher.getEagerRegion());
211         BlockBuilder blockBuilder = VARCHAR.createBlockBuilder(new BlockBuilderStatus(), 32);
212         int group = Ints.checkedCast(groupIndex);
213 
214         int nextStart = 0;
215         while (true) {
216             int offset = matcher.search(nextStart, source.length(), Option.DEFAULT);
217             if (offset == -1) {
218                 break;
219             }
220             if (matcher.getEnd() == matcher.getBegin()) {
221                 nextStart = matcher.getEnd() + 1;
222             }
223             else {
224                 nextStart = matcher.getEnd();
225             }
226             Region region = matcher.getEagerRegion();
227             int beg = region.beg[group];
228             int end = region.end[group];
229             if (beg == -1 || end == -1) {
230                 blockBuilder.appendNull();
231             }
232             else {
233                 Slice slice = source.slice(beg, end - beg);
234                 VARCHAR.writeSlice(blockBuilder, slice);
235             }
236         }
237         return blockBuilder.build();
238     }

249     @Nullable
250     @Description("returns regex group of extracted string with a pattern")
251     @ScalarFunction
252     @SqlType(StandardTypes.VARCHAR)
253     public static Slice regexpExtract(@SqlType(StandardTypes.VARCHAR) Slice source, @SqlType(RegexpType.NAME) Regex pattern, @SqlType(StandardTypes.BIGINT) long groupIndex)
254     {
255         Matcher matcher = pattern.matcher(source.getBytes());
256         validateGroup(groupIndex, matcher.getEagerRegion());
257         int group = Ints.checkedCast(groupIndex);
258 
259         int offset = matcher.search(0, source.length(), Option.DEFAULT);
260         if (offset == -1) {
261             return null;
262         }
263         Region region = matcher.getEagerRegion();
264         int beg = region.beg[group];
265         int end = region.end[group];
266         if (beg == -1) {
267             // end == -1 must be true
268             return null;
269         }
270 
271         Slice slice = source.slice(beg, end - beg);
272         return slice;
273     }

275     @ScalarFunction
276     @Description("returns array of strings split by pattern")
277     @SqlType("array<varchar>")
278     public static Block regexpSplit(@SqlType(StandardTypes.VARCHAR) Slice source, @SqlType(RegexpType.NAME) Regex pattern)
279     {
280         Matcher matcher = pattern.matcher(source.getBytes());
281         BlockBuilder blockBuilder = VARCHAR.createBlockBuilder(new BlockBuilderStatus(), 32);
282 
283         int lastEnd = 0;
284         int nextStart = 0;
285         while (true) {
286             int offset = matcher.search(nextStart, source.length(), Option.DEFAULT);
287             if (offset == -1) {
288                 break;
289             }
290             if (matcher.getEnd() == matcher.getBegin()) {
291                 nextStart = matcher.getEnd() + 1;
292             }
293             else {
294                 nextStart = matcher.getEnd();
295             }
296             Slice slice = source.slice(lastEnd, matcher.getBegin() - lastEnd);
297             lastEnd = matcher.getEnd();
298             VARCHAR.writeSlice(blockBuilder, slice);
299         }
300         VARCHAR.writeSlice(blockBuilder, source.slice(lastEnd, source.length() - lastEnd));
301 
302         return blockBuilder.build();
303     }

305     private static void validateGroup(long group, Region region)
306     {
307         if (group < 0) {
308             throw new PrestoException(INVALID_FUNCTION_ARGUMENT, "Group cannot be negative");
309         }
310         if (group > region.numRegs - 1) {
311             throw new PrestoException(INVALID_FUNCTION_ARGUMENT, format("Pattern has %d groups. Cannot access group %d", region.numRegs - 1, group));
312         }
313     }
