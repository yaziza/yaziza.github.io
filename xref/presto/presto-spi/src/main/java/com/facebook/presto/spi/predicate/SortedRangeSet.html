37  /**
38   * A set containing zero or more Ranges of the same type over a continuous space of possible values.
39   * Ranges are coalesced into the most compact representation of non-overlapping Ranges. This structure
40   * allows iteration across these compacted Ranges in increasing order, as well as other common
41   * set-related operation.
42   */
43  public final class SortedRangeSet
44          implements ValueSet
45  {
46      private final Type type;
47      private final NavigableMap<Marker, Range> lowIndexedRanges;
48  
49      private SortedRangeSet(Type type, NavigableMap<Marker, Range> lowIndexedRanges)
50      {
51          requireNonNull(type, "type is null");
52          requireNonNull(lowIndexedRanges, "lowIndexedRanges is null");
53  
54          if (!type.isOrderable()) {
55              throw new IllegalArgumentException("Type is not orderable: " + type);
56          }
57          this.type = type;
58          this.lowIndexedRanges = lowIndexedRanges;
59      }
60  
61      static SortedRangeSet none(Type type)
62      {
63          return copyOf(type, Collections.<Range>emptyList());
64      }
65  
66      static SortedRangeSet all(Type type)
67      {
68          return copyOf(type, Collections.singletonList(Range.all(type)));
69      }
70  
71      /**
72       * Provided discrete values that are unioned together to form the SortedRangeSet
73       */
74      static SortedRangeSet of(Type type, Object first, Object... rest)
75      {
76          List<Range> ranges = new ArrayList<>(rest.length + 1);
77          ranges.add(Range.equal(type, first));
78          for (Object value : rest) {
79              ranges.add(Range.equal(type, value));
80          }
81          return copyOf(type, ranges);
82      }
83  
84      /**
85       * Provided Ranges are unioned together to form the SortedRangeSet
86       */
87      static SortedRangeSet of(Range first, Range... rest)
88      {
89          List<Range> rangeList = new ArrayList<>(rest.length + 1);
90          rangeList.add(first);
91          for (Range range : rest) {
92              rangeList.add(range);
93          }
94          return copyOf(first.getType(), rangeList);
95      }
96  
97      /**
98       * Provided Ranges are unioned together to form the SortedRangeSet
99       */
100     static SortedRangeSet copyOf(Type type, Iterable<Range> ranges)
101     {
102         return new Builder(type).addAll(ranges).build();
103     }

108             @JsonProperty("ranges") List<Range> ranges)

149     @Override
150     public Object getSingleValue()
151     {
152         if (!isSingleValue()) {
153             throw new IllegalStateException("SortedRangeSet does not have just a single value");
154         }
155         return lowIndexedRanges.values().iterator().next().getSingleValue();
156     }
157 
158     @Override
159     public boolean containsValue(Object value)
160     {
161         return includesMarker(Marker.exactly(type, value));
162     }
163 
164     boolean includesMarker(Marker marker)
165     {
166         requireNonNull(marker, "marker is null");
167         checkTypeCompatibility(marker);
168 
169         Map.Entry<Marker, Range> floorEntry = lowIndexedRanges.floorEntry(marker);
170         return floorEntry != null && floorEntry.getValue().includes(marker);
171     }
172 
173     public Range getSpan()
174     {
175         if (lowIndexedRanges.isEmpty()) {
176             throw new IllegalStateException("Can not get span if no ranges exist");
177         }
178         return lowIndexedRanges.firstEntry().getValue().span(lowIndexedRanges.lastEntry().getValue());
179     }

225     @Override
226     public SortedRangeSet intersect(ValueSet other)
227     {
228         SortedRangeSet otherRangeSet = checkCompatibility(other);
229 
230         Builder builder = new Builder(type);
231 
232         Iterator<Range> iterator1 = getOrderedRanges().iterator();
233         Iterator<Range> iterator2 = otherRangeSet.getOrderedRanges().iterator();
234 
235         if (iterator1.hasNext() && iterator2.hasNext()) {
236             Range range1 = iterator1.next();
237             Range range2 = iterator2.next();
238 
239             while (true) {
240                 if (range1.overlaps(range2)) {
241                     builder.add(range1.intersect(range2));
242                 }
243 
244                 if (range1.getHigh().compareTo(range2.getHigh()) <= 0) {
245                     if (!iterator1.hasNext()) {
246                         break;
247                     }
248                     range1 = iterator1.next();
249                 }
250                 else {
251                     if (!iterator2.hasNext()) {
252                         break;
253                     }
254                     range2 = iterator2.next();
255                 }
256             }
257         }
258 
259         return builder.build();
260     }

262     @Override
263     public SortedRangeSet union(ValueSet other)
264     {
265         SortedRangeSet otherRangeSet = checkCompatibility(other);
266         return new Builder(type)
267                 .addAll(this.getOrderedRanges())
268                 .addAll(otherRangeSet.getOrderedRanges())
269                 .build();
270     }
271 
272     @Override
273     public SortedRangeSet union(Collection<ValueSet> valueSets)
274     {
275         Builder builder = new Builder(type);
276         builder.addAll(this.getOrderedRanges());
277         for (ValueSet valueSet : valueSets) {
278             builder.addAll(checkCompatibility(valueSet).getOrderedRanges());
279         }
280         return builder.build();
281     }
282 
283     @Override
284     public SortedRangeSet complement()
285     {
286         Builder builder = new Builder(type);
287 
288         if (lowIndexedRanges.isEmpty()) {
289             return builder.add(Range.all(type)).build();
290         }
291 
292         Iterator<Range> rangeIterator = lowIndexedRanges.values().iterator();
293 
294         Range firstRange = rangeIterator.next();
295         if (!firstRange.getLow().isLowerUnbounded()) {
296             builder.add(new Range(Marker.lowerUnbounded(type), firstRange.getLow().lesserAdjacent()));
297         }
298 
299         Range previousRange = firstRange;
300         while (rangeIterator.hasNext()) {
301             Range currentRange = rangeIterator.next();
302 
303             Marker lowMarker = previousRange.getHigh().greaterAdjacent();
304             Marker highMarker = currentRange.getLow().lesserAdjacent();
305             builder.add(new Range(lowMarker, highMarker));
306 
307             previousRange = currentRange;
308         }
309 
310         Range lastRange = previousRange;
311         if (!lastRange.getHigh().isUpperUnbounded()) {
312             builder.add(new Range(lastRange.getHigh().greaterAdjacent(), Marker.upperUnbounded(type)));
313         }
314 
315         return builder.build();
316     }
317 
318     private SortedRangeSet checkCompatibility(ValueSet other)
319     {
320         if (!getType().equals(other.getType())) {
321             throw new IllegalStateException(String.format("Mismatched types: %s vs %s", getType(), other.getType()));
322         }
323         if (!(other instanceof SortedRangeSet)) {
324             throw new IllegalStateException(String.format("ValueSet is not a SortedRangeSet: %s", other.getClass()));
325         }
326         return (SortedRangeSet) other;
327     }
328 
329     private void checkTypeCompatibility(Marker marker)
330     {
331         if (!getType().equals(marker.getType())) {
332             throw new IllegalStateException(String.format("Marker of %s does not match SortedRangeSet of %s", marker.getType(), getType()));
333         }
334     }

342     @Override
343     public boolean equals(Object obj)
344     {
345         if (this == obj) {
346             return true;
347         }
348         if (obj == null || getClass() != obj.getClass()) {
349             return false;
350         }
351         final SortedRangeSet other = (SortedRangeSet) obj;
352         return Objects.equals(this.lowIndexedRanges, other.lowIndexedRanges);
353     }

355     @Override
356     public String toString(ConnectorSession session)
357     {
358         return "[" + lowIndexedRanges.values().stream()
359                 .map(range -> range.toString(session))
360                 .collect(Collectors.joining(", ")) + "]";
361     }
362 
363     static class Builder
364     {
365         private final Type type;
366         private final List<Range> ranges = new ArrayList<>();
367 
368         Builder(Type type)
369         {
370             requireNonNull(type, "type is null");
371 
372             if (!type.isOrderable()) {
373                 throw new IllegalArgumentException("Type is not orderable: " + type);
374             }
375             this.type = type;
376         }
377 
378         Builder add(Range range)
379         {
380             if (!type.equals(range.getType())) {
381                 throw new IllegalArgumentException(String.format("Range type %s does not match builder type %s", range.getType(), type));
382             }
383 
384             ranges.add(range);
385             return this;
386         }
387 
388         Builder addAll(Iterable<Range> ranges)
389         {
390             for (Range range : ranges) {
391                 add(range);
392             }
393             return this;
394         }
395 
396         SortedRangeSet build()
397         {
398             Collections.sort(ranges, Comparator.comparing(Range::getLow));
399 
400             NavigableMap<Marker, Range> result = new TreeMap<>();
401 
402             Range current = null;
403             for (Range next : ranges) {
404                 if (current == null) {
405                     current = next;
406                     continue;
407                 }
408 
409                 if (current.overlaps(next) || current.getHigh().isAdjacent(next.getLow())) {
410                     current = current.span(next);
411                 }
412                 else {
413                     result.put(current.getLow(), current);
414                     current = next;
415                 }
416             }
417 
418             if (current != null) {
419                 result.put(current.getLow(), current);
420             }
421 
422             return new SortedRangeSet(type, result);
423         }
424     }
