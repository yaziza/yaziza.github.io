98      public Stripe readStripe(StripeInformation stripe, AggregatedMemoryContext systemMemoryUsage)
99              throws IOException
100     {
101         // read the stripe footer
102         StripeFooter stripeFooter = readStripeFooter(stripe, systemMemoryUsage);
103         List<ColumnEncoding> columnEncodings = stripeFooter.getColumnEncodings();
104 
105         // get streams for selected columns
106         Map<StreamId, Stream> streams = new HashMap<>();
107         boolean hasRowGroupDictionary = false;
108         for (Stream stream : stripeFooter.getStreams()) {
109             if (includedOrcColumns.contains(stream.getColumn())) {
110                 streams.put(new StreamId(stream), stream);
111 
112                 ColumnEncodingKind columnEncoding = columnEncodings.get(stream.getColumn()).getColumnEncodingKind();
113                 if (columnEncoding == DICTIONARY && stream.getStreamKind() == StreamKind.IN_DICTIONARY) {
114                     hasRowGroupDictionary = true;
115                 }
116             }
117         }
118 
119         // handle stripes with more than one row group or a dictionary
120         if ((stripe.getNumberOfRows() > rowsInRowGroup) || hasRowGroupDictionary) {
121             // determine ranges of the stripe to read
122             Map<StreamId, DiskRange> diskRanges = getDiskRanges(stripeFooter.getStreams());
123             diskRanges = Maps.filterKeys(diskRanges, Predicates.in(streams.keySet()));
124 
125             // read the file regions
126             Map<StreamId, OrcInputStream> streamsData = readDiskRanges(stripe.getOffset(), diskRanges, systemMemoryUsage);
127 
128             // read the row index for each column
129             Map<Integer, List<RowGroupIndex>> columnIndexes = readColumnIndexes(streams, streamsData);
130 
131             // select the row groups matching the tuple domain
132             Set<Integer> selectedRowGroups = selectRowGroups(stripe, columnIndexes);
133 
134             // if all row groups are skipped, return null
135             if (selectedRowGroups.isEmpty()) {
136                 // set accounted memory usage to zero
137                 systemMemoryUsage.close();
138                 return null;
139             }
140 
141             // value streams
142             Map<StreamId, ValueStream<?>> valueStreams = createValueStreams(streams, streamsData, columnEncodings);
143 
144             // build the dictionary streams
145             StreamSources dictionaryStreamSources = createDictionaryStreamSources(streams, valueStreams, columnEncodings);
146 
147             // build the row groups
148             try {
149                 List<RowGroup> rowGroups = createRowGroups(
150                         stripe.getNumberOfRows(),
151                         streams,
152                         valueStreams,
153                         columnIndexes,
154                         selectedRowGroups,
155                         columnEncodings);
156 
157                 return new Stripe(stripe.getNumberOfRows(), columnEncodings, rowGroups, dictionaryStreamSources);
158             }
159             catch (InvalidCheckpointException e) {
160                 // The ORC file contains a corrupt checkpoint stream
161                 // If the file does not have a row group dictionary, treat the stripe as a single row group. Otherwise,
162                 // we must fail because the length of the row group dictionary is contained in the checkpoint stream.
163                 if (hasRowGroupDictionary) {
164                     throw new OrcCorruptionException(e, "ORC file %s has corrupt checkpoints", orcDataSource);
165                 }
166             }
167         }
168 
169         // stripe only has one row group and no dictionary
170         ImmutableMap.Builder<StreamId, DiskRange> diskRangesBuilder = ImmutableMap.builder();
171         for (Entry<StreamId, DiskRange> entry : getDiskRanges(stripeFooter.getStreams()).entrySet()) {
172             StreamId streamId = entry.getKey();
173             if (streamId.getStreamKind() != ROW_INDEX && streams.keySet().contains(streamId)) {
174                 diskRangesBuilder.put(entry);
175             }
176         }
177         ImmutableMap<StreamId, DiskRange> diskRanges = diskRangesBuilder.build();
178 
179         // read the file regions
180         Map<StreamId, OrcInputStream> streamsData = readDiskRanges(stripe.getOffset(), diskRanges, systemMemoryUsage);
181 
182         // value streams
183         Map<StreamId, ValueStream<?>> valueStreams = createValueStreams(streams, streamsData, columnEncodings);
184 
185         // build the dictionary streams
186         StreamSources dictionaryStreamSources = createDictionaryStreamSources(streams, valueStreams, columnEncodings);
187 
188         // build the row group
189         ImmutableMap.Builder<StreamId, StreamSource<?>> builder = ImmutableMap.builder();
190         for (Entry<StreamId, ValueStream<?>> entry : valueStreams.entrySet()) {
191             builder.put(entry.getKey(), new ValueStreamSource<>(entry.getValue()));
192         }
193         RowGroup rowGroup = new RowGroup(0, 0, stripe.getNumberOfRows(), new StreamSources(builder.build()));
194 
195         return new Stripe(stripe.getNumberOfRows(), columnEncodings, ImmutableList.of(rowGroup), dictionaryStreamSources);
196     }

198     public Map<StreamId, OrcInputStream> readDiskRanges(long stripeOffset, Map<StreamId, DiskRange> diskRanges, AbstractAggregatedMemoryContext systemMemoryUsage)
199             throws IOException
200     {
201         //
202         // Note: this code does not use the Java 8 stream APIs to avoid any extra object allocation
203         //
204 
205         // transform ranges to have an absolute offset in file
206         ImmutableMap.Builder<StreamId, DiskRange> diskRangesBuilder = ImmutableMap.builder();
207         for (Entry<StreamId, DiskRange> entry : diskRanges.entrySet()) {
208             DiskRange diskRange = entry.getValue();
209             diskRangesBuilder.put(entry.getKey(), new DiskRange(stripeOffset + diskRange.getOffset(), diskRange.getLength()));
210         }
211         diskRanges = diskRangesBuilder.build();
212 
213         // read ranges
214         Map<StreamId, FixedLengthSliceInput> streamsData = orcDataSource.readFully(diskRanges);
215 
216         // transform streams to OrcInputStream
217         String sourceName = orcDataSource.toString();
218         ImmutableMap.Builder<StreamId, OrcInputStream> streamsBuilder = ImmutableMap.builder();
219         for (Entry<StreamId, FixedLengthSliceInput> entry : streamsData.entrySet()) {
220             streamsBuilder.put(entry.getKey(), new OrcInputStream(sourceName, entry.getValue(), compressionKind, bufferSize, systemMemoryUsage));
221         }
222         return streamsBuilder.build();
223     }

225     private Map<StreamId, ValueStream<?>> createValueStreams(Map<StreamId, Stream> streams, Map<StreamId, OrcInputStream> streamsData, List<ColumnEncoding> columnEncodings)
226     {
227         ImmutableMap.Builder<StreamId, ValueStream<?>> valueStreams = ImmutableMap.builder();
228         for (Entry<StreamId, Stream> entry : streams.entrySet()) {
229             StreamId streamId = entry.getKey();
230             Stream stream = entry.getValue();
231             ColumnEncodingKind columnEncoding = columnEncodings.get(stream.getColumn()).getColumnEncodingKind();
232 
233             // skip index and empty streams
234             if (isIndexStream(stream) || stream.getLength() == 0) {
235                 continue;
236             }
237 
238             OrcInputStream inputStream = streamsData.get(streamId);
239             OrcTypeKind columnType = types.get(stream.getColumn()).getOrcTypeKind();
240 
241             valueStreams.put(streamId, ValueStreams.createValueStreams(streamId, inputStream, columnType, columnEncoding, stream.isUseVInts()));
242         }
243         return valueStreams.build();
244     }

246     public StreamSources createDictionaryStreamSources(Map<StreamId, Stream> streams, Map<StreamId, ValueStream<?>> valueStreams, List<ColumnEncoding> columnEncodings)
247     {
248         ImmutableMap.Builder<StreamId, StreamSource<?>> dictionaryStreamBuilder = ImmutableMap.builder();
249         for (Entry<StreamId, Stream> entry : streams.entrySet()) {
250             StreamId streamId = entry.getKey();
251             Stream stream = entry.getValue();
252             int column = stream.getColumn();
253 
254             // only process dictionary streams
255             ColumnEncodingKind columnEncoding = columnEncodings.get(column).getColumnEncodingKind();
256             if (!isDictionary(stream, columnEncoding)) {
257                 continue;
258             }
259 
260             // skip streams without data
261             ValueStream<?> valueStream = valueStreams.get(streamId);
262             if (valueStream == null) {
263                 continue;
264             }
265 
266             OrcTypeKind columnType = types.get(stream.getColumn()).getOrcTypeKind();
267             StreamCheckpoint streamCheckpoint = getDictionaryStreamCheckpoint(streamId, columnType, columnEncoding);
268 
269             StreamSource<?> streamSource = createCheckpointStreamSource(valueStream, streamCheckpoint);
270             dictionaryStreamBuilder.put(streamId, streamSource);
271         }
272         return new StreamSources(dictionaryStreamBuilder.build());
273     }

275     private List<RowGroup> createRowGroups(
276             int rowsInStripe,
277             Map<StreamId, Stream> streams,
278             Map<StreamId, ValueStream<?>> valueStreams,
279             Map<Integer, List<RowGroupIndex>> columnIndexes,
280             Set<Integer> selectedRowGroups,
281             List<ColumnEncoding> encodings)
282             throws InvalidCheckpointException
283     {
284         ImmutableList.Builder<RowGroup> rowGroupBuilder = ImmutableList.builder();
285 
286         for (int rowGroupId : selectedRowGroups) {
287             Map<StreamId, StreamCheckpoint> checkpoints = getStreamCheckpoints(includedOrcColumns, types, compressionKind, rowGroupId, encodings, streams, columnIndexes);
288             int rowOffset = rowGroupId * rowsInRowGroup;
289             int rowsInGroup = Math.min(rowsInStripe - rowOffset, rowsInRowGroup);
290             rowGroupBuilder.add(createRowGroup(rowGroupId, rowOffset, rowsInGroup, valueStreams, checkpoints));
291         }
292 
293         return rowGroupBuilder.build();
294     }

296     public static RowGroup createRowGroup(int groupId, int rowOffset, int rowCount, Map<StreamId, ValueStream<?>> valueStreams, Map<StreamId, StreamCheckpoint> checkpoints)
297     {
298         ImmutableMap.Builder<StreamId, StreamSource<?>> builder = ImmutableMap.builder();
299         for (Entry<StreamId, StreamCheckpoint> entry : checkpoints.entrySet()) {
300             StreamId streamId = entry.getKey();
301             StreamCheckpoint checkpoint = entry.getValue();
302 
303             // skip streams without data
304             ValueStream<?> valueStream = valueStreams.get(streamId);
305             if (valueStream == null) {
306                 continue;
307             }
308 
309             builder.put(streamId, createCheckpointStreamSource(valueStream, checkpoint));
310         }
311         StreamSources rowGroupStreams = new StreamSources(builder.build());
312         return new RowGroup(groupId, rowOffset, rowCount, rowGroupStreams);
313     }

315     public StripeFooter readStripeFooter(StripeInformation stripe, AbstractAggregatedMemoryContext systemMemoryUsage)
316             throws IOException
317     {
318         long offset = stripe.getOffset() + stripe.getIndexLength() + stripe.getDataLength();
319         int tailLength = Ints.checkedCast(stripe.getFooterLength());
320 
321         // read the footer
322         byte[] tailBuffer = new byte[tailLength];
323         orcDataSource.readFully(offset, tailBuffer);
324         try (InputStream inputStream = new OrcInputStream(orcDataSource.toString(), Slices.wrappedBuffer(tailBuffer).getInput(), compressionKind, bufferSize, systemMemoryUsage)) {
325             return metadataReader.readStripeFooter(types, inputStream);
326         }
327     }

329     private Map<Integer, List<RowGroupIndex>> readColumnIndexes(Map<StreamId, Stream> streams, Map<StreamId, OrcInputStream> streamsData)
330             throws IOException
331     {
332         ImmutableMap.Builder<Integer, List<RowGroupIndex>> columnIndexes = ImmutableMap.builder();
333         for (Entry<StreamId, Stream> entry : streams.entrySet()) {
334             Stream stream = entry.getValue();
335             if (stream.getStreamKind() == ROW_INDEX) {
336                 OrcInputStream inputStream = streamsData.get(entry.getKey());
337                 columnIndexes.put(stream.getColumn(), metadataReader.readRowIndexes(inputStream));
338             }
339         }
340         return columnIndexes.build();
341     }

343     private Set<Integer> selectRowGroups(StripeInformation stripe,  Map<Integer, List<RowGroupIndex>> columnIndexes)
344             throws IOException
345     {
346         int rowsInStripe = Ints.checkedCast(stripe.getNumberOfRows());
347         int groupsInStripe = ceil(rowsInStripe, rowsInRowGroup);
348 
349         ImmutableSet.Builder<Integer> selectedRowGroups = ImmutableSet.builder();
350         int remainingRows = rowsInStripe;
351         for (int rowGroup = 0; rowGroup < groupsInStripe; ++rowGroup) {
352             int rows = Math.min(remainingRows, rowsInRowGroup);
353             Map<Integer, ColumnStatistics> statistics = getRowGroupStatistics(types.get(0), columnIndexes, rowGroup);
354             if (predicate.matches(rows, statistics)) {
355                 selectedRowGroups.add(rowGroup);
356             }
357             remainingRows -= rows;
358         }
359         return selectedRowGroups.build();
360     }

362     private static Map<Integer, ColumnStatistics> getRowGroupStatistics(OrcType rootStructType, Map<Integer, List<RowGroupIndex>> columnIndexes, int rowGroup)
363     {
364         requireNonNull(rootStructType, "rootStructType is null");
365         checkArgument(rootStructType.getOrcTypeKind() == OrcTypeKind.STRUCT);
366         requireNonNull(columnIndexes, "columnIndexes is null");
367         checkArgument(rowGroup >= 0, "rowGroup is negative");
368 
369         ImmutableMap.Builder<Integer, ColumnStatistics> statistics = ImmutableMap.builder();
370         for (int ordinal = 0; ordinal < rootStructType.getFieldCount(); ordinal++) {
371             List<RowGroupIndex> rowGroupIndexes = columnIndexes.get(rootStructType.getFieldTypeIndex(ordinal));
372             if (rowGroupIndexes != null) {
373                 statistics.put(ordinal, rowGroupIndexes.get(rowGroup).getColumnStatistics());
374             }
375         }
376         return statistics.build();
377     }

379     private static boolean isIndexStream(Stream stream)
380     {
381         return stream.getStreamKind() == ROW_INDEX || stream.getStreamKind() == DICTIONARY_COUNT;
382     }
383 
384     private static boolean isDictionary(Stream stream, ColumnEncodingKind columnEncoding)
385     {
386         return stream.getStreamKind() == DICTIONARY_DATA || (stream.getStreamKind() == LENGTH && (columnEncoding == DICTIONARY || columnEncoding == DICTIONARY_V2));
387     }

389     private static Map<StreamId, DiskRange> getDiskRanges(List<Stream> streams)
390     {
391         ImmutableMap.Builder<StreamId, DiskRange> streamDiskRanges = ImmutableMap.builder();
392         long stripeOffset = 0;
393         for (Stream stream : streams) {
394             int streamLength = Ints.checkedCast(stream.getLength());
395             streamDiskRanges.put(new StreamId(stream), new DiskRange(stripeOffset, streamLength));
396             stripeOffset += streamLength;
397         }
398         return streamDiskRanges.build();
399     }

401     private static Set<Integer> getIncludedOrcColumns(List<OrcType> types, Set<Integer> includedColumns)
402     {
403         Set<Integer> includes = new LinkedHashSet<>();
404 
405         OrcType root = types.get(0);
406         for (int includedColumn : includedColumns) {
407             includeOrcColumnsRecursive(types, includes, root.getFieldTypeIndex(includedColumn));
408         }
409 
410         return includes;
411     }
412 
413     private static void includeOrcColumnsRecursive(List<OrcType> types, Set<Integer> result, int typeId)
414     {
415         result.add(typeId);
416         OrcType type = types.get(typeId);
417         int children = type.getFieldCount();
418         for (int i = 0; i < children; ++i) {
419             includeOrcColumnsRecursive(types, result, type.getFieldTypeIndex(i));
420         }
421     }

423     /**
424      * Ceiling of integer division
425      */
426     private static int ceil(int dividend, int divisor)
427     {
428         return ((dividend + divisor) - 1) / divisor;
