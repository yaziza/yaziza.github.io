37  /**
38   * An {@link InputStream} that converts characters from a {@link Reader} into bytes using an
39   * arbitrary Charset.
40   *
41   * <p>This is an alternative to copying the data to an {@code OutputStream} via a {@code Writer},
42   * which is necessarily blocking. By implementing an {@code InputStream} it allows consumers to
43   * "pull" as much data as they can handle, which is more convenient when dealing with flow
44   * controlled, async APIs.
45   *
46   * @author Chris Nokleberg
47   */
48  final class ReaderInputStream extends InputStream {
49    private final Reader reader;
50    private final CharsetEncoder encoder;
51    private final byte[] singleByte = new byte[1];
52  
53    /**
54     * charBuffer holds characters that have been read from the Reader but not encoded yet. The buffer
55     * is perpetually "flipped" (unencoded characters between position and limit).
56     */
57    private CharBuffer charBuffer;
58  
59    /**
60     * byteBuffer holds encoded characters that have not yet been sent to the caller of the input
61     * stream. When encoding it is "unflipped" (encoded bytes between 0 and position) and when
62     * draining it is flipped (undrained bytes between position and limit).
63     */
64    private ByteBuffer byteBuffer;
65  
66    /** Whether we've finished reading the reader. */
67    private boolean endOfInput;
68    /** Whether we're copying encoded bytes to the caller's buffer. */
69    private boolean draining;
70    /** Whether we've successfully flushed the encoder. */
71    private boolean doneFlushing;
72  
73    /**
74     * Creates a new input stream that will encode the characters from {@code reader} into bytes using
75     * the given character set. Malformed input and unmappable characters will be replaced.
76     *
77     * @param reader input source
78     * @param charset character set used for encoding chars to bytes
79     * @param bufferSize size of internal input and output buffers
80     * @throws IllegalArgumentException if bufferSize is non-positive
81     */
82    ReaderInputStream(Reader reader, Charset charset, int bufferSize) {
83      this(reader,
84          charset.newEncoder()
85              .onMalformedInput(CodingErrorAction.REPLACE)
86              .onUnmappableCharacter(CodingErrorAction.REPLACE),
87          bufferSize);
88    }
89  
90    /**
91     * Creates a new input stream that will encode the characters from {@code reader} into bytes using
92     * the given character set encoder.
93     *
94     * @param reader input source
95     * @param encoder character set encoder used for encoding chars to bytes
96     * @param bufferSize size of internal input and output buffers
97     * @throws IllegalArgumentException if bufferSize is non-positive
98     */
99    ReaderInputStream(Reader reader, CharsetEncoder encoder, int bufferSize) {
100     this.reader = checkNotNull(reader);
101     this.encoder = checkNotNull(encoder);
102     checkArgument(bufferSize > 0, "bufferSize must be positive: %s", bufferSize);
103     encoder.reset();
104 
105     charBuffer = CharBuffer.allocate(bufferSize);
106     charBuffer.flip();
107 
108     byteBuffer = ByteBuffer.allocate(bufferSize);
109   }
110 
111   @Override public void close() throws IOException {
112     reader.close();
113   }
114 
115   @Override public int read() throws IOException {
116     return (read(singleByte) == 1) ? UnsignedBytes.toInt(singleByte[0]) : -1;
117   }
118 
119   // TODO(chrisn): Consider trying to encode/flush directly to the argument byte
120   // buffer when possible.
121   @Override public int read(byte[] b, int off, int len) throws IOException {
122     // Obey InputStream contract.
123     checkPositionIndexes(off, off + len, b.length);
124     if (len == 0) {
125       return 0;
126     }
127 
128     // The rest of this method implements the process described by the CharsetEncoder javadoc.
129     int totalBytesRead = 0;
130     boolean doneEncoding = endOfInput;
131 
132     DRAINING: while (true) {
133       // We stay in draining mode until there are no bytes left in the output buffer.  Then we go
134       // back to encoding/flushing.
135       if (draining) {
136         totalBytesRead += drain(b, off + totalBytesRead, len - totalBytesRead);
137         if (totalBytesRead == len || doneFlushing) {
138           return (totalBytesRead > 0) ? totalBytesRead : -1;
139         }
140         draining = false;
141         byteBuffer.clear();
142       }
143 
144       while (true) {
145         // We call encode until there is no more input. The last call to encode will have endOfInput
146         // == true. Then there is a final call to flush.
147         CoderResult result;
148         if (doneFlushing) {
149           result = CoderResult.UNDERFLOW;
150         } else if (doneEncoding) {
151           result = encoder.flush(byteBuffer);
152         } else {
153           result = encoder.encode(charBuffer, byteBuffer, endOfInput);
154         }
155 
156         if (result.isOverflow()) {
157           // Not enough room in output buffer--drain it, creating a bigger buffer if necessary.
158           startDraining(true);
159           continue DRAINING;
160         } else if (result.isUnderflow()) {
161           // If encoder underflows, it means either:
162           // a) the final flush() succeeded; next drain (then done)
163           // b) we encoded all of the input; next flush
164           // c) we ran of out input to encode; next read more input
165           if (doneEncoding) { // (a)
166             doneFlushing = true;
167             startDraining(false);
168             continue DRAINING;
169           } else if (endOfInput) { // (b)
170             doneEncoding = true;
171           } else { // (c)
172             readMoreChars();
173           }
174         } else if (result.isError()) {
175           // Only reach here if a CharsetEncoder with non-REPLACE settings is used.
176           result.throwException();
177           return 0; // Not called.
178         }
179       }
180     }
181   }
182 
183   /** Returns a new CharBuffer identical to buf, except twice the capacity. */
184   private static CharBuffer grow(CharBuffer buf) {
185     char[] copy = Arrays.copyOf(buf.array(), buf.capacity() * 2);
186     CharBuffer bigger = CharBuffer.wrap(copy);
187     bigger.position(buf.position());
188     bigger.limit(buf.limit());
189     return bigger;
190   }
191 
192   /** Handle the case of underflow caused by needing more input characters. */
193   private void readMoreChars() throws IOException {
194     // Possibilities:
195     // 1) array has space available on right hand side (between limit and capacity)
196     // 2) array has space available on left hand side (before position)
197     // 3) array has no space available
198     //
199     // In case 2 we shift the existing chars to the left, and in case 3 we create a bigger
200     // array, then they both become case 1.
201 
202     if (availableCapacity(charBuffer) == 0) {
203       if (charBuffer.position() > 0) {
204         // (2) There is room in the buffer. Move existing bytes to the beginning.
205         charBuffer.compact().flip();
206       } else {
207         // (3) Entire buffer is full, need bigger buffer.
208         charBuffer = grow(charBuffer);
209       }
210     }
211 
212     // (1) Read more characters into free space at end of array.
213     int limit = charBuffer.limit();
214     int numChars = reader.read(charBuffer.array(), limit, availableCapacity(charBuffer));
215     if (numChars == -1) {
216       endOfInput = true;
217     } else {
218       charBuffer.limit(limit + numChars);
219     }
220   }
221 
222   /** Returns the number of elements between the limit and capacity. */
223   private static int availableCapacity(Buffer buffer) {
224     return buffer.capacity() - buffer.limit();
225   }
226 
227   /**
228    * Flips the buffer output buffer so we can start reading bytes from it.  If we are starting to
229    * drain because there was overflow, and there aren't actually any characters to drain, then the
230    * overflow must be due to a small output buffer.
231    */
232   private void startDraining(boolean overflow) {
233     byteBuffer.flip();
234     if (overflow && byteBuffer.remaining() == 0) {
235       byteBuffer = ByteBuffer.allocate(byteBuffer.capacity() * 2);
236     } else {
237       draining = true;
238     }
239   }
240 
241   /**
242    * Copy as much of the byte buffer into the output array as possible, returning the (positive)
243    * number of characters copied.
244    */
245   private int drain(byte[] b, int off, int len) {
246     int remaining = Math.min(len, byteBuffer.remaining());
247     byteBuffer.get(b, off, remaining);
248     return remaining;
249   }
