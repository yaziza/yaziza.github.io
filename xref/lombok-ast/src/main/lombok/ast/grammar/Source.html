97  	public String getOverviewProfileInformation() {
98  		clear();
99  		preProcess();
100 		ParserGroup group = new ParserGroup(this);
101 		ProfilerParseRunner<Node> runner = new ProfilerParseRunner<Node>(group.structures.compilationUnitEoi(), preprocessed);
102 		this.parsingResult = runner.run();
103 		StringBuilder out = new StringBuilder();
104 		out.append(runner.getOverviewReport());
105 		postProcess();
106 		return out.toString();

122 	private List<Integer> calculateLineEndings() {
123 		ImmutableList.Builder<Integer> builder = ImmutableList.builder();
124 		
125 		boolean atCR = false;
126 		for (int i = 0; i < rawInput.length(); i++) {
127 			char c = rawInput.charAt(i);
128 			if (c == '\n' && !atCR) builder.add(i);
129 			atCR = c == '\r';
130 			if (atCR) builder.add(i);
131 		}
132 		return builder.build();
133 	}

174 	private void postProcess() {
175 		for (ParseError error : parsingResult.parseErrors) {
176 			int errStart = error.getStartIndex();
177 			int errEnd = error.getEndIndex();
178 			problems.add(new ParseProblem(new Position(mapPosition(errStart), mapPosition(errEnd)), error.toString()));
179 		}
180 		
181 		if (parsingResult.parseTreeRoot != null) {
182 			nodes.add(parsingResult.parseTreeRoot.getValue());
183 			gatherComments(parsingResult.parseTreeRoot);
184 		}
185 		
186 		comments = Collections.unmodifiableList(comments);
187 		nodes = Collections.unmodifiableList(nodes);
188 		problems = Collections.unmodifiableList(problems);
189 		
190 		rtrimPositions(nodes, comments);
191 		
192 		//TODO Write test case with javadoc intermixed with empty declares.
193 		//TODO test javadoc on a package declaration.
194 		//TODO javadoc in between keywords.
195 		
196 		associateJavadoc(comments, nodes);
197 		
198 		fixPositions(nodes);
199 		fixPositions(comments);
200 		
201 		parsed = true;
202 	}

208 	public Map<Node, Collection<SourceStructure>> getSourceStructures() {
209 		if (cachedSourceStructures != null) return cachedSourceStructures;
210 		parseCompilationUnit();
211 		ListMultimap<Node, SourceStructure> map = LinkedListMultimap.create();
212 		
213 		org.parboiled.Node<Node> pNode = parsingResult.parseTreeRoot;
214 		
215 		buildSourceStructures(pNode, null, map);
216 		
217 		Map<Node, Collection<SourceStructure>> result = map.asMap();
218 		
219 		for (Collection<SourceStructure> structures : result.values()) {
220 			for (SourceStructure structure : structures) {
221 				structure.setPosition(new Position(
222 						mapPosition(structure.getPosition().getStart()),
223 						mapPosition(structure.getPosition().getEnd())));
224 			}
225 		}
226 		
227 		return cachedSourceStructures = result;
228 	}

230 	private void addSourceStructure(ListMultimap<Node, SourceStructure> map, Node node, SourceStructure structure) {
231 		if (structure.getPosition().size() > 0 && structure.getContent().trim().length() > 0 &&
232 				!structure.getPosition().equals(node.getPosition())) {
233 			
234 			map.put(node, structure);
235 		}
236 	}

238 	private void buildSourceStructures(org.parboiled.Node<Node> pNode, Node owner, ListMultimap<Node, SourceStructure> map) {
239 		Node target = registeredStructures.remove(pNode);
240 		if (target != null || pNode.getChildren().isEmpty()) {
241 			int start = pNode.getStartIndex();
242 			int end = pNode.getEndIndex();
243 			String text = preprocessed.substring(start, end);
244 			SourceStructure structure = new SourceStructure(new Position(start, end), text);
245 			if (target != null) addSourceStructure(map, target, structure);
246 			else if (pNode.getValue() != null && !(pNode.getValue() instanceof TemporaryNode)) addSourceStructure(map, pNode.getValue(), structure);
247 			else if (owner != null) addSourceStructure(map, owner, structure);
248 		} else {
249 			Node possibleOwner = pNode.getValue();
250 			if (possibleOwner instanceof TemporaryNode) possibleOwner = null;
251 			for (org.parboiled.Node<Node> child : pNode.getChildren()) {
252 				if (child.getValue() == null || child.getValue() instanceof TemporaryNode) continue;
253 				/* If the next if holds true, then we aren't the true generator; the child generated the node and this pNode adopted it */
254 				if (child.getValue() == possibleOwner) possibleOwner = null;
255 			}
256 			
257 			if (possibleOwner != null) owner = possibleOwner;
258 			
259 			for (org.parboiled.Node<Node> child : pNode.getChildren()) {
260 				buildSourceStructures(child, owner, map);
261 			}
262 		}
263 	}

265 	/**
266 	 * The end positions of all nodes include their trailing whitespace which isn't very convenient.
267 	 * We'll 'fix' the end marker of each node by trimming it back. This is somewhat complicated as comments also need to be trimmed across.
268 	 * We also adjust all positions to conform with the raw input (undoing any positional shifts caused by preprocessing).
269 	 */
270 	private void rtrimPositions(List<Node> nodes, List<Comment> comments) {
271 		final boolean[] whitespace = new boolean[preprocessed.length()];
272 		for (Comment comment : comments) {
273 			Position p = comment.getPosition();
274 			if (!p.isUnplaced()) {
275 				for (int i = p.getStart(); i < p.getEnd(); i++) whitespace[i] = true;
276 			}
277 		}
278 		
279 		/* Process actual whitespace in preprocessed source data */ {
280 			char[] chars = preprocessed.toCharArray();
281 			for (int i = 0; i < chars.length; i++) if (Character.isWhitespace(chars[i])) whitespace[i] = true;
282 		}
283 		
284 		for (Node node : nodes) node.accept(new ForwardingAstVisitor() {
285 			@Override public boolean visitNode(Node node) {
286 				Position p = node.getPosition();
287 				if (p.isUnplaced()) return false;
288 				
289 				int trimmed = Math.min(whitespace.length, p.getEnd());
290 				while (trimmed > 0 && whitespace[trimmed-1]) trimmed--;
291 				
292 				int start, end;
293 				
294 				if (p.getEnd() - p.getStart() == 0) {
295 					if (node.getParent() != null) {
296 						start = Math.min(node.getParent().getPosition().getEnd(), Math.max(node.getParent().getPosition().getStart(), p.getStart()));
297 						end = start;
298 					} else {
299 						start = p.getStart();
300 						end = start;
301 					}
302 				} else {
303 					start = p.getStart();
304 					end = Math.max(trimmed, start);
305 				}
306 				
307 				node.setPosition(new Position(start, end));
308 				
309 				return false;
310 			}
311 		});
312 	}

314 	private void fixPositions(List<? extends Node> nodes) {
315 		for (Node node : nodes) node.accept(new ForwardingAstVisitor() {
316 			@Override public boolean visitNode(Node node) {
317 				Position p = node.getPosition();
318 				if (!p.isUnplaced()) {
319 					node.setPosition(new Position(mapPosition(p.getStart()), mapPosition(p.getEnd())));
320 				}
321 				if (node instanceof Expression) {
322 					List<Position> list = ((Expression)node).astParensPositions();
323 					if (list != null) {
324 						ListIterator<Position> li = list.listIterator();
325 						while (li.hasNext()) {
326 							Position parenPos = li.next();
327 							if (!parenPos.isUnplaced()) {
328 								parenPos = new Position(mapPosition(parenPos.getStart()), mapPosition(parenPos.getEnd()));
329 								li.set(parenPos);
330 							}
331 						}
332 					}
333 				}
334 				return false;
335 			}
336 		});
337 	}

339 	/**
340 	 * Associates comments that are javadocs to the node they belong to, by checking if the node that immediately follows a javadoc node is a JavadocContainer.
341 	 */
342 	private void associateJavadoc(List<Comment> comments, List<Node> nodes) {
343 		final TreeMap<Integer, Node> startPosMap = Maps.newTreeMap();
344 		for (Node node : nodes) node.accept(new ForwardingAstVisitor() {
345 			@Override public boolean visitNode(Node node) {
346 				if (node.isGenerated()) return false;
347 				int startPos = node.getPosition().getStart();
348 				Node current = startPosMap.get(startPos);
349 				if (current == null || !(current instanceof JavadocContainer)) {
350 					startPosMap.put(startPos, node);
351 				}
352 				
353 				return false;
354 			}
355 		});
356 		
357 		for (Comment comment : comments) {
358 			if (!comment.isJavadoc()) continue;
359 			Map<Integer, Node> tailMap = startPosMap.tailMap(comment.getPosition().getEnd());
360 			if (tailMap.isEmpty()) continue;
361 			Node assoc = tailMap.values().iterator().next();
362 			if (!(assoc instanceof JavadocContainer)) continue;
363 			JavadocContainer jc = (JavadocContainer) assoc;
364 			if (jc.rawJavadoc() != null) {
365 				if (jc.rawJavadoc().getPosition().getEnd() >= comment.getPosition().getEnd()) continue;
366 			}
367 			jc.rawJavadoc(comment);
368 		}
369 	}

371 	void registerComment(Context<Node> context, Comment c) {
372 		List<Comment> list = registeredComments.get(context);
373 		if (list == null) {
374 			list = Lists.newArrayList();
375 			registeredComments.put(context.getSubNodes().get(0), list);
376 		}
377 		list.add(c);
378 	}

380 	/**
381 	 * Delves through the parboiled node tree to find comments.
382 	 */
383 	private boolean gatherComments(org.parboiled.Node<Node> parsed) {
384 		boolean foundComments = false;
385 		for (org.parboiled.Node<Node> child : parsed.getChildren()) {
386 			foundComments |= gatherComments(child);
387 		}
388 		
389 		List<Comment> cmts = registeredComments.get(parsed);
390 		if (cmts != null) for (Comment c : cmts) {
391 			comments.add(c);
392 			return true;
393 		}
394 		
395 		return foundComments;
396 	}

398 	private void setPositionDelta(int position, int delta) {
399 		Integer i = positionDeltas.get(position);
400 		if (i == null) i = 0;
401 		positionDeltas.put(position, i + delta);
402 	}

408 	public long lineColumn(int index) {
409 		//Possible efficiency improvement: Store in a list the index into the line table with the first line that's over 500, 1000, 1500, ... chars.
410 		//Or just binary search.
411 		int oldIdx = 0;
412 		int line = 0;
413 		
414 		for (; line < lineEndings.size(); line++) {
415 			int pos = lineEndings.get(line);
416 			if (pos > index) break;
417 			oldIdx = pos;
418 		}
419 		return ((long) line << 32 | index - oldIdx);
420 	}

422 	/**
423 	 * Maps a position in the {@code preprocessed} string to the equivalent character in the {@code rawInput}.
424 	 * 
425 	 * The difference is caused by decoding backslash-U unicode escapes, for example.
426 	 */
427 	int mapPosition(int position) {
428 		int out = position;
429 		for (int delta : positionDeltas.headMap(position, true).values()) {
430 			out += delta;
431 		}
432 		return out;
433 	}

435 	private String preProcess() {
436 		preprocessed = rawInput;
437 		this.lineEndings = calculateLineEndings();
438 		applyBackslashU();
439 //		applyBraceMatching();
440 		return preprocessed;
441 	}

443 	/**
444 	 * @see <a href="http://java.sun.com/docs/books/jls/third_edition/html/lexical.html#3.3">JLS section 3.3</a>
445 	 */
446 	private void applyBackslashU() {
447 		StringBuilder buffer = new StringBuilder();
448 		StringBuilder out = new StringBuilder();
449 		
450 		int state = 0;
451 		int idx = 0;
452 		for (char c : preprocessed.toCharArray()) {
453 			idx++;
454 			switch (state) {
455 			case 0:	//normal mode. Anything that isn't a backslash is not interesting.
456 				if (c != '\\') {
457 					out.append(c);
458 					break;
459 				}
460 				
461 				state = 1;
462 				break;
463 			case 1:	//Last character read is an (uneven amount of) backslash.
464 				if (c != 'u') {
465 					out.append('\\');
466 					out.append(c);
467 					state = 0;
468 				} else {
469 					buffer.setLength(0);
470 					buffer.append("\\u");
471 					state = 2;
472 				}
473 				break;
474 				//TODO add a test for more backslash-U stuff.
475 			default:
476 				//Gobbling hex digits. state-2 is our current position. We want 4.
477 				buffer.append(c);
478 				if (c == 'u') {
479 					//JLS Puzzler: backslash-u-u-u-u-u-u-u-u-u-4hexdigits means the same thing as just 1 u.
480 					//So, we just keep going as if nothing changed.
481 				} else if ((c >= '0' && c <= '9') || (c >= 'a' && c <= 'f') || (c >= 'A' && c <= 'F')) {
482 					state++;
483 					if (state == 6) {
484 						//We've got our 4 hex digits.
485 						out.append((char)Integer.parseInt(buffer.substring(buffer.length()-4), 0x10));
486 						int delta = buffer.length() -1;	//Buffer goes away but 1 character appears in its place.
487 						setPositionDelta(idx - delta, delta);
488 						buffer.setLength(0);
489 						//We don't have to check if this char is a backslash and set state to 1; JLS says backslash-u is not recursively applied.
490 						state = 0;
491 					}
492 				} else {
493 					//Invalid unicode escape.
494 					problems.add(new ParseProblem(new Position(idx-buffer.length(), idx), "Invalid backslash-u escape: \\u is supposed to be followed by 4 hex digits."));
495 					out.append(buffer.toString());
496 					state = 0;
497 				}
498 				break;
499 			}
500 		}
501 		
502 		preprocessed = out.toString();
503 	}
