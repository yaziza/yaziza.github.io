35  /**
36   * Utility methods for runtime code generation and class loading. We use this stuff for {@link
37   * net.sf.cglib.reflect.FastClass faster reflection}, {@link net.sf.cglib.proxy.Enhancer method
38   * interceptors} and to proxy circular dependencies.
39   *
40   * <p>When loading classes, we need to be careful of:
41   * <ul>
42   *   <li><strong>Memory leaks.</strong> Generated classes need to be garbage collected in long-lived
43   *       applications. Once an injector and any instances it created can be garbage collected, the
44   *       corresponding generated classes should be collectable.
45   *   <li><strong>Visibility.</strong> Containers like <code>OSGi</code> use class loader boundaries
46   *       to enforce modularity at runtime.
47   * </ul>
48   *
49   * <p>For each generated class, there's multiple class loaders involved:
50   * <ul>
51   *    <li><strong>The related class's class loader.</strong> Every generated class services exactly
52   *        one user-supplied class. This class loader must be used to access members with protected
53   *        and package visibility.
54   *    <li><strong>Guice's class loader.</strong>
55   *    <li><strong>Our bridge class loader.</strong> This is a child of the user's class loader. It
56   *        selectively delegates to either the user's class loader (for user classes) or the Guice
57   *        class loader (for internal classes that are used by the generated classes). This class
58   *        loader that owns the classes generated by Guice.
59   * </ul>
60   *
61   * @author mcculls@gmail.com (Stuart McCulloch)
62   * @author jessewilson@google.com (Jesse Wilson)
63   */
64  public final class BytecodeGen {
65

123   /**
124    * Weak cache of bridge class loaders that make the Guice implementation
125    * classes visible to various code-generated proxies of client classes.
126    */
127   private static final LoadingCache<ClassLoader, ClassLoader> CLASS_LOADER_CACHE;
128 
129   static {
130     CacheBuilder<Object, Object> builder = CacheBuilder.newBuilder().weakKeys().weakValues();
131     if (getCustomClassLoadingOption() == CustomClassLoadingOption.OFF) {
132       builder.maximumSize(0);
133     }
134     CLASS_LOADER_CACHE = builder.build(
135         new CacheLoader<ClassLoader, ClassLoader>() {
136           @Override public ClassLoader load(final ClassLoader typeClassLoader) {
137             logger.fine("Creating a bridge ClassLoader for " + typeClassLoader);
138             return AccessController.doPrivileged(new PrivilegedAction<ClassLoader>() {
139               public ClassLoader run() {
140                 return new BridgeClassLoader(typeClassLoader);
141               }
142             });
143           }
144         });
145   }

147   /**
148    * Attempts to canonicalize null references to the system class loader.
149    * May return null if for some reason the system loader is unavailable.
150    */
151   private static ClassLoader canonicalize(ClassLoader classLoader) {
152     return classLoader != null ? classLoader : SystemBridgeHolder.SYSTEM_BRIDGE.getParent();
153   }
154 
155   /**
156    * Returns the class loader to host generated classes for {@code type}.
157    */
158   public static ClassLoader getClassLoader(Class<?> type) {
159     return getClassLoader(type, type.getClassLoader());
160   }
161 
162   private static ClassLoader getClassLoader(Class<?> type, ClassLoader delegate) {
163 
164     // simple case: do nothing!
165     if (getCustomClassLoadingOption() == CustomClassLoadingOption.OFF) {
166       return delegate;
167     }
168 
169     // java.* types can be seen everywhere
170     if (type.getName().startsWith("java.")) {
171       return GUICE_CLASS_LOADER;
172     }
173 
174     delegate = canonicalize(delegate);
175 
176     // no need for a bridge if using same class loader, or it's already a bridge
177     if (delegate == GUICE_CLASS_LOADER || delegate instanceof BridgeClassLoader) {
178       return delegate;
179     }
180 
181     // don't try bridging private types as it won't work
182     if (Visibility.forType(type) == Visibility.PUBLIC) {
183       if (delegate != SystemBridgeHolder.SYSTEM_BRIDGE.getParent()) {
184         // delegate guaranteed to be non-null here
185         return CLASS_LOADER_CACHE.getUnchecked(delegate);
186       }
187       // delegate may or may not be null here
188       return SystemBridgeHolder.SYSTEM_BRIDGE;
189     }
190 
191     return delegate; // last-resort: do nothing!
192   }

206   /**
207    * Returns a FastClass proxy for invoking the given member or {@code null} if access rules 
208    * disallow it.
209    * 
210    * <p>FastClass works by generating a type in the same package as the target {@code type}. This
211    * may or may not work depending on the access level of the class/member. It breaks down into the
212    * following cases depending on accessibility:
213    * <ul>
214    *   <li>Public: This always works since we can generate the type into the 
215    *   {@link BridgeClassLoader} which ensures there are no versioning issues.
216    *   <li>Package private and Protected: This works as long as:
217    *   <ul> 
218    *     <li>We can generate into the same classloader as the type.  This is not possible for JDK
219    *     types which use the 'bootstrap' loader.
220    *     <li>The classloader of the type has the same version of {@code FastClass} as we do.  This
221    *     may be violated when running in OSGI bundles.
222    *   </ul>
223    *   <li>Private: This never works.
224    * </ul>
225    * 
226    * If we are unable to generate the type, then we return null and callers should work around by
227    * using normal java reflection.
228    */
229   public static net.sf.cglib.reflect.FastClass newFastClassForMember(Class<?> type, Member member) {
230     if (!new net.sf.cglib.core.VisibilityPredicate(type, false).evaluate(member)) {
231       // the member cannot be indexed by fast class.  Bail out.
232       return null;
233     }
234 
235     boolean publiclyCallable = isPubliclyCallable(member);
236     if (!publiclyCallable && !hasSameVersionOfCglib(type.getClassLoader())) {
237       // The type is in a classloader with a different version of cglib and is not publicly visible
238       // (so we can't use the bridge classloader to work around).  Bail out.
239       return null;
240     }
241     net.sf.cglib.reflect.FastClass.Generator generator
242         = new net.sf.cglib.reflect.FastClass.Generator();
243     if (publiclyCallable) {
244       // Use the bridge classloader if we can
245       generator.setClassLoader(getClassLoader(type));
246     }
247     generator.setType(type);
248     generator.setNamingPolicy(FASTCLASS_NAMING_POLICY);
249     if (logger.isLoggable(Level.FINE)) {
250       logger.fine("Loading " + type + " FastClass with " + generator.getClassLoader());
251     }
252     return generator.create();
253   }

255   /**
256    * Returns true if the types classloader has the same version of cglib that BytecodeGen has.  This
257    * only returns false in strange OSGI situations, but it prevents us from using FastClass for non
258    * public members.
259    */
260   private static boolean hasSameVersionOfCglib(ClassLoader classLoader) {
261     Class<?> fc = net.sf.cglib.reflect.FastClass.class;
262     try {
263       return classLoader.loadClass(fc.getName()) == fc;
264     } catch (ClassNotFoundException e) {
265       return false;
266     }
267   }

269   /**
270    * Returns true if the member can be called by a fast class generated in a different classloader.
271    */
272   private static boolean isPubliclyCallable(Member member) {
273     if (!Modifier.isPublic(member.getModifiers())) {
274       return false;
275     }
276     Class<?>[] parameterTypes;
277     if (member instanceof Constructor) {
278       parameterTypes = ((Constructor) member).getParameterTypes();
279     } else {
280       Method method = (Method) member;
281       if (!Modifier.isPublic(method.getReturnType().getModifiers())) {
282         return false;
283       }
284       parameterTypes = method.getParameterTypes();
285     }
286 
287     for (Class<?> type : parameterTypes) {
288       if (!Modifier.isPublic(type.getModifiers())) {
289         return false;
290       }
291     }
292     return true;
293   }

295   public static net.sf.cglib.proxy.Enhancer newEnhancer(Class<?> type, Visibility visibility) {
296     net.sf.cglib.proxy.Enhancer enhancer = new net.sf.cglib.proxy.Enhancer();
297     enhancer.setSuperclass(type);
298     enhancer.setUseFactory(false);
299     if (visibility == Visibility.PUBLIC) {
300       enhancer.setClassLoader(getClassLoader(type));
301     }
302     enhancer.setNamingPolicy(ENHANCER_NAMING_POLICY);
303     logger.fine("Loading " + type + " Enhancer with " + enhancer.getClassLoader());
304     return enhancer;
305   }

308   /**
309    * The required visibility of a user's class from a Guice-generated class. Visibility of
310    * package-private members depends on the loading classloader: only if two classes were loaded by
311    * the same classloader can they see each other's package-private members. We need to be careful
312    * when choosing which classloader to use for generated classes. We prefer our bridge classloader,
313    * since it's OSGi-safe and doesn't leak permgen space. But often we cannot due to visibility.
314    */
315   public enum Visibility {
316 
317     /**
318      * Indicates that Guice-generated classes only need to call and override public members of the
319      * target class. These generated classes may be loaded by our bridge classloader.
320      */
321     PUBLIC {
322       @Override
323       public Visibility and(Visibility that) {
324         return that;
325       }
326     },
327 
328     /**
329      * Indicates that Guice-generated classes need to call or override package-private members.
330      * These generated classes must be loaded in the same classloader as the target class. They
331      * won't work with OSGi, and won't get garbage collected until the target class' classloader is
332      * garbage collected.
333      */
334     SAME_PACKAGE {
335       @Override
336       public Visibility and(Visibility that) {
337         return this;
338       }
339     };
340 
341     public static Visibility forMember(Member member) {
342       if ((member.getModifiers() & (Modifier.PROTECTED | Modifier.PUBLIC)) == 0) {
343         return SAME_PACKAGE;
344       }
345 
346       Class[] parameterTypes;
347       if (member instanceof Constructor) {
348         parameterTypes = ((Constructor) member).getParameterTypes();
349       } else {
350         Method method = (Method) member;
351         if (forType(method.getReturnType()) == SAME_PACKAGE) {
352           return SAME_PACKAGE;
353         }
354         parameterTypes = method.getParameterTypes();
355       }
356 
357       for (Class<?> type : parameterTypes) {
358         if (forType(type) == SAME_PACKAGE) {
359           return SAME_PACKAGE;
360         }
361       }
362 
363       return PUBLIC;
364     }
365 
366     public static Visibility forType(Class<?> type) {
367       return (type.getModifiers() & (Modifier.PROTECTED | Modifier.PUBLIC)) != 0
368           ? PUBLIC
369           : SAME_PACKAGE;
370     }
371 
372     public abstract Visibility and(Visibility that);
373   }

375   /**
376    * Loader for Guice-generated classes. For referenced classes, this delegates to either either the
377    * user's classloader (which is the parent of this classloader) or Guice's class loader.
378    */
379   private static class BridgeClassLoader extends ClassLoader {
380 
381     BridgeClassLoader() {
382       // use system loader as parent
383     }
384 
385     BridgeClassLoader(ClassLoader usersClassLoader) {
386       super(usersClassLoader);
387     }
388 
389     @Override protected Class<?> loadClass(String name, boolean resolve)
390         throws ClassNotFoundException {
391 
392       if (name.startsWith("sun.reflect")) {
393         // these reflection classes must be loaded from bootstrap class loader
394         return SystemBridgeHolder.SYSTEM_BRIDGE.classicLoadClass(name, resolve);
395       }
396 
397       if (name.startsWith(GUICE_INTERNAL_PACKAGE) || name.startsWith(CGLIB_PACKAGE)) {
398         if (null == GUICE_CLASS_LOADER) {
399           // use special system bridge to load classes from bootstrap class loader
400           return SystemBridgeHolder.SYSTEM_BRIDGE.classicLoadClass(name, resolve);
401         }
402         try {
403           Class<?> clazz = GUICE_CLASS_LOADER.loadClass(name);
404           if (resolve) {
405             resolveClass(clazz);
406           }
407           return clazz;
408         } catch (Throwable e) {
409           // fall-back to classic delegation
410         }
411       }
412 
413       return classicLoadClass(name, resolve);
414     }
415 
416     // make the classic delegating loadClass method visible
417     Class<?> classicLoadClass(String name, boolean resolve)
418       throws ClassNotFoundException {
419       return super.loadClass(name, resolve);
420     }
421   }
