74      protected static DefaultConfiguration createCheckConfig(Class<?> clazz) {
75          return new DefaultConfiguration(clazz.getName());
76      }

78      protected Checker createChecker(Configuration checkConfig)
79              throws Exception {
80          final DefaultConfiguration dc = createCheckerConfig(checkConfig);
81          final Checker checker = new Checker();
82          // make sure the tests always run with English error messages
83          // so the tests don't fail in supported locales like German
84          final Locale locale = Locale.ENGLISH;
85          checker.setLocaleCountry(locale.getCountry());
86          checker.setLocaleLanguage(locale.getLanguage());
87          checker.setModuleClassLoader(Thread.currentThread().getContextClassLoader());
88          checker.configure(dc);
89          checker.addListener(new BriefUtLogger(stream));
90          return checker;
91      }
92  
93      protected DefaultConfiguration createCheckerConfig(Configuration config) {
94          final DefaultConfiguration dc =
95                  new DefaultConfiguration("configuration");
96          final DefaultConfiguration twConf = createCheckConfig(TreeWalker.class);
97          // make sure that the tests always run with this charset
98          dc.addAttribute("charset", "iso-8859-1");
99          dc.addChild(twConf);
100         twConf.addChild(config);
101         return dc;
102     }

109     protected void verify(Configuration config, String fileName, String[] expected,

116     protected void verify(Checker checker,
117             File[] processedFiles,
118             String messageFileName,
119             String[] expected,
120             Integer... warnsExpected)
121         throws Exception {
122         stream.flush();
123         final List<File> theFiles = Lists.newArrayList();
124         Collections.addAll(theFiles, processedFiles);
125         final List<Integer> theWarnings = Lists.newArrayList();
126         Collections.addAll(theWarnings, warnsExpected);
127         final int errs = checker.process(theFiles);
128 
129         // process each of the lines
130         final ByteArrayInputStream inputStream =
131                 new ByteArrayInputStream(stream.toByteArray());
132         try (final LineNumberReader lnr = new LineNumberReader(
133                 new InputStreamReader(inputStream, StandardCharsets.UTF_8))) {
134 
135             int previousLineNumber = 0;
136             for (int i = 0; i < expected.length; i++) {
137                 final String expectedResult = messageFileName + ":" + expected[i];
138                 final String actual = lnr.readLine();
139                 assertEquals("error message " + i, expectedResult, actual);
140 
141                 String parseInt = removeDeviceFromPathOnWindows(actual);
142                 parseInt = parseInt.substring(parseInt.indexOf(':') + 1);
143                 parseInt = parseInt.substring(0, parseInt.indexOf(':'));
144                 final int lineNumber = Integer.parseInt(parseInt);
145                 assertTrue("input file is expected to have a warning comment on line number "
146                         + lineNumber, theWarnings.remove((Integer) lineNumber)
147                             || previousLineNumber == lineNumber);
148                 previousLineNumber = lineNumber;
149             }
150 
151             assertEquals("unexpected output: " + lnr.readLine(),
152                     expected.length, errs);
153             assertEquals("unexpected warnings " + theWarnings, 0, theWarnings.size());
154         }
155 
156         checker.destroy();
157     }

159     /**
160      * Gets the check message 'as is' from appropriate 'messages.properties'
161      * file.
162      *
163      * @param messageKey the key of message in 'messages.properties' file.
164      * @param arguments  the arguments of message in 'messages.properties' file.
165      */
166     protected String getCheckMessage(Class<? extends AbstractViolationReporter> aClass,
167             String messageKey, Object... arguments) {
168         final Properties pr = new Properties();
169         try {
170             pr.load(aClass.getResourceAsStream("messages.properties"));
171         }
172         catch (IOException ex) {
173             return null;
174         }
175         final MessageFormat formatter = new MessageFormat(pr.getProperty(messageKey),
176                 Locale.ROOT);
177         return formatter.format(arguments);
178     }
179 
180     /**
181      * Gets the check message 'as is' from appropriate 'messages.properties' file.
182      * @param messageKey the key of message in 'messages.properties' file.
183      * @param arguments the arguments of message in 'messages.properties' file.
184      */
185     protected String getCheckMessage(Map<String, String> messages, String messageKey,
186             Object... arguments) {
187         for (Map.Entry<String, String> entry : messages.entrySet()) {
188             if (messageKey.equals(entry.getKey())) {
189                 final MessageFormat formatter = new MessageFormat(entry.getValue(), Locale.ROOT);
190                 return formatter.format(arguments);
191             }
192         }
193         return null;
194     }

196     protected static Configuration getCheckConfig(String checkName) throws CheckstyleException {
197         Configuration result = null;
198         for (Configuration currentConfig : getConfiguration().getChildren()) {
199             if ("TreeWalker".equals(currentConfig.getName())) {
200                 for (Configuration checkConfig : currentConfig.getChildren()) {
201                     if (checkName.equals(checkConfig.getName())) {
202                         result = checkConfig;
203                         break;
204                     }
205                 }
206             }
207             else if (checkName.equals(currentConfig.getName())) {
208                 result = currentConfig;
209                 break;
210             }
211         }
212         return result;
213     }
214 
215     private static String removeDeviceFromPathOnWindows(String path) {
216         final String os = System.getProperty("os.name", "Unix");
217         if (os.startsWith("Windows")) {
218             return path.substring(path.indexOf(':') + 1);
219         }
220         return path;
221     }

223     protected Integer[] getLinesWithWarn(String fileName) throws IOException {
224         final List<Integer> result = new ArrayList<>();
225         try (BufferedReader br = new BufferedReader(new InputStreamReader(
226                 new FileInputStream(fileName), StandardCharsets.UTF_8))) {
227             int lineNumber = 1;
228             while (true) {
229                 final String line = br.readLine();
230                 if (line == null) {
231                     break;
232                 }
233                 if (WARN_PATTERN.matcher(line).find()) {
234                     result.add(lineNumber);
235                 }
236                 lineNumber++;
237             }
238         }
239         return result.toArray(new Integer[result.size()]);
240     }
