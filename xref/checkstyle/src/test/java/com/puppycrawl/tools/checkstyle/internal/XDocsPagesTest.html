156                         .replace("...", "").trim();

161                 }

169             }

173     private static void buildAndValidateXml(String fileName, String unserializedSource)
174             throws IOException, ParserConfigurationException {
175         // not all examples come with the full xml structure
176         String code = unserializedSource;
177 
178         if (!hasFileSetClass(code)) {
179             code = "<module name=\"TreeWalker\">\n" + code + "\n</module>";
180         }
181         if (!code.contains("name=\"Checker\"")) {
182             code = "<module name=\"Checker\">\n" + code + "\n</module>";
183         }
184         if (!code.startsWith("<?xml")) {
185             final String dtdPath = new File(
186                     "src/main/resources/com/puppycrawl/tools/checkstyle/configuration_1_3.dtd")
187                     .getCanonicalPath();
188 
189             code = "<?xml version=\"1.0\"?>\n<!DOCTYPE module PUBLIC "
190                     + "\"-//Puppy Crawl//DTD Check Configuration 1.3//EN\" \"" + dtdPath + "\">\n"
191                     + code;
192         }
193 
194         // validate only
195         XmlUtil.getRawXml(fileName, code, unserializedSource);
196 
197         // can't test ant structure, or old and outdated checks
198         if (!fileName.startsWith("anttask") && !fileName.startsWith("releasenotes")) {
199             validateCheckstyleXml(fileName, code, unserializedSource);
200         }
201     }

203     private static boolean hasFileSetClass(String xml) {
204         boolean found = false;
205 
206         for (String find : XML_FILESET_LIST) {
207             if (xml.contains(find)) {
208                 found = true;
209                 break;
210             }
211         }
212 
213         return found;
214     }

216     private static void validateCheckstyleXml(String fileName, String code,
217             String unserializedSource) throws IOException {
218         // can't process non-existent examples, or out of context snippets
219         if (code.contains("com.mycompany") || code.contains("checkstyle-packages")
220                 || code.contains("MethodLimit") || code.contains("<suppress ")
221                 || code.contains("<import-control ") || unserializedSource.startsWith("<property ")
222                 || unserializedSource.startsWith("<taskdef ")) {
223             return;
224         }
225 
226         // validate checkstyle structure and contents
227         try {
228             final Properties properties = new Properties();
229 
230             properties.setProperty("checkstyle.header.file",
231                     new File("config/java.header").getCanonicalPath());
232 
233             final PropertiesExpander expander = new PropertiesExpander(properties);
234             final Configuration config = ConfigurationLoader.loadConfiguration(new InputSource(
235                     new StringReader(code)), expander, false);
236             final Checker checker = new Checker();
237 
238             try {
239                 final ClassLoader moduleClassLoader = Checker.class.getClassLoader();
240                 checker.setModuleClassLoader(moduleClassLoader);
241                 checker.configure(config);
242             }
243             finally {
244                 checker.destroy();
245             }
246         }
247         catch (CheckstyleException ex) {
248             Assert.fail(fileName + " has invalid Checkstyle xml (" + ex.getMessage() + "): "
249                     + unserializedSource);
250         }
251     }

257         for (Path path : XDocUtil.getXdocsConfigFilePaths(XDocUtil.getXdocsFilePaths())) {

263             }

266             final Document document = XmlUtil.getRawXml(fileName, input, input);
267             final NodeList sources = document.getElementsByTagName("section");
268             String lastSectioName = null;

271                 final Node section = sources.item(position);
272                 final String sectionName = section.getAttributes().getNamedItem("name")

283                 if (lastSectioName != null) {
284                     Assert.assertTrue(
285                             fileName + " section '" + sectionName
286                                     + "' is out of order compared to '" + lastSectioName + "'",
287                             sectionName.toLowerCase(Locale.ENGLISH).compareTo(
288                                     lastSectioName.toLowerCase(Locale.ENGLISH)) >= 0);

298     private static void validateCheckSection(ModuleFactory moduleFactory, String fileName,
299             String sectionName, Node section) {
300         Object instance = null;
301 
302         try {
303             instance = moduleFactory.createModule(sectionName);
304         }
305         catch (CheckstyleException ex) {
306             Assert.fail(fileName + " couldn't find class: " + sectionName);
307         }
308 
309         int subSectionPos = 0;
310         for (Node subSection : XmlUtil.getChildrenElements(section)) {
311             final String subSectionName = subSection.getAttributes().getNamedItem("name")
312                     .getNodeValue();
313 
314             // can be in different orders, and completely optional
315             if ("Notes".equals(subSectionName)
316                     || "Rule Description".equals(subSectionName)) {
317                 continue;
318             }
319 
320             if (subSectionPos == 1 && !"Properties".equals(subSectionName)) {
321                 validatePropertySection(fileName, sectionName, null, instance);
322                 subSectionPos++;
323             }
324 
325             Assert.assertEquals(fileName + " section '" + sectionName
326                     + "' should be in order", getSubSectionName(subSectionPos),
327                     subSectionName);
328 
329             switch (subSectionPos) {
330                 case 0:
331                     break;
332                 case 1:
333                     validatePropertySection(fileName, sectionName, subSection, instance);
334                     break;
335                 case 2:
336                     break;
337                 case 3:
338                     validateUsageExample(fileName, sectionName, subSection);
339                     break;
340                 case 4:
341                     validatePackageSection(fileName, sectionName, subSection, instance);
342                     break;
343                 case 5:
344                     validateParentSection(fileName, sectionName, subSection);
345                     break;
346                 default:
347                     break;
348             }
349 
350             subSectionPos++;
351         }
352     }

354     private static Object getSubSectionName(int subSectionPos) {
355         final String result;
356 
357         switch (subSectionPos) {
358             case 0:
359                 result = "Description";
360                 break;
361             case 1:
362                 result = "Properties";
363                 break;
364             case 2:
365                 result = "Examples";
366                 break;
367             case 3:
368                 result = "Example of Usage";
369                 break;
370             case 4:
371                 result = "Package";
372                 break;
373             case 5:
374                 result = "Parent Module";
375                 break;
376             default:
377                 result = null;
378                 break;
379         }
380 
381         return result;
382     }

384     private static void validatePropertySection(String fileName, String sectionName,
385             Node subSection, Object instance) {
386         final Set<String> properties = getProperties(instance.getClass());
387         final Class<?> clss = instance.getClass();
388 
389         // remove global properties that don't need documentation
390         if (hasParentModule(sectionName)) {
391             properties.removeAll(CHECK_PROPERTIES);
392         }
393         else if (AbstractFileSetCheck.class.isAssignableFrom(clss)) {
394             properties.removeAll(FILESET_PROPERTIES);
395 
396             // override
397             properties.add("fileExtensions");
398         }
399 
400         // remove undocumented properties
401         for (String p : new HashSet<>(properties)) {
402             if (UNDOCUMENTED_PROPERTIES.contains(clss.getSimpleName() + "." + p)) {
403                 properties.remove(p);
404             }
405         }
406 
407         final Check check;
408 
409         if (Check.class.isAssignableFrom(clss)) {
410             check = (Check) instance;
411 
412             if (!Arrays.equals(check.getAcceptableTokens(), check.getDefaultTokens())
413                     || !Arrays.equals(check.getAcceptableTokens(), check.getRequiredTokens())) {
414                 properties.add("tokens");
415             }
416         }
417         else {
418             check = null;
419         }
420 
421         if (subSection != null) {
422             Assert.assertTrue(fileName + " section '" + sectionName
423                     + "' should have no properties to show", !properties.isEmpty());
424 
425             validatePropertySectionProperties(fileName, sectionName, subSection, check,
426                     properties);
427         }
428 
429         Assert.assertTrue(fileName + " section '" + sectionName + "' should show properties: "
430                 + properties, properties.isEmpty());
431     }

433     private static void validatePropertySectionProperties(String fileName, String sectionName,
434             Node subSection, Check check, Set<String> properties) {
435         boolean skip = true;
436         boolean didTokens = false;
437 
438         for (Node row : XmlUtil.getChildrenElements(XmlUtil.getFirstChildElement(subSection))) {
439             if (skip) {
440                 skip = false;
441                 continue;
442             }
443             Assert.assertFalse(fileName + " section '" + sectionName
444                     + "' should have token properties last", didTokens);
445 
446             final List<Node> columns = new ArrayList<>(XmlUtil.getChildrenElements(row));
447 
448             final String propertyName = columns.get(0).getTextContent();
449             Assert.assertTrue(fileName + " section '" + sectionName
450                     + "' should not contain the property: " + propertyName,
451                     properties.remove(propertyName));
452 
453             if ("tokens".equals(propertyName)) {
454                 Assert.assertEquals(fileName + " section '" + sectionName
455                         + "' should have the basic token description", "tokens to check", columns
456                         .get(1).getTextContent());
457                 Assert.assertEquals(
458                         fileName + " section '" + sectionName
459                                 + "' should have all the acceptable tokens",
460                         "subset of tokens "
461                                 + CheckUtil.getTokenText(check.getAcceptableTokens(),
462                                         check.getRequiredTokens()), columns.get(2).getTextContent()
463                                 .replaceAll("\\s+", " ").trim());
464                 Assert.assertEquals(
465                         fileName + " section '" + sectionName
466                                 + "' should have all the default tokens",
467                         CheckUtil.getTokenText(check.getDefaultTokens(), check.getRequiredTokens()),
468                         columns.get(3).getTextContent().replaceAll("\\s+", " ").trim());
469                 didTokens = true;
470             }
471             else {
472                 Assert.assertFalse(fileName + " section '" + sectionName
473                         + "' should have a description for " + propertyName, columns.get(1)
474                         .getTextContent().trim().isEmpty());
475                 Assert.assertFalse(fileName + " section '" + sectionName
476                         + "' should have a type for " + propertyName, columns.get(2)
477                         .getTextContent().trim().isEmpty());
478                 // default can be empty string
479             }
480         }
481     }

483     private static void validateUsageExample(String fileName, String sectionName, Node subSection) {
484         final String text = subSection.getTextContent().replace("Checkstyle Style", "")
485                 .replace("Google Style", "").replace("Sun Style", "").trim();
486 
487         Assert.assertTrue(fileName + " section '" + sectionName
488                 + "' has unknown text in 'Example of Usage': " + text, text.isEmpty());
489 
490         for (Node node : XmlUtil.findChildElementsByTag(subSection, "a")) {
491             final String url = node.getAttributes().getNamedItem("href").getTextContent();
492             final String linkText = node.getTextContent().trim();
493             String expectedUrl = null;
494 
495             if ("Checkstyle Style".equals(linkText)) {
496                 expectedUrl = "https://github.com/search?q="
497                         + "path%3Aconfig+filename%3Acheckstyle_checks.xml+"
498                         + "repo%3Acheckstyle%2Fcheckstyle+" + sectionName;
499             }
500             else if ("Google Style".equals(linkText)) {
501                 expectedUrl = "https://github.com/search?q="
502                         + "path%3Asrc%2Fmain%2Fresources+filename%3Agoogle_checks.xml+"
503                         + "repo%3Acheckstyle%2Fcheckstyle+"
504                         + sectionName;
505 
506                 Assert.assertTrue(fileName + " section '" + sectionName
507                         + "' should be in google_checks.xml or not reference 'Google Style'",
508                         GOOGLE_CHECKS.contains(sectionName));
509             }
510             else if ("Sun Style".equals(linkText)) {
511                 expectedUrl = "https://github.com/search?q="
512                         + "path%3Asrc%2Fmain%2Fresources+filename%3Asun_checks.xml+"
513                         + "repo%3Acheckstyle%2Fcheckstyle+"
514                         + sectionName;
515 
516                 Assert.assertTrue(fileName + " section '" + sectionName
517                         + "' should be in sun_checks.xml or not reference 'Sun Style'",
518                         SUN_CHECKS.contains(sectionName));
519             }
520 
521             Assert.assertEquals(fileName + " section '" + sectionName
522                     + "' should have matching url", expectedUrl, url);
523         }
524     }

614     private static void validateStyleChecks(Set<Node> checks, Set<Node> configs, String fileName,
615             String ruleName) {
616         final Iterator<Node> itrChecks = checks.iterator();
617         final Iterator<Node> itrConfigs = configs.iterator();
618 
619         while (itrChecks.hasNext()) {
620             final Node check = itrChecks.next();
621             final String checkName = check.getTextContent().trim();
622 
623             if (!check.getAttributes().getNamedItem("href").getTextContent()
624                     .startsWith("config_")) {
625                 continue;
626             }
627 
628             Assert.assertTrue(fileName + " rule '" + ruleName + "' check '" + checkName
629                     + "' shouldn't end with 'Check'", !checkName.endsWith("Check"));
630 
631             for (String configName : new String[] {"config", "test"}) {
632                 Node config = null;
633 
634                 try {
635                     config = itrConfigs.next();
636                 }
637                 catch (NoSuchElementException ignore) {
638                     Assert.fail(fileName + " rule '" + ruleName + "' check '" + checkName
639                             + "' is missing the config link: " + configName);
640                 }
641 
642                 Assert.assertEquals(fileName + " rule '" + ruleName + "' check '" + checkName
643                         + "' has mismatched config/test links", configName, config.getTextContent()
644                         .trim());
645 
646                 final String configUrl = config.getAttributes().getNamedItem("href")
647                         .getTextContent();
648 
649                 if ("config".equals(configName)) {
650                     final String expectedUrl = "https://github.com/search?q="
651                             + "path%3Asrc%2Fmain%2Fresources+filename%3Agoogle_checks.xml+"
652                             + "repo%3Acheckstyle%2Fcheckstyle+" + checkName;
653 
654                     Assert.assertEquals(fileName + " rule '" + ruleName + "' check '" + checkName
655                             + "' should have matching " + configName + " url", expectedUrl,
656                             configUrl);
657                 }
658                 else if ("test".equals(configName)) {
659                     Assert.assertTrue(fileName + " rule '" + ruleName + "' check '" + checkName
660                             + "' should have matching " + configName + " url",
661                             configUrl.startsWith("https://github.com/checkstyle/checkstyle/"
662                                     + "blob/master/src/it/java/com/google/checkstyle/test/"));
663                     Assert.assertTrue(fileName + " rule '" + ruleName + "' check '" + checkName
664                             + "' should have matching " + configName + " url",
665                             configUrl.endsWith("/" + checkName + "Test.java"));
666 
667                     Assert.assertTrue(fileName + " rule '" + ruleName + "' check '" + checkName
668                             + "' should have a test that exists", new File(configUrl.substring(53)
