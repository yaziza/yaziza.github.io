113     public Object[] getChildren(Object parent) {
114         if (grouping == null) {
115             // on initialization
116             return EMPTY;
117         }
118         Object[] children = EMPTY;
119         if (parent instanceof BugGroup) {
120             BugGroup group = (BugGroup) parent;
121             children = group.getChildren();
122             if (rootElement == parent && rootElement.size() == 0) {
123                 if (DEBUG) {
124                     System.out.println("Root is empty...");
125                 }
126             }
127         } else {
128             if (parent instanceof IWorkspaceRoot || parent instanceof IWorkingSet) {
129                 if (input == parent) {
130                     Object[] objects = rootElement.getChildren();
131                     if (objects.length > 0) {
132                         if (bugFilterActive != isBugFilterActive()) {
133                             refreshFilters();
134                         }
135                         return objects;
136                     }
137                 }
138                 BugGroup root = new BugGroup(null, parent, GroupType.getType(parent));
139                 clearFilters();
140                 children = createChildren(grouping.getFirstType(), getResources(parent), root);
141                 if (input == parent) {
142                     // remember the root
143                     rootElement = root;
144                 }
145             }
146         }
147         return children;
148     }

150     private Set<IResource> getResources(Object parent) {
151         Set<IResource> resources = new HashSet<IResource>();
152         if (parent instanceof IWorkingSet) {
153             IWorkingSet workingSet = (IWorkingSet) parent;
154             IAdaptable[] elements = workingSet.getElements();
155             // elements may contain NON-resource elements, which we have to
156             // convert to
157             // resources
158             for (IAdaptable adaptable : elements) {
159                 IResource resource = (IResource) adaptable.getAdapter(IResource.class);
160                 // TODO get only java projects or children of them
161                 if (resource != null) {
162                     resources.add(resource);
163                 }
164             }
165         } else if (parent instanceof IWorkspaceRoot) {
166             IWorkspaceRoot workspaceRoot = (IWorkspaceRoot) parent;
167             // TODO get only java projects
168             IProject[] projects = workspaceRoot.getProjects();
169             for (IProject project : projects) {
170                 resources.add(project);
171             }
172         }
173         return resources;
174     }

215     public void inputChanged(Viewer newViewer, Object oldInput, Object newInput) {
216         viewer = (CommonViewer) newViewer;
217         if (newInput == null || newInput instanceof IWorkingSet || newInput instanceof IWorkspaceRoot) {
218             rootElement.dispose();
219         }
220         input = newInput;

225             bugFilterActive = isBugFilterActive();

256     public void refreshFilters() {
257         clearFilters();
258         bugFilterActive = isBugFilterActive();
259         if (!bugFilterActive) {
260             return;
261         }
262         if (DEBUG) {
263             System.out.println("Refreshing filters!");
264         }
265         Set<String> patternFilter = getPatternFilter();
266         for (IMarker marker : rootElement.getAllMarkers()) {
267             if (MarkerUtil.isFiltered(marker, patternFilter)) {
268                 filteredMarkers.add(marker);
269             }
270         }
271     }

273     public synchronized int getFilteredMarkersCount(BugGroup bugGroup) {
274         if (!isBugFilterActive()) {
275             return 0;
276         }
277         Integer bugCount;
278         bugCount = filteredMarkersMap.get(bugGroup);
279         if (bugCount == null) {
280             int count = 0;
281             for (IMarker marker : bugGroup.getAllMarkers()) {
282                 if (isFiltered(marker)) {
283                     count++;
284                 }
285             }
286             bugCount = Integer.valueOf(count);
287             filteredMarkersMap.put(bugGroup, bugCount);
288         }
289         return bugCount.intValue();
290     }

312     /**
313      * @param desiredType
314      *            type of the child groups
315      * @param parents
316      *            all the common parents
317      * @param parent
318      *            "empty" root node, which must be filled in with markers from
319      *            set of given parents
320      * @return
321      */
322     private synchronized Object[] createChildren(GroupType desiredType, Set<IResource> parents, BugGroup parent) {
323         Set<IMarker> markerSet = new HashSet<IMarker>();
324         boolean filterActive = isBugFilterActive();
325         Set<String> patternFilter = getPatternFilter();
326         for (IResource resource : parents) {
327             IMarker[] markers = getMarkers(resource);
328             for (IMarker marker : markers) {
329                 boolean added = markerSet.add(marker);
330                 if (filterActive && added && MarkerUtil.isFiltered(marker, patternFilter)) {
331                     filteredMarkers.add(marker);
332                 }
333             }
334         }
335         parent.setMarkers(markerSet);
336         return createGroups(parent, desiredType.getMapper());
337     }

339     /**
340      * @param <Identifier>
341      *            object type, like String or Integer
342      * @param mapper
343      *            maps between BugInstance and group
344      */
345     private synchronized <Identifier> Object[] createGroups(BugGroup parent, MarkerMapper<Identifier> mapper) {
346         if (mapper == MarkerMapper.NO_MAPPING) {
347             return parent.getAllMarkers().toArray(new IMarker[0]);
348         }
349         Set<IMarker> allMarkers = parent.getAllMarkers();
350         GroupType childType = grouping.getChildType(mapper.getType());
351         Map<Identifier, Set<IMarker>> groupIds = new HashMap<Identifier, Set<IMarker>>();
352         UserPreferences prefs = FindbugsPlugin.getCorePreferences(null, false);
353         Set<String> disabledPlugins = prefs.getCustomPlugins(false);
354 
355         // first, sort all bugs to the sets with same identifier type
356         Set<String> errorMessages = new HashSet<String>();
357         for (IMarker marker : allMarkers) {
358             Identifier id = mapper.getIdentifier(marker);
359             if (id == null) {
360                 String pluginId = MarkerUtil.getPluginId(marker);
361                 if(pluginId.length() == 0 || disabledPlugins.contains(pluginId)){
362                     // do not report errors for disabled plugins
363                     continue;
364                 }
365                 try {
366                     String debugDescription = mapper.getDebugDescription(marker);
367                     if(errorMessages.contains(debugDescription)) {
368                         continue;
369                     }
370                     errorMessages.add(debugDescription);
371                     if(FindbugsPlugin.getDefault().isDebugging()){
372                         FindbugsPlugin.getDefault().logWarning(
373                             "BugContentProvider.createGroups: failed to find " + debugDescription + " for marker on file "
374                                     + marker.getResource());
375                     }
376                 } catch (CoreException e) {
377                     FindbugsPlugin.getDefault().logException(e, "Exception on retrieving debug data for: " + mapper.getType());
378                 }
379                 continue;
380             }
381             if (!groupIds.containsKey(id)) {
382                 groupIds.put(id, new HashSet<IMarker>());
383             }
384             groupIds.get(id).add(marker);
385         }
386 
387         // now create groups from the sorted bug sets
388         Set<Entry<Identifier, Set<IMarker>>> typesSet = groupIds.entrySet();
389         BugGroup[] children = new BugGroup[typesSet.size()];
390         boolean lastlevel = grouping.getChildType(mapper.getType()) == GroupType.Marker;
391         int i = 0;
392         for (Entry<Identifier, Set<IMarker>> entry : typesSet) {
393             Identifier groupId = entry.getKey();
394             children[i] = new BugGroup(parent, groupId, mapper.getType());
395             children[i].setMarkers(entry.getValue());
396             i++;
397         }
398         if (!lastlevel) {
399             for (BugGroup bugGroup : children) {
400                 // recursive call
401                 createGroups(bugGroup, childType.getMapper());
402             }
403         }
404         return children;
405     }

466     /**
467      * @return list of the *visible* parents with changed children to refresh
468      *         the viewer. Returns empty list if the full refresh is needed
469      */
470     public synchronized Set<BugGroup> updateContent(List<DeltaInfo> deltas) {
471         int oldRootSize = rootElement.getChildren().length;
472         Set<BugGroup> changedParents = new HashSet<BugGroup>();
473         bugFilterActive = isBugFilterActive();
474         Set<String> patternFilter = getPatternFilter();
475         for (DeltaInfo delta : deltas) {
476             if (DEBUG) {
477                 System.out.println(delta + " (contentProvider.updateContent)");
478             }
479             IMarker changedMarker = delta.marker;
480             switch (delta.changeKind) {
481             case IResourceDelta.REMOVED:
482                 BugGroup parent = findParent(changedMarker);
483                 if (parent == null) {
484                     if (DEBUG) {
485                         System.out.println(delta + " IGNORED because marker does not have parent!");
486                     }
487                     continue;
488                 }
489                 removeMarker(parent, changedMarker, changedParents);
490                 break;
491             case IResourceDelta.ADDED:
492                 if (!changedMarker.exists()) {
493                     if (DEBUG) {
494                         System.out.println(delta + " IGNORED because marker does not exists anymore!");
495                     }
496                     continue;
497                 }
498                 addMarker(changedMarker, changedParents, patternFilter);
499                 break;
500             default:
501                     FindbugsPlugin.getDefault()
502                     .logWarning("UKNOWN delta change kind" + delta.changeKind);
503 
504             }
505         }
506         if (rootElement.getMarkersCount() == 0 || rootElement.getChildren().length != oldRootSize) {
507             if (oldRootSize == 0 || rootElement.getChildren().length == 0) {
508                 changedParents.clear();
509             } else {
510                 changedParents.add(rootElement);
511             }
512             return changedParents;
513         }
514         // XXX this is a fix for not updating of children on incremental build
515         // and
516         // for "empty" groups which can never be empty... I don't know where the
517         // bug is...
518         changedParents.add(rootElement);
519         return changedParents;
520     }

522     private void removeMarker(BugGroup parent, IMarker marker, Set<BugGroup> changedParents) {
523         BugGroup accessibleParent = getFirstAccessibleParent(parent);
524         changedParents.add(accessibleParent);
525         removeMarker(parent, marker);
526     }

528     private void addMarker(IMarker toAdd, Set<BugGroup> changedParents, Set<String> patternFilter) {
529         MarkerMapper<?> mapper = grouping.getFirstType().getMapper();
530         // filter through working set
531         IMarker marker = toAdd;
532         if (!resourceFilter.contains(marker.getResource())) {
533             return;
534         }
535         rootElement.addMarker(marker);
536         addMarker(marker, mapper, rootElement, changedParents, patternFilter);
537     }

539     private <Identifier> void addMarker(IMarker marker, MarkerMapper<Identifier> mapper, BugGroup parent,
540             Set<BugGroup> changedParents, Set<String> patternFilter) {
541 
542         if (mapper == MarkerMapper.NO_MAPPING) {
543             return;
544         }
545 
546         Identifier id = mapper.getIdentifier(marker);
547         if (id == null) {
548             if(FindbugsPlugin.getDefault().isDebugging()){
549                 FindbugsPlugin.getDefault().logWarning(
550                     "BugContentProvider.createPatternGroups: " + "Failed to find bug id of type " + mapper.getType()
551                             + " for marker on file " + marker.getResource());
552             }
553             return;
554         }
555 
556         // search for the right node to insert. there cannot be more then one
557         BugGroup matchingChild = null;
558         for (Object object : parent.getChildren()) {
559             if (!(object instanceof BugGroup)) {
560                 break;
561             }
562             BugGroup group = (BugGroup) object;
563             if (id.equals(group.getData())) {
564                 matchingChild = group;
565                 break;
566             }
567         }
568 
569         GroupType childType = grouping.getChildType(mapper.getType());
570         boolean filtered = bugFilterActive && MarkerUtil.isFiltered(marker, patternFilter);
571         if (filtered) {
572             filteredMarkers.add(marker);
573         }
574 
575         if (matchingChild != null) {
576             // node exists? fine, add marker and update children
577             matchingChild.addMarker(marker);
578             if (filtered) {
579                 Integer count = filteredMarkersMap.get(matchingChild);
580                 if (count == null) {
581                     count = Integer.valueOf(0);
582                 }
583                 filteredMarkersMap.put(matchingChild, Integer.valueOf(count.intValue() + 1));
584             }
585 
586             if (isAccessible(matchingChild)) {
587                 // update only first visible parent element to avoid multiple
588                 // refreshes
589                 changedParents.add(getFirstAccessibleParent(matchingChild));
590             }
591             boolean lastlevel = childType == GroupType.Marker;
592             if (!lastlevel) {
593                 addMarker(marker, childType.getMapper(), matchingChild, changedParents, patternFilter);
594             }
595         } else {
596             // if there is no node, create one and recursively all children to
597             // the last
598             // level
599             BugGroup group = new BugGroup(parent, id, mapper.getType());
600             group.addMarker(marker);
601             if (filtered) {
602                 filteredMarkersMap.put(group, Integer.valueOf(1));
603             }
604             createGroups(group, childType.getMapper());
605         }
606     }

608     /**
609      * @return true if the element is accessible from the content viewer point
610      *         of view. After "go into" action parent elements can became
611      *         inaccessible to the viewer
612      */
613     public boolean isAccessible(BugGroup group) {
614         BugGroup rootGroup;
615         if (!(input instanceof BugGroup)) {
616             rootGroup = rootElement;
617         } else {
618             rootGroup = (BugGroup) input;
619         }
620 
621         if (grouping.compare(rootGroup.getType(), group.getType()) < 0) {
622             return true;
623         }
624         return false;
625     }

627     private void removeMarker(BugGroup parent, IMarker marker) {
628         parent.removeMarker(marker);
629         List<BugGroup> parents = getSelfAndParents(parent);
630         boolean filtered = isFiltered(marker);
631         for (BugGroup group : parents) {
632             if (group.getMarkersCount() == 0 && group.getParent() instanceof BugGroup) {
633                 removeGroup((BugGroup) group.getParent(), group);
634             }
635             if (filtered) {
636                 Integer count = filteredMarkersMap.get(group);
637                 if (count != null && count.intValue() > 0) {
638                     filteredMarkersMap.put(group, Integer.valueOf(count.intValue() - 1));
639                 }
640             }
641         }
642         filteredMarkers.remove(marker);
643     }
644 
645     private void removeGroup(BugGroup parent, BugGroup child) {
646         if (parent.removeChild(child)) {
647             filteredMarkersMap.remove(child);
648             if (parent.getMarkersCount() == 0 && parent.getParent() instanceof BugGroup) {
649                 removeGroup((BugGroup) parent.getParent(), parent);
650             }
651         }
652     }
653 
654     private List<BugGroup> getSelfAndParents(BugGroup child) {
655         List<BugGroup> parents = new ArrayList<BugGroup>();
656         parents.add(child);
657         while (child.getParent() instanceof BugGroup) {
658             child = (BugGroup) child.getParent();
659             parents.add(child);
660         }
661         return parents;
662     }

664     private BugGroup getFirstAccessibleParent(BugGroup element) {
665         if (element.getParent() instanceof BugGroup) {
666             BugGroup parent = (BugGroup) element.getParent();
667             if (!isAccessible(parent)) {
668                 return element;
669             }
670             return getFirstAccessibleParent(parent);
671         }
672         return element;
673     }

675     private BugGroup findParent(IMarker marker) {
676 
677         Object[] rootObjects = rootElement.getChildren();
678 
679         GroupType parentType = grouping.getParentType(GroupType.Marker);
680         if (parentType == GroupType.Undefined) {
681             return null;
682         }
683         // traverse through all groups looking for one with marker
684         for (int i = 0; i < rootObjects.length; i++) {
685             Object object = rootObjects[i];
686             if (!(object instanceof BugGroup)) {
687                 break;
688             }
689             BugGroup group = (BugGroup) object;
690             if (group.contains(marker)) {
691                 if (group.getType() == parentType) {
692                     return group;
693                 }
694                 rootObjects = group.getChildren();
695                 i = -1;
696             }
697         }
698         return null;
699     }

724     public Set<Object> getShowInTargets(Object obj) {
725         Set<Object> supported = new HashSet<Object>();
726         if (obj instanceof BugGroup) {
727             supported.add(obj);
728         } else if (obj instanceof IMarker) {
729             supported.add(obj);
730         } else if (obj instanceof IJavaProject) {
731             return getShowInTargets(((IJavaProject) obj).getProject());
732         } else if (obj instanceof IClassFile) {
733             return getShowInTargets(((IClassFile) obj).getType());
734         } else if (obj instanceof IFile) {
735             IJavaElement javaElement = JavaCore.create((IFile) obj);
736             return getShowInTargets(javaElement);
737         } else if (obj instanceof IFolder) {
738             IJavaElement javaElement = JavaCore.create((IFolder) obj);
739             return getShowInTargets(javaElement);
740         } else if (obj instanceof IStructuredSelection) {
741             IStructuredSelection selection = (IStructuredSelection) obj;
742             Iterator<?> iter = selection.iterator();
743             while (iter.hasNext()) {
744                 Object object = iter.next();
745                 supported.add(getShowInTargets(object));
746             }
747         } else {
748             // TODO think how improve performance for project/package objects?
749             Set<IMarker> markers = MarkerUtil.getMarkers(obj);
750             boolean found = false;
751             main: for (IMarker marker : markers) {
752                 BugGroup group = findParent(marker);
753                 if (group == null) {
754                     continue;
755                 }
756                 List<BugGroup> selfAndParents = getSelfAndParents(group);
757                 for (BugGroup bugGroup : selfAndParents) {
758                     if (obj.equals(bugGroup.getData())) {
759                         supported.add(bugGroup);
760                         found = true;
761                         break main;
762                     }
763                 }
764             }
765             if (!found) {
766                 supported.addAll(markers);
767             }
768         }
769         return supported;
770     }
