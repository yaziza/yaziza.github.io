112     /**
113      * Create an Eclipse marker for given BugInstance.
114      *
115      * @param javaProject
116      *            the project
117      * @param monitor
118      */
119     public static void createMarkers(final IJavaProject javaProject, final SortedBugCollection theCollection, final ISchedulingRule rule, IProgressMonitor monitor) {
120         if (monitor.isCanceled()) {
121             return;
122         }
123         final List<MarkerParameter> bugParameters = createBugParameters(javaProject, theCollection, monitor);
124         if (monitor.isCanceled()) {
125             return;
126         }
127         WorkspaceJob wsJob = new WorkspaceJob("Creating FindBugs markers") {
128 
129             @Override
130             public IStatus runInWorkspace(IProgressMonitor monitor1) throws CoreException {
131                 IProject project = javaProject.getProject();
132                 try {
133                     new MarkerReporter(bugParameters, theCollection, project).run(monitor1);
134                 } catch (CoreException e) {
135                     FindbugsPlugin.getDefault().logException(e, "Core exception on add marker");
136                     return e.getStatus();
137                 }
138                 return monitor1.isCanceled()? Status.CANCEL_STATUS : Status.OK_STATUS;
139             }
140         };
141         wsJob.setRule(rule);
142         wsJob.setSystem(true);
143         wsJob.setUser(false);
144         wsJob.schedule();
145     }

147     /**
148      * As a side-effect this method updates missing line information for some
149      * bugs stored in the given bug collection
150      *
151      * @param project
152      * @param theCollection
153      * @return never null
154      */
155     public static List<MarkerParameter> createBugParameters(IJavaProject project, BugCollection theCollection,
156             IProgressMonitor monitor) {
157         List<MarkerParameter> bugParameters = new ArrayList<MarkerParameter>();
158         if (project == null) {
159             FindbugsPlugin.getDefault().logException(new NullPointerException("project is null"), "project is null");
160             return bugParameters;
161         }
162         Iterator<BugInstance> iterator = theCollection.iterator();
163         while (iterator.hasNext() && !monitor.isCanceled()) {
164             BugInstance bug = iterator.next();
165             DetectorFactory detectorFactory = bug.getDetectorFactory();
166             if(detectorFactory != null && !detectorFactory.getPlugin().isGloballyEnabled()){
167                 continue;
168             }
169             MarkerParameter mp = createMarkerParameter(project, bug);
170             if (mp != null) {
171                 bugParameters.add(mp);
172             }
173         }
174         return bugParameters;
175     }

177     private static MarkerParameter createMarkerParameter(IJavaProject project, BugInstance bug) {
178         IJavaElement type = null;
179         WorkItem resource = null;
180         try {
181             type = getJavaElement(bug, project);
182             if (type != null) {
183                 resource = new WorkItem(type);
184             }
185         } catch (JavaModelException e1) {
186             FindbugsPlugin.getDefault().logException(e1, "Could not find Java type for FindBugs warning");
187         }
188         if (resource == null) {
189             if (Reporter.DEBUG) {
190                 reportNoResourceFound(bug);
191             }
192             return null;
193         }
194 
195         // default - first class line
196         int primaryLine = bug.getPrimarySourceLineAnnotation().getStartLine();
197 
198         // FindBugs needs originally generated primary line in order to find the
199         // bug again.
200         // Sometimes this primary line is <= 0, which causes Eclipse editor to
201         // ignore it
202         // So we check if we can replace the "wrong" primary line with a
203         // "better" start line
204         // If not, we just provide two values - one for Eclipse, another for
205         // FindBugs itself.
206 
207         int startLine = DONT_KNOW_LINE;
208 
209         // only update if we don't already tried it
210         if(primaryLine != DONT_KNOW_LINE) {
211             // XXX "first line of a file" is too simplistic. What if we have inner types?
212             if (primaryLine <= 1 && type instanceof IType) {
213                 IType iType = (IType) type;
214                 try {
215                     startLine = getLineStart(iType);
216                     if(startLine > 0) {
217                         if (Reporter.DEBUG) {
218                             System.out.println("4. Fixed start line to: " + startLine + " on " + type.getElementName());
219                         }
220                     }
221                 } catch (JavaModelException e1) {
222                     FindbugsPlugin.getDefault().logException(e1, "Could not find source line for Java type " + type
223                             + "for FindBugs warning: " + bug);
224                 }
225             }
226         }
227 
228         if (primaryLine <= 0 && startLine <= 0) {
229             // We have to provide line number, otherwise editor wouldn't show it
230             startLine = 1;
231             if (Reporter.DEBUG) {
232                 System.out.println("5. Fixed start line to *default* (1) on " + type);
233             }
234         }
235 
236         MarkerParameter parameter;
237         if (startLine > 0) {
238             parameter = new MarkerParameter(bug, resource, startLine, primaryLine);
239         } else {
240             parameter = new MarkerParameter(bug, resource, primaryLine, primaryLine);
241         }
242 //        if (Reporter.DEBUG) {
243 //            System.out
244 //            .println("Creating marker for " + resource.getPath() + ": line " + parameter.primaryLine + bug.getMessage());
245 //        }
246         return parameter;
247     }

251         String packageName = null;
252         ClassAnnotation primaryClass = bug.getPrimaryClass();
253         if (primaryClass != null) {
254             className = primaryClass.getClassName();
255             packageName = primaryClass.getPackageName();
256         }
257         if (Reporter.DEBUG) {
258             System.out.println("NOT found resource for a BUG in BUG in class: " //$NON-NLS-1$
259                     + packageName + "." //$NON-NLS-1$
260                     + className + ": \n\t" //$NON-NLS-1$
261                     + bug.getMessage() + " / Annotation: " //$NON-NLS-1$
262                     + bug.getAnnotationText() + " / Source Line: " //$NON-NLS-1$
263                     + bug.getPrimarySourceLineAnnotation());
264         }

267     /**
268      * Get the underlying resource (Java class) for given BugInstance.
269      *
270      * @param bug
271      *            the BugInstance
272      * @param project
273      *            the project
274      * @return the IResource representing the Java class
275      */
276     private static @CheckForNull
277     IJavaElement getJavaElement(BugInstance bug, IJavaProject project) throws JavaModelException {
278 
279         SourceLineAnnotation primarySourceLineAnnotation = bug.getPrimarySourceLineAnnotation();
280         String qualifiedClassName = primarySourceLineAnnotation.getClassName();
281 
282 //        if (Reporter.DEBUG) {
283 //            System.out.println("Looking up class: " + packageName + ", " + qualifiedClassName);
284 //        }
285 
286         Matcher m = fullName.matcher(qualifiedClassName);
287         IType type;
288         String innerName = null;
289         if (m.matches() && m.group(2).length() > 0) {
290 
291             String outerQualifiedClassName = m.group(1).replace('$', '.');
292             innerName = m.group(2).substring(1);
293             // second argument is required to find also secondary types
294             type = project.findType(outerQualifiedClassName, (IProgressMonitor) null);
295 
296             /*
297              * code below only points to the first line of inner class even if
298              * this is not a class bug but field bug
299              */
300             if (type != null && !hasLineInfo(primarySourceLineAnnotation)) {
301                 completeInnerClassInfo(qualifiedClassName, innerName, type, bug);
302             }
303         } else {
304             // second argument is required to find also secondary types
305             type = project.findType(qualifiedClassName.replace('$', '.'), (IProgressMonitor) null);
306 
307             // for inner classes, some detectors does not properly report source
308             // lines:
309             // instead of reporting the first line of inner class, they report
310             // first line of parent class
311             // in this case we will try to fix this here and point to the right
312             // start line
313             if (type != null && type.isMember()) {
314                 if (!hasLineInfo(primarySourceLineAnnotation)) {
315                     completeInnerClassInfo(qualifiedClassName, type.getElementName(), type, bug);
316                 }
317             }
318         }
319 
320         // reassign it as it may be changed for inner classes
321         primarySourceLineAnnotation = bug.getPrimarySourceLineAnnotation();
322 
323         /*
324          * Eclipse can help us find the line number for fields => we trying to
325          * add line info for fields here
326          */
327         int startLine = primarySourceLineAnnotation.getStartLine();
328         // TODO don't use "1", use "0" ?
329         if (startLine <= 1 && type != null) {
330             FieldAnnotation primaryField = bug.getPrimaryField();
331             if(primaryField != null) {
332                 completeFieldInfo(qualifiedClassName, type, bug, primaryField);
333             }
334         }
335 
336         return type;
337     }

339     private static boolean hasLineInfo(SourceLineAnnotation annotation) {
340         // XXX don't use "1", use "0" ?
341         return annotation != null && annotation.getStartLine() > 1;
342     }

344     private static void completeFieldInfo(String qualifiedClassName,
345             @Nonnull IType type, @Nonnull BugInstance bug, @Nonnull FieldAnnotation field) {
346 
347         IField ifield = type.getField(field.getFieldName());
348         ISourceRange sourceRange = null;
349         JavaModelException ex = null;
350         try {
351             sourceRange = ifield.getNameRange();
352         } catch (JavaModelException e) {
353             ex = e;
354         }
355         IScanner scanner = null;
356         try {
357             // second try...
358             if (sourceRange == null) {
359                 sourceRange = ifield.getSourceRange();
360             }
361             scanner = initScanner(type, sourceRange);
362         } catch (JavaModelException e) {
363             String message = "Can not complete field annotation " + field + " for the field: " + ifield + " in class: "
364                     + qualifiedClassName + ", type " + type + ", bug " + bug;
365             if (ex != null) {
366                 // report only first one
367                 e = ex;
368             }
369             FindbugsPlugin.getDefault().logMessage(IStatus.WARNING, message, e);
370         }
371         if (scanner == null || sourceRange == null) {
372             return;
373         }
374         int lineNbr = scanner.getLineNumber(sourceRange.getOffset());
375         if(lineNbr > 0){
376             String sourceFileStr = getSourceFileHint(type, qualifiedClassName);
377             field.setSourceLines(new SourceLineAnnotation(qualifiedClassName, sourceFileStr, lineNbr, lineNbr, 0, 0));
378             if (Reporter.DEBUG) {
379                 System.out.println("2. Fixed start line to: " + lineNbr + " on " + qualifiedClassName);
380             }
381         }
382     }

393     private static void completeInnerClassInfo(String qualifiedClassName, String innerName, @Nonnull IType type, BugInstance bug) throws JavaModelException {
394         int lineNbr = findChildSourceLine(type, innerName, bug);
395         if(lineNbr > 0){
396             String sourceFileStr = getSourceFileHint(type, qualifiedClassName);
397             if (sourceFileStr != null && sourceFileStr.length() > 0) {
398                 bug.addSourceLine(new SourceLineAnnotation(qualifiedClassName, sourceFileStr, lineNbr, lineNbr, 0, 0));
399                 if (Reporter.DEBUG) {
400                     System.out.println("1. Fixed start line to: " + lineNbr + " on " + qualifiedClassName + "$" + innerName);
401                 }
402             }
403         }
404     }

406     /**
407      * @return start line of given type, or {@link #DONT_KNOW_LINE} if line could not be found
408      */
409     private static int getLineStart(IType source) throws JavaModelException {
410         ISourceRange range = source.getNameRange();
411         if (range == null) {
412             range = source.getSourceRange();
413         }
414         IScanner scanner = initScanner(source, range);
415         if (scanner != null && range != null) {
416             return scanner.getLineNumber(range.getOffset());
417         }
418         return DONT_KNOW_LINE;
419     }

421     /**
422      * @param source
423      *            must be not null
424      * @param range
425      *            can be null
426      * @return may return null, otherwise an initialized scanner which may
427      *         answer which source offset index belongs to which source line
428      * @throws JavaModelException
429      */
430     private static IScanner initScanner(IType source, ISourceRange range) throws JavaModelException {
431         if (range == null) {
432             return null;
433         }
434         char[] charContent = getContent(source);
435         if (charContent == null) {
436             return null;
437         }
438         IScanner scanner = ToolFactory.createScanner(false, false, false, true);
439         scanner.setSource(charContent);
440         int offset = range.getOffset();
441         try {
442             while (scanner.getNextToken() != ITerminalSymbols.TokenNameEOF) {
443                 // do nothing, just wait for the end of stream
444                 if (offset <= scanner.getCurrentTokenEndPosition()) {
445                     break;
446                 }
447             }
448         } catch (InvalidInputException e) {
449             FindbugsPlugin.getDefault().logException(e, "Could not init scanner for type: " + source);
450         }
451         return scanner;
452     }

454     @SuppressWarnings("restriction")
455     private static char[] getContent(IType source) throws JavaModelException {
456         char[] charContent = null;
457         IOpenable op = source.getOpenable();
458         if (op instanceof CompilationUnit) {
459             charContent = ((CompilationUnit) (op)).getContents();
460         }
461         if (charContent == null) {
462             String content = source.getSource();
463             if (content != null) {
464                 charContent = content.toCharArray();
465             }
466         }
467         return charContent;
468     }

470     private static int findChildSourceLine(IType parentType, String name, BugInstance bug) throws JavaModelException {
471         if (parentType == null) {
472             return -1;
473         }
474         char firstChar = name.charAt(0);
475         boolean firstIsDigit = Character.isDigit(firstChar);
476         if (!firstIsDigit) {
477             return findInnerClassSourceLine(parentType, name);
478         }
479         boolean innerFromMember = firstIsDigit && name.length() > 1 && !Character.isDigit(name.charAt(1));
480         if (innerFromMember) {
481             return findInnerClassSourceLine(parentType, name.substring(1));
482         }
483         return findInnerAnonymousClassSourceLine(parentType, name);
484     }

486     private static int findInnerAnonymousClassSourceLine(IJavaElement parentType, String innerName) throws JavaModelException {
487         IType anon = JdtUtils.findAnonymous((IType) parentType, innerName);
488         if (anon != null) {
489             return getLineStart(anon);
490         }
491         return DONT_KNOW_LINE;
492     }

494     private static int findInnerClassSourceLine(IJavaElement type, String name) throws JavaModelException {
495         String elemName = type.getElementName();
496         if (name.equals(elemName)) {
497             if (type instanceof IType) {
498                 return getLineStart((IType) type);
499             }
500         }
501         if (type instanceof IParent) {
502             IJavaElement[] children = ((IParent) type).getChildren();
503             for (int i = 0; i < children.length; i++) {
504                 // recursive call
505                 int line = findInnerClassSourceLine(children[i], name);
506                 if (line > 0) {
507                     return line;
508                 }
509             }
510         }
511         return DONT_KNOW_LINE;
512     }

553     public static void redisplayMarkers(final IJavaProject javaProject) {
554         final IProject project = javaProject.getProject();
555         FindBugsJob job = new FindBugsJob("Refreshing FindBugs markers", project) {
556             @Override
557             protected void runWithProgress(IProgressMonitor monitor) throws CoreException {
558                 // TODO in case we removed some of previously available
559                 // detectors, we should
560                 // throw away bugs reported by them
561 
562                 // Get the saved bug collection for the project
563                 SortedBugCollection bugs = FindbugsPlugin.getBugCollection(project, monitor);
564                 // Remove old markers
565                 project.deleteMarkers(FindBugsMarker.NAME, true, IResource.DEPTH_INFINITE);
566                 // Display warnings
567                 createMarkers(javaProject, bugs, project, monitor);
568             }
569         };
570         job.setRule(project);
571         job.scheduleInteractive();
572     }

650     public static Set<IMarker> findMarkerForJavaElement(IJavaElement elt, IMarker[] possibleCandidates, boolean recursive) {
651         String id = elt.getHandleIdentifier();
652         Set<IMarker> markers = new HashSet<IMarker>();
653         for (IMarker marker : possibleCandidates) {
654             try {
655                 Object elementId = marker.getAttribute(FindBugsMarker.UNIQUE_JAVA_ID);
656                 // UNIQUE_JAVA_ID exists first since 1.3.9 as FB attribute
657                 if (!(elementId instanceof String)) {
658                     continue;
659                 }
660                 String stringId = (String) elementId;
661                 if (!recursive) {
662                     if (stringId.equals(id)) {
663                         // exact match
664                         markers.add(marker);
665                     } else if (isDirectChild(id, stringId)) {
666                         // direct child: class in the package, but not in the
667                         // sub-package
668                         markers.add(marker);
669                     }
670                 } else if (stringId.startsWith(id)) {
671                     markers.add(marker);
672                 }
673             } catch (CoreException e) {
674                 FindbugsPlugin.getDefault().logException(e, "Marker does not contain valid java element id");
675                 continue;
676             }
677         }
678         return markers;
679     }

681     /**
682      *
683      * @param parentId
684      *            java element id of a parent element
685      * @param childId
686      *            java element id of possible child
687      * @return true if the second string represents a java element which is a
688      *         direct child of the parent element.
689      */
690     @SuppressWarnings("restriction")
691     private static boolean isDirectChild(String parentId, String childId) {
692         return childId.startsWith(parentId) && (childId.length() > (parentId.length() + 1))
693                 // if there is NOT a class file separator, then it's not a direct child
694                 && childId.charAt(parentId.length()) == JavaElement.JEM_CLASSFILE;
695     }

744     /**
745      * Find the BugCollectionAndInstance associated with given FindBugs marker.
746      *
747      * @param marker
748      *            a FindBugs marker
749      * @return the BugInstance associated with the marker, or null if we can't
750      *         find the BugInstance
751      */
752     public static @CheckForNull
753     BugCollectionAndInstance findBugCollectionAndInstanceForMarker(IMarker marker) {
754 
755         IResource resource = marker.getResource();
756         IProject project = resource.getProject();
757         if (project == null) {
758             // Also shouldn't happen.
759             FindbugsPlugin.getDefault().logError("No project for warning marker");
760             return null;
761         }
762         if (!isFindBugsMarker(marker)) {
763             // log disabled because otherwise each selection in problems view
764             // generates
765             // 6 new errors (we need refactor all bug views to get rid of this).
766             // FindbugsPlugin.getDefault().logError("Selected marker is not a FindBugs marker");
767             // FindbugsPlugin.getDefault().logError(marker.getType());
768             // FindbugsPlugin.getDefault().logError(FindBugsMarker.NAME);
769             return null;
770         }
771 
772         // We have a FindBugs marker. Get the corresponding BugInstance.
773         String bugId = marker.getAttribute(FindBugsMarker.UNIQUE_ID, null);
774         if (bugId == null) {
775             FindbugsPlugin.getDefault().logError("Marker does not contain unique id for warning");
776             return null;
777         }
778 
779         try {
780             BugCollection bugCollection = FindbugsPlugin.getBugCollection(project, null);
781             if (bugCollection == null) {
782                 FindbugsPlugin.getDefault().logError("Could not get BugCollection for FindBugs marker");
783                 return null;
784             }
785 
786             String bugType = (String) marker.getAttribute(FindBugsMarker.BUG_TYPE);
787             Integer primaryLineNumber = (Integer) marker.getAttribute(FindBugsMarker.PRIMARY_LINE);
788 
789             // compatibility
790             if (primaryLineNumber == null) {
791                 primaryLineNumber = Integer.valueOf(getEditorLine(marker));
792             }
793 
794             if (bugType == null) {
795                 FindbugsPlugin.getDefault().logError(
796                         "Could not get find attributes for marker " + marker + ": (" + bugId + ", " + primaryLineNumber + ")");
797                 return null;
798             }
799             BugInstance bug = bugCollection.findBug(bugId, bugType, primaryLineNumber.intValue());
800             if(bug == null) {
801                 FindbugsPlugin.getDefault().logError(
802                         "Could not get find bug for marker on " + resource + ": (" + bugId + ", " + primaryLineNumber + ")");
803                 return null;
804             }
805             return new BugCollectionAndInstance(bugCollection, bug);
806         } catch (CoreException e) {
807             FindbugsPlugin.getDefault().logException(e, "Could not get BugInstance for FindBugs marker");
808             return null;
809         }
810     }

816     /**
817      * Fish an IMarker out of given selection.
818      *
819      * @param selection
820      *            the selection
821      * @return the selected IMarker, or null if we can't find an IMarker in the
822      *         selection
823      */
824     public static Set<IMarker> getMarkerFromSelection(ISelection selection) {
825         Set<IMarker> markers = new HashSet<IMarker>();
826         if (!(selection instanceof IStructuredSelection)) {
827             return markers;
828         }
829         IStructuredSelection sSelection = (IStructuredSelection) selection;
830         for (Iterator<?> iter = sSelection.iterator(); iter.hasNext();) {
831             Object next = iter.next();
832             markers.addAll(getMarkers(next));
833         }
834         return markers;
835     }

837     public static Set<IMarker> getMarkers(Object obj) {
838         Set<IMarker> markers = new HashSet<IMarker>();
839         if (obj instanceof IMarker) {
840             IMarker marker = (IMarker) obj;
841             if (isFindBugsMarker(marker)) {
842                 markers.add(marker);
843             }
844         } else if (obj instanceof BugGroup) {
845             BugGroup group = (BugGroup) obj;
846             markers.addAll(group.getAllMarkers());
847         } else if (obj instanceof IResource) {
848             IResource res = (IResource) obj;
849             IMarker[] markers2 = MarkerUtil.getAllMarkers(res);
850             for (IMarker marker : markers2) {
851                 markers.add(marker);
852             }
853         } else if (obj instanceof IJavaElement) {
854             markers.addAll(new WorkItem((IJavaElement) obj).getMarkers(true));
855         } else if (obj instanceof IAdaptable) {
856             IAdaptable adapter = (IAdaptable) obj;
857             IMarker marker = (IMarker) adapter.getAdapter(IMarker.class);
858             if (marker == null) {
859                 IResource resource = (IResource) adapter.getAdapter(IResource.class);
860                 if (resource == null) {
861                     return markers;
862                 }
863                 IMarker[] markers2 = getMarkers(resource, IResource.DEPTH_INFINITE);
864                 markers.addAll(Arrays.asList(markers2));
865             } else if (isFindBugsMarker(marker)) {
866                 markers.add(marker);
867             }
868         }
869         return markers;
870     }

872     /**
873      * Tries to retrieve right bug marker for given selection. If there are many
874      * markers for given editor, and text selection doesn't match any of them,
875      * return null. If there is only one marker for given editor, returns this
876      * marker in any case.
877      *
878      * @param selection
879      * @param editor
880      * @return may return null
881      */
882     public static IMarker getMarkerFromEditor(ITextSelection selection, IEditorPart editor) {
883         IResource resource = (IResource) editor.getEditorInput().getAdapter(IFile.class);
884         IMarker[] allMarkers;
885         if (resource != null) {
886             allMarkers = getMarkers(resource, IResource.DEPTH_ZERO);
887         } else {
888             IClassFile classFile = (IClassFile) editor.getEditorInput().getAdapter(IClassFile.class);
889             if (classFile == null) {
890                 return null;
891             }
892             Set<IMarker> markers = getMarkers(classFile.getType());
893             allMarkers = markers.toArray(new IMarker[markers.size()]);
894         }
895         // if editor contains only one FB marker, do some cheating and always
896         // return it.
897         if (allMarkers.length == 1) {
898             return allMarkers[0];
899         }
900         // +1 because it counts real lines, but editor shows lines + 1
901         int startLine = selection.getStartLine() + 1;
902         for (IMarker marker : allMarkers) {
903             int line = getEditorLine(marker);
904             if (startLine == line) {
905                 return marker;
906             }
907         }
908         return null;
909     }

911     public static IMarker getMarkerFromSingleSelection(ISelection selection) {
912         if (selection instanceof ITextSelection) {
913             IEditorPart editor = FindbugsPlugin.getActiveWorkbenchWindow().getActivePage().getActiveEditor();
914             if (!(editor instanceof ITextEditor)) {
915                 return null;
916             }
917             IMarker marker = MarkerUtil.getMarkerFromEditor((ITextSelection) selection, (ITextEditor) editor);
918             if (marker != null) {
919                 selection = new StructuredSelection(marker);
920             } else {
921                 selection = new StructuredSelection();
922             }
923         }
924 
925         if (!(selection instanceof IStructuredSelection)) {
926             return null;
927         }
928         IStructuredSelection sSelection = (IStructuredSelection) selection;
929         if (sSelection.size() != 1) {
930             return null;
931         }
932 
933         Object next = sSelection.getFirstElement();
934         if (next instanceof IMarker) {
935             IMarker marker = (IMarker) next;
936             if (!isFindBugsMarker(marker)) {
937                 return null;
938             }
939             return marker;
940         } else if (next instanceof BugGroup) {
941             return null;
942         } else if (next instanceof IResource) {
943             return null;
944         } else if (next instanceof IAdaptable) {
945             IAdaptable adapter = (IAdaptable) next;
946             IMarker marker = (IMarker) adapter.getAdapter(IMarker.class);
947             if (!isFindBugsMarker(marker)) {
948                 return null;
949             }
950             return marker;
951         }
952         return null;
953     }

961             FindbugsPlugin.getDefault().logException(e, "Exception while checking FindBugs type on marker.");
962         }
963         return false;

1009     /**
1010      * @param marker
1011      *            might be null
1012      * @param bugIdToFilter
1013      *            might be null
1014      * @return true if marker should be filtered
1015      */
1016     public static boolean isFiltered(IMarker marker, Set<String> bugIdToFilter) {
1017         if (marker == null) {
1018             return true;
1019         }
1020         if (bugIdToFilter == null) {
1021             return false;
1022         }
1023         String pattern = marker.getAttribute(FindBugsMarker.BUG_TYPE, "not found");
1024         String patternType = marker.getAttribute(FindBugsMarker.PATTERN_TYPE, "not found");
1025         for (String badId : bugIdToFilter) {
1026             if (badId.equals(patternType) || badId.equals(pattern)) {
1027                 return true;
1028             }
1029         }
1030         return false;
1031     }
