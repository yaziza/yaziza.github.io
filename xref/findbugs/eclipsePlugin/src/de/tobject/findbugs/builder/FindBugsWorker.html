92      public FindBugsWorker(IResource resource, IProgressMonitor monitor) throws CoreException {
93          super();
94          this.resource = resource;
95          this.project = resource.getProject();
96          this.javaProject = JavaCore.create(project);
97          if (javaProject == null || !javaProject.exists() || !javaProject.getProject().isOpen()) {
98              throw new CoreException(FindbugsPlugin.createErrorStatus("Java project is not open or does not exist: " + project,
99                      null));
100         }
101         this.monitor = monitor;
102         // clone is required because we rewrite project relative references to absolute
103         this.userPrefs = FindbugsPlugin.getUserPreferences(project).clone();
104     }

121     /**
122      * Run FindBugs on the given collection of resources from same project
123      * (note: This is currently not thread-safe)
124      *
125      * @param resources
126      *            files or directories which should be on the project classpath.
127      *            All resources must belong to the same project, and no one of
128      *            the elements can contain another one. Ergo, if the list
129      *            contains a project itself, then it must have only one element.
130      * @throws CoreException
131      */
132     public void work(List<WorkItem> resources) throws CoreException {
133         if (resources == null || resources.isEmpty()) {
134             if (DEBUG) {
135                 FindbugsPlugin.getDefault().logInfo("No resources to analyse for project " + project);
136             }
137             return;
138         }
139         if (DEBUG) {
140             System.out.println(resources);
141         }
142         st = new StopTimer();
143         st.newPoint("initPlugins");
144 
145         // make sure it's initialized
146         FindbugsPlugin.applyCustomDetectors(false);
147 
148         st.newPoint("clearMarkers");
149 
150         // clear markers
151         clearMarkers(resources);
152 
153         st.newPoint("configureOutputFiles");
154 
155         final Project findBugsProject = new Project();
156         findBugsProject.setProjectName(javaProject.getElementName());
157         final Reporter bugReporter = new Reporter(javaProject, findBugsProject, monitor);
158         if (FindBugsConsole.getConsole() != null) {
159             bugReporter.setReportingStream(FindBugsConsole.getConsole().newOutputStream());
160         }
161         bugReporter.setPriorityThreshold(userPrefs.getUserDetectorThreshold());
162 
163         FindBugs.setHome(FindbugsPlugin.getFindBugsEnginePluginLocation());
164 
165         Map<IPath, IPath> outLocations = createOutputLocations();
166 
167         // collect all related class/jar/war etc files for analysis
168         collectClassFiles(resources, outLocations, findBugsProject);
169 
170         // attach source directories (can be used by some detectors, see
171         // SwitchFallthrough)
172         configureSourceDirectories(findBugsProject, outLocations);
173 
174         if (findBugsProject.getFileCount() == 0) {
175             if (DEBUG) {
176                 FindbugsPlugin.getDefault().logInfo("No resources to analyse for project " + project);
177             }
178             return;
179         }
180 
181         st.newPoint("createAuxClasspath");
182 
183         String[] classPathEntries = createAuxClasspath();
184         // add to findbugs classpath
185         for (String entry : classPathEntries) {
186             findBugsProject.addAuxClasspathEntry(entry);
187         }
188         String cloudId = userPrefs.getCloudId();
189         if (cloudId != null) {
190             findBugsProject.setCloudId(cloudId);
191         }
192 
193 
194         st.newPoint("configureProps");
195         IPreferenceStore store = FindbugsPlugin.getPluginPreferences(project);
196         boolean cacheClassData = store.getBoolean(FindBugsConstants.KEY_CACHE_CLASS_DATA);
197 
198         final FindBugs2 findBugs = new FindBugs2Eclipse(project, cacheClassData, bugReporter);
199         findBugs.setNoClassOk(true);
200         findBugs.setProject(findBugsProject);
201         findBugs.setBugReporter(bugReporter);
202         findBugs.setProgressCallback(bugReporter);
203 
204         findBugs.setDetectorFactoryCollection(DetectorFactoryCollection.instance());
205 
206         // configure detectors.
207         userPrefs.setIncludeFilterFiles(relativeToAbsolute(userPrefs.getIncludeFilterFiles()));
208         userPrefs.setExcludeFilterFiles(relativeToAbsolute(userPrefs.getExcludeFilterFiles()));
209         userPrefs.setExcludeBugsFiles(relativeToAbsolute(userPrefs.getExcludeBugsFiles()));
210         findBugs.setUserPreferences(userPrefs);
211 
212         // configure extended preferences
213         findBugs.setAnalysisFeatureSettings(userPrefs.getAnalysisFeatureSettings());
214         findBugs.setMergeSimilarWarnings(false);
215 
216         if(cacheClassData) {
217             FindBugs2Eclipse.checkClassPathChanges(findBugs.getProject().getAuxClasspathEntryList(), project);
218         }
219 
220         st.newPoint("runFindBugs");
221         if (DEBUG) {
222             FindbugsPlugin.log("Running findbugs");
223         }
224 
225         runFindBugs(findBugs);
226         if (DEBUG) {
227             FindbugsPlugin.log("Done running findbugs");
228         }
229 
230         // Merge new results into existing results
231         // if the argument is project, then it's not incremental
232         boolean incremental = !(resources.get(0) instanceof IProject);
233         updateBugCollection(findBugsProject, bugReporter, incremental);
234         st.newPoint("done");
235         st = null;
236         monitor.done();
237     }

241     private void configureSourceDirectories(Project findBugsProject, Map<IPath, IPath> outLocations) {
242         Set<IPath> srcDirs = outLocations.keySet();
243         for (IPath iPath : srcDirs) {
244             findBugsProject.addSourceDir(iPath.toOSString());
245         }
246     }

248     /**
249      * Load existing FindBugs xml report for the given collection of files.
250      *
251      * @param fileName
252      *            xml file name to load bugs from
253      * @throws CoreException
254      */
255     public void loadXml(String fileName) throws CoreException {
256         if (fileName == null) {
257             return;
258         }
259         st = new StopTimer();
260 
261         // clear markers
262         clearMarkers(null);
263 
264         final Project findBugsProject = new Project();
265         final Reporter bugReporter = new Reporter(javaProject, findBugsProject, monitor);
266         bugReporter.setPriorityThreshold(userPrefs.getUserDetectorThreshold());
267 
268         reportFromXml(fileName, findBugsProject, bugReporter);
269         // Merge new results into existing results.
270         updateBugCollection(findBugsProject, bugReporter, false);
271         monitor.done();
272     }

275      * Clear associated markers
276      *
277      * @param files
278      */
279     private void clearMarkers(List<WorkItem> files) throws CoreException {
280         if (files == null) {
281             project.deleteMarkers(FindBugsMarker.NAME, true, IResource.DEPTH_INFINITE);
282             return;
283         }
284         for (WorkItem item : files) {
285             if (item != null) {
286                 item.clearMarkers();
287             }
288         }
289     }

307     /**
308      * this method will block current thread until the findbugs is running
309      *
310      * @param findBugs
311      *            fb engine, which will be <b>disposed</b> after the analysis is
312      *            done
313      */
314     private void runFindBugs(final FindBugs2 findBugs) {
315         if (DEBUG) {
316             FindbugsPlugin.log("Running findbugs in thread " + Thread.currentThread().getName());
317         }
318         System.setProperty("findbugs.progress", "true");
319         try {
320             // Perform the analysis! (note: This is not thread-safe)
321             findBugs.execute();
322         } catch (InterruptedException e) {
323             if (DEBUG) {
324                 FindbugsPlugin.getDefault().logException(e, "Worker interrupted");
325             }
326             Thread.currentThread().interrupt();
327         } catch (IOException e) {
328             FindbugsPlugin.getDefault().logException(e, "Error performing FindBugs analysis");
329         } finally {
330             findBugs.dispose();
331         }
332     }

334     void logDirty(SortedBugCollection bugCollection) {
335         if (true) {
336             return;
337         }
338         int count = 0;
339         for(BugInstance b : bugCollection) {
340             BugDesignation bd = b.getUserDesignation();
341             if (bd == null) {
342                 continue;
343             }
344             if (bd.isDirty()) {
345                 count++;
346             }
347 
348         }
349         if (count > 0) {
350             new RuntimeException("Found " + count + " dirty designations").printStackTrace(System.out);
351         }
352     }
353     /**
354      * Update the BugCollection for the project.
355      *
356      * @param findBugsProject
357      *            FindBugs project representing analyzed classes
358      * @param bugReporter
359      *            Reporter used to collect the new warnings
360      */
361     private void updateBugCollection(Project findBugsProject, Reporter bugReporter, boolean incremental) {
362         SortedBugCollection newBugCollection = bugReporter.getBugCollection();
363         logDirty(newBugCollection);
364         try {
365             st.newPoint("getBugCollection");
366             SortedBugCollection oldBugCollection = FindbugsPlugin.getBugCollection(project, monitor, false);
367             logDirty(oldBugCollection);
368 
369             st.newPoint("mergeBugCollections");
370             SortedBugCollection resultCollection = mergeBugCollections(oldBugCollection, newBugCollection, incremental);
371             logDirty(resultCollection);
372              resultCollection.getProject().setGuiCallback(new EclipseGuiCallback(project));
373             resultCollection.setTimestamp(System.currentTimeMillis());
374             resultCollection.setDoNotUseCloud(false);
375             resultCollection.reinitializeCloud();
376             logDirty(resultCollection);
377 
378             // will store bugs in the default FB file + Eclipse project session
379             // props
380             st.newPoint("storeBugCollection");
381             FindbugsPlugin.storeBugCollection(project, resultCollection, monitor);
382         } catch (IOException e) {
383             FindbugsPlugin.getDefault().logException(e, "Error performing FindBugs results update");
384         } catch (CoreException e) {
385             FindbugsPlugin.getDefault().logException(e, "Error performing FindBugs results update");
386         }
387 
388         // will store bugs as markers in Eclipse workspace
389         st.newPoint("createMarkers");
390         MarkerUtil.createMarkers(javaProject, newBugCollection, resource, monitor);
391     }

393     private SortedBugCollection mergeBugCollections(SortedBugCollection firstCollection, SortedBugCollection secondCollection,
394             boolean incremental) {
395         Update update = new Update();
396         // TODO copyDeadBugs must be true, otherwise incremental compile leads
397         // to
398         // unknown bug instances appearing (merged collection doesn't contain
399         // all bugs)
400         boolean copyDeadBugs = incremental;
401         SortedBugCollection merged = (SortedBugCollection) (update.mergeCollections(firstCollection, secondCollection,
402                 copyDeadBugs, incremental));
403         return merged;
404     }

406     private Map<String, Boolean> relativeToAbsolute(Map<String, Boolean> map) {
407         Map<String, Boolean> resultMap = new TreeMap<String, Boolean>();
408         for (Entry<String, Boolean> entry : map.entrySet()) {
409             if(!entry.getValue().booleanValue()) {
410                 continue;
411             }
412             String filePath = entry.getKey();
413             IPath path = getFilterPath(filePath, project);
414             if (!path.toFile().exists()) {
415                 FindbugsPlugin.getDefault().logWarning("Filter not found: " + filePath);
416                 continue;
417             }
418             String filterName = path.toOSString();
419             resultMap.put(filterName, Boolean.TRUE);
420         }
421         return resultMap;
422     }

424     /**
425      * Checks the given path and convert it to absolute path if it is specified
426      * relative to the given project or workspace
427      *
428      * @param filePath
429      *            project relative OR workspace relative OR absolute OS file
430      *            path (1.3.8+ version)
431      * @param project
432      *            might be null (only for workspace relative or absolute paths)
433      * @return absolute path which matches given relative or absolute path,
434      *         never null
435      */
436     public static IPath getFilterPath(String filePath, IProject project) {
437         IPath path = new Path(filePath);
438         if (path.isAbsolute()) {
439             return path;
440         }
441         if (project != null) {
442             // try first project relative location
443             IPath newPath = project.getLocation().append(path);
444             if (newPath.toFile().exists()) {
445                 return newPath;
446             }
447         }
448 
449         // try to resolve relative to workspace (if we use workspace properties
450         // for project)
451         IPath wspLocation = ResourcesPlugin.getWorkspace().getRoot().getLocation();
452         IPath newPath = wspLocation.append(path);
453         if (newPath.toFile().exists()) {
454             return newPath;
455         }
456 
457         // something which we have no idea what it can be (or missing/wrong file
458         // path)
459         return path;
460     }

462     /**
463      * Checks the given absolute path and convert it to relative path if it is
464      * relative to the given project or workspace. This representation can be
465      * used to store filter paths in user preferences file
466      *
467      * @param filePath
468      *            absolute OS file path
469      * @param project
470      *            might be null
471      * @return filter file path as stored in preferences which matches given
472      *         path
473      */
474     public static IPath toFilterPath(String filePath, IProject project) {
475         IPath path = new Path(filePath);
476         IPath commonPath;
477         if (project != null) {
478             commonPath = project.getLocation();
479             IPath relativePath = getRelativePath(path, commonPath);
480             if (!relativePath.equals(path)) {
481                 return relativePath;
482             }
483         }
484         commonPath = ResourcesPlugin.getWorkspace().getRoot().getLocation();
485         return getRelativePath(path, commonPath);
486     }

488     /**
489      * @param filePath
490      *            path (eventually absolute) to a file
491      * @param commonPath
492      *            absolute path of some common location
493      * @return path relative to common root if given path is contained under the
494      *         common directory, otherwise unchanged path
495      */
496     private static IPath getRelativePath(IPath filePath, IPath commonPath) {
497         if (!filePath.isAbsolute()) {
498             return filePath;
499         }
500         // since Equinox 3.5 we can use IPath.makeRelativeTo(IPath)
501         IPath relativeTo = filePath.makeRelativeTo(commonPath);
502         return relativeTo;
503     }

512     /**
513      * @return map of all source folders to output folders, for current java
514      *         project, where both are represented by absolute IPath objects
515      *
516      * @throws CoreException
517      */
518     private Map<IPath, IPath> createOutputLocations() throws CoreException {
519 
520         Map<IPath, IPath> srcToOutputMap = new HashMap<IPath, IPath>();
521 
522         // get the default location => relative to wsp
523         IPath defaultOutputLocation = ResourceUtils.relativeToAbsolute(javaProject.getOutputLocation());
524         IWorkspaceRoot root = ResourcesPlugin.getWorkspace().getRoot();
525         // path to the project without project name itself
526         IClasspathEntry entries[] = javaProject.getResolvedClasspath(true);
527         for (int i = 0; i < entries.length; i++) {
528             IClasspathEntry classpathEntry = entries[i];
529             if (classpathEntry.getEntryKind() == IClasspathEntry.CPE_SOURCE) {
530                 IPath outputLocation = ResourceUtils.getOutputLocation(classpathEntry, defaultOutputLocation);
531                 if(outputLocation == null) {
532                     continue;
533                 }
534                 IResource cpeResource = root.findMember(classpathEntry.getPath());
535                 // patch from 2891041: do not analyze derived "source" folders
536                 // because they probably contain auto-generated classes
537                 if (cpeResource != null && cpeResource.isDerived()) {
538                     continue;
539                 }
540                 // TODO not clear if it is absolute in workspace or in global FS
541                 IPath srcLocation = ResourceUtils.relativeToAbsolute(classpathEntry.getPath());
542                 if(srcLocation != null) {
543                     srcToOutputMap.put(srcLocation, outputLocation);
544                 }
545             }
546         }
547 
548         return srcToOutputMap;
549     }

551     private void reportFromXml(final String xmlFileName, final Project findBugsProject, final Reporter bugReporter) {
552         if (!"".equals(xmlFileName)) {
553             FileInputStream input = null;
554             try {
555                 input = new FileInputStream(xmlFileName);
556                 bugReporter.reportBugsFromXml(input, findBugsProject);
557             } catch (FileNotFoundException e) {
558                 FindbugsPlugin.getDefault().logException(e, "XML file not found: " + xmlFileName);
559             } catch (DocumentException e) {
560                 FindbugsPlugin.getDefault().logException(e, "Invalid XML file: " + xmlFileName);
561             } catch (IOException e) {
562                 FindbugsPlugin.getDefault().logException(e, "Error loading FindBugs results xml file: " + xmlFileName);
563             } finally {
564                 IO.closeQuietly(input);
565             }
566         }
567     }
