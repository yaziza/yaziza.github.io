209     @Override
210     public void start(BundleContext context) throws Exception {
211         super.start(context);
212 
213         Version.registerApplication("FindBugs-Eclipse", Version.RELEASE);
214 
215          // configure debugging
216         configurePluginDebugOptions();
217 
218         // initialize resource strings
219         try {
220             resourceBundle = ResourceBundle.getBundle("de.tobject.findbugs.messages"); //this is correct //$NON-NLS-1$
221         } catch (MissingResourceException x) {
222             resourceBundle = null;
223         }
224 
225         if (System.getProperty("findbugs.home") == null) {
226             // TODO workaround for findbugs home property
227             // - see de.tobject.findbugs.builder.FindBugsWorker.work() too
228             String findBugsHome = getFindBugsEnginePluginLocation();
229             if (DEBUG) {
230                 logInfo("FindBugs home is: " + findBugsHome);
231             }
232             System.setProperty("findbugs.home", findBugsHome);
233         }
234         if (System.getProperty("findbugs.cloud.default") == null) {
235             // TODO workaround for findbugs default cloud property
236             // - see edu.umd.cs.findbugs.cloud.CloudFactory and messages.xml
237             String defCloud = DEFAULT_CLOUD_ID;
238             if (DEBUG) {
239                 logInfo("Using default cloud: " + defCloud);
240             }
241             System.setProperty("findbugs.cloud.default", defCloud);
242         }
243 
244         /** Don't load main classes */
245         FindBugs.setNoMains();
246 
247         // enable source searching for "fall" (ignore case) in switch statements
248         SystemProperties.setProperty("findbugs.sf.comment", "true");
249 
250         // Register our save participant
251         FindbugsSaveParticipant saveParticipant = new FindbugsSaveParticipant();
252         ResourcesPlugin.getWorkspace().addSaveParticipant(PLUGIN_ID, saveParticipant);
253     }

255     public static void dumpClassLoader(Class<?> c) {
256         System.out.printf("Class loaders for %s:%n", c.getName());
257         ClassLoader loader = c.getClassLoader();
258         while (loader != null) {
259             System.out.printf("  %s %s%n", loader.toString(),  loader.getClass().getSimpleName());
260             loader = loader.getParent();
261         }

265     /**
266      * The complexity of the code below is partly caused by the fact that we
267      * might have multiple ways to install and/or enable custom plugins. There
268      * are plugins discovered by FB itself, plugins contributed to Eclipse and
269      * plugins added by user manually via properties. Plugins can be disabled
270      * via code or properties. The code below is still work in progress, see
271      * also {@link DetectorProvider#getPluginElements(UserPreferences)}.
272      *
273      * @param detectorPaths
274      *            list of possible detector plugins
275      * @param force
276      *            true if we MUST set plugins even if the given list is empty
277      */
278     public static synchronized void applyCustomDetectors(boolean force) {
279         if(customDetectorsInitialized && !force) {
280             return;
281         }
282         customDetectorsInitialized = true;
283         DetectorValidator validator = new DetectorValidator();
284         final SortedSet<String> detectorPaths = new TreeSet<String>();
285         SortedMap<String, String> contributedDetectors = DetectorsExtensionHelper.getContributedDetectors();
286         UserPreferences corePreferences = getCorePreferences(null, force);
287         detectorPaths.addAll(corePreferences.getCustomPlugins(true));
288         if(DEBUG) {
289             dumpClassLoader(FindbugsPlugin.class);
290             dumpClassLoader(Plugin.class);
291             System.out.println("applyCustomDetectors - going to add " + detectorPaths.size() + " plugin urls...");
292             for (String url : detectorPaths) {
293                 System.out.println("\t" + url);
294             }
295         }
296 
297         // disable custom plugins configured via properties, if they are already loaded
298         Set<String> disabledPlugins = corePreferences.getCustomPlugins(false);
299         Map<URI, Plugin> allPlugins = Plugin.getAllPluginsMap();
300         for (Entry<URI, Plugin> entry : allPlugins.entrySet()) {
301             Plugin fbPlugin = entry.getValue();
302             String pluginId = fbPlugin.getPluginId();
303             // ignore all custom plugins with the same plugin id as already loaded
304             if(contributedDetectors.containsKey(pluginId)) {
305                 contributedDetectors.remove(pluginId);
306                 detectorPaths.remove(pluginId);
307             }
308 
309             if (fbPlugin.isCorePlugin() || fbPlugin.isInitialPlugin()) {
310                 continue;
311             }
312             if (disabledPlugins.contains(entry.getKey().getPath())
313                     || disabledPlugins.contains(pluginId)) {
314                 fbPlugin.setGloballyEnabled(false);
315                 Plugin.removeCustomPlugin(fbPlugin);
316                 if (DEBUG) {
317                     System.out.println("Removed plugin: " + fbPlugin + " loaded from " + entry.getKey());
318                 }
319             }
320         }
321 
322         HashSet<Plugin> enabled = new HashSet<Plugin>();
323 
324         // adding FindBugs *Eclipse* plugins, key plugin id, value is path
325         for (Entry<String, String> entry : contributedDetectors.entrySet()) {
326             String pluginId = entry.getKey();
327             String pluginPath = entry.getValue();
328             URI uri = new File(pluginPath).toURI();
329             if (disabledPlugins.contains(pluginId)
330                     || disabledPlugins.contains(pluginPath)
331                     || allPlugins.containsKey(uri)) {
332                 continue;
333             }
334             addCustomPlugin(enabled, uri);
335         }
336 
337         // adding custom plugins configured via properties, but only if they are not loaded yet
338         for (String path : detectorPaths) {
339             // this is plugin id, so we can't use it as URL
340             if(new Path(path).segmentCount() == 1) {
341                 continue;
342             }
343             path = FindBugsWorker.getFilterPath(path, null).toOSString();
344             URI uri = new File(path).toURI();
345             if(allPlugins.containsKey(uri)) {
346                 continue;
347             }
348             ValidationStatus status = validator.validate(path);
349             if (status.isOK()) {
350                 addCustomPlugin(enabled, uri);
351             } else {
352                 getDefault().getLog().log(status);
353             }
354         }
355 
356         if (DEBUG) {
357             System.out.println("applyCustomDetectors - there was " + detectorPaths.size()
358                     + " extra FB plugin urls with " + enabled.size()
359                     + " valid FB plugins and " + allPlugins.size()
360                     + " total plugins registered by FB.");
361             for (Entry<URI, Plugin> entry : allPlugins.entrySet()) {
362                 Plugin fbPlugin = entry.getValue();
363                 if (fbPlugin.isGloballyEnabled()) {
364                     System.out.println("IS  enabled:\t" + fbPlugin.getPluginId());
365                 } else {
366                     System.out.println("NOT enabled:\t" + fbPlugin.getPluginId());
367                 }
368             }
369         }
370     }

372     protected static void addCustomPlugin(HashSet<Plugin> enabled, URI uri) {
373         try {
374             // bug 3117769 - we must provide our own classloader
375             // to allow third-party plugins extend the classpath via
376             // "Buddy" classloading
377             // see also: Eclipse-BuddyPolicy attribute in MANIFEST.MF
378             Plugin fbPlugin = Plugin.addCustomPlugin(uri, FindbugsPlugin.class.getClassLoader());
379             if(fbPlugin != null) {
380                 // TODO line below required to enable this *optional* plugin
381                 // but it should be taken by FB core from the findbugs.xml,
382                 // which currently only works for *core* plugins only
383                 fbPlugin.setGloballyEnabled(true);
384                 enabled.add(fbPlugin);
385             }
386         } catch (PluginException e) {
387             getDefault().logException(e, "Failed to load plugin for custom detector: " + uri);
388         } catch (DuplicatePluginIdException e) {
389             getDefault().logException(e, e.getPluginId() + " already loaded from " + e.getPreviouslyLoadedFrom()
390                     + ", ignoring: " + uri);
391         }
392     }

474     public void configurePluginDebugOptions() {
475         if (isDebugging()) {
476             // debugging for the plugin itself
477             String option = Platform.getDebugOption(PLUGIN_DEBUG);
478             FindbugsPlugin.DEBUG = Boolean.valueOf(option).booleanValue();
479 
480             // debugging for the builder and friends
481             option = Platform.getDebugOption(BUILDER_DEBUG);
482             FindBugsBuilder.DEBUG = Boolean.valueOf(option).booleanValue();
483             FindBugsWorker.DEBUG = FindBugsBuilder.DEBUG;
484 
485             // debugging for the nature
486             option = Platform.getDebugOption(NATURE_DEBUG);
487             FindBugsNature.DEBUG = Boolean.valueOf(option).booleanValue();
488 
489             // debugging for the reporter
490             option = Platform.getDebugOption(REPORTER_DEBUG);
491             Reporter.DEBUG = Boolean.valueOf(option).booleanValue();
492 
493             // debugging for the content provider
494             option = Platform.getDebugOption(CONTENT_DEBUG);
495             BugContentProvider.DEBUG = Boolean.valueOf(option).booleanValue();
496 
497             option = Platform.getDebugOption(PROFILER_DEBUG);
498             if (Boolean.valueOf(option).booleanValue()) {
499                 System.setProperty("profiler.report", "true");
500             }
501         }
502     }

504     /**
505      * Find the filesystem path of the FindBugs plugin directory.
506      *
507      * @return the filesystem path of the FindBugs plugin directory, or null if
508      *         the FindBugs plugin directory cannot be found
509      */
510     public static String getFindBugsEnginePluginLocation() {
511         // findbugs.home should be set to the directory the plugin is
512         // installed in.
513         URL u = plugin.getBundle().getEntry("/");
514         try {
515             URL bundleRoot = FileLocator.resolve(u);
516             String path = bundleRoot.getPath();
517             if (FindBugsBuilder.DEBUG) {
518                 System.out.println("Pluginpath: " + path); //$NON-NLS-1$
519             }
520             if (path.endsWith("/eclipsePlugin/")) {
521                 File f = new File(path);
522                 f = f.getParentFile();
523                 f = new File(f, "findbugs");
524                 path = f.getPath() + "/";
525             }
526 
527             return path;
528         } catch (IOException e) {
529             FindbugsPlugin.getDefault().logException(e, "IO Exception locating engine plugin");
530         }
531         return null;
532     }

584     public void logMessage(int severity, String message, Throwable e) {
585         if (DEBUG) {
586             String what = (severity == IStatus.ERROR) ? (e != null ? "Exception" : "Error") : "Warning";
587             System.out.println(what + " in FindBugs plugin: " + message);
588             if (e != null) {
589                 e.printStackTrace();
590             }
591         }
592         IStatus status = createStatus(severity, message, e);
593         getLog().log(status);
594     }

697     /**
698      * Read saved bug collection and findbugs project from file. Will populate
699      * the bug collection and findbugs project session properties if successful.
700      * If there is no saved bug collection and project for the eclipse project,
701      * then FileNotFoundException will be thrown.
702      *
703      * @param project
704      *            the eclipse project
705      * @param monitor
706      *            a progress monitor
707      * @throws java.io.FileNotFoundException
708      *             the saved bug collection doesn't exist
709      * @throws IOException
710      * @throws DocumentException
711      * @throws CoreException
712      */
713     private static void readBugCollectionAndProject(IProject project, IProgressMonitor monitor, boolean useCloud)
714             throws IOException, DocumentException, CoreException {
715         SortedBugCollection bugCollection;
716 
717         IPath bugCollectionPath = getBugCollectionFile(project);
718         // Don't turn the path to an IFile because it isn't local to the
719         // project.
720         // see the javadoc for org.eclipse.core.runtime.Plugin
721         File bugCollectionFile = bugCollectionPath.toFile();
722         if (!bugCollectionFile.exists()) {
723             // throw new
724             // FileNotFoundException(bugCollectionFile.getLocation().toOSString());
725             getDefault().logInfo("creating new bug collection: " + bugCollectionPath.toOSString());
726             createDefaultEmptyBugCollection(project); // since we no longer
727             // throw, have to do this
728             // here
729             return;
730         }
731 
732         UserPreferences prefs = getUserPreferences(project);
733         bugCollection = new SortedBugCollection();
734         bugCollection.getProject().setGuiCallback(new EclipseGuiCallback(project));
735         bugCollection.setDoNotUseCloud(!useCloud);
736 
737         bugCollection.readXML(bugCollectionFile);
738         if (useCloud) {
739             String cloudId = prefs.getCloudId();
740             if (cloudId != null) {
741                 bugCollection.getProject().setCloudId(cloudId);
742             }
743         }
744 
745         cacheBugCollectionAndProject(project, bugCollection, bugCollection.getProject());
746     }

831     public static boolean isProjectSettingsEnabled(IProject project) {
832         // fast path: read from session, if available
833         Boolean enabled;
834         try {
835             enabled = (Boolean) project.getSessionProperty(SESSION_PROPERTY_SETTINGS_ON);
836         } catch (CoreException e) {
837             enabled = null;
838         }
839         if (enabled != null) {
840             return enabled.booleanValue();
841         }
842 
843         // legacy support: before 1.3.8, there was ONLY project preferences in
844         // .fbprefs
845         // so check if the file is there...
846         IFile file = getUserPreferencesFile(project);
847         boolean projectPropsEnabled = file.isAccessible();
848         if (projectPropsEnabled) {
849             ScopedPreferenceStore store = new ScopedPreferenceStore(new ProjectScope(project), PLUGIN_ID);
850             // so if the file is there, we can check if after 1.3.8 the flag is
851             // set
852             // to use workspace properties instead
853             projectPropsEnabled = !store.contains(FindBugsConstants.PROJECT_PROPS_DISABLED)
854                     || !store.getBoolean(FindBugsConstants.PROJECT_PROPS_DISABLED);
855         }
856         // remember in the session to speedup access, don't touch the store
857         setProjectSettingsEnabled(project, null, projectPropsEnabled);
858         return projectPropsEnabled;
859     }

970     /**
971      * Save current UserPreferences for given project or workspace.
972      *
973      * @param project
974      *            the project or null for workspace
975      * @throws CoreException
976      */
977     public static void saveUserPreferences(IProject project, final UserPreferences userPrefs) throws CoreException {
978 
979         FileOutput userPrefsOutput = new FileOutput() {
980             @Override
981             public void writeFile(OutputStream os) throws IOException {
982                 userPrefs.write(os);
983             }
984 
985             @Override
986             public String getTaskDescription() {
987                 return "writing user preferences";
988             }
989         };
990 
991         if (project != null) {
992             // Make the new user preferences current for the project
993             project.setSessionProperty(SESSION_PROPERTY_USERPREFS, userPrefs);
994             IFile userPrefsFile = getUserPreferencesFile(project);
995             ensureReadWrite(userPrefsFile);
996             IO.writeFile(userPrefsFile, userPrefsOutput, null);
997             if (project.getFile(DEPRECATED_PREFS_PATH).equals(userPrefsFile)) {
998                 String message = "Found old style FindBugs preferences for project '" + project.getName()
999                         + "'. This preferences are not at the default location: '" + DEFAULT_PREFS_PATH + "'." + " Please move '"
1000                         + DEPRECATED_PREFS_PATH + "' to '" + DEFAULT_PREFS_PATH + "'.";
1001                 getDefault().logWarning(message);
1002             }
1003         } else {
1004             // write the workspace preferences to the eclipse preference store
1005             ByteArrayOutputStream bos = new ByteArrayOutputStream(10000);
1006             try {
1007                 userPrefs.write(bos);
1008             } catch (IOException e) {
1009                 getDefault().logException(e, "Failed to write user preferences");
1010                 return;
1011             }
1012             Properties props = new Properties();
1013             try {
1014                 props.load(new ByteArrayInputStream(bos.toByteArray()));
1015             } catch (IOException e) {
1016                 getDefault().logException(e, "Failed to save user preferences");
1017                 return;
1018             }
1019             IPreferenceStore store = getDefault().getPreferenceStore();
1020             // Reset any existing custom group entries
1021             resetStore(store, UserPreferences.KEY_PLUGIN);
1022             resetStore(store, UserPreferences.KEY_EXCLUDE_BUGS);
1023             resetStore(store, UserPreferences.KEY_EXCLUDE_FILTER);
1024             resetStore(store, UserPreferences.KEY_INCLUDE_FILTER);
1025             for (Entry<Object, Object> entry : props.entrySet()) {
1026                 store.putValue((String) entry.getKey(), (String) entry.getValue());
1027             }
1028             if(store instanceof IPersistentPreferenceStore){
1029                 IPersistentPreferenceStore store2 = (IPersistentPreferenceStore) store;
1030                 try {
1031                     store2.save();
1032                 } catch (IOException e) {
1033                     getDefault().logException(e, "Failed to save user preferences");
1034                 }
1035             }
1036         }
1037     }

1039     /**
1040      * Removes all consequent enumerated keys from given store staring with given prefix
1041      */
1042     private static void resetStore(IPreferenceStore store, String prefix) {
1043         int start = 0;
1044         // 99 is paranoia.
1045         while(start < 99){
1046             String name = prefix + start;
1047             if(store.contains(name)){
1048                 store.setToDefault(name);
1049             } else {
1050                 break;
1051             }
1052             start ++;
1053         }
1054     }

1076     /**
1077      * Read UserPreferences for project from the file in the project directory.
1078      * Returns null if the preferences have not been saved to a file, or if
1079      * there is an error reading the preferences file.
1080      *
1081      * @param project
1082      *            the project to get the UserPreferences for
1083      * @return the UserPreferences, or null if the UserPreferences file could
1084      *         not be read
1085      * @throws CoreException
1086      */
1087     private static UserPreferences readUserPreferences(IProject project) throws CoreException {
1088         IFile userPrefsFile = getUserPreferencesFile(project);
1089         if (!userPrefsFile.exists()) {
1090             return null;
1091         }
1092         try {
1093             // force is preventing us for out-of-sync exception if file was
1094             // changed externally
1095             InputStream in = userPrefsFile.getContents(true);
1096             UserPreferences userPrefs = FindBugsPreferenceInitializer.createDefaultUserPreferences();
1097             userPrefs.read(in);
1098             return userPrefs;
1099         } catch (IOException e) {
1100             FindbugsPlugin.getDefault().logException(e, "Could not read user preferences for project");
1101             return null;
1102         }
1103     }

1106         IWorkbenchPage page = PlatformUI.getWorkbench().getActiveWorkbenchWindow().getActivePage();
1107         IViewPart view = page.findView(viewId);
1108         if (!page.isPartVisible(view)) {
1109             try {
1110                 view = page.showView(viewId);

1114             }
1115         }
1116         if (view instanceof IMarkerSelectionHandler) {
1117             IMarkerSelectionHandler handler = (IMarkerSelectionHandler) view;

1120             ISelectionListener listener = (ISelectionListener) view;
1121             listener.selectionChanged(source, new StructuredSelection(marker));
1122         }
