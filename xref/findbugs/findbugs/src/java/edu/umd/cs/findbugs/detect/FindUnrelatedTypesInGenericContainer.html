132      * correspondence, then the value is -1.

143         String call = methodName+sig;
144         Info info = new Info(c, argumentParameterIndex, typeParameterIndex);
145         callMap.add(call, info);

243     /**
244      * Visit the class context
245      *
246      * @see edu.umd.cs.findbugs.Detector#visitClassContext(edu.umd.cs.findbugs.ba.ClassContext)
247      */
248     @Override
249     public void visitClassContext(ClassContext classContext) {
250         JavaClass javaClass = classContext.getJavaClass();
251         Method[] methodList = javaClass.getMethods();
252 
253         for (Method method : methodList) {
254             if (method.getCode() == null) {
255                 continue;
256             }
257 
258             try {
259                 analyzeMethod(classContext, method);
260             } catch (MethodUnprofitableException e) {
261                 assert true; // move along; nothing to see
262             } catch (CFGBuilderException e) {
263                 String msg = "Detector " + this.getClass().getName() + " caught exception while analyzing "
264                         + javaClass.getClassName() + "." + method.getName() + " : " + method.getSignature();
265                 bugReporter.logError(msg, e);
266             } catch (DataflowAnalysisException e) {
267                 String msg = "Detector " + this.getClass().getName() + " caught exception while analyzing "
268                         + javaClass.getClassName() + "." + method.getName() + " : " + method.getSignature();
269                 bugReporter.logError(msg, e);
270             }
271         }
272     }

274     /**
275      * Use this to screen out methods that do not contain invocations.
276      */
277     public boolean prescreen(ClassContext classContext, Method method) {
278         BitSet bytecodeSet = classContext.getBytecodeSet(method);
279         return bytecodeSet != null
280                 && (bytecodeSet.get(Constants.INVOKEINTERFACE) || bytecodeSet.get(Constants.INVOKEVIRTUAL)
281                         || bytecodeSet.get(Constants.INVOKESPECIAL) || bytecodeSet.get(Constants.INVOKESTATIC) || bytecodeSet
282                         .get(Constants.INVOKENONVIRTUAL));
283     }

285     /**
286      * Methods marked with the "Synthetic" attribute do not appear in the source
287      * code
288      */
289     private boolean isSynthetic(Method m) {
290         if ((m.getAccessFlags() & Constants.ACC_SYNTHETIC) != 0) {
291             return true;
292         }
293         Attribute[] attrs = m.getAttributes();
294         for (Attribute attr : attrs) {
295             if (attr instanceof Synthetic) {
296                 return true;
297             }
298         }
299         return false;
300     }

310     private boolean isGenericCollection(ClassDescriptor operandClass) {
311         String dottedClassName = operandClass.getDottedClassName();
312 
313         if (baseGenericTypes.contains(dottedClassName)) {
314             return true;
315         }
316 
317         String found = null;
318         for(String c : baseGenericTypes) {
319             if (Subtypes2.instanceOf(operandClass, c)) {
320                 found = c;
321                 break;
322             }
323         }
324         if (found == null) {
325             return false;
326         }
327         if (dottedClassName.startsWith("java.util.") || dottedClassName.startsWith("com.google.common.collect.") ) {
328             return true;
329         }
330         try {
331             XClass xclass = Global.getAnalysisCache().getClassAnalysis(XClass.class, operandClass);
332 
333             String sig = xclass.getSourceSignature();
334             if (sig == null) {
335                 return false;
336             }
337 
338             String typeParameter = null;
339             List<String> split = GenericUtilities.split(sig, true);
340             if (sig.charAt(0) == '<') {
341                 int end = sig.indexOf(':');
342                 if (end > 0) {
343                     typeParameter = sig.substring(1, end);
344                 }
345 
346             }
347             if (DEBUG) {
348                 System.out.println(dottedClassName + " " + typeParameter + " " + split);
349             }
350             for (String s : split) {
351                 int i = s.indexOf('<');
352                 if (i < 0) {
353                     continue;
354                 }
355                 if (s.charAt(0) != 'L') {
356                     throw new IllegalStateException("unexpected non signature: " + s);
357                 }
358                 ClassDescriptor c = DescriptorFactory.createClassDescriptor(s.substring(1, i));
359                 String superTypeParameter = s.substring(i+1);
360                 if (isGenericCollection(c) && (typeParameter == null || superTypeParameter.startsWith("T" + typeParameter))) {
361                     if (DEBUG) {
362                         System.out.println(operandClass + " is a subtype of " + s);
363                     }
364                     return true;
365                 }
366             }
367             if (DEBUG) {
368                 System.out.println("Not a subtype");
369             }
370 
371 
372         } catch (CheckedAnalysisException e1) {
373             AnalysisContext.logError("Error checking for weird generic parameterization of " + operandClass, e1);
374         }
375         return false;
376     }

378     private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
379         if (isSynthetic(method) || !prescreen(classContext, method)) {
380             return;
381         }
382         XMethod xmethod = XFactory.createXMethod(classContext.getJavaClass(), method);
383         if (xmethod.isSynthetic()) {
384             return;
385         }
386 
387         BugAccumulator accumulator = new BugAccumulator(bugReporter);
388 
389         CFG cfg = classContext.getCFG(method);
390         TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
391         ValueNumberDataflow vnDataflow = classContext.getValueNumberDataflow(method);
392 
393         ConstantPoolGen cpg = classContext.getConstantPoolGen();
394         MethodGen methodGen = classContext.getMethodGen(method);
395         if (methodGen == null) {
396             return;
397         }
398         String fullMethodName = methodGen.getClassName() + "." + methodGen.getName();
399 
400         String sourceFile = classContext.getJavaClass().getSourceFileName();
401         if (DEBUG) {
402             System.out.println("\n" + fullMethodName);
403         }
404 
405         // Process each instruction
406         for (Iterator<Location> iter = cfg.locationIterator(); iter.hasNext();) {
407             Location location = iter.next();
408             InstructionHandle handle = location.getHandle();
409             Instruction ins = handle.getInstruction();
410 
411             // Only consider invoke instructions
412             if (!(ins instanceof InvokeInstruction)) {
413                 continue;
414             }
415 
416             InvokeInstruction inv = (InvokeInstruction) ins;
417 
418             XMethod invokedMethod = XFactory.createXMethod(inv, cpg);
419 
420             String invokedMethodName = invokedMethod.getName();
421             String argSignature = invokedMethod.getSignature();
422             argSignature = argSignature.substring(0, argSignature.indexOf(')') + 1);
423             String call = invokedMethodName+argSignature;
424             SignatureParser sigParser = new SignatureParser(inv.getSignature(cpg));
425 
426             Collection<Info> collection = callMap.get(call);
427             if (!callMap.containsKey(call)) {
428                 continue;
429             }
430             for(Info info : collection) {
431                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
432                 if (DEBUG) {
433                     System.out.println("at " + handle.getPosition() + " Checking call to " + info.interfaceForCall + " : " + invokedMethod);
434                 }
435                 try {
436                     if (!subtypes2.isSubtype(invokedMethod.getClassDescriptor(), info.interfaceForCall)) {
437                         continue;
438                     }
439                 } catch (ClassNotFoundException e) {
440                     if ("java/util/Collection".equals(info.interfaceForCall.getClassName())
441                             && "com.google.common.collect.Multiset".equals(invokedMethod.getClassName())) {
442                         assert true;
443                         // we know this is OK without needing to find definition of Multiset
444                     } else {
445                         AnalysisContext.reportMissingClass(e);
446                         continue;
447                     }
448                 }
449 
450                 boolean allMethod;
451 
452                 int typeArgument;
453                 if (info.typeIndex >= 0) {
454                     allMethod = false;
455                     typeArgument = info.typeIndex;
456                 } else {
457                     allMethod = true;
458                     typeArgument = -(1+info.typeIndex);
459                 }
460                 int pos = info.argumentIndex;
461 
462 
463                 int lhsPos;
464                 if (inv instanceof INVOKESTATIC) {
465                     lhsPos = sigParser.getSlotsFromTopOfStackForParameter(0);
466                 } else {
467                     lhsPos = sigParser.getTotalArgumentSize();
468                 }
469 
470                 int stackPos = sigParser.getSlotsFromTopOfStackForParameter(pos);
471 
472                 TypeFrame frame = typeDataflow.getFactAtLocation(location);
473                 if (!frame.isValid()) {
474                     // This basic block is probably dead
475                     continue;
476                 }
477 
478 
479                 Type operandType = frame.getStackValue(stackPos);
480                 if (operandType.equals(TopType.instance())) {
481                     // unreachable
482                     continue;
483                 }
484 
485                 if (operandType.equals(NullType.instance())) {
486                     // ignore
487                     continue;
488                 }
489 
490                 ValueNumberFrame vnFrame = vnDataflow.getFactAtLocation(location);
491 
492                 if (!vnFrame.isValid()) {
493                     AnalysisContext.logError("Invalid value number frame in " + xmethod);
494                     continue;
495                 }
496 
497                 ValueNumber objectVN = vnFrame.getStackValue(lhsPos);
498                 ValueNumber argVN = vnFrame.getStackValue(stackPos);
499 
500                 if (objectVN.equals(argVN)) {
501                     String bugPattern = "DMI_COLLECTIONS_SHOULD_NOT_CONTAIN_THEMSELVES";
502                     int priority = HIGH_PRIORITY;
503                     if ("removeAll".equals(invokedMethodName)) {
504                         bugPattern = "DMI_USING_REMOVEALL_TO_CLEAR_COLLECTION";
505                         priority = NORMAL_PRIORITY;
506                     } else if (invokedMethodName.endsWith("All")) {
507                         bugPattern = "DMI_VACUOUS_SELF_COLLECTION_CALL";
508                         priority = NORMAL_PRIORITY;
509                     }
510                     if (invokedMethodName.startsWith("contains")) {
511                         InstructionHandle next = handle.getNext();
512                         if (next != null) {
513                             Instruction nextIns = next.getInstruction();
514 
515                             if (nextIns instanceof InvokeInstruction) {
516                                 XMethod nextMethod = XFactory.createXMethod((InvokeInstruction) nextIns, cpg);
517                                 if ("assertFalse".equals(nextMethod.getName())) {
518                                     continue;
519                                 }
520                             }
521                         }
522                     }
523                     accumulator.accumulateBug(
524                             new BugInstance(this, bugPattern, priority)
525                             .addClassAndMethod(methodGen, sourceFile)
526                             .addCalledMethod(methodGen, (InvokeInstruction) ins)
527                             .addOptionalAnnotation(
528                                     ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, objectVN,
529                                             vnFrame, "INVOKED_ON")), SourceLineAnnotation.fromVisitedInstruction(
530                                                     classContext, methodGen, sourceFile, handle));
531                 }
532 
533                 // Only consider generic...
534                 Type objectType = frame.getStackValue(lhsPos);
535                 if (!(objectType instanceof GenericObjectType)) {
536                     continue;
537                 }
538 
539                 GenericObjectType operand = (GenericObjectType) objectType;
540 
541                 int expectedTypeParameters = 1;
542                 String simpleName = info.interfaceForCall.getSimpleName();
543                 if ( simpleName.toLowerCase().endsWith("map") || "Hashtable".equals(simpleName)) {
544                     expectedTypeParameters = 2;
545                 } else if ("Table".equals(simpleName)) {
546                     expectedTypeParameters = 3;
547                 }
548 
549                 // ... containers
550                 if (!operand.hasParameters()) {
551                     continue;
552                 }
553                 if (operand.getNumParameters() != expectedTypeParameters) {
554                     continue;
555                 }
556                 ClassDescriptor operandClass = DescriptorFactory.getClassDescriptor(operand);
557                 if (!isGenericCollection(operandClass)) {
558                     continue;
559                 }
560 
561                 if (expectedTypeParameters == 2 &&
562                         Subtypes2.instanceOf(operandClass, Map.class)
563                         && !TypeFrameModelingVisitor.isStraightGenericMap(operandClass)) {
564                     continue;
565                 }
566                 Type expectedType;
567                 if (allMethod) {
568                     expectedType = operand;
569                 } else {
570                     expectedType = operand.getParameterAt(typeArgument);
571                 }
572                 Type actualType = frame.getStackValue(stackPos);
573                 Type equalsType = actualType;
574                 if (allMethod) {
575                     if (!(actualType instanceof GenericObjectType)) {
576                         continue;
577                     }
578                     equalsType = ((GenericObjectType)actualType).getParameterAt(typeArgument);
579                 }
580 
581 
582                 IncompatibleTypes matchResult = compareTypes(expectedType, actualType, allMethod);
583 
584                 boolean parmIsObject = "Ljava/lang/Object;".equals(expectedType.getSignature());
585                 boolean selfOperation = !allMethod && operand.equals(actualType) && !parmIsObject;
586                 if (!allMethod && !parmIsObject && actualType instanceof GenericObjectType) {
587 
588                     GenericObjectType p2 = (GenericObjectType) actualType;
589                     List<? extends ReferenceType> parameters = p2.getParameters();
590                     if (parameters != null && parameters.equals(operand.getParameters())) {
591                         selfOperation = true;
592                     }
593                 }
594 
595                 if (!selfOperation && ( matchResult == IncompatibleTypes.SEEMS_OK || matchResult.getPriority() == Priorities.IGNORE_PRIORITY)) {
596                     continue;
597                 }
598 
599                 if (invokedMethodName.startsWith("contains") || "remove".equals(invokedMethodName)) {
600                     InstructionHandle next = handle.getNext();
601                     if (next != null) {
602                         Instruction nextIns = next.getInstruction();
603 
604                         if (nextIns instanceof InvokeInstruction) {
605                             XMethod nextMethod = XFactory.createXMethod((InvokeInstruction) nextIns, cpg);
606                             if ("assertFalse".equals(nextMethod.getName())) {
607                                 continue;
608                             }
609                         }
610                     }
611                 } else if ("get".equals(invokedMethodName) || "remove".equals(invokedMethodName)) {
612                     InstructionHandle next = handle.getNext();
613                     if (next != null) {
614                         Instruction nextIns = next.getInstruction();
615 
616                         if (nextIns instanceof InvokeInstruction) {
617                             XMethod nextMethod = XFactory.createXMethod((InvokeInstruction) nextIns, cpg);
618                             if ("assertNull".equals(nextMethod.getName())) {
619                                 continue;
620                             }
621                         }
622                     }
623                 }
624                 boolean noisy = false;
625                 if ("get".equals(invokedMethodName)) {
626                     UnconditionalValueDerefDataflow unconditionalValueDerefDataflow = classContext
627                             .getUnconditionalValueDerefDataflow(method);
628 
629                     UnconditionalValueDerefSet unconditionalDeref = unconditionalValueDerefDataflow.getFactAtLocation(location);
630                     ValueNumberFrame vnAfter = vnDataflow.getFactAfterLocation(location);
631                     ValueNumber top = vnAfter.getTopValue();
632                     noisy = unconditionalDeref.getValueNumbersThatAreUnconditionallyDereferenced().contains(top);
633                 }
634                 // Prepare bug report
635                 SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
636                         sourceFile, handle);
637 
638                 // Report a bug that mentions each of the failed arguments in
639                 // matches
640 
641                 if (expectedType instanceof GenericObjectType) {
642                     expectedType = ((GenericObjectType) expectedType).getUpperBound();
643                 }
644 
645                 int priority = matchResult.getPriority();
646                 if (!operandClass.getClassName().startsWith("java/util") && priority == Priorities.HIGH_PRIORITY) {
647                     priority = Math.max(priority, Priorities.NORMAL_PRIORITY);
648                 }
649                 if (TestCaseDetector.likelyTestCase(xmethod)) {
650                     priority = Math.max(priority, Priorities.NORMAL_PRIORITY);
651                 } else if (selfOperation) {
652                     priority = Priorities.HIGH_PRIORITY;
653                 }
654                 ClassDescriptor expectedClassDescriptor = DescriptorFactory
655                         .createClassOrObjectDescriptorFromSignature(expectedType.getSignature());
656                 ClassDescriptor actualClassDescriptor = DescriptorFactory.createClassOrObjectDescriptorFromSignature(equalsType
657                         .getSignature());
658                 ClassSummary classSummary = AnalysisContext.currentAnalysisContext().getClassSummary();
659                 Set<XMethod> targets = null;
660                 try {
661                     targets = Hierarchy2.resolveVirtualMethodCallTargets(actualClassDescriptor, "equals",
662                             "(Ljava/lang/Object;)Z", false, false);
663                     boolean allOk = targets.size() > 0;
664                     for (XMethod m2 : targets) {
665                         if (!classSummary.mightBeEqualTo(m2.getClassDescriptor(), expectedClassDescriptor)) {
666                             allOk = false;
667                         }
668                     }
669                     if (allOk) {
670                         priority += 2;
671                     }
672                 } catch (ClassNotFoundException e) {
673                     AnalysisContext.reportMissingClass(e);
674                 }
675                 String bugPattern = "GC_UNRELATED_TYPES";
676 
677                 BugInstance bug = new BugInstance(this, bugPattern, priority)
678                 .addClassAndMethod(methodGen, sourceFile)
679                 .addFoundAndExpectedType(actualType, expectedType)
680                 .addCalledMethod(methodGen, (InvokeInstruction) ins)
681                 .addOptionalAnnotation(
682                         ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, objectVN, vnFrame,
683                                 "INVOKED_ON"))
684                                 .addOptionalAnnotation(
685                                         ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, argVN, vnFrame, "ARGUMENT"))
686                                         .addEqualsMethodUsed(targets);
687                 if (noisy) {
688                     WarningPropertySet<WarningProperty> propertySet = new WarningPropertySet<WarningProperty>();
689 
690                     propertySet.addProperty(GeneralWarningProperty.NOISY_BUG);
691                     propertySet.decorateBugInstance(bug);
692                 }
693                 accumulator.accumulateBug(bug, sourceLineAnnotation);
694             }
695         }
696         accumulator.reportAccumulatedBugs();
697     }

699     /**
700      * Compare to see if the argument <code>argType</code> passed to the method
701      * matches the type of the corresponding parameter. The simplest case is
702      * when both are equal.
703      * <p>
704      * This is a conservative comparison - returns true if it cannot decide. If
705      * the parameter type is a type variable (e.g. <code>T</code>) then we don't
706      * know enough (yet) to decide if they do not match so return true.
707      *
708      * @param ignoreBaseType
709      *            TODO
710      */
711     private IncompatibleTypes compareTypes(Type expectedType, Type actualType, boolean ignoreBaseType) {
712         // XXX equality not implemented for GenericObjectType
713         // if (parmType.equals(argType)) return true;
714         if (expectedType == actualType) {
715             return IncompatibleTypes.SEEMS_OK;
716         }
717         // Compare type signatures instead
718         String expectedString = GenericUtilities.getString(expectedType);
719         String actualString = GenericUtilities.getString(actualType);
720         if (expectedString.equals(actualString)) {
721             return IncompatibleTypes.SEEMS_OK;
722         }
723 
724         if (expectedType.equals(Type.OBJECT))
725         {
726             return IncompatibleTypes.SEEMS_OK;
727             // if either type is java.lang.Object, then automatically true!
728             // again compare strings...
729         }
730 
731         String objString = GenericUtilities.getString(Type.OBJECT);
732 
733         if (expectedString.equals(objString)) {
734             return IncompatibleTypes.SEEMS_OK;
735         }
736 
737         // get a category for each type
738         TypeCategory expectedCat = GenericUtilities.getTypeCategory(expectedType);
739         TypeCategory argCat = GenericUtilities.getTypeCategory(actualType);
740         if (actualString.equals(objString) && expectedCat == TypeCategory.TYPE_VARIABLE) {
741             return IncompatibleTypes.SEEMS_OK;
742         }
743         if (expectedCat == TypeCategory.WILDCARD) {
744             return IncompatibleTypes.SEEMS_OK;
745         }
746         if (ignoreBaseType) {
747             if (expectedCat == TypeCategory.PARAMETERIZED && argCat == TypeCategory.PARAMETERIZED) {
748                 GenericObjectType parmGeneric = (GenericObjectType) expectedType;
749                 GenericObjectType argGeneric = (GenericObjectType) actualType;
750                 return compareTypeParameters(parmGeneric, argGeneric);
751             }
752             return IncompatibleTypes.SEEMS_OK;
753         }
754 
755         if (actualType.equals(Type.OBJECT) && expectedCat == TypeCategory.ARRAY_TYPE) {
756             return IncompatibleTypes.ARRAY_AND_OBJECT;
757         }
758 
759         // -~- plain objects are easy
760         if (expectedCat == TypeCategory.PLAIN_OBJECT_TYPE && argCat == TypeCategory.PLAIN_OBJECT_TYPE) {
761             return IncompatibleTypes.getPriorityForAssumingCompatible(expectedType, actualType, false);
762         }
763 
764         if (expectedCat == TypeCategory.PARAMETERIZED && argCat == TypeCategory.PLAIN_OBJECT_TYPE) {
765             return IncompatibleTypes.getPriorityForAssumingCompatible((GenericObjectType) expectedType, actualType);
766         }
767         if (expectedCat == TypeCategory.PLAIN_OBJECT_TYPE && argCat == TypeCategory.PARAMETERIZED) {
768             return IncompatibleTypes.getPriorityForAssumingCompatible((GenericObjectType) actualType, expectedType);
769         }
770 
771         // -~- parmType is: "? extends Another Type" OR "? super Another Type"
772         if (expectedCat == TypeCategory.WILDCARD_EXTENDS || expectedCat == TypeCategory.WILDCARD_SUPER) {
773             return compareTypes(((GenericObjectType) expectedType).getExtension(), actualType, ignoreBaseType);
774         }
775 
776         // -~- Not handling type variables
777         if (expectedCat == TypeCategory.TYPE_VARIABLE || argCat == TypeCategory.TYPE_VARIABLE) {
778             return IncompatibleTypes.SEEMS_OK;
779         }
780 
781         // -~- Array Types: compare dimensions, then base type
782         if (expectedCat == TypeCategory.ARRAY_TYPE && argCat == TypeCategory.ARRAY_TYPE) {
783             ArrayType parmArray = (ArrayType) expectedType;
784             ArrayType argArray = (ArrayType) actualType;
785 
786             if (parmArray.getDimensions() != argArray.getDimensions()) {
787                 return IncompatibleTypes.ARRAY_AND_NON_ARRAY;
788             }
789 
790             return compareTypes(parmArray.getBasicType(), argArray.getBasicType(), ignoreBaseType);
791         }
792         // If one is an Array Type and the other is not, then they
793         // are incompatible. (We already know neither is java.lang.Object)
794         if (expectedCat == TypeCategory.ARRAY_TYPE ^ argCat == TypeCategory.ARRAY_TYPE) {
795             return IncompatibleTypes.ARRAY_AND_NON_ARRAY;
796         }
797 
798         // -~- Parameter Types: compare base type then parameters
799         if (expectedCat == TypeCategory.PARAMETERIZED && argCat == TypeCategory.PARAMETERIZED) {
800             GenericObjectType parmGeneric = (GenericObjectType) expectedType;
801             GenericObjectType argGeneric = (GenericObjectType) actualType;
802 
803             // base types should be related
804             {
805                 IncompatibleTypes result = compareTypes(parmGeneric.getObjectType(), argGeneric.getObjectType(), ignoreBaseType);
806                 if (!result.equals(IncompatibleTypes.SEEMS_OK)) {
807                     return result;
808                 }
809             }
810             return compareTypeParameters(parmGeneric, argGeneric);
811 
812             // XXX More to come
813         }
814         // If one is a Parameter Type and the other is not, then they
815         // are incompatible. (We already know neither is java.lang.Object)
816         if (false) {
817             // not true. Consider class Foo extends ArrayList<String>
818             if (expectedCat == TypeCategory.PARAMETERIZED ^ argCat == TypeCategory.PARAMETERIZED) {
819                 return IncompatibleTypes.SEEMS_OK; // fix this when we know what
820                 // we are doing here
821             }
822         }
823 
824         // -~- Wildcard e.g. List<*>.contains(...)
825         if (expectedCat == TypeCategory.WILDCARD) {
826             return IncompatibleTypes.SEEMS_OK;
827         }
828 
829         // -~- Non Reference types
830         // if ( parmCat == TypeCategory.NON_REFERENCE_TYPE ||
831         // argCat == TypeCategory.NON_REFERENCE_TYPE )
832         if (expectedType instanceof BasicType || actualType instanceof BasicType) {
833             // this should not be possible, compiler will complain (pre 1.5)
834             // or autobox primitive types (1.5 +)
835             throw new IllegalArgumentException("checking for compatibility of " + expectedType + " with " + actualType);
836         }
837 
838         return IncompatibleTypes.SEEMS_OK;
839 
840     }

842     private IncompatibleTypes compareTypeParameters(GenericObjectType parmGeneric, GenericObjectType argGeneric) {
843         int p = parmGeneric.getNumParameters();
844         if (p != argGeneric.getNumParameters()) {
845             return IncompatibleTypes.SEEMS_OK;
846         }
847         for (int x = 0; x < p; x++) {
848             IncompatibleTypes result = compareTypes(parmGeneric.getParameterAt(x), argGeneric.getParameterAt(x), false);
849             if (result != IncompatibleTypes.SEEMS_OK) {
850                 return result;
851             }
852         }
853         return IncompatibleTypes.SEEMS_OK;
854     }

856     // old version of compare types
857     private boolean compareTypesOld(Type parmType, Type argType) {
858         // XXX equality not implemented for GenericObjectType
859         // if (parmType.equals(argType)) return true;
860         // Compare type signatures instead
861         if (GenericUtilities.getString(parmType).equals(GenericUtilities.getString(argType))) {
862             return true;
863         }
864 
865         if (parmType instanceof GenericObjectType) {
866             GenericObjectType o = (GenericObjectType) parmType;
867             if (o.getTypeCategory() == GenericUtilities.TypeCategory.WILDCARD_EXTENDS) {
868                 return compareTypesOld(o.getExtension(), argType);
869             }
870         }
871         // ignore type variables for now
872         if (parmType instanceof GenericObjectType && !((GenericObjectType) parmType).hasParameters()) {
873             return true;
874         }
875         if (argType instanceof GenericObjectType && !((GenericObjectType) argType).hasParameters()) {
876             return true;
877         }
878 
879         // Case: Both are generic containers
880         if (parmType instanceof GenericObjectType && argType instanceof GenericObjectType) {
881             return true;
882         } else {
883             // Don't consider non reference types (should not be possible)
884             if (!(parmType instanceof ReferenceType && argType instanceof ReferenceType)) {
885                 return true;
886             }
887 
888             // Don't consider non object types (for now)
889             if (!(parmType instanceof ObjectType && argType instanceof ObjectType)) {
890                 return true;
891             }
892 
893             // Otherwise, compare base types ignoring generic information
894             try {
895                 return Repository.instanceOf(((ObjectType) argType).getClassName(), ((ObjectType) parmType).getClassName());
896             } catch (ClassNotFoundException e) {
897             }
898         }
899 
900         return true;
901     }
