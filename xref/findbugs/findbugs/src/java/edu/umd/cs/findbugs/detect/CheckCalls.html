42  /**
43   * This is just for debugging method call resolution.
44   *
45   * @author David Hovemeyer
46   */
47  public class CheckCalls implements Detector, NonReportingDetector {
48  
49      private static final String METHOD = SystemProperties.getProperty("checkcalls.method");
50  
51      private static final String TARGET_METHOD = SystemProperties.getProperty("checkcalls.targetmethod");
52  
53      BugReporter bugReporter;
54  
55      public CheckCalls(BugReporter bugReporter) {
56          this.bugReporter = bugReporter;
57      }
58  
59      /*
60       * (non-Javadoc)
61       *
62       * @see
63       * edu.umd.cs.findbugs.Detector#visitClassContext(edu.umd.cs.findbugs.ba
64       * .ClassContext)
65       */
66      @Override
67      public void visitClassContext(ClassContext classContext) {
68          Method[] methodList = classContext.getJavaClass().getMethods();
69          for (Method method : methodList) {
70              if (method.getCode() == null) {
71                  continue;
72              }
73  
74              // System.out.println("--> " + method.getName());
75              if (METHOD != null && !method.getName().equals(METHOD)) {
76                  continue;
77              }
78  
79              try {
80                  System.out.println("Analyzing " + SignatureConverter.convertMethodSignature(classContext.getJavaClass(), method));
81                  analyzeMethod(classContext, method);
82              } catch (CFGBuilderException e) {
83                  bugReporter.logError("Error", e);
84              } catch (DataflowAnalysisException e) {
85                  bugReporter.logError("Error", e);
86              } catch (ClassNotFoundException e) {
87                  bugReporter.reportMissingClass(e);
88              }
89          }
90      }
91  
92      private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, ClassNotFoundException,
93      DataflowAnalysisException {
94          CFG cfg = classContext.getCFG(method);
95          for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
96              Location location = i.next();
97              Instruction ins = location.getHandle().getInstruction();
98  
99              if (ins instanceof InvokeInstruction) {
100                 if (TARGET_METHOD != null
101                         && !((InvokeInstruction) ins).getMethodName(classContext.getConstantPoolGen()).equals(TARGET_METHOD)) {
102                     continue;
103                 }
104 
105                 System.out.println("\n*******************************************************\n");
106 
107                 System.out.println("Method invocation: " + location.getHandle());
108                 System.out.println("\tInvoking: "
109                         + SignatureConverter.convertMethodSignature((InvokeInstruction) ins, classContext.getConstantPoolGen()));
110 
111                 JavaClassAndMethod proto = Hierarchy.findInvocationLeastUpperBound((InvokeInstruction) ins,
112                         classContext.getConstantPoolGen());
113                 if (proto == null) {
114                     System.out.println("\tUnknown prototype method");
115                 } else {
116                     System.out.println("\tPrototype method: class=" + proto.getJavaClass().getClassName() + ", method="
117                             + proto.getMethod());
118                 }
119                 Set<JavaClassAndMethod> calledMethodSet = Hierarchy.resolveMethodCallTargets((InvokeInstruction) ins,
120                         classContext.getTypeDataflow(method).getFactAtLocation(location), classContext.getConstantPoolGen());
121                 System.out.println("\tTarget method set: " + calledMethodSet);
122             }
123         }
124     }
125 
126     /*
127      * (non-Javadoc)
128      *
129      * @see edu.umd.cs.findbugs.Detector#report()
130      */
131     @Override
132     public void report() {
133     }
