45  public abstract class CommandLine {

49      private final List<String> optionList;

65      public CommandLine() {
66          this.unlistedOptions = new HashSet<String>();
67          this.optionList = new LinkedList<String>();
68          this.optionGroups = new HashMap<Integer, String>();
69          this.requiresArgumentSet = new HashSet<String>();
70          this.optionDescriptionMap = new HashMap<String, String>();
71          this.optionExtraPartSynopsisMap = new HashMap<String, String>();
72          this.argumentDescriptionMap = new HashMap<String, String>();
73          this.maxWidth = 0;
74      }

76      /**
77       * Start a new group of related command-line options.
78       *
79       * @param description
80       *            description of the group
81       */
82      public void startOptionGroup(String description) {
83          optionGroups.put(optionList.size(), description);
84      }

86      /**
87       * Add a command line switch. This method is for adding options that do not
88       * require an argument.
89       *
90       * @param option
91       *            the option, must start with "-"
92       * @param description
93       *            single line description of the option
94       */
95      public void addSwitch(String option, String description) {
96          optionList.add(option);
97          optionDescriptionMap.put(option, description);
98  
99          if (option.length() > maxWidth) {
100             maxWidth = option.length();
101         }
102     }
103 
104     /**
105      * Add a command line switch that allows optional extra information to be
106      * specified as part of it.
107      *
108      * @param option
109      *            the option, must start with "-"
110      * @param optionExtraPartSynopsis
111      *            synopsis of the optional extra information
112      * @param description
113      *            single-line description of the option
114      */
115     public void addSwitchWithOptionalExtraPart(String option, String optionExtraPartSynopsis, String description) {
116         optionList.add(option);
117         optionExtraPartSynopsisMap.put(option, optionExtraPartSynopsis);
118         optionDescriptionMap.put(option, description);
119 
120         // Option will display as -foo[:extraPartSynopsis]
121         int length = option.length() + optionExtraPartSynopsis.length() + 3;
122         if (length > maxWidth) {
123             maxWidth = length;
124         }
125     }
126 
127     /**
128      * Add an option requiring an argument.
129      *
130      * @param option
131      *            the option, must start with "-"
132      * @param argumentDesc
133      *            brief (one or two word) description of the argument
134      * @param description
135      *            single line description of the option
136      */
137     public void addOption(String option, String argumentDesc, String description) {
138         optionList.add(option);
139         optionDescriptionMap.put(option, description);
140         requiresArgumentSet.add(option);
141         argumentDescriptionMap.put(option, argumentDesc);
142 
143         int width = option.length() + 3 + argumentDesc.length();
144         if (width > maxWidth) {
145             maxWidth = width;
146         }
147     }

150      * Don't list this option when printing Usage information
151      *
152      * @param option
153      */
154     public void makeOptionUnlisted(String option) {
155         unlistedOptions.add(option);

158     /**
159      * Expand option files in given command line. Any token beginning with "@"
160      * is assumed to be an option file. Option files contain one command line
161      * option per line.
162      *
163      * @param argv
164      *            the original command line
165      * @param ignoreComments
166      *            ignore comments (lines starting with "#")
167      * @param ignoreBlankLines
168      *            ignore blank lines
169      * @return the expanded command line
170      */
171 
172     public String[] expandOptionFiles(String[] argv, boolean ignoreComments, boolean ignoreBlankLines) throws IOException,
173     HelpRequestedException {
174         // Add all expanded options at the end of the options list, before the
175         // list of
176         // jar/zip/class files and directories.
177         // At the end of the options to preserve the order of the options (e.g.
178         // -adjustPriority
179         // must always come after -pluginList).
180         int lastOptionIndex = parse(argv, true);
181         ArrayList<String> resultList = new ArrayList<String>();
182         ArrayList<String> expandedOptionsList = getAnalysisOptionProperties(ignoreComments, ignoreBlankLines);
183         for (int i = 0; i < lastOptionIndex; i++) {
184             String arg = argv[i];
185             if (!arg.startsWith("@")) {
186                 resultList.add(arg);
187                 continue;
188             }
189 
190             BufferedReader reader = null;
191             try {
192                 reader = UTF8.bufferedReader(new FileInputStream(arg.substring(1)));
193                 addCommandLineOptions(expandedOptionsList, reader, ignoreComments, ignoreBlankLines);
194             } finally {
195                 Util.closeSilently(reader);
196             }
197         }
198         resultList.addAll(expandedOptionsList);
199         for (int i = lastOptionIndex; i < argv.length; i++) {
200             resultList.add(argv[i]);
201         }
202 
203         return resultList.toArray(new String[resultList.size()]);
204     }
205 
206     public static ArrayList<String> getAnalysisOptionProperties(boolean ignoreComments, boolean ignoreBlankLines) {
207         ArrayList<String> resultList = new ArrayList<String>();
208         URL u = DetectorFactoryCollection.getCoreResource("analysisOptions.properties");
209         if (u != null) {
210             BufferedReader reader = null;
211             try {
212                 reader = UTF8.bufferedReader(u.openStream());
213                 addCommandLineOptions(resultList, reader, ignoreComments, ignoreBlankLines);
214             } catch (IOException e) {
215                 AnalysisContext.logError("unable to load analysisOptions.properties", e);
216             } finally {
217                 Util.closeSilently(reader);
218             }
219         }
220         return resultList;
221     }
222 
223     private static void addCommandLineOptions(ArrayList<String> resultList, BufferedReader reader, boolean ignoreComments,
224             boolean ignoreBlankLines) throws IOException {
225         String line;
226         while ((line = reader.readLine()) != null) {
227             line = line.trim();
228 
229             if (ignoreComments && line.startsWith("#")) {
230                 continue;
231             }
232 
233             if (ignoreBlankLines && "".equals(line)) {
234                 continue;
235             }
236             if (line.length() >= 2 && line.charAt(0) == '"' && line.charAt(line.length() - 1) == '"') {
237                 resultList.add(line.substring(0, line.length() - 1));
238             } else {
239                 for (String segment : line.split(" ")) {
240                     resultList.add(segment);
241                 }
242             }
243         }
244     }

250     /**
251      * Parse switches/options, showing usage information if they can't be
252      * parsed, or if we have the wrong number of remaining arguments after
253      * parsing. Calls parse(String[]).
254      *
255      * @param argv
256      *            command line arguments
257      * @param minArgs
258      *            allowed minimum number of arguments remaining after
259      *            switches/options are parsed
260      * @param maxArgs
261      *            allowed maximum number of arguments remaining after
262      *            switches/options are parsed
263      * @param usage
264      *            usage synopsis
265      * @return number of arguments parsed
266      */
267     @SuppressFBWarnings("DM_EXIT")
268     public int parse(String argv[], int minArgs, int maxArgs, String usage) {
269         try {
270             int count = parse(argv);
271             int remaining = argv.length - count;
272             if (remaining < minArgs || remaining > maxArgs) {
273                 System.out.println(usage);
274                 System.out.println("Expected " + minArgs + "..." + maxArgs + " file arguments, found " + remaining);
275                 System.out.println("Options:");
276                 printUsage(System.out);
277                 System.exit(1);
278             }
279             return count;
280         } catch (HelpRequestedException e) {
281             // fall through
282         } catch (RuntimeException e) {
283             e.printStackTrace();
284         } catch (IOException e) {
285             e.printStackTrace();
286         }
287         System.out.println(usage);
288         System.out.println("Options:");
289         printUsage(System.out);
290         System.exit(1);
291         return -1;
292     }

294     /**
295      * Parse a command line. Calls down to handleOption() and
296      * handleOptionWithArgument() methods. Stops parsing when it reaches the end
297      * of the command line, or when a command line argument not starting with
298      * "-" is seen.
299      *
300      * @param argv
301      *            the arguments
302      * @return the number of arguments parsed; if equal to argv.length, then the
303      *         entire command line was parsed
304      * @throws HelpRequestedException
305      */
306     public int parse(String argv[]) throws IOException, HelpRequestedException {
307         return parse(argv, false);
308     }
309 
310     private int parse(String argv[], boolean dryRun) throws IOException, HelpRequestedException {
311         int arg = 0;
312 
313         while (arg < argv.length) {
314             String option = argv[arg];
315             if ("-help".equals(option) || "-h".equals(option)) {
316                 throw new HelpRequestedException();
317             }
318             if (!option.startsWith("-")) {
319                 break;
320             }
321 
322             String optionExtraPart = "";
323             int colon = option.indexOf(':');
324             if (colon >= 0) {
325                 optionExtraPart = option.substring(colon + 1);
326                 option = option.substring(0, colon);
327             }
328 
329             if (optionDescriptionMap.get(option) == null) {
330                 throw new IllegalArgumentException("Unknown option: " + option);
331             }
332 
333             if (requiresArgumentSet.contains(option)) {
334                 ++arg;
335                 if (arg >= argv.length) {
336                     throw new IllegalArgumentException("Option " + option + " requires an argument");
337                 }
338                 String argument = argv[arg];
339                 if (!dryRun) {
340                     handleOptionWithArgument(option, argument);
341                 }
342                 ++arg;
343             } else {
344                 if (!dryRun) {
345                     handleOption(option, optionExtraPart);
346                 }
347                 ++arg;
348             }
349         }
350 
351         return arg;
352     }

355      * Callback method for handling an option.
356      *
357      * @param option
358      *            the option
359      * @param optionExtraPart

361      *            e.g., "withMessages" in "-xml:withMessages"); the empty string
362      *            if there was no extra part

376     /**
377      * Print command line usage information to given stream.
378      *
379      * @param os
380      *            the output stream
381      */
382     public void printUsage(OutputStream os) {
383         int count = 0;
384         PrintStream out = UTF8.printStream(os);
385         for (String option : optionList) {
386 
387             if (optionGroups.containsKey(count)) {
388                 out.println("  " + optionGroups.get(count));
389             }
390             count++;
391 
392             if (unlistedOptions.contains(option)) {
393                 continue;
394             }
395             out.print("    ");
396 
397             StringBuilder buf = new StringBuilder();
398             buf.append(option);
399             if (optionExtraPartSynopsisMap.get(option) != null) {
400                 String optionExtraPartSynopsis = optionExtraPartSynopsisMap.get(option);
401                 buf.append("[:");
402                 buf.append(optionExtraPartSynopsis);
403                 buf.append("]");
404             }
405             if (requiresArgumentSet.contains(option)) {
406                 buf.append(" <");
407                 buf.append(argumentDescriptionMap.get(option));
408                 buf.append(">");
409             }
410             printField(out, buf.toString(), maxWidth + 1);
411 
412             out.println(optionDescriptionMap.get(option));
413         }
414         out.flush();
415     }
416 
417     private static void printField(PrintStream out, String s, int width) {
418         if (s.length() > width) {
419             throw new IllegalArgumentException();
420         }
421         int nSpaces = width - s.length();
422         out.print(s);
423         while (nSpaces > 0) {
424             int n = Math.min(SPACES.length(), nSpaces);
425             out.print(SPACES.substring(0, n));
426             nSpaces -= n;
427         }
428     }
