129     @Override
130     public void visitClassContext(ClassContext classContext) {
131         this.classContext = classContext;
132 
133         String currentMethod = null;
134 
135         JavaClass jclass = classContext.getJavaClass();
136         String className = jclass.getClassName();
137         String superClassName = jclass.getSuperclassName();
138         if (superClassName.endsWith("ProtocolMessage")) {
139             return;
140         }
141         if (CLASS != null && !className.equals(CLASS)) {
142             return;
143         }
144         Method[] methodList = jclass.getMethods();
145         for (Method method : methodList) {
146             try {
147                 if (method.isAbstract() || method.isNative() || method.getCode() == null) {
148                     continue;
149                 }
150 
151                 currentMethod = SignatureConverter.convertMethodSignature(jclass, method);
152 
153                 if (METHOD != null && !method.getName().equals(METHOD)) {
154                     continue;
155                 }
156                 if (DEBUG || DEBUG_NULLARG) {
157                     System.out.println("Checking for NP in " + currentMethod);
158                 }
159                 analyzeMethod(classContext, method);
160             } catch (MissingClassException e) {
161                 bugReporter.reportMissingClass(e.getClassNotFoundException());
162             } catch (DataflowAnalysisException e) {
163                 bugReporter.logError("While analyzing " + currentMethod + ": FindNullDeref caught dae exception", e);
164             } catch (CFGBuilderException e) {
165                 bugReporter.logError("While analyzing " + currentMethod + ": FindNullDeref caught cfgb exception", e);
166             }
167             bugAccumulator.reportAccumulatedBugs();
168         }
169     }

171     private void analyzeMethod(ClassContext classContext, Method method) throws DataflowAnalysisException, CFGBuilderException {
172         if (DEBUG || DEBUG_NULLARG) {
173             System.out.println("Pre FND ");
174         }
175 
176         MethodGen methodGen = classContext.getMethodGen(method);
177         if (methodGen == null) {
178             return;
179         }
180 
181         // UsagesRequiringNonNullValues uses =
182         // classContext.getUsagesRequiringNonNullValues(method);
183         this.method = method;
184 
185         if (DEBUG || DEBUG_NULLARG) {
186             System.out.println("FND: " + SignatureConverter.convertMethodSignature(methodGen));
187         }
188 
189         findPreviouslyDeadBlocks();
190 
191         vnaDataflow = classContext.getValueNumberDataflow(method);
192 
193         // Create a NullDerefAndRedundantComparisonFinder object to do the
194         // actual
195         // work. It will call back to report null derefs and redundant null
196         // comparisons
197         // through the NullDerefAndRedundantComparisonCollector interface we
198         // implement.
199         NullDerefAndRedundantComparisonFinder worker = new NullDerefAndRedundantComparisonFinder(classContext, method, this);
200         worker.execute();
201 
202     }

204     /**
205      * Find set of blocks which were known to be dead before doing the null
206      * pointer analysis.
207      *
208      * @return set of previously dead blocks, indexed by block id
209      * @throws CFGBuilderException
210      * @throws DataflowAnalysisException
211      */
212     private BitSet findPreviouslyDeadBlocks() throws DataflowAnalysisException, CFGBuilderException {
213         BitSet deadBlocks = new BitSet();
214         ValueNumberDataflow vnaDataflow = classContext.getValueNumberDataflow(method);
215         for (Iterator<BasicBlock> i = vnaDataflow.getCFG().blockIterator(); i.hasNext();) {
216             BasicBlock block = i.next();
217             ValueNumberFrame vnaFrame = vnaDataflow.getStartFact(block);
218             if (vnaFrame.isTop()) {
219                 deadBlocks.set(block.getLabel());
220             }
221         }
222 
223         return deadBlocks;
224     }

249     @Override
250     public void foundNullDeref(Location location, ValueNumber valueNumber, IsNullValue refValue, ValueNumberFrame vnaFrame,
251             boolean isConsistent) {
252         if (!refValue.isNullOnComplicatedPath23()) {
253             return;
254         }
255         WarningPropertySet<WarningProperty> propertySet = new WarningPropertySet<WarningProperty>();
256         if (valueNumber.hasFlag(ValueNumber.CONSTANT_CLASS_OBJECT)) {
257             return;
258         }
259 
260         boolean onExceptionPath = refValue.isException();
261         if (onExceptionPath) {
262             propertySet.addProperty(GeneralWarningProperty.ON_EXCEPTION_PATH);
263         }
264         BugAnnotation variable = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, valueNumber, vnaFrame,
265                 "VALUE_OF");
266         addPropertiesForDereferenceLocations(propertySet, Collections.singleton(location));
267         Instruction ins = location.getHandle().getInstruction();
268         BugAnnotation cause;
269         final ConstantPoolGen cpg = classContext.getConstantPoolGen();
270 
271         if (ins instanceof InvokeInstruction) {
272             InvokeInstruction iins = (InvokeInstruction) ins;
273             XMethod invokedMethod = XFactory.createXMethod((InvokeInstruction) ins, cpg);
274             cause = MethodAnnotation.fromXMethod(invokedMethod);
275             cause.setDescription(MethodAnnotation.METHOD_CALLED);
276 
277             if ("close".equals(iins.getMethodName(cpg)) && "()V".equals(iins.getSignature(cpg))) {
278                 propertySet.addProperty(NullDerefProperty.CLOSING_NULL);
279             }
280         } else if (ins instanceof FieldInstruction) {
281             FieldInstruction fins = (FieldInstruction) ins;
282             XField referencedField = XFactory.createXField(fins, cpg);
283             cause = FieldAnnotation.fromXField(referencedField);
284 
285         } else {
286             cause = new StringAnnotation(ins.getName());
287         }
288 
289         boolean caught = inCatchNullBlock(location);
290         if (caught && skipIfInsideCatchNull()) {
291             return;
292         }
293 
294         int basePriority = Priorities.NORMAL_PRIORITY;
295 
296         if (!refValue.isNullOnComplicatedPath2()) {
297             basePriority--;
298         }
299         reportNullDeref(propertySet, location, "NOISE_NULL_DEREFERENCE", basePriority, cause, variable);
300 
301     }

303     private void reportNullDeref(WarningPropertySet<WarningProperty> propertySet, Location location, String type, int priority,
304             BugAnnotation cause, @CheckForNull BugAnnotation variable) {
305 
306         BugInstance bugInstance = new BugInstance(this, type, priority).addClassAndMethod(classContext.getJavaClass(), method);
307         bugInstance.add(cause);
308         if (variable != null) {
309             bugInstance.add(variable);
310         } else {
311             bugInstance.add(new LocalVariableAnnotation("?", -1, -1));
312         }
313         bugInstance.addSourceLine(classContext, method, location).describe("SOURCE_LINE_DEREF");
314 
315         if (FindBugsAnalysisFeatures.isRelaxedMode()) {
316             WarningPropertyUtil.addPropertiesForDataMining(propertySet, classContext, method, location);
317         }
318         addPropertiesForDereferenceLocations(propertySet, Collections.singleton(location));
319 
320         propertySet.decorateBugInstance(bugInstance);
321 
322         bugReporter.reportBug(bugInstance);
323     }

325     public static boolean isThrower(BasicBlock target) {
326         InstructionHandle ins = target.getFirstInstruction();
327         int maxCount = 7;
328         while (ins != null) {
329             if (maxCount-- <= 0) {
330                 break;
331             }
332             Instruction i = ins.getInstruction();
333             if (i instanceof ATHROW) {
334                 return true;
335             }
336             if (i instanceof InstructionTargeter || i instanceof ReturnInstruction) {
337                 return false;
338             }
339             ins = ins.getNext();
340         }
341         return false;
342     }

355     private void addPropertiesForDereferenceLocations(WarningPropertySet<WarningProperty> propertySet,
356             Collection<Location> derefLocationSet) {
357         boolean derefOutsideCatchBlock = false;
358         boolean allDerefsAtDoomedLocations = true;
359         for (Location loc : derefLocationSet) {
360             if (!inCatchNullBlock(loc)) {
361                 derefOutsideCatchBlock = true;
362             }
363 
364             if (!isDoomed(loc)) {
365                 allDerefsAtDoomedLocations = false;
366             }
367         }
368 
369         if (allDerefsAtDoomedLocations) {
370             // Add a WarningProperty
371             propertySet.addProperty(DoomedCodeWarningProperty.DOOMED_CODE);
372         }
373         boolean uniqueDereferenceLocations = uniqueLocations(derefLocationSet);
374 
375         if (!derefOutsideCatchBlock) {
376             if (!uniqueDereferenceLocations || skipIfInsideCatchNull()) {
377                 propertySet.addProperty(GeneralWarningProperty.FALSE_POSITIVE);
378             } else {
379                 propertySet.addProperty(NullDerefProperty.DEREFS_IN_CATCH_BLOCKS);
380             }
381         }
382         if (!uniqueDereferenceLocations) {
383             // Add a WarningProperty
384             propertySet.addProperty(NullDerefProperty.DEREFS_ARE_CLONED);
385         }
386 
387         addPropertiesForMethodContainingWarning(propertySet);
388     }

390     private boolean uniqueLocations(Collection<Location> derefLocationSet) {
391         boolean uniqueDereferenceLocations = false;
392         LineNumberTable table = method.getLineNumberTable();
393         if (table == null) {
394             uniqueDereferenceLocations = true;
395         } else {
396             BitSet linesMentionedMultipleTimes = classContext.linesMentionedMultipleTimes(method);
397             for (Location loc : derefLocationSet) {
398                 int lineNumber = table.getSourceLine(loc.getHandle().getPosition());
399                 if (!linesMentionedMultipleTimes.get(lineNumber)) {
400                     uniqueDereferenceLocations = true;
401                 }
402             }
403         }
404         return uniqueDereferenceLocations;
405     }

407     private void addPropertiesForMethodContainingWarning(WarningPropertySet<WarningProperty> propertySet) {
408         XMethod xMethod = XFactory.createXMethod(classContext.getJavaClass(), method);
409 
410         boolean uncallable = !AnalysisContext.currentXFactory().isCalledDirectlyOrIndirectly(xMethod) && xMethod.isPrivate();
411 
412         if (uncallable) {
413             propertySet.addProperty(GeneralWarningProperty.IN_UNCALLABLE_METHOD);
414         }
415     }

417     private boolean isDoomed(Location loc) {
418         if (!MARK_DOOMED) {
419             return false;
420         }
421 
422         ReturnPathTypeDataflow rptDataflow;
423         try {
424             rptDataflow = classContext.getReturnPathTypeDataflow(method);
425 
426             ReturnPathType rpt = rptDataflow.getFactAtLocation(loc);
427 
428             return !rpt.canReturnNormally();
429         } catch (CheckedAnalysisException e) {
430             AnalysisContext.logError("Error getting return path type", e);
431             return false;
432         }
433     }

435     String getDescription(Location loc, ValueNumber refValue) {
436         PointerUsageRequiringNonNullValue pu;
437         try {
438             UsagesRequiringNonNullValues usages = classContext.getUsagesRequiringNonNullValues(method);
439             pu = usages.get(loc, refValue, vnaDataflow);
440             if (pu == null) {
441                 return "SOURCE_LINE_DEREF";
442             }
443             return pu.getDescription();
444         } catch (DataflowAnalysisException e) {
445             AnalysisContext.logError("Error getting UsagesRequiringNonNullValues for " + method, e);
446             return "SOURCE_LINE_DEREF";
447         } catch (CFGBuilderException e) {
448             AnalysisContext.logError("Error getting UsagesRequiringNonNullValues for " + method, e);
449             return "SOURCE_LINE_DEREF";
450         }
451

454     boolean inCatchNullBlock(Location loc) {
455         int pc = loc.getHandle().getPosition();
456         int catchSize = Util.getSizeOfSurroundingTryBlock(classContext.getJavaClass().getConstantPool(), method.getCode(),
457                 "java/lang/NullPointerException", pc);
458         if (catchSize < Integer.MAX_VALUE) {
459             return true;
460         }
461         catchSize = Util.getSizeOfSurroundingTryBlock(classContext.getJavaClass().getConstantPool(), method.getCode(),
462                 "java/lang/Exception", pc);
463         if (catchSize < 5) {
464             return true;
465         }
466         catchSize = Util.getSizeOfSurroundingTryBlock(classContext.getJavaClass().getConstantPool(), method.getCode(),
467                 "java/lang/RuntimeException", pc);
468         if (catchSize < 5) {
469             return true;
470         }
471         catchSize = Util.getSizeOfSurroundingTryBlock(classContext.getJavaClass().getConstantPool(), method.getCode(),
472                 "java/lang/Throwable", pc);
473         if (catchSize < 5) {
474             return true;
475         }
476         return false;
