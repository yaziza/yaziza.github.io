182     @Override
183     public void visit(JavaClass obj) {
184         data.calledFromConstructors.clear();
185         hasNativeMethods = false;
186         sawSelfCallInConstructor = false;
187         publicOrProtectedConstructor = false;
188         isSerializable = false;
189         if (obj.isAbstract()) {
190             data.abstractClasses.add(getDottedClassName());
191         } else {
192             String superClass = obj.getSuperclassName();
193             if (superClass != null) {
194                 data.hasNonAbstractSubClass.add(superClass);
195             }
196         }
197         data.classesScanned.add(getDottedClassName());
198         boolean superClassIsObject = "java.lang.Object".equals(obj.getSuperclassName());
199         if (getSuperclassName().indexOf('$') >= 0 || getSuperclassName().indexOf('+') >= 0
200                 || withinAnonymousClass.matcher(getDottedClassName()).find()) {
201             // System.out.println("hicfsc: " + betterClassName);
202             data.innerClassCannotBeStatic.add(getDottedClassName());
203             // System.out.println("hicfsc: " + betterSuperclassName);
204             data.innerClassCannotBeStatic.add(getDottedSuperclassName());
205         }
206         // Does this class directly implement Serializable?
207         String[] interface_names = obj.getInterfaceNames();
208         for (String interface_name : interface_names) {
209             if ("java.io.Externalizable".equals(interface_name)) {
210                 isSerializable = true;
211             } else if ("java.io.Serializable".equals(interface_name)) {
212                 isSerializable = true;
213                 break;
214             }
215         }
216 
217         // Does this class indirectly implement Serializable?
218         if ((!superClassIsObject || interface_names.length > 0) && !isSerializable) {
219             try {
220                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
221                 ClassDescriptor desc = DescriptorFactory.createClassDescriptor(obj);
222                 if (subtypes2.getSubtypes(serializable).contains(desc) || subtypes2.getSubtypes(externalizable).contains(desc)
223                         || subtypes2.getSubtypes(remote).contains(desc)) {
224                     isSerializable = true;
225                 }
226             } catch (ClassNotFoundException e) {
227                 bugReporter.reportMissingClass(e);
228             }
229         }
230 
231         // System.out.println(getDottedClassName() + " is serializable: " +
232         // isSerializable);
233         super.visit(obj);
234     }

238             if (a instanceof Signature) {

301     public static boolean isInjectionAttribute(@DottedClassName String annotationClass) {
302         if (annotationClass.startsWith("javax.annotation.")
303                 || annotationClass.startsWith("javax.ejb")
304                 || "org.apache.tapestry5.annotations.Persist".equals(annotationClass)
305                 || "org.jboss.seam.annotations.In".equals(annotationClass)
306                 || annotationClass.startsWith("javax.persistence")
307                 || annotationClass.endsWith("SpringBean")
308                 || "com.google.inject.Inject".equals(annotationClass)
309                 || annotationClass.startsWith("com.google.") && annotationClass.endsWith(".Bind")
310                 && annotationClass.hashCode() == -243168318
311                 || annotationClass.startsWith("org.nuxeo.common.xmap.annotation")
312                 || annotationClass.startsWith("com.google.gwt.uibinder.client")
313                 || annotationClass.startsWith("org.springframework.beans.factory.annotation")
314                 || "javax.ws.rs.core.Context".equals(annotationClass)) {
315             return true;
316         }
317         int lastDot = annotationClass.lastIndexOf('.');
318         String lastPart = annotationClass.substring(lastDot + 1);
319         if (lastPart.startsWith("Inject")) {
320             return true;
321         }
322         return false;
323     }

386     @Override
387     public void sawOpcode(int seen) {
388         if (DEBUG) {
389             System.out.println(getPC() + ": " + OPCODE_NAMES[seen] + " " + saState);
390         }
391         if (seen == MONITORENTER) {
392             seenMonitorEnter = true;
393         }
394         switch (saState) {
395         case 0:
396             if (seen == ALOAD_0) {
397                 saState = 1;
398             }
399             break;
400         case 1:
401             if (seen == ALOAD_0) {
402                 saState = 2;
403             } else {
404                 saState = 0;
405             }
406             break;
407         case 2:
408             if (seen == GETFIELD) {
409                 saState = 3;
410             } else {
411                 saState = 0;
412             }
413             break;
414         case 3:
415             if (seen == PUTFIELD) {
416                 saState = 4;
417             } else {
418                 saState = 0;
419             }
420             break;
421         default:
422             break;
423         }
424         boolean selfAssignment = false;
425         if (pendingGetField != null) {
426             if (seen != PUTFIELD && seen != PUTSTATIC) {
427                 data.readFields.add(pendingGetField);
428             } else if (XFactory.createReferencedXField(this).equals(pendingGetField) && (saState == 4 || seen == PUTSTATIC)) {
429                 selfAssignment = true;
430             } else {
431                 data.readFields.add(pendingGetField);
432             }
433             pendingGetField = null;
434         }
435         if (saState == 4) {
436             saState = 0;
437         }
438 
439         if (seen == INVOKESTATIC && "java/util/concurrent/atomic/AtomicReferenceFieldUpdater".equals(getClassConstantOperand())
440                 && "newUpdater".equals(getNameConstantOperand())) {
441             String fieldName = (String) stack.getStackItem(0).getConstant();
442             String fieldSignature = (String) stack.getStackItem(1).getConstant();
443             String fieldClass = (String) stack.getStackItem(2).getConstant();
444             if (fieldName != null && fieldSignature != null && fieldClass != null) {
445                 XField f = XFactory.createXField(fieldClass.replace('/', '.'), fieldName, ClassName.toSignature(fieldSignature),
446                         false);
447                 data.reflectiveFields.add(f);
448             }
449 
450         }
451         if (seen == INVOKESTATIC && "java/util/concurrent/atomic/AtomicIntegerFieldUpdater".equals(getClassConstantOperand())
452                 && "newUpdater".equals(getNameConstantOperand())) {
453             String fieldName = (String) stack.getStackItem(0).getConstant();
454             String fieldClass = (String) stack.getStackItem(1).getConstant();
455             if (fieldName != null && fieldClass != null) {
456                 XField f = XFactory.createXField(fieldClass.replace('/', '.'), fieldName, "I", false);
457                 data.reflectiveFields.add(f);
458             }
459 
460         }
461         if (seen == INVOKESTATIC && "java/util/concurrent/atomic/AtomicLongFieldUpdater".equals(getClassConstantOperand())
462                 && "newUpdater".equals(getNameConstantOperand())) {
463             String fieldName = (String) stack.getStackItem(0).getConstant();
464             String fieldClass = (String) stack.getStackItem(1).getConstant();
465             if (fieldName != null && fieldClass != null) {
466                 XField f = XFactory.createXField(fieldClass.replace('/', '.'), fieldName, "J", false);
467                 data.reflectiveFields.add(f);
468             }
469 
470         }
471 
472         if (seen == GETSTATIC) {
473             XField f = XFactory.createReferencedXField(this);
474             data.staticFieldsReadInThisMethod.add(f);
475         } else if (seen == INVOKESTATIC) {
476             seenInvokeStatic = true;
477         } else if (seen == PUTSTATIC && !getMethod().isStatic()) {
478             XField f = XFactory.createReferencedXField(this);
479             OpcodeStack.Item valuePut = getStack().getStackItem(0);
480 
481             checkWriteToStaticFromInstanceMethod: if (f.getName().indexOf("class$") != 0) {
482                 int priority = LOW_PRIORITY;
483                 if (f.isReferenceType()) {
484                     try {
485                         ValueNumberDataflow vnaDataflow = getClassContext().getValueNumberDataflow(getMethod());
486                         IsNullValueDataflow invDataflow = getClassContext().getIsNullValueDataflow(getMethod());
487                         ValueNumberFrame vFrame = vnaDataflow.getAnalysis().getFactAtPC(vnaDataflow.getCFG(), getPC());
488                         IsNullValueFrame iFrame = invDataflow.getAnalysis().getFactAtPC(invDataflow.getCFG(), getPC());
489                         AvailableLoad l = new AvailableLoad(f);
490                         ValueNumber[] availableLoads = vFrame.getAvailableLoad(l);
491                         if (availableLoads != null && iFrame.isTrackValueNumbers()) {
492                             for (ValueNumber v : availableLoads) {
493                                 IsNullValue knownValue = iFrame.getKnownValue(v);
494                                 if (knownValue == null) {
495                                     continue;
496                                 }
497                                 if (knownValue.isDefinitelyNotNull()) {
498                                     if (valuePut.isNull()) {
499                                         priority++;
500                                     } else {
501                                         priority--;
502                                     }
503                                     break;
504                                 } else if (knownValue.isDefinitelyNull()) {
505                                     break checkWriteToStaticFromInstanceMethod;
506                                 }
507                             }
508                         }
509 
510                     } catch (CheckedAnalysisException e) {
511                         AnalysisContext.logError("foo", e);
512                     }
513                 }
514 
515                 if (!publicOrProtectedConstructor) {
516                     priority--;
517                 }
518                 if (seenMonitorEnter) {
519                     priority++;
520                 }
521                 if (!seenInvokeStatic && data.staticFieldsReadInThisMethod.isEmpty()) {
522                     priority--;
523                 }
524                 if (getThisClass().isPublic() && getMethod().isPublic()) {
525                     priority--;
526                 }
527                 if (getThisClass().isPrivate() || getMethod().isPrivate()) {
528                     priority++;
529                 }
530                 if (getClassName().indexOf('$') != -1 || BCELUtil.isSynthetic(getMethod()) || f.isSynthetic()
531                         || f.getName().indexOf('$') >= 0) {
532                     priority++;
533                 }
534 
535                 // Decrease priority for boolean fields used to control
536                 // debug/test settings
537                 if (f.getName().indexOf("DEBUG") >= 0 || f.getName().indexOf("VERBOSE") >= 0 && "Z".equals(f.getSignature())) {
538                     priority++;
539                     priority++;
540                 }
541                 // Eclipse bundles which implements start/stop *very* often
542                 // assigns static instances there
543                 if (("start".equals(getMethodName()) || "stop".equals(getMethodName()))
544                         && "(Lorg/osgi/framework/BundleContext;)V".equals(getMethodSig())) {
545                     try {
546                         JavaClass bundleClass = Repository.lookupClass("org.osgi.framework.BundleActivator");
547                         if (getThisClass().instanceOf(bundleClass)) {
548                             priority++;
549                         }
550                         if (f.isReferenceType()) {
551                             FieldDescriptor fieldInfo = f.getFieldDescriptor();
552                             String dottedClass = DeepSubtypeAnalysis.getComponentClass(fieldInfo.getSignature());
553                             JavaClass fieldClass = Repository.lookupClass(dottedClass);
554                             if (fieldClass != null && fieldClass.instanceOf(bundleClass)) {
555                                 // the code "plugin = this;" unfortunately
556                                 // exists in the
557                                 // template for new Eclipse plugin classes, so
558                                 // nearly every one
559                                 // plugin has this pattern => decrease to very
560                                 // low prio
561                                 priority = Priorities.IGNORE_PRIORITY;
562                             }
563                         }
564                     } catch (ClassNotFoundException e) {
565                         bugReporter.reportMissingClass(e);
566                     }
567                 }
568                 bugAccumulator.accumulateBug(new BugInstance(this, "ST_WRITE_TO_STATIC_FROM_INSTANCE_METHOD", priority)
569                 .addClassAndMethod(this).addField(f), this);
570 
571             }
572         }
573 
574         if (seen == INVOKEVIRTUAL || seen == INVOKEINTERFACE || seen == INVOKESPECIAL || seen == INVOKESTATIC) {
575 
576             String sig = getSigConstantOperand();
577             String invokedClassName = getClassConstantOperand();
578             if (invokedClassName.equals(getClassName())
579                     && ("<init>".equals(getMethodName()) || "<clinit>".equals(getMethodName()))) {
580 
581                 data.calledFromConstructors.add(getNameConstantOperand() + ":" + sig);
582             }
583             int pos = PreorderVisitor.getNumberArguments(sig);
584             if (stack.getStackDepth() > pos) {
585                 OpcodeStack.Item item = stack.getStackItem(pos);
586                 boolean superCall = seen == INVOKESPECIAL && !invokedClassName.equals(getClassName());
587 
588                 if (DEBUG) {
589                     System.out.println("In " + getFullyQualifiedMethodName() + " saw call on " + item);
590                 }
591 
592                 boolean selfCall = item.getRegisterNumber() == 0 && !superCall;
593                 if (selfCall && "<init>".equals(getMethodName())) {
594                     sawSelfCallInConstructor = true;
595                     if (DEBUG) {
596                         System.out.println("Saw self call in " + getFullyQualifiedMethodName() + " to " + invokedClassName + "."
597                                 + getNameConstantOperand());
598                     }
599                 }
600             }
601         }
602 
603         if ((seen == IFNULL || seen == IFNONNULL) && stack.getStackDepth() > 0) {
604             OpcodeStack.Item item = stack.getStackItem(0);
605             XField f = item.getXField();
606             if (f != null) {
607                 data.nullTested.add(f);
608                 if (DEBUG) {
609                     System.out.println(f + " null checked in " + getFullyQualifiedMethodName());
610                 }
611             }
612         }
613 
614         if ((seen == IF_ACMPEQ || seen == IF_ACMPNE) && stack.getStackDepth() >= 2) {
615             OpcodeStack.Item item0 = stack.getStackItem(0);
616             OpcodeStack.Item item1 = stack.getStackItem(1);
617             XField field1 = item1.getXField();
618             if (item0.isNull() && field1 != null) {
