45  public class PruneUnconditionalExceptionThrowerEdges implements EdgeTypes {
46      private static final boolean DEBUG = SystemProperties.getBoolean("cfg.prune.throwers.debug");
47  
48      private static final boolean DEBUG_DIFFERENCES = SystemProperties.getBoolean("cfg.prune.throwers.differences.debug");

50      private static final String UNCONDITIONAL_THROWER_METHOD_NAMES = SystemProperties.getProperty(
51              "findbugs.unconditionalThrower", " ").replace(',', '|');

53      private final MethodGen methodGen;
54  
55      private final CFG cfg;
56

65      private static final Pattern unconditionalThrowerPattern;
66  
67      private static final BitSet RETURN_OPCODE_SET = new BitSet();
68      static {
69          RETURN_OPCODE_SET.set(Constants.ARETURN);
70          RETURN_OPCODE_SET.set(Constants.IRETURN);
71          RETURN_OPCODE_SET.set(Constants.LRETURN);
72          RETURN_OPCODE_SET.set(Constants.DRETURN);
73          RETURN_OPCODE_SET.set(Constants.FRETURN);
74          RETURN_OPCODE_SET.set(Constants.RETURN);
75          Pattern p;
76          try {
77              p = Pattern.compile(UNCONDITIONAL_THROWER_METHOD_NAMES);
78              if (DEBUG) {
79                  System.out.println("Pattern is '" + p + "'");
80                  System.out.println(p.matcher("showInvalidPage").matches());
81              }
82          } catch (RuntimeException e) {
83              AnalysisContext.logError("Error compiling unconditional thrower pattern " + UNCONDITIONAL_THROWER_METHOD_NAMES, e);
84              p = Pattern.compile(" ");
85          }
86          unconditionalThrowerPattern = p;
87      }

89      public PruneUnconditionalExceptionThrowerEdges(/* ClassContext classContext, */JavaClass javaClass, Method method,
90              MethodGen methodGen, CFG cfg, ConstantPoolGen cpg, TypeDataflow typeDataflow, AnalysisContext analysisContext) {
91          // this.classContext = classContext;
92          this.methodGen = methodGen;
93          this.cfg = cfg;
94          this.cpg = cpg;
95          this.typeDataflow = typeDataflow;
96          this.analysisContext = analysisContext;
97      }

99      public void execute() throws DataflowAnalysisException {
100         AnalysisContext currentAnalysisContext = AnalysisContext.currentAnalysisContext();
101         if (currentAnalysisContext.getBoolProperty(AnalysisFeatures.CONSERVE_SPACE)) {
102             throw new IllegalStateException("This should not happen");
103         }
104         boolean foundInexact = false;
105         Set<Edge> deletedEdgeSet = new HashSet<Edge>();
106         // TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
107 
108         if (DEBUG) {
109             System.out.println("PruneUnconditionalExceptionThrowerEdges: examining "
110                     + SignatureConverter.convertMethodSignature(methodGen));
111         }
112 
113         for (Iterator<BasicBlock> i = cfg.blockIterator(); i.hasNext();) {
114             BasicBlock basicBlock = i.next();
115             if (!basicBlock.isExceptionThrower()) {
116                 continue;
117             }
118 
119             InstructionHandle instructionHandle = basicBlock.getExceptionThrower();
120             Instruction exceptionThrower = instructionHandle.getInstruction();
121             if (!(exceptionThrower instanceof InvokeInstruction)) {
122                 continue;
123             }
124 
125             InvokeInstruction inv = (InvokeInstruction) exceptionThrower;
126             boolean foundThrower = false;
127             boolean foundNonThrower = false;
128             boolean isExact = true;
129             XMethod primaryXMethod = XFactory.createXMethod(inv, cpg);
130             final String methodName = primaryXMethod.getName();
131             final boolean matches = unconditionalThrowerPattern.matcher(methodName).matches();
132             if (DEBUG) {
133                 System.out.println("Checking '" + methodName + "' is " + matches);
134             }
135 
136             if (matches) {
137                 if (DEBUG) {
138                     System.out.println("\tmatched for " + instructionHandle + " : " + primaryXMethod);
139                 }
140 
141                 foundThrower = true;
142             } else if (inv instanceof INVOKEINTERFACE) {
143                 continue;
144             } else if (inv instanceof INVOKESTATIC) {
145                 foundThrower = isUnconditionalThrower(primaryXMethod);
146             } else {
147                 String className = inv.getClassName(cpg);
148                 if (DEBUG) {
149                     System.out.println("\tlooking up method for " + instructionHandle + " : " + primaryXMethod);
150                 }
151 
152                 Location loc = new Location(instructionHandle, basicBlock);
153                 TypeFrame typeFrame = typeDataflow.getFactAtLocation(loc);
154                 // if (primaryXMethod.isAbstract()) continue;
155                 Set<XMethod> targetSet = null;
156                 try {
157 
158                     if (className.startsWith("[")) {
159                         continue;
160                     }
161                     String methodSig = inv.getSignature(cpg);
162                     if (!methodSig.endsWith("V") && !methodSig.endsWith("Exception;")
163                             && !methodSig.endsWith("Error;") && !methodSig.endsWith(")Ljava/lang/Object;")) {
164                         continue;
165                     }
166 
167                     targetSet = Hierarchy2.resolveMethodCallTargets(inv, typeFrame, cpg);
168 
169                     for (XMethod xMethod : targetSet) {
170                         if (DEBUG) {
171                             System.out.println("\tFound " + xMethod);
172                         }
173 
174                         // Ignore abstract and native methods
175                         boolean isUnconditionalThrower = isUnconditionalThrower(xMethod);
176                         if (isUnconditionalThrower) {
177                             if (!(xMethod.isFinal() || xMethod.isStatic() || xMethod.isPrivate())) {
178                                 try {
179                                     isExact = false;
180                                     XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class,
181                                             xMethod.getClassDescriptor());
182                                     if (xClass.isAbstract()) {
183                                         continue;
184                                     }
185                                 } catch (CheckedAnalysisException e) {
186                                     AnalysisContext.logError("Unable to resolve class for " + xMethod, e);
187                                 }
188                             }
189                             foundThrower = true;
190                             if (DEBUG) {
191                                 System.out.println("Found thrower");
192                             }
193                         } else {
194                             foundNonThrower = true;
195                             if (DEBUG) {
196                                 System.out.println("Found non thrower");
197                             }
198                         }
199 
200                     }
201                 } catch (ClassNotFoundException e) {
202                     analysisContext.getLookupFailureCallback().reportMissingClass(e);
203                 }
204             }
205             boolean newResult = foundThrower && !foundNonThrower;
206             if (newResult) {
207                 if (!isExact) {
208                     foundInexact = true;
209                 }
210                 // Method always throws an unhandled exception
211                 // Remove the normal control flow edge from the CFG.
212                 Edge fallThrough = cfg.getOutgoingEdgeWithType(basicBlock, FALL_THROUGH_EDGE);
213                 if (fallThrough != null) {
214                     if (DEBUG) {
215                         System.out.println("\tREMOVING normal return for: " + primaryXMethod);
216                     }
217                     deletedEdgeSet.add(fallThrough);
218                 }
219             }
220 
221         }
222 
223         if (!deletedEdgeSet.isEmpty()) {
224             cfgModified = true;
225             if (foundInexact) {
226                 cfg.setFlag(CFG.FOUND_INEXACT_UNCONDITIONAL_THROWERS);
227             }
228             // Remove all edges marked for deletion
229             for (Edge edge : deletedEdgeSet) {
230                 cfg.removeEdge(edge);
231 
232             }
233         }
234     }

236     private boolean isUnconditionalThrower(XMethod xMethod) {
237         return xMethod.isUnconditionalThrower() && !xMethod.isUnsupported() && !xMethod.isSynthetic();
238     }
