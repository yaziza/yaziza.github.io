41  public class LoadOfKnownNullValue implements Detector {
42  
43      private final BugReporter bugReporter;
44  
45      private final BugAccumulator bugAccumulator;
46  
47      public LoadOfKnownNullValue(BugReporter bugReporter) {
48          this.bugReporter = bugReporter;
49          this.bugAccumulator = new BugAccumulator(bugReporter);
50      }
51  
52      @Override
53      public void visitClassContext(ClassContext classContext) {
54          Method[] methodList = classContext.getJavaClass().getMethods();
55  
56          for (Method method : methodList) {
57              if (method.getCode() == null) {
58                  continue;
59              }
60  
61              try {
62                  analyzeMethod(classContext, method);
63              } catch (MethodUnprofitableException mue) {
64                  if (SystemProperties.getBoolean("unprofitable.debug")) {
65                      // don't
66                      // report
67                      bugReporter.logError("skipping unprofitable method in " + getClass().getName());
68                  }
69              } catch (CFGBuilderException e) {
70                  bugReporter.logError("Detector " + this.getClass().getName() + " caught exception", e);
71              } catch (DataflowAnalysisException e) {
72                  bugReporter.logError("Detector " + this.getClass().getName() + " caught exception", e);
73              }
74              bugAccumulator.reportAccumulatedBugs();
75          }
76      }

78      private void analyzeMethod(ClassContext classContext, Method method) throws CFGBuilderException, DataflowAnalysisException {
79          BitSet lineMentionedMultipleTimes = classContext.linesMentionedMultipleTimes(method);
80          BitSet linesWithLoadsOfNotDefinitelyNullValues = null;
81  
82          CFG cfg = classContext.getCFG(method);
83          IsNullValueDataflow nullValueDataflow = classContext.getIsNullValueDataflow(method);
84          MethodGen methodGen = classContext.getMethodGen(method);
85          String sourceFile = classContext.getJavaClass().getSourceFileName();
86  
87          if (lineMentionedMultipleTimes.cardinality() > 0) {
88              linesWithLoadsOfNotDefinitelyNullValues = new BitSet();
89              LineNumberTable lineNumbers = method.getLineNumberTable();
90              for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
91                  Location location = i.next();
92  
93                  InstructionHandle handle = location.getHandle();
94                  Instruction ins = handle.getInstruction();
95                  if (!(ins instanceof ALOAD)) {
96                      continue;
97                  }
98  
99                  IsNullValueFrame frame = nullValueDataflow.getFactAtLocation(location);
100                 if (!frame.isValid()) {
101                     // This basic block is probably dead
102                     continue;
103                 }
104                 // System.out.println(handle.getPosition() + "\t" +
105                 // ins.getName() + "\t" + frame);
106 
107                 ALOAD load = (ALOAD) ins;
108 
109                 int index = load.getIndex();
110                 IsNullValue v = frame.getValue(index);
111                 if (!v.isDefinitelyNull()) {
112                     int sourceLine = lineNumbers.getSourceLine(handle.getPosition());
113                     if (sourceLine > 0) {
114                         linesWithLoadsOfNotDefinitelyNullValues.set(sourceLine);
115                     }
116                 }
117             }
118         }
119 
120         IdentityHashMap<InstructionHandle, Object> sometimesGood = new IdentityHashMap<InstructionHandle, Object>();
121 
122         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
123             Location location = i.next();
124             InstructionHandle handle = location.getHandle();
125             Instruction ins = handle.getInstruction();
126             if (!(ins instanceof ALOAD)) {
127                 continue;
128             }
129             IsNullValueFrame frame = nullValueDataflow.getFactAtLocation(location);
130             if (!frame.isValid()) {
131                 // This basic block is probably dead
132                 continue;
133             }
134             // System.out.println(handle.getPosition() + "\t" + ins.getName() +
135             // "\t" + frame);
136 
137             ALOAD load = (ALOAD) ins;
138 
139             int index = load.getIndex();
140             IsNullValue v = frame.getValue(index);
141             if (!v.isDefinitelyNull()) {
142                 sometimesGood.put(handle, null);
143             }
144         }
145 
146         // System.out.println(nullValueDataflow);
147         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
148             Location location = i.next();
149 
150             InstructionHandle handle = location.getHandle();
151             Instruction ins = handle.getInstruction();
152             if (!(ins instanceof ALOAD)) {
153                 continue;
154             }
155 
156             if (sometimesGood.containsKey(handle)) {
157                 continue;
158             }
159             IsNullValueFrame frame = nullValueDataflow.getFactAtLocation(location);
160             if (!frame.isValid()) {
161                 // This basic block is probably dead
162                 continue;
163             }
164             // System.out.println(handle.getPosition() + "\t" + ins.getName() +
165             // "\t" + frame);
166 
167             ALOAD load = (ALOAD) ins;
168 
169             int index = load.getIndex();
170             IsNullValue v = frame.getValue(index);
171             if (v.isDefinitelyNull()) {
172                 InstructionHandle nextHandle = handle.getNext();
173                 Instruction next = nextHandle.getInstruction();
174                 int position = location
175                         .getHandle().getPosition();
176                 int catchSizeANY = Util.getSizeOfSurroundingTryBlock(method, "", position);
177                 if (catchSizeANY < Integer.MAX_VALUE && isNullTestedClose( classContext, load, nextHandle, next)) {
178                     continue;
179                 }
180                 InstructionHandle prevHandle = handle.getPrev();
181                 SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
182                         sourceFile, handle);
183                 SourceLineAnnotation prevSourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext,
184                         methodGen, sourceFile, prevHandle);
185 
186                 if (next instanceof ARETURN) {
187                     // probably stored for duration of finally block
188                     continue;
189                 }
190                 if (next instanceof GOTO) {
191                     InstructionHandle targ = ((BranchInstruction) next).getTarget();
192                     if (targ.getInstruction() instanceof ARETURN) {
193                         // Skip for the same reason we would skip if
194                         // (next instanceof ARETURN) were true. This
195                         // is necessary because the bytecode compiler
196                         // compiles the ternary ? operator with a GOTO
197                         // to an ARETURN instead of just an ARETURN.
198                         continue;
199                     }
200                 }
201                 int startLine = sourceLineAnnotation.getStartLine();
202                 if (startLine > 0 && lineMentionedMultipleTimes.get(startLine)
203                         && linesWithLoadsOfNotDefinitelyNullValues.get(startLine)) {
204                     continue;
205                 }
206 
207                 int previousLine = prevSourceLineAnnotation.getEndLine();
208                 if (startLine < previousLine) {
209                     // probably stored for duration of finally block
210                     // System.out.println("Inverted line");
211                     continue;
212                 }
213                 int priority = NORMAL_PRIORITY;
214                 if (!v.isChecked()) {
215                     priority++;
216                 }
217 
218                 BugAnnotation variableAnnotation = null;
219                 try {
220                     // Get the value number
221                     ValueNumberFrame vnaFrame = classContext.getValueNumberDataflow(method).getFactAfterLocation(location);
222                     if (vnaFrame.isValid()) {
223 
224                         ValueNumber valueNumber = vnaFrame.getTopValue();
225                         if (valueNumber.hasFlag(ValueNumber.CONSTANT_CLASS_OBJECT)) {
226                             return;
227                         }
228                         variableAnnotation = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, valueNumber, vnaFrame,
229                                 "VALUE_OF");
230                         if (variableAnnotation instanceof LocalVariableAnnotation) {
231                             LocalVariableAnnotation local = (LocalVariableAnnotation) variableAnnotation;
232                             if (!local.isNamed()) {
233                                 priority++;
234                             }
235                         }
236 
237                     }
238                 } catch (DataflowAnalysisException e) {
239                     // ignore
240                 } catch (CFGBuilderException e) {
241                     // ignore
242                 }
243 
244                 // System.out.println("lineMentionedMultipleTimes: " +
245                 // lineMentionedMultipleTimes);
246                 // System.out.println("linesWithLoadsOfNonNullValues: " +
247                 // linesWithLoadsOfNotDefinitelyNullValues);
248 
249                 bugAccumulator.accumulateBug(
250                         new BugInstance(this, "NP_LOAD_OF_KNOWN_NULL_VALUE", priority).addClassAndMethod(methodGen, sourceFile)
251                         .addOptionalAnnotation(variableAnnotation),
252                         sourceLineAnnotation);
253             }
254 
255         }
256     }

258     /**
259      * @param classContext
260      * @param nextHandle
261      * @param next
262      */
263     private boolean isNullTestedClose(ClassContext classContext, ALOAD load, InstructionHandle nextHandle, Instruction next) {
264         if (!(next instanceof IFNULL)) {
265             return false;
266         }
267 
268         IFNULL ifNull = (IFNULL) next;
269         InstructionHandle nextNextHandle = nextHandle.getNext(); // aload
270         if (nextNextHandle == null) {
271             return false;
272         }
273         Instruction nextInstruction = nextNextHandle.getInstruction();
274 
275         if (!(nextInstruction instanceof ALOAD)) {
276             return false;
277         }
278         ALOAD nextLoad = (ALOAD) nextInstruction;
279         if (load.getIndex() != nextLoad.getIndex()) {
280             return false;
281         }
282         InstructionHandle nextNextNextHandle = nextNextHandle.getNext(); // invoke
283         if (nextNextNextHandle == null) {
284             return false;
285         }
286         Instruction nextNextNextInstruction = nextNextNextHandle.getInstruction();
287         if (!(nextNextNextInstruction instanceof INVOKEVIRTUAL)) {
288             return false;
289         }
290         INVOKEVIRTUAL invokeVirtual = (INVOKEVIRTUAL) nextNextNextInstruction;
291         String methodName = invokeVirtual.getMethodName(classContext.getConstantPoolGen());
292         String methodSig = invokeVirtual.getSignature(classContext.getConstantPoolGen());
293         if (!"close".equals(methodName)) {
294             return false;
295         }
296         if (!"()V".equals(methodSig)) {
297             return false;
298         }
299         InstructionHandle nextNextNextNextHandle = nextNextNextHandle.getNext(); // after
300         if (ifNull.getTarget() != nextNextNextNextHandle) {
301             return false;
302         }
303 
304         return true;
305 
306     }
