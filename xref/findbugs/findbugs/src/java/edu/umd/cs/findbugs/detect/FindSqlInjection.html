105     private static class StringAppendState {
106         // remember the smallest position at which we saw something that
107         // concerns us
108         int sawOpenQuote = Integer.MAX_VALUE;
109 
110         int sawCloseQuote = Integer.MAX_VALUE;
111 
112         int sawComma = Integer.MAX_VALUE;
113 
114         int sawAppend = Integer.MAX_VALUE;

116         int sawUnsafeAppend = Integer.MAX_VALUE;

120         int sawSeriousTaint = Integer.MAX_VALUE;

122         public boolean getSawOpenQuote(InstructionHandle handle) {
123             return sawOpenQuote <= handle.getPosition();
124         }
125 
126         public boolean getSawCloseQuote(InstructionHandle handle) {
127             return sawCloseQuote <= handle.getPosition();
128         }
129 
130         public boolean getSawComma(InstructionHandle handle) {
131             return sawComma <= handle.getPosition();
132         }
133 
134         public boolean getSawAppend(InstructionHandle handle) {
135             return sawAppend <= handle.getPosition();
136         }
137 
138         public boolean getSawUnsafeAppend(InstructionHandle handle) {
139             return sawUnsafeAppend <= handle.getPosition();
140         }
141 
142         public boolean getSawTaint(InstructionHandle handle) {
143             return sawTaint <= handle.getPosition();
144         }
145 
146         public boolean getSawSeriousTaint(InstructionHandle handle) {
147             return sawSeriousTaint <= handle.getPosition();

151             sawOpenQuote = Math.min(sawOpenQuote, handle.getPosition());
152         }

155             sawCloseQuote = Math.min(sawCloseQuote, handle.getPosition());

192     private final boolean testingEnabled;
193 
194     public FindSqlInjection(BugReporter bugReporter) {
195         this.bugReporter = bugReporter;
196         this.bugAccumulator = new BugAccumulator(bugReporter);
197         testingEnabled = SystemProperties.getBoolean("report_TESTING_pattern_in_standard_detectors");
198         Set<MethodParameter> baseExecuteMethods = new HashSet<>();
199         for(MethodDescriptor executeMethod : EXECUTE_METHODS) {
200             baseExecuteMethods.add(new MethodParameter(executeMethod, 0));
201         }
202         executeMethods = Global.getAnalysisCache().getDatabase(StringPassthruDatabase.class).findLinkedMethods(baseExecuteMethods);
203         Set<MethodParameter> basePrepareMethods = new HashSet<>();
204         for(String signature : PREPARE_STATEMENT_SIGNATURES) {
205             basePrepareMethods.add(new MethodParameter(new MethodDescriptor("java/sql/Connection", "prepareStatement", signature), 0));
206         }
207         preparedStatementMethods = Global.getAnalysisCache().getDatabase(StringPassthruDatabase.class).findLinkedMethods(basePrepareMethods);
208         allMethods.addAll(executeMethods.keySet());
209         allMethods.addAll(preparedStatementMethods.keySet());
210     }

212     @Override
213     public void visitClassContext(ClassContext classContext) {
214         JavaClass javaClass = classContext.getJavaClass();
215         if(!PreorderVisitor.hasInterestingMethod(javaClass.getConstantPool(), allMethods)) {
216             return;
217         }
218         Method[] methodList = javaClass.getMethods();
219 
220         for (Method method : methodList) {
221             MethodGen methodGen = classContext.getMethodGen(method);
222             if (methodGen == null) {
223                 continue;
224             }
225 
226             try {
227                 analyzeMethod(classContext, method);
228             } catch (DataflowAnalysisException e) {
229                 bugReporter.logError(
230                         "FindSqlInjection caught exception while analyzing " + classContext.getFullyQualifiedMethodName(method),
231                         e);
232             } catch (CFGBuilderException e) {
233                 bugReporter.logError(
234                         "FindSqlInjection caught exception while analyzing " + classContext.getFullyQualifiedMethodName(method),
235                         e);
236             } catch (RuntimeException e) {
237                 bugReporter.logError(
238                         "FindSqlInjection caught exception while analyzing " + classContext.getFullyQualifiedMethodName(method),
239                         e);
240             }
241         }
242     }

244     private boolean isStringAppend(Instruction ins, ConstantPoolGen cpg) {
245         if (ins instanceof INVOKEVIRTUAL) {
246             INVOKEVIRTUAL invoke = (INVOKEVIRTUAL) ins;
247 
248             if ("append".equals(invoke.getMethodName(cpg)) && invoke.getClassName(cpg).startsWith("java.lang.StringB")) {
249                 String sig = invoke.getSignature(cpg);
250                 char firstChar = sig.charAt(1);
251                 return firstChar == '[' || firstChar == 'L';
252             }
253         }
254 
255         return false;
256     }

258     private boolean isConstantStringLoad(Location location, ConstantPoolGen cpg)  {
259         Instruction ins = location.getHandle().getInstruction();
260         if (ins instanceof LDC) {
261             LDC load = (LDC) ins;
262             Object value = load.getValue(cpg);
263             if (value instanceof String) {
264                 return true;
265             }
266         }
267 
268         return false;
269     }

271     static final Pattern openQuotePattern = Pattern.compile("((^')|[^\\p{Alnum}]')$");
272 
273     public static boolean isOpenQuote(String s) {
274         return openQuotePattern.matcher(s).find();
275     }

277     static final Pattern closeQuotePattern = Pattern.compile("^'($|[^\\p{Alnum}])");
278 
279     public static boolean isCloseQuote(String s) {
280         return closeQuotePattern.matcher(s).find();
281     }

283     private StringAppendState updateStringAppendState(Location location, ConstantPoolGen cpg, StringAppendState stringAppendState)
284     {
285         InstructionHandle handle = location.getHandle();
286         Instruction ins = handle.getInstruction();
287         if (!isConstantStringLoad(location, cpg)) {
288             throw new IllegalArgumentException("instruction must be LDC");
289         }
290 
291         LDC load = (LDC) ins;
292         Object value = load.getValue(cpg);
293         String stringValue = ((String) value).trim();
294         if (stringValue.startsWith(",") || stringValue.endsWith(",")) {
295             stringAppendState.setSawComma(handle);
296         }
297         if (isCloseQuote(stringValue) && stringAppendState.getSawOpenQuote(handle)) {
298             stringAppendState.setSawCloseQuote(handle);
299         }
300         if (isOpenQuote(stringValue)) {
301             stringAppendState.setSawOpenQuote(handle);
302         }
303 
304         return stringAppendState;
305     }

307     private StringAppendState getStringAppendState(CFG cfg, ConstantPoolGen cpg) throws CFGBuilderException {
308         StringAppendState stringAppendState = new StringAppendState();
309         String sig = method.getSignature();
310         sig = sig.substring(0, sig.indexOf(')'));
311 
312         if (sig.indexOf("java/lang/String") >= 0) {
313             stringAppendState.setSawInitialTaint();
314         }
315         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
316             Location location = i.next();
317             InstructionHandle handle = location.getHandle();
318             Instruction ins = handle.getInstruction();
319             if (isConstantStringLoad(location, cpg)) {
320                 stringAppendState = updateStringAppendState(location, cpg, stringAppendState);
321             } else if (isStringAppend(ins, cpg)) {
322                 stringAppendState.setSawAppend(handle);
323 
324                 Location prevLocation = getPreviousLocation(cfg, location, true);
325                 if (prevLocation != null && !isSafeValue(prevLocation, cpg)) {
326                     stringAppendState.setSawUnsafeAppend(handle);
327                 }
328 
329             } else if (ins instanceof InvokeInstruction) {
330                 InvokeInstruction inv = (InvokeInstruction) ins;
331                 String sig1 = inv.getSignature(cpg);
332                 String sig2 = sig1.substring(sig1.indexOf(')'));
333 
334                 if (sig2.indexOf("java/lang/String") >= 0) {
335                     String methodName = inv.getMethodName(cpg);
336                     String className = inv.getClassName(cpg);
337                     if ("valueOf".equals(methodName) && "java.lang.String".equals(className)
338                             && "(Ljava/lang/Object;)Ljava/lang/String;".equals(sig1)) {
339                         try {
340                             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
341                             TypeFrame frame = typeDataflow.getFactAtLocation(location);
342                             if (!frame.isValid()) {
343                                 // This basic block is probably dead
344                                 continue;
345                             }
346                             Type operandType = frame.getTopValue();
347                             if (operandType.equals(TopType.instance())) {
348                                 // unreachable
349                                 continue;
350                             }
351                             String sig3 = operandType.getSignature();
352                             if (!"Ljava/lang/String;".equals(sig3)) {
353                                 stringAppendState.setSawTaint(handle);
354                             }
355                         } catch (CheckedAnalysisException e) {
356                             stringAppendState.setSawTaint(handle);
357                         }
358                     } else if (className.startsWith("java.lang.String") || "java.lang.Long".equals(className)
359                             || "java.lang.Integer".equals(className) || "java.lang.Float".equals(className)
360                             || "java.lang.Double".equals(className) || "java.lang.Short".equals(className)
361                             || "java.lang.Byte".equals(className) || "java.lang.Character".equals(className)) {
362                         // ignore it
363                         assert true;
364                     } else if (methodName.startsWith("to") && methodName.endsWith("String") && methodName.length() > 8) {
365                         // ignore it
366                         assert true;
367                     } else if (className.startsWith("javax.servlet") && methodName.startsWith("get")) {
368                         stringAppendState.setSawTaint(handle);
369                         stringAppendState.setSawSeriousTaint(handle);
370                     } else {
371                         stringAppendState.setSawTaint(handle);
372                     }
373 
374                 }
375             } else if (ins instanceof GETFIELD) {
376                 GETFIELD getfield = (GETFIELD) ins;
377                 String sig2 = getfield.getSignature(cpg);
378                 if (sig2.indexOf("java/lang/String") >= 0) {
379                     stringAppendState.setSawTaint(handle);
380                 }
381             }
382         }
383 
384         return stringAppendState;
385     }

387     private boolean isSafeValue(Location location, ConstantPoolGen cpg) throws CFGBuilderException {
388         Instruction prevIns = location.getHandle().getInstruction();
389         if (prevIns instanceof LDC || prevIns instanceof GETSTATIC) {
390             return true;
391         }
392         if (prevIns instanceof InvokeInstruction) {
393             String methodName = ((InvokeInstruction) prevIns).getMethodName(cpg);
394             if (methodName.startsWith("to") && methodName.endsWith("String") && methodName.length() > 8) {
395                 return true;
396             }
397         }
398         if (prevIns instanceof AALOAD) {
399             CFG cfg = classContext.getCFG(method);
400 
401             Location prev = getPreviousLocation(cfg, location, true);
402             if (prev != null) {
403                 Location prev2 = getPreviousLocation(cfg, prev, true);
404                 if (prev2 != null && prev2.getHandle().getInstruction() instanceof GETSTATIC) {
405                     GETSTATIC getStatic = (GETSTATIC) prev2.getHandle().getInstruction();
406                     if ("[Ljava/lang/String;".equals(getStatic.getSignature(cpg))) {
407                         return true;
408                     }
409                 }
410             }
411         }
412         return false;
413     }

415     private @CheckForNull
416     InstructionHandle getPreviousInstruction(InstructionHandle handle, boolean skipNops) {
417         while (handle.getPrev() != null) {
418             handle = handle.getPrev();
419             Instruction prevIns = handle.getInstruction();
420             if (!(prevIns instanceof NOP && skipNops)) {
421                 return handle;
422             }
423         }
424         return null;
425     }

427     private @CheckForNull
428     Location getPreviousLocation(CFG cfg, Location startLocation, boolean skipNops) {
429         Location loc = startLocation;
430         InstructionHandle prev = getPreviousInstruction(loc.getHandle(), skipNops);
431         if (prev != null) {
432             return new Location(prev, loc.getBasicBlock());
433         }
434         BasicBlock block = loc.getBasicBlock();
435         while (true) {
436             block = cfg.getPredecessorWithEdgeType(block, EdgeTypes.FALL_THROUGH_EDGE);
437             if (block == null) {
438                 return null;
439             }
440             InstructionHandle lastInstruction = block.getLastInstruction();
441             if (lastInstruction != null) {
442                 return new Location(lastInstruction, block);
443             }
444         }
445     }

447     private BugInstance generateBugInstance(JavaClass javaClass, MethodGen methodGen, InstructionHandle handle,
448             StringAppendState stringAppendState, boolean isExecute) {
449         int priority = LOW_PRIORITY;
450         boolean sawSeriousTaint = false;
451         if (stringAppendState.getSawAppend(handle)) {
452             if (stringAppendState.getSawOpenQuote(handle) && stringAppendState.getSawCloseQuote(handle)) {
453                 priority = HIGH_PRIORITY;
454             } else if (stringAppendState.getSawComma(handle)) {
455                 priority = NORMAL_PRIORITY;
456             }
457 
458             if (!stringAppendState.getSawUnsafeAppend(handle)) {
459                 priority += 2;
460             } else if (stringAppendState.getSawSeriousTaint(handle)) {
461                 priority--;
462                 sawSeriousTaint = true;
463             } else if (!stringAppendState.getSawTaint(handle)) {
464                 priority++;
465             }
466         }
467 
468         String description = "TESTING";
469         if (isExecute) {
470             description = "SQL_NONCONSTANT_STRING_PASSED_TO_EXECUTE";
471         } else {
472             description = "SQL_PREPARED_STATEMENT_GENERATED_FROM_NONCONSTANT_STRING";
473         }
474 
475         BugInstance bug = new BugInstance(this, description, priority);
476         bug.addClassAndMethod(methodGen, javaClass.getSourceFileName());
477         if ("TESTING".equals(description)) {
478             bug.addString("Incomplete report invoking non-constant SQL string");
479         }
480         if (sawSeriousTaint) {
481             bug.addString("non-constant SQL string involving HTTP taint");
482         }
483 
484         return bug;
485     }

491     private void analyzeMethod(ClassContext classContext, Method method) throws DataflowAnalysisException, CFGBuilderException {
492         JavaClass javaClass = classContext.getJavaClass();
493         this.method = method;
494         this.classContext = classContext;
495         MethodGen methodGen = classContext.getMethodGen(method);
496         if (methodGen == null) {
497             return;
498         }
499 
500         ConstantPoolGen cpg = methodGen.getConstantPool();
501         CFG cfg = classContext.getCFG(method);
502 
503         StringAppendState stringAppendState = getStringAppendState(cfg, cpg);
504 
505         ConstantDataflow dataflow = classContext.getConstantDataflow(method);
506         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
507             Location location = i.next();
508             Instruction ins = location.getHandle().getInstruction();
509             if (!(ins instanceof InvokeInstruction)) {
510                 continue;
511             }
512             InvokeInstruction invoke = (InvokeInstruction) ins;
513             MethodDescriptor md = new MethodDescriptor(invoke, cpg);
514             boolean executeMethod;
515             int[] params = preparedStatementMethods.get(md);
516             int paramNumber;
517             // Currently only one method parameter is checked, though it's the most common case
518             // TODO: support methods which take several SQL statements
519             if(params != null) {
520                 executeMethod = false;
521                 paramNumber = params[0];
522             } else {
523                 params = executeMethods.get(md);
524                 if(params != null) {
525                     executeMethod = true;
526                     paramNumber = params[0];
527                 } else {
528                     continue;
529                 }
530             }
531             ConstantFrame frame = dataflow.getFactAtLocation(location);
532             int numArguments = frame.getNumArguments(invoke, cpg);
533             Constant value = frame.getStackValue(numArguments - 1 - paramNumber);
534 
535             if (!value.isConstantString()) {
536                 // TODO: verify it's the same string represented by
537                 // stringAppendState
538                 // FIXME: will false positive on const/static strings
539                 // returns by methods
540                 Location prev = getPreviousLocation(cfg, location, true);
541                 if (prev == null || !isSafeValue(prev, cpg)) {
542                     BugInstance bug = generateBugInstance(javaClass, methodGen, location.getHandle(), stringAppendState, executeMethod);
543                     if(!testingEnabled && "TESTING".equals(bug.getType())){
544                         continue;
545                     }
546                     bugAccumulator.accumulateBug(
547                             bug,
548                             SourceLineAnnotation.fromVisitedInstruction(classContext, methodGen,
549                                     javaClass.getSourceFileName(), location.getHandle()));
550                 }
551             }
552         }
553         bugAccumulator.reportAccumulatedBugs();
554     }
