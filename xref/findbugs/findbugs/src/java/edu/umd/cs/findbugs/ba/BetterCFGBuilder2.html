287         /**
288          * Add an instruction to the subroutine. We keep track of which
289          * instructions are part of which subroutines. No instruction may be
290          * part of more than one subroutine.
291          *
292          * @param handle
293          *            the instruction to be added to the subroutine
294          */
295         public void addInstruction(InstructionHandle handle) throws CFGBuilderException {
296             int position = handle.getPosition();
297             if (usedInstructionSet.get(position)) {
298                 throw new CFGBuilderException("Instruction " + handle + " visited in multiple subroutines");
299             }
300             instructionSet.set(position);
301             usedInstructionSet.set(position);
302         }
303 
304         /**
305          * Is the given instruction part of this subroutine?
306          */
307         public boolean containsInstruction(InstructionHandle handle) {
308             return instructionSet.get(handle.getPosition());
309         }

311         /**
312          * Get the basic block in the subroutine for the given instruction. If
313          * the block doesn't exist yet, it is created, and a work list item is
314          * added which will populate it. Note that if start is an exception
315          * thrower, the block returned will be its ETB.
316          *
317          * @param start
318          *            the start instruction for the block
319          * @return the basic block for the instruction
320          */
321         public BasicBlock getBlock(InstructionHandle start) {
322             BasicBlock block = blockMap.get(start);
323             if (block == null) {
324                 block = allocateBasicBlock();
325                 blockMap.put(start, block);
326 
327                 // Block is an exception handler?
328                 CodeExceptionGen exceptionGen = exceptionHandlerMap.getHandlerForStartInstruction(start);
329                 if (exceptionGen != null) {
330                     block.setExceptionGen(null, exceptionGen);
331                 }
332 
333                 addItem(new WorkListItem(start, block));
334             }
335             return block;
336         }

390         /**
391          * Add a control flow edge to the subroutine. If the control target has
392          * not yet been added to the subroutine, a new work list item is added.
393          * If the control target is in another subroutine, an EscapeTarget is
394          * added.
395          *
396          * @param sourceBlock
397          *            the source basic block
398          * @param target
399          *            the control target
400          * @param edgeType
401          *            the type of control edge
402          */
403         public void addEdgeAndExplore(BasicBlock sourceBlock, InstructionHandle target, @Edge.Type int edgeType) {
404             if (usedInstructionSet.get(target.getPosition()) && !containsInstruction(target)) {
405                 // Control escapes this subroutine
406                 List<EscapeTarget> escapeTargetList = escapeTargetListMap.get(sourceBlock);
407                 if (escapeTargetList == null) {
408                     escapeTargetList = new LinkedList<EscapeTarget>();
409                     escapeTargetListMap.put(sourceBlock, escapeTargetList);
410                 }
411                 escapeTargetList.add(new EscapeTarget(target, edgeType));
412             } else {
413                 // Edge within the current subroutine
414                 BasicBlock targetBlock = getBlock(target);
415                 addEdge(sourceBlock, targetBlock, edgeType);
416             }
417         }
418 
419         /**
420          * Add an edge to the subroutine's CFG.
421          *
422          * @param sourceBlock
423          *            the source basic block
424          * @param destBlock
425          *            the destination basic block
426          * @param edgeType
427          *            the type of edge
428          */
429         public void addEdge(BasicBlock sourceBlock, BasicBlock destBlock, @Edge.Type int edgeType) {
430             if (VERIFY_INTEGRITY) {
431                 if (destBlock.isExceptionHandler() && edgeType != HANDLED_EXCEPTION_EDGE) {
432                     throw new IllegalStateException("In method " + SignatureConverter.convertMethodSignature(methodGen)
433                             + ": exception handler " + destBlock.getFirstInstruction() + " reachable by non exception edge type "
434                             + edgeType);
435                 }
436             }
437             cfgSub.createEdge(sourceBlock, destBlock, edgeType);
438         }

545         /**
546          * Get the block in the result CFG corresponding to the given subroutine
547          * block.
548          *
549          * @param subBlock
550          *            the subroutine block
551          * @return the result CFG block
552          */
553         public BasicBlock getBlock(BasicBlock subBlock) {
554             BasicBlock resultBlock = blockMap.get(subBlock);
555             if (resultBlock == null) {
556                 resultBlock = result.allocate();
557                 blockMap.put(subBlock, resultBlock);
558                 workList.add(subBlock);
559             }
560             return resultBlock;
561         }

563         /**
564          * Check to ensure that this context is not the result of recursion.
565          */
566         public void checkForRecursion() throws CFGBuilderException {
567             Context callerContext = caller;
568 
569             while (callerContext != null) {
570                 if (callerContext.subroutine == this.subroutine) {
571                     throw new CFGBuilderException("JSR recursion detected!");
572                 }
573                 callerContext = callerContext.caller;
574             }
575         }

613     public BetterCFGBuilder2(@Nonnull MethodDescriptor descriptor, @Nonnull MethodGen methodGen) {
614         this.methodGen = methodGen;
615         this.cpg = methodGen.getConstantPool();
616         IAnalysisCache analysisCache = Global.getAnalysisCache();
617         StandardTypeMerger merger = null;
618         ExceptionSetFactory exceptionSetFactory;
619         try {
620             exceptionSetFactory = analysisCache.getMethodAnalysis(ExceptionSetFactory.class, descriptor);
621             merger = new StandardTypeMerger( AnalysisContext.currentAnalysisContext()
622                     .getLookupFailureCallback(), exceptionSetFactory);
623         } catch (CheckedAnalysisException e) {
624             AnalysisContext.logError("Unable to generate exceptionSetFactory for " + descriptor, e);
625         }
626 
627 
628         this.exceptionHandlerMap = new ExceptionHandlerMap(methodGen, merger);
629         this.usedInstructionSet = new BitSet();
630         this.jsrSubroutineMap = new IdentityHashMap<InstructionHandle, Subroutine>();
631         this.subroutineWorkList = new LinkedList<Subroutine>();
632     }

634     public int getIndex(FieldDescriptor f) {
635         Integer i = addedFields.get(f);
636         if (i != null) {
637             return i;
638         }
639         int index = cpg.addFieldref(f.getSlashedClassName(), f.getName(), f.getSignature());
640         addedFields.put(f, index);
641         return index;

644     public void optimize(InstructionList instructionList) {
645         InstructionHandle head = instructionList.getStart();
646 
647         while (head != null) {
648             Instruction i = head.getInstruction();
649 
650 
651             if (i instanceof INVOKESTATIC) {
652                 INVOKESTATIC is = (INVOKESTATIC) i;
653                 String name = is.getMethodName(cpg);
654                 String signature = is.getSignature(cpg);
655                 if (name.startsWith("access$")) {
656                     XMethod invoked = XFactory.createXMethod(is, cpg);
657                     FieldDescriptor field = invoked.getAccessMethodForField();
658                     if (field != null) {
659                         boolean isSetter = signature.endsWith("V");
660                         Instruction replacement;
661                         int index = getIndex(field);
662                         if (field.isStatic()) {
663                             if (isSetter) {
664                                 replacement = new PUTSTATIC(index);
665                             } else {
666                                 replacement = new GETSTATIC(index);
667                             }
668                         } else {
669                             if (isSetter) {
670                                 replacement = new PUTFIELD(index);
671                             } else {
672                                 replacement = new GETFIELD(index);
673                             }
674                         }
675                         head.swapInstruction(replacement);
676                         /*
677                             if (false)
678                                 System.out.println("Substituting " + (isSetter ? "set" : "get") + " of " + field + " for call of "
679                                     + invoked + " in " + methodGen.getClassName() + "." + methodGen.getName()
680                                     + methodGen.getSignature());
681                          */
682 
683                     }
684 
685                 }
686             }
687             if (i instanceof IfInstruction) {
688                 IfInstruction ii = (IfInstruction) i;
689                 InstructionHandle target = ii.getTarget();
690                 InstructionHandle next = head.getNext();
691                 if (target.equals(next)) {
692                     int consumed = ii.consumeStack(methodGen.getConstantPool());
693                     if (consumed != 1 && consumed != 2) {
694                         throw new IllegalStateException();
695                     }
696                     head.swapInstruction(consumed == 1 ? new POP() : new POP2());
697                 }
698 
699             }
700             if (i instanceof IFNULL || i instanceof IFNONNULL) {
701                 IfInstruction ii = (IfInstruction) i;
702                 InstructionHandle target = ii.getTarget();
703                 InstructionHandle next1 = head.getNext(); // ICONST
704                 if (next1 == null) {
705                     break;
706                 }
707                 if (next1.getInstruction() instanceof ICONST) {
708                     InstructionHandle next2 = next1.getNext(); // GOTO
709                     if (next2 == null) {
710                         break;
711                     }
712                     InstructionHandle next3 = next2.getNext(); // ICONST
713                     if (next3== null) {
714                         break;
715                     }
716                     InstructionHandle next4 = next3.getNext();
717                     if (next4 == null) {
718                         break;
719                     }
720                     if (target.equals(next3)  && next2.getInstruction() instanceof GOTO
721                             && next3.getInstruction() instanceof ICONST && next1.getTargeters().length == 0
722                             && next2.getTargeters().length == 0 && next3.getTargeters().length == 1
723                             && next4.getTargeters().length == 1) {
724                         int c1 = ((ICONST) next1.getInstruction()).getValue().intValue();
725                         GOTO g = (GOTO) next2.getInstruction();
726                         int c2 = ((ICONST) next3.getInstruction()).getValue().intValue();
727                         if (g.getTarget().equals(next4) && (c1 == 1 && c2 == 0 || c1 == 0 && c2 == 1)) {
728                             boolean nullIsTrue = i instanceof IFNULL && c2 == 1 || i instanceof IFNONNULL && c2 == 0;
729 
730                             if (nullIsTrue) {
731                                 // System.out.println("Found NULL2Z instruction");
732                                 head.swapInstruction(new NULL2Z());
733 
734                             } else {
735                                 // System.out.println("Found NONNULL2Z instruction");
736                                 head.swapInstruction(new NONNULL2Z());
737                             }
738                             next3.removeAllTargeters();
739                             next4.removeAllTargeters();
740                             next1.swapInstruction(new NOP());
741                             next2.swapInstruction(new NOP());
742                             next3.swapInstruction(new NOP());
743                         }
744                     }
745                 }
746 
747             }
748             if (i instanceof ACONST_NULL) {
749                 InstructionHandle next = head.getNext();
750                 assert next != null;
751                 InstructionHandle next2 = next.getNext();
752                 if (next2 != null && next.getInstruction() instanceof ALOAD) {
753                     Instruction check = next2.getInstruction();
754                     if (check instanceof IF_ACMPNE || check instanceof IF_ACMPEQ) {
755                         // need to update
756                         head.swapInstruction(new NOP());
757                         IfInstruction ifTest = (IfInstruction) check;
758                         if (check instanceof IF_ACMPNE) {
759                             next2.swapInstruction(new IFNONNULL(ifTest.getTarget()));
760                         } else {
761                             next2.swapInstruction(new IFNULL(ifTest.getTarget()));
762                         }
763                     }
764                 }
765             }
766             head = head.getNext();
767         }

770     @Override
771     public void build() throws CFGBuilderException {
772         InstructionList instructionList = methodGen.getInstructionList();
773         optimize(instructionList);
774         topLevelSubroutine = new Subroutine(instructionList.getStart());
775         subroutineWorkList.add(topLevelSubroutine);
776 
777         // Build top level subroutine and all JSR subroutines
778         while (!subroutineWorkList.isEmpty()) {
779             Subroutine subroutine = subroutineWorkList.removeFirst();
780             if (DEBUG) {
781                 System.out.println("Starting subroutine " + subroutine.getStartInstruction());
782             }
783             build(subroutine);
784         }
785 
786         // Inline everything into the top level subroutine
787         cfg = inlineAll();
788 
789         // Add a NOP instruction to the entry block.
790         // This allows analyses to construct a Location
791         // representing the entry to the method.
792         BasicBlock entryBlock = cfg.getEntry();
793         InstructionList il = new InstructionList();
794         entryBlock.addInstruction(il.append(new NOP()));
795 
796         if (VERIFY_INTEGRITY) {
797             cfg.checkIntegrity();
798         }
799 
800         if (true) {
801             cfg.checkIntegrity();
802 
803         }
804     }

817     /**
818      * Build a subroutine. We iteratively add basic blocks to the subroutine
819      * until there are no more blocks reachable from the calling context. As JSR
820      * instructions are encountered, new Subroutines are added to the subroutine
821      * work list.
822      *
823      * @param subroutine
824      *            the subroutine
825      */
826     private void build(Subroutine subroutine) throws CFGBuilderException {
827         // Prime the work list
828         subroutine.addEdgeAndExplore(subroutine.getEntry(), subroutine.getStartInstruction(), START_EDGE);
829 
830         // Keep going until all basic blocks in the subroutine have been added
831         while (subroutine.hasMoreWork()) {
832             WorkListItem item = subroutine.nextItem();
833 
834             InstructionHandle handle = item.getStartInstruction();
835             BasicBlock basicBlock = item.getBasicBlock();
836 
837             // Add exception handler block (ETB) for exception-throwing
838             // instructions
839             if (isPEI(handle)) {
840                 if (DEBUG) {
841                     System.out.println("ETB block " + basicBlock.getLabel() + " for " + handle);
842                 }
843                 handleExceptions(subroutine, handle, basicBlock);
844                 BasicBlock body = subroutine.allocateBasicBlock();
845                 subroutine.addEdge(basicBlock, body, FALL_THROUGH_EDGE);
846                 basicBlock = body;
847             }
848 
849             if (DEBUG) {
850                 System.out.println("BODY block " + basicBlock.getLabel() + " for " + handle);
851             }
852 
853             if (!basicBlock.isEmpty()) {
854                 throw new IllegalStateException("Block isn't empty!");
855             }
856 
857             // Add instructions until we get to the end of the block
858             boolean endOfBasicBlock = false;
859             do {
860                 Instruction ins = handle.getInstruction();
861 
862                 // Add the instruction to the block
863                 if (DEBUG) {
864                     System.out.println("BB " + basicBlock.getLabel() + ": adding" + handle);
865                 }
866                 basicBlock.addInstruction(handle);
867                 subroutine.addInstruction(handle);
868 
869                 short opcode = ins.getOpcode();
870 
871                 // TODO: should check instruction to ensure that in a JSR
872                 // subroutine
873                 // no assignments are made to the local containing the return
874                 // address.
875                 // if (ins instanceof ASTORE) ...
876 
877                 if (opcode == Constants.JSR || opcode == Constants.JSR_W) {
878                     // Find JSR subroutine, add it to subroutine work list if
879                     // we haven't built a CFG for it yet
880                     JsrInstruction jsr = (JsrInstruction) ins;
881                     InstructionHandle jsrTarget = jsr.getTarget();
882                     Subroutine jsrSubroutine = jsrSubroutineMap.get(jsrTarget);
883                     if (jsrSubroutine == null) {
884                         jsrSubroutine = new Subroutine(jsrTarget);
885                         jsrSubroutineMap.put(jsrTarget, jsrSubroutine);
886                         subroutineWorkList.add(jsrSubroutine);
887                     }
888 
889                     // This ends the basic block.
890                     // Add a JSR_EDGE to the successor.
891                     // It will be replaced later by the inlined JSR subroutine.
892                     subroutine.addEdgeAndExplore(basicBlock, handle.getNext(), JSR_EDGE);
893                     endOfBasicBlock = true;
894                 } else if (opcode == Constants.RET) {
895                     // End of JSR subroutine
896                     subroutine.addEdge(basicBlock, subroutine.getExit(), RET_EDGE);
897                     endOfBasicBlock = true;
898                 } else {
899                     TargetEnumeratingVisitor visitor = new TargetEnumeratingVisitor(handle, cpg);
900                     if (visitor.isEndOfBasicBlock()) {
901                         endOfBasicBlock = true;
902 
903                         // Add control edges as appropriate
904                         if (visitor.instructionIsThrow()) {
905                             handleExceptions(subroutine, handle, basicBlock);
906                         } else if (visitor.instructionIsExit()) {
907                             subroutine.setExitBlock(basicBlock);
908                         } else if (visitor.instructionIsReturn()) {
909                             subroutine.setReturnBlock(basicBlock);
910                         } else {
911                             Iterator<Target> i = visitor.targetIterator();
912                             while (i.hasNext()) {
913                                 Target target = i.next();
914                                 subroutine.addEdgeAndExplore(basicBlock, target.getTargetInstruction(), target.getEdgeType());
915                             }
916                         }
917                     }
918                 }
919 
920                 if (!endOfBasicBlock) {
921                     InstructionHandle next = handle.getNext();
922                     if (next == null) {
923                         throw new CFGBuilderException("Control falls off end of method: " + handle);
924                     }
925 
926                     // Is the next instruction a control merge or a PEI?
927                     if (isMerge(next) || isPEI(next)) {
928                         subroutine.addEdgeAndExplore(basicBlock, next, FALL_THROUGH_EDGE);
929                         endOfBasicBlock = true;
930                     } else {
931                         // Basic block continues
932                         handle = next;
933                     }
934                 }
935 
936             } while (!endOfBasicBlock);
937         }
938     }

940     /**
941      * Add exception edges for given instruction.
942      *
943      * @param subroutine
944      *            the subroutine containing the instruction
945      * @param pei
946      *            the instruction which throws an exception
947      * @param etb
948      *            the exception thrower block (ETB) for the instruction
949      */
950     private void handleExceptions(Subroutine subroutine, InstructionHandle pei, BasicBlock etb) {
951         etb.setExceptionThrower(pei);
952 
953         // Remember whether or not a universal exception handler
954         // is reachable. If so, then we know that exceptions raised
955         // at this instruction cannot propagate out of the method.
956         boolean sawUniversalExceptionHandler = false;
957 
958         List<CodeExceptionGen> exceptionHandlerList = exceptionHandlerMap.getHandlerList(pei);
959         if (exceptionHandlerList != null) {
960             for (CodeExceptionGen exceptionHandler : exceptionHandlerList) {
961                 InstructionHandle handlerStart = exceptionHandler.getHandlerPC();
962                 subroutine.addEdgeAndExplore(etb, handlerStart, HANDLED_EXCEPTION_EDGE);
963 
964                 if (Hierarchy.isUniversalExceptionHandler(exceptionHandler.getCatchType())) {
965                     sawUniversalExceptionHandler = true;
966                 }
967             }
968         }
969 
970         // If required, mark this block as throwing an unhandled exception.
971         // For now, we assume that if there is no reachable handler that handles
972         // ANY exception type, then the exception can be thrown out of the
973         // method.
974         if (!sawUniversalExceptionHandler) {
975             if (DEBUG) {
976                 System.out.println("Adding unhandled exception edge from " + pei);
977             }
978             subroutine.setUnhandledExceptionBlock(etb);
979         }
980     }

982     /**
983      * Return whether or not the given instruction can throw exceptions.
984      *
985      * @param handle
986      *            the instruction
987      * @return true if the instruction can throw an exception, false otherwise
988      * @throws CFGBuilderException
989      */
990     private boolean isPEI(InstructionHandle handle) throws CFGBuilderException {
991         Instruction ins = handle.getInstruction();
992 
993         if (!(ins instanceof ExceptionThrower)) {
994             return false;
995         }
996 
997         if (ins instanceof NEW) {
998             return false;
999         }
1000         // if (ins instanceof ATHROW) return false;
1001         if (ins instanceof GETSTATIC) {
1002             return false;
1003         }
1004         if (ins instanceof PUTSTATIC) {
1005             return false;
1006         }
1007         if (ins instanceof ReturnInstruction) {
1008             return false;
1009         }
1010         if (ins instanceof INSTANCEOF) {
1011             return false;
1012         }
1013         if (ins instanceof MONITOREXIT) {
1014             return false;
1015         }
1016         if (ins instanceof LDC) {
1017             return false;
1018         }
1019         if (ins instanceof GETFIELD && !methodGen.isStatic()) {
1020             // Assume that GETFIELD on this object is not PEI
1021             return !isSafeFieldSource(handle.getPrev());
1022         }
1023         if (ins instanceof PUTFIELD && !methodGen.isStatic()) {
1024             // Assume that PUTFIELD on this object is not PEI
1025             int depth = ins.consumeStack(cpg);
1026             for(InstructionHandle prev = handle.getPrev(); prev != null; prev = prev.getPrev()) {
1027                 Instruction prevInst = prev.getInstruction();
1028                 if(prevInst instanceof BranchInstruction) {
1029                     if(prevInst instanceof GotoInstruction) {
1030                         // Currently we support only jumps to the PUTFIELD itself
1031                         // This will cover simple cases like this.a = flag ? foo : bar
1032                         if(((BranchInstruction) prevInst).getTarget() == handle) {
1033                             depth = ins.consumeStack(cpg);
1034                         } else {
1035                             return true;
1036                         }
1037                     } else if (!(prevInst instanceof IfInstruction)) {
1038                         // As IF instructions may fall through then the stack depth remains unchanged
1039                         // Actually we should not go here for normal Java bytecode: switch or jsr should not appear in this context
1040                         return true;
1041                     }
1042                 }
1043                 depth = depth - prevInst.produceStack(cpg) + prevInst.consumeStack(cpg);
1044                 if(depth < 1) {
1045                     throw new CFGBuilderException("Invalid stack at "+prev+" when checking "+handle);
1046                 }
1047                 if(depth == 1) {
1048                     InstructionHandle prevPrev = prev.getPrev();
1049                     if(prevPrev != null && prevPrev.getInstruction() instanceof BranchInstruction) {
1050                         continue;
1051                     }
1052                     return !isSafeFieldSource(prevPrev);
1053                 }
1054             }
1055         }
1056         return true;
1057     }

1059     /**
1060      * @param handle instruction handle which loads the object for further GETFIELD/PUTFIELD operation
1061      * @return true if this object is known to be non-null
1062      */
1063     private boolean isSafeFieldSource(InstructionHandle handle) {
1064         while(handle != null && handle.getInstruction().getOpcode() == Constants.DUP) {
1065             // Some compilers generate DUP for field increment code like
1066             // ALOAD_0 / DUP / GETFIELD x / ICONST_1 / IADD / PUTFIELD x
1067             handle = handle.getPrev();
1068         }
1069         if(handle == null) {
1070             return false;
1071         }
1072         Instruction inst = handle.getInstruction();
1073         if(inst.getOpcode() == Constants.ALOAD_0) {
1074             return true;
1075         }
1076         if(inst instanceof GETFIELD && ((GETFIELD)inst).getFieldName(cpg).startsWith("this$")) {
1077             return true;
1078         }
1079         return false;
1080     }

1082     /**
1083      * Determine whether or not the given instruction is a control flow merge.
1084      *
1085      * @param handle
1086      *            the instruction
1087      * @return true if the instruction is a control merge, false otherwise
1088      */
1089     private static boolean isMerge(InstructionHandle handle) {
1090         if (handle.hasTargeters()) {
1091             // Check all targeters of this handle to see if any
1092             // of them are branches. If so, the instruction is a merge.
1093             InstructionTargeter[] targeterList = handle.getTargeters();
1094             for (InstructionTargeter targeter : targeterList) {
1095                 if (targeter instanceof BranchInstruction) {
1096                     return true;
1097                 }
1098             }
1099         }
1100         return false;
1101     }

1103     /**
1104      * Inline all JSR subroutines into the top-level subroutine. This produces a
1105      * complete CFG for the entire method, in which all JSR subroutines are
1106      * inlined.
1107      *
1108      * @return the CFG for the method
1109      */
1110     private CFG inlineAll() throws CFGBuilderException {
1111         CFG result = new CFG();
1112 
1113         Context rootContext = new Context(null, topLevelSubroutine, result);
1114         rootContext.mapBlock(topLevelSubroutine.getEntry(), result.getEntry());
1115         rootContext.mapBlock(topLevelSubroutine.getExit(), result.getExit());
1116 
1117         BasicBlock resultStartBlock = rootContext.getBlock(topLevelSubroutine.getStartBlock());
1118         result.createEdge(result.getEntry(), resultStartBlock, START_EDGE);
1119 
1120         inline(rootContext);
1121 
1122         return result;
1123     }

1125     /**
1126      * Inline a subroutine into a calling context.
1127      *
1128      * @param context
1129      *            the Context
1130      */
1131     public void inline(Context context) throws CFGBuilderException {
1132 
1133         CFG result = context.getResult();
1134 
1135         // Check to ensure we're not trying to inline something that is
1136         // recursive
1137         context.checkForRecursion();
1138 
1139         Subroutine subroutine = context.getSubroutine();
1140         CFG subCFG = subroutine.getCFG();
1141 
1142         while (context.hasMoreWork()) {
1143             BasicBlock subBlock = context.nextItem();
1144             BasicBlock resultBlock = context.getBlock(subBlock);
1145 
1146             // Mark blocks which are in JSR subroutines
1147             resultBlock.setInJSRSubroutine(context.getCaller() != null);
1148 
1149             // Copy instructions into the result block
1150             BasicBlock.InstructionIterator insIter = subBlock.instructionIterator();
1151             while (insIter.hasNext()) {
1152                 InstructionHandle handle = insIter.next();
1153                 resultBlock.addInstruction(handle);
1154             }
1155 
1156             // Set exception thrower status
1157             if (subBlock.isExceptionThrower()) {
1158                 resultBlock.setExceptionThrower(subBlock.getExceptionThrower());
1159             }
1160 
1161             // Set exception handler status
1162             if (subBlock.isExceptionHandler()) {
1163                 resultBlock.setExceptionGen(null, subBlock.getExceptionGen());
1164             }
1165 
1166             // Add control edges (including inlining JSR subroutines)
1167             Iterator<Edge> edgeIter = subCFG.outgoingEdgeIterator(subBlock);
1168             while (edgeIter.hasNext()) {
1169                 Edge edge = edgeIter.next();
1170                 int edgeType = edge.getType();
1171 
1172                 if (edgeType == JSR_EDGE) {
1173                     // Inline a JSR subroutine...
1174 
1175                     // Create a new Context
1176                     InstructionHandle jsrHandle = subBlock.getLastInstruction();
1177                     JsrInstruction jsr = (JsrInstruction) jsrHandle.getInstruction();
1178                     Subroutine jsrSub = jsrSubroutineMap.get(jsr.getTarget());
1179                     Context jsrContext = new Context(context, jsrSub, context.getResult());
1180 
1181                     // The start block in the JSR subroutine maps to the first
1182                     // inlined block in the result CFG.
1183                     BasicBlock resultJSRStartBlock = jsrContext.getBlock(jsrSub.getStartBlock());
1184                     result.createEdge(resultBlock, resultJSRStartBlock, GOTO_EDGE);
1185 
1186                     // The exit block in the JSR subroutine maps to the result
1187                     // block
1188                     // corresponding to the instruction following the JSR.
1189                     // (I.e., that is where control returns after the execution
1190                     // of
1191                     // the JSR subroutine.)
1192                     BasicBlock subJSRSuccessorBlock = subroutine.getBlock(jsrHandle.getNext());
1193                     BasicBlock resultJSRSuccessorBlock = context.getBlock(subJSRSuccessorBlock);
1194                     jsrContext.mapBlock(jsrSub.getExit(), resultJSRSuccessorBlock);
1195 
1196                     // Inline the JSR subroutine
1197                     inline(jsrContext);
1198                 } else {
1199                     // Ordinary control edge
1200                     BasicBlock resultTarget = context.getBlock(edge.getTarget());
1201                     result.createEdge(resultBlock, resultTarget, edge.getType());
1202                 }
1203             }
1204 
1205             // Add control edges for escape targets
1206             Iterator<EscapeTarget> escapeTargetIter = subroutine.escapeTargetIterator(subBlock);
1207             while (escapeTargetIter.hasNext()) {
1208                 EscapeTarget escapeTarget = escapeTargetIter.next();
1209                 InstructionHandle targetInstruction = escapeTarget.getTarget();
1210 
1211                 // Look for the calling context which has the target instruction
1212                 Context caller = context.getCaller();
1213                 while (caller != null) {
1214                     if (caller.getSubroutine().containsInstruction(targetInstruction)) {
1215                         break;
1216                     }
1217                     caller = caller.getCaller();
1218                 }
1219 
1220                 if (caller == null) {
1221                     throw new CFGBuilderException("Unknown caller for escape target " + targetInstruction + " referenced by "
1222                             + context.getSubroutine().getStartInstruction());
1223                 }
1224 
1225                 // Find result block in caller
1226                 BasicBlock subCallerTargetBlock = caller.getSubroutine().getBlock(targetInstruction);
1227                 BasicBlock resultCallerTargetBlock = caller.getBlock(subCallerTargetBlock);
1228 
1229                 // Add an edge to caller context
1230                 result.createEdge(resultBlock, resultCallerTargetBlock, escapeTarget.getEdgeType());
1231             }
1232 
1233             // If the block returns from the method, add a return edge
1234             if (subroutine.isReturnBlock(subBlock)) {
1235                 result.createEdge(resultBlock, result.getExit(), RETURN_EDGE);
1236             }
1237 
1238             // If the block calls System.exit(), add an exit edge
1239             if (subroutine.isExitBlock(subBlock)) {
1240                 result.createEdge(resultBlock, result.getExit(), EXIT_EDGE);
1241             }
1242 
1243             // If the block throws an unhandled exception, add an unhandled
1244             // exception edge
1245             if (subroutine.isUnhandledExceptionBlock(subBlock)) {
1246                 result.createEdge(resultBlock, result.getExit(), UNHANDLED_EXCEPTION_EDGE);
1247             }
1248 
1249         }
1250 
1251         /*
1252          * while (blocks are left) {
1253          *
1254          * get block from subroutine get corresponding block from result copy
1255          * instructions into result block
1256          *
1257          * if (block terminated by JSR) { get JSR subroutine create new context
1258          * create GOTO edge from current result block to start block of new
1259          * inlined context map subroutine exit block to result JSR successor
1260          * block inline (new context, result) } else { for each outgoing edge {
1261          * map each target to result blocks (add block to to work list if
1262          * needed) add edges to result }
1263          *
1264          * for each outgoing escape target { add edges into blocks in outer
1265          * contexts (adding those blocks to outer work list if needed) }
1266          *
1267          * if (block returns) { add return edge from result block to result CFG
1268          * exit block }
1269          *
1270          * if (block calls System.exit()) { add exit edge from result block to
1271          * result CFG exit block }
1272          *
1273          * if (block throws unhandled exception) { add unhandled exception edge
1274          * from result block to result CFG exit block } }
1275          *
1276          * }
1277          */
1278     }

1283     public static void main(String[] argv) throws Exception {

1305             MethodGen methodGen = new MethodGen(method, jclass.getClassName(), classGen.getConstantPool());

1307             CFGBuilder cfgBuilder = new BetterCFGBuilder2(descriptor, methodGen);
1308             cfgBuilder.build();
1309 
1310             CFG cfg = cfgBuilder.getCFG();
1311 
1312             CFGPrinter cfgPrinter = new CFGPrinter(cfg);
1313             System.out.println("---------------------------------------------------------------------");
1314             System.out.println("Method: " + SignatureConverter.convertMethodSignature(methodGen));
1315             System.out.println("---------------------------------------------------------------------");
1316             cfgPrinter.print(System.out);
