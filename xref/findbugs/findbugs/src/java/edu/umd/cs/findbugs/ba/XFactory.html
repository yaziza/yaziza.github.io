166     public boolean isCalledDirectlyOrIndirectly(XMethod m) {
167         if (isCalled(m)) {
168             return true;
169         }
170         if (m.isStatic() || m.isPrivate() || "<init>".equals(m.getName())) {
171             return false;
172         }
173         try {
174             IAnalysisCache analysisCache = Global.getAnalysisCache();
175             XClass clazz = analysisCache.getClassAnalysis(XClass.class, m.getClassDescriptor());
176             if (isCalledDirectlyOrIndirectly(clazz.getSuperclassDescriptor(), m)) {
177                 return true;
178             }
179             for (ClassDescriptor i : clazz.getInterfaceDescriptorList()) {
180                 if (isCalledDirectlyOrIndirectly(i, m)) {
181                     return true;
182                 }
183             }
184 
185             return false;
186         } catch (edu.umd.cs.findbugs.classfile.MissingClassException e) {
187             // AnalysisContext.reportMissingClass(e.getClassNotFoundException());
188             return false;
189         } catch (MissingClassException e) {
190             AnalysisContext.reportMissingClass(e.getClassNotFoundException());
191             return false;
192         } catch (Exception e) {
193             AnalysisContext.logError("Error checking to see if " + m + " is called (" + e.getClass().getCanonicalName() + ")", e);
194             return false;
195         }
196     }

198     private boolean isCalledDirectlyOrIndirectly(@CheckForNull ClassDescriptor clazzDescriptor, XMethod m)
199             throws CheckedAnalysisException {
200         if (clazzDescriptor == null) {
201             return false;
202         }
203         IAnalysisCache analysisCache = Global.getAnalysisCache();
204         XClass clazz = analysisCache.getClassAnalysis(XClass.class, clazzDescriptor);
205         XMethod m2 = clazz.findMethod(m.getName(), m.getSignature(), m.isStatic());
206         if (m2 != null && isCalled(m2)) {
207             return true;
208         }
209         if (isCalledDirectlyOrIndirectly(clazz.getSuperclassDescriptor(), m)) {
210             return true;
211         }
212         for (ClassDescriptor i : clazz.getInterfaceDescriptorList()) {
213             if (isCalledDirectlyOrIndirectly(i, m)) {
214                 return true;
215             }
216         }
217 
218         return false;
219 
220     }

321     public static XMethod createXMethod(MethodDescriptor desc) {
322         XFactory xFactory = AnalysisContext.currentXFactory();
323 
324         XMethod m = xFactory.methods.get(desc);
325         if (m != null) {
326             return m;
327         }
328         m = xFactory.resolveXMethod(desc);
329         if (m instanceof MethodDescriptor) {
330             xFactory.methods.put((MethodDescriptor) m, m);
331             DescriptorFactory.instance().canonicalize((MethodDescriptor) m);
332         } else {
333             xFactory.methods.put(desc, m);
334         }
335         return m;
336     }

351     private XMethod resolveXMethod(MethodDescriptor originalDescriptor) {
352         MethodDescriptor desc = originalDescriptor;
353         try {
354             while (true) {
355                 XMethod m = methods.get(desc);
356                 if (m != null) {
357                     return m;
358                 }
359                 XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, desc.getClassDescriptor());
360                 if (xClass == null) {
361                     break;
362                 }
363                 ClassDescriptor superClass = xClass.getSuperclassDescriptor();
364                 if (superClass == null) {
365                     break;
366                 }
367                 desc = DescriptorFactory.instance().getMethodDescriptor(superClass.getClassName(), desc.getName(),
368                         desc.getSignature(), desc.isStatic());
369             }
370         } catch (CheckedAnalysisException e) {
371             assert true;
372         } catch (RuntimeException e) {
373             assert true;
374         }
375         UnresolvedXMethod xmethod = new UnresolvedXMethod(originalDescriptor);
376 
377 
378         ObligationPolicyDatabase database = Global.getAnalysisCache().getOptionalDatabase(ObligationPolicyDatabase.class);
379 
380         if (BuildObligationPolicyDatabase.INFER_CLOSE_METHODS && database != null
381                 && !xmethod.getClassName().startsWith("java")) {
382             boolean methodHasCloseInName = false;
383             String methodName = xmethod.getName();
384             SplitCamelCaseIdentifier splitter = new SplitCamelCaseIdentifier(methodName);
385             methodHasCloseInName = splitter.split().contains("close");
386             Obligation[] paramObligationTypes = database.getFactory().getParameterObligationTypes(xmethod);
387 
388             for (int i = 0; i < xmethod.getNumParams(); i++) {
389                 Obligation obligationType = paramObligationTypes[i];
390                 if (obligationType == null) {
391                     continue;
392                 }
393                 if (methodHasCloseInName) {
394                     // Method has "close" in its name.
395                     // Assume that it deletes the obligation.
396                     ObligationPolicyDatabaseEntry entry = database.addParameterDeletesObligationDatabaseEntry(xmethod,
397                             obligationType, ObligationPolicyDatabaseEntryType.STRONG);
398 
399                 } else {
400 
401                     /*
402                      * // Interesting case: we have a parameter which is // an
403                      * Obligation type, but no annotation or other indication //
404                      * what is done by the method with the obligation. // We'll
405                      * create a "weak" database entry deleting the //
406                      * obligation. If strict checking is performed, // weak
407                      * entries are ignored.
408                      */
409                     if ("<init>".equals(methodName) || methodName.startsWith("access$") || xmethod.isStatic()
410                             || methodName.toLowerCase().indexOf("close") >= 0
411                             || xmethod.getSignature().toLowerCase().indexOf("Closeable") >= 0) {
412                         ObligationPolicyDatabaseEntry entry = database.addParameterDeletesObligationDatabaseEntry(xmethod,
413                                 obligationType, ObligationPolicyDatabaseEntryType.WEAK);
414                     }
415                 }
416             }
417 
418         }
419 
420         return xmethod;
421     }

563     private XField resolveXField(FieldDescriptor originalDescriptor) {
564         FieldDescriptor desc = originalDescriptor;
565         LinkedList<ClassDescriptor> worklist = new LinkedList<ClassDescriptor>();
566         ClassDescriptor originalClassDescriptor = desc.getClassDescriptor();
567         worklist.add(originalClassDescriptor);
568         try {
569             while (!worklist.isEmpty()) {
570                 ClassDescriptor d = worklist.removeFirst();
571                 if (!d.equals(originalClassDescriptor)) {
572                     desc = DescriptorFactory.instance().getFieldDescriptor(d.getClassName(), desc.getName(), desc.getSignature(),
573                             desc.isStatic());
574                 }
575 
576                 XField f = fields.get(desc);
577                 if (f != null) {
578                     return f;
579                 }
580                 XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, d);
581                 if (xClass == null) {
582                     break;
583                 }
584                 ClassDescriptor superClass = xClass.getSuperclassDescriptor();
585                 if (superClass != null) {
586                     worklist.add(superClass);
587                 }
588                 if (originalDescriptor.isStatic()) {
589                     for (ClassDescriptor i : xClass.getInterfaceDescriptorList()) {
590                         worklist.add(i);
591                     }
592                 }
593 
594             }
595         } catch (CheckedAnalysisException e) {
596             AnalysisContext.logError("Error resolving " + originalDescriptor, e);
597         }
598         return new UnresolvedXField(originalDescriptor);
599     }

691     public static <E extends ClassMember> int compare(E lhs, E rhs) {
692         int cmp;
693 
694         cmp = lhs.getClassName().compareTo(rhs.getClassName());
695         if (cmp != 0) {
696             return cmp;
697         }
698 
699         cmp = lhs.getName().compareTo(rhs.getName());
700         if (cmp != 0) {
701             return cmp;
702         }
703 
704         cmp = lhs.getSignature().compareTo(rhs.getSignature());
705         if (cmp != 0) {
706             return cmp;
707         }
708 
709         return (lhs.isStatic() ? 1 : 0) - (rhs.isStatic() ? 1 : 0);
710     }
