20  package edu.umd.cs.findbugs.workflow;
21  
22  import java.util.ArrayList;
23  import java.util.Map;
24  
25  import edu.umd.cs.findbugs.BugCollection;
26  import edu.umd.cs.findbugs.BugInstance;
27  import edu.umd.cs.findbugs.DetectorFactoryCollection;
28  import edu.umd.cs.findbugs.ProjectPackagePrefixes;
29  import edu.umd.cs.findbugs.SortedBugCollection;
30  import edu.umd.cs.findbugs.util.Bag;
31  
32  public class TestingGround {
33      BugCollection bugCollection;
34  
35      public TestingGround() {
36      }
37  
38      public TestingGround(BugCollection bugCollection) {
39          this.bugCollection = bugCollection;
40      }
41  
42      public void setBugCollection(BugCollection bugCollection) {
43          this.bugCollection = bugCollection;
44      }
45  
46      public TestingGround execute() {
47          ProjectPackagePrefixes foo = new ProjectPackagePrefixes();
48  
49          for (BugInstance b : bugCollection.getCollection()) {
50              foo.countBug(b);
51          }
52          foo.report();
53  
54          return this;
55      }
56  
57      static class CommandLine extends edu.umd.cs.findbugs.config.CommandLine {
58  
59          @Override
60          public void handleOption(String option, String optionalExtraPart) {
61              throw new IllegalArgumentException("unknown option: " + option);
62          }
63  
64          @Override
65          public void handleOptionWithArgument(String option, String argument) {
66              throw new IllegalArgumentException("unknown option: " + option);
67          }
68      }
69  
70      public static void main(String[] args) throws Exception {
71          DetectorFactoryCollection.instance(); // load plugins
72  
73          CommandLine commandLine = new CommandLine();
74          int argCount = commandLine.parse(args, 0, 2, "Usage: " + TestingGround.class.getName() + " [options] [<xml results>] ");
75  
76          SortedBugCollection bugCollection = new SortedBugCollection();
77          if (argCount < args.length) {
78              bugCollection.readXML(args[argCount++]);
79          } else {
80              bugCollection.readXML(System.in);
81          }
82          ArrayList<Bag<String>> live = new ArrayList<Bag<String>>();
83          ArrayList<Bag<String>> died = new ArrayList<Bag<String>>();
84          Bag<String> allBugs = new Bag<String>();
85          for (int i = 0; i <= bugCollection.getSequenceNumber(); i++) {
86              live.add(new Bag<String>());
87              died.add(new Bag<String>());
88          }
89          for (BugInstance b : bugCollection) {
90              int first = (int) b.getFirstVersion();
91              int buried = (int) b.getLastVersion() + 1;
92              int finish = buried;
93              if (finish == 0) {
94                  finish = (int) bugCollection.getSequenceNumber();
95              }
96  
97              String bugPattern = b.getBugPattern().getType();
98              allBugs.add(bugPattern);
99  
100             for (int i = first; i <= finish; i++) {
101                 live.get(i).add(bugPattern);
102             }
103             if (buried > 0) {
104                 died.get(buried).add(bugPattern);
105             }
106         }
107         for (int i = 0; i < bugCollection.getSequenceNumber(); i++) {
108             for (Map.Entry<String, Integer> e : died.get(i).entrySet()) {
109                 Integer buried = e.getValue();
110                 int total = live.get(i).getCount(e.getKey());
111                 if (buried > 30 && buried * 3 > total) {
112                     System.out.printf("%d/%d died at %d for %s%n", buried, total, i, e.getKey());
113                 }
114             }
115 
116         }
117         SortedBugCollection results = bugCollection.createEmptyCollectionWithMetadata();
118         for (BugInstance b : bugCollection) {
119             int buried = (int) b.getLastVersion() + 1;
120             String bugPattern = b.getBugPattern().getType();
121 
122             if (buried > 0) {
123                 int buriedCount = died.get(buried).getCount(bugPattern);
124                 int total = live.get(buried).getCount(bugPattern);
125                 if (buriedCount > 30 && buriedCount * 3 > total) {
126                     continue;
127                 }
128             }
129             int survied = live.get((int) bugCollection.getSequenceNumber()).getCount(bugPattern);
130             if (survied == 0 && allBugs.getCount(bugPattern) > 100) {
131                 continue;
132             }
133 
134             results.add(b, false);
135         }
136         if (argCount == args.length) {
137             results.writeXML(System.out);
138         } else {
139             results.writeXML(args[argCount++]);
140 
141         }
142 
143     }
144 }
