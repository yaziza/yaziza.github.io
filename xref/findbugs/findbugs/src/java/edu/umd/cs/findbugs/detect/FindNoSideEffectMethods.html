150     private static enum SideEffectStatus {
151         SIDE_EFFECT, UNSURE_OBJECT_ONLY, OBJECT_ONLY, UNSURE, NO_SIDE_EFFECT;
152 
153         boolean unsure() {
154             return this == UNSURE || this == UNSURE_OBJECT_ONLY;
155         }
156 
157         SideEffectStatus toObjectOnly() {
158             switch(this) {
159             case UNSURE:
160                 return UNSURE_OBJECT_ONLY;
161             case NO_SIDE_EFFECT:
162                 return OBJECT_ONLY;
163             default:
164                 return this;
165             }
166         }
167 
168         SideEffectStatus toUnsure() {
169             switch(this) {
170             case OBJECT_ONLY:
171                 return UNSURE_OBJECT_ONLY;
172             case NO_SIDE_EFFECT:
173                 return UNSURE;
174             default:
175                 return this;
176             }
177         }
178 
179         SideEffectStatus toSure() {
180             switch(this) {
181             case UNSURE_OBJECT_ONLY:
182                 return OBJECT_ONLY;
183             case UNSURE:
184                 return NO_SIDE_EFFECT;
185             default:
186                 return this;
187             }
188         }
189     }

191     private static class MethodCall {
192         private final MethodDescriptor method;
193         private final FieldDescriptor target;
194 
195         public MethodCall(MethodDescriptor method, FieldDescriptor target) {
196             this.method = method;
197             this.target = target;
198         }
199 
200         public MethodDescriptor getMethod() {
201             return method;
202         }
203 
204         public FieldDescriptor getTarget() {
205             return target;
206         }
207 
208         @Override
209         public int hashCode() {
210             throw new UnsupportedOperationException();
211         }
212 
213         @Override
214         public boolean equals(Object obj) {
215             if (this == obj) {
216                 return true;
217             }
218             if (obj == null) {
219                 return false;
220             }
221             if (getClass() != obj.getClass()) {
222                 return false;
223             }
224             MethodCall other = (MethodCall) obj;
225             if (!method.equals(other.method)) {
226                 return false;
227             }
228             if (!target.equals(other.target)) {
229                 return false;
230             }
231             return true;
232         }
233     }

239     public static enum MethodSideEffectStatus {
240         NSE, // Non-void method has no side effect
241         NSE_EX, // No side effect method which result value might be ignored for some reason
242         CHECK, // (unimplemented yet) No side effect method which just checks the arguments, throws exceptions and returns one of arguments (or void) like assert or precondition
243         USELESS, // Void method which seems to be useless
244         SE_CLINIT, // Method has no side effect, but it's a constructor or static method of the class having side effect
245         OBJ, // Non-static method which changes only its object
246         SE // Method has side effect or side-effect status for the method is unknown
247     }

249     public static class NoSideEffectMethodsDatabase {
250         private final Map<MethodDescriptor, MethodSideEffectStatus> map = new HashMap<>();
251 
252         void add(MethodDescriptor m, MethodSideEffectStatus s) {
253             map.put(m, s);
254         }
255 
256         public @Nonnull MethodSideEffectStatus status(MethodDescriptor m) {
257             MethodSideEffectStatus s = map.get(m);
258             return s == null ? MethodSideEffectStatus.SE : s;
259         }
260 
261         /**
262          * @param m method to check
263          * @param statuses allowed statuses
264          * @return true if method status is one of the statuses
265          */
266         public boolean is(MethodDescriptor m, MethodSideEffectStatus... statuses) {
267             MethodSideEffectStatus s = status(m);
268             for(MethodSideEffectStatus status : statuses) {
269                 if(s == status) {
270                     return true;
271                 }
272             }
273             return false;
274         }
275 
276         public boolean hasNoSideEffect(MethodDescriptor m) {
277             return status(m) == MethodSideEffectStatus.NSE;
278         }
279 
280         public boolean useless(MethodDescriptor m) {
281             return status(m) == MethodSideEffectStatus.USELESS;
282         }
283 
284         public boolean excluded(MethodDescriptor m) {
285             return is(m, MethodSideEffectStatus.NSE_EX, MethodSideEffectStatus.SE_CLINIT);
286         }
287     }

316     @Override
317     public void visit(JavaClass obj) {
318         super.visit(obj);
319         allowedFields = new HashSet<>();
320         fieldsModifyingMethods = new HashSet<>();
321         subtypes = null;
322         if (!obj.isFinal() && !obj.isEnum()) {
323             try {
324                 Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
325                 subtypes = new HashSet<>(subtypes2.getSubtypes(getClassDescriptor()));
326                 subtypes.remove(getClassDescriptor());
327             } catch (ClassNotFoundException e) {
328             }
329         }
330     }

332     @Override
333     public void visit(Method method) {
334         constructor = method.getName().equals("<init>");
335         classInit = method.getName().equals("<clinit>");
336         calledMethods = new ArrayList<>();
337         status = SideEffectStatus.NO_SIDE_EFFECT;
338         if (hasNoSideEffect(getMethodDescriptor())) {
339             handleStatus();
340             return;
341         }
342         if(isObjectOnlyMethod(getMethodDescriptor())) {
343             status = SideEffectStatus.OBJECT_ONLY;
344         }
345         if (method.isNative() || changedArg(getMethodDescriptor()) != -1) {
346             status = SideEffectStatus.SIDE_EFFECT;
347             handleStatus();
348             return;
349         }
350         boolean sawImplementation = false;
351         if (classInit) {
352             superClinitCall();
353         }
354         if (!method.isStatic() && !method.isPrivate() && !method.isFinal() && !constructor && subtypes != null) {
355             for (ClassDescriptor subtype : subtypes) {
356                 try {
357                     XClass xClass = Global.getAnalysisCache().getClassAnalysis(XClass.class, subtype);
358                     XMethod matchingMethod = xClass.findMatchingMethod(getMethodDescriptor());
359                     if (matchingMethod != null) {
360                         sawImplementation = true;
361                         sawCall(new MethodCall(matchingMethod.getMethodDescriptor(), TARGET_THIS), false);
362                     }
363                 } catch (CheckedAnalysisException e) {
364                 }
365             }
366         }
367         if (method.isAbstract() || method.isInterface()) {
368             if (!sawImplementation
369                     || getClassName().endsWith("Visitor") || getClassName().endsWith("Listener")
370                     || getClassName().startsWith("java/sql/")
371                     || (getClassName().equals("java/util/concurrent/Future") && !method.getName().startsWith("is"))
372                     || (getClassName().equals("java/lang/Process") && method.getName().equals("exitValue"))) {
373                 status = SideEffectStatus.SIDE_EFFECT;
374             } else if(isObjectOnlyMethod(getMethodDescriptor())) {
375                 status = SideEffectStatus.OBJECT_ONLY;
376             } else {
377                 String[] thrownExceptions = getXMethod().getThrownExceptions();
378                 if(thrownExceptions != null && thrownExceptions.length > 0) {
379                     status = SideEffectStatus.SIDE_EFFECT;
380                 }
381             }
382         }
383         if ((status == SideEffectStatus.SIDE_EFFECT || status == SideEffectStatus.OBJECT_ONLY) || method.isAbstract()
384                 || method.isInterface() || method.isNative()) {
385             handleStatus();
386         }
387     }

397     @Override
398     public void visitAfter(JavaClass obj) {
399         for(MethodDescriptor method : fieldsModifyingMethods) {
400             List<MethodCall> calls = callGraph.get(method);
401             SideEffectStatus prevStatus = statusMap.get(method);
402             status = prevStatus.toSure();
403             calledMethods = new ArrayList<>();
404             for(MethodCall methodCall : calls) {
405                 FieldDescriptor target = methodCall.getTarget();
406                 if(target != TARGET_NEW && target != TARGET_OTHER && target != TARGET_THIS) {
407                     if(allowedFields.contains(target)) {
408                         methodCall = new MethodCall(methodCall.getMethod(), TARGET_THIS);
409                     } else {
410                         methodCall = new MethodCall(methodCall.getMethod(), TARGET_OTHER);
411                     }
412                 }
413                 sawCall(methodCall, false);
414                 if(status == SideEffectStatus.SIDE_EFFECT) {
415                     break;
416                 }
417             }
418             if (status != prevStatus) {
419                 statusMap.put(method, status);
420             }
421             if(status.unsure()) {
422                 calledMethods.trimToSize();
423                 callGraph.put(method, calledMethods);
424             } else {
425                 callGraph.remove(method);
426             }
427         }
428         MethodDescriptor clinit = new MethodDescriptor(getClassName(), "<clinit>", "()V", true);
429         if(!statusMap.containsKey(clinit)) {
430             status = SideEffectStatus.NO_SIDE_EFFECT;
431             calledMethods = new ArrayList<>();
432             superClinitCall();
433             statusMap.put(clinit, status);
434             if(status == SideEffectStatus.UNSURE || status == SideEffectStatus.UNSURE_OBJECT_ONLY) {
435                 calledMethods.trimToSize();
436                 callGraph.put(clinit, calledMethods);
437             }
438         }
439     }

441     private void superClinitCall() {
442         ClassDescriptor superclassDescriptor = getXClass().getSuperclassDescriptor();
443         if(superclassDescriptor != null && !superclassDescriptor.getClassName().equals("java/lang/Object")) {
444             sawCall(new MethodCall(new MethodDescriptor(superclassDescriptor.getClassName(), "<clinit>", "()V", true), TARGET_THIS), false);
445         }
446     }

448     private void handleStatus() {
449         statusMap.put(getMethodDescriptor(), status);
450         if(status == SideEffectStatus.UNSURE || status == SideEffectStatus.UNSURE_OBJECT_ONLY) {
451             calledMethods.trimToSize();
452             callGraph.put(getMethodDescriptor(), calledMethods);
453         } else {
454             fieldsModifyingMethods.remove(getMethodDescriptor());
455         }
456     }

458     @Override
459     public void visit(Code obj) {
460         uselessVoidCandidate = !classInit && !constructor && !getXMethod().isSynthetic() && Type.getReturnType(getMethodSig()) == Type.VOID;
461         byte[] code = obj.getCode();
462         if(code.length == 4 && (code[0] & 0xFF) == GETSTATIC && (code[3] & 0xFF) == ARETURN) {
463             getStaticMethods.add(getMethodDescriptor());
464             handleStatus();
465             return;
466         }
467 
468         if (code.length <= 2 && !getXMethod().isStatic() && (getXMethod().isPublic() || getXMethod().isProtected())
469                 && !getXMethod().isFinal() && (getXClass().isPublic() || getXClass().isProtected())) {
470             for(byte[] stubMethod : STUB_METHODS) {
471                 if (Arrays.equals(stubMethod, code)
472                         && (getClassName().endsWith("Visitor") || getClassName().endsWith("Listener") || !hasOtherImplementations(getXMethod()))) {
473                     // stub method which can be extended: assume it can be extended with possible side-effect
474                     status = SideEffectStatus.SIDE_EFFECT;
475                     handleStatus();
476                     return;
477                 }
478             }
479         }
480         if (statusMap.containsKey(getMethodDescriptor())) {
481             return;
482         }
483         finallyTargets = new HashSet<>();
484         for(CodeException ex : getCode().getExceptionTable()) {
485             if(ex.getCatchType() == 0) {
486                 finallyTargets.add(ex.getHandlerPC());
487             }
488         }
489         finallyExceptionRegisters = new HashSet<>();
490         try {
491             super.visit(obj);
492         } catch (EarlyExitException e) {
493             // Ignore
494         }
495         if (uselessVoidCandidate && code.length > 1
496                 && (status == SideEffectStatus.UNSURE || status == SideEffectStatus.NO_SIDE_EFFECT)) {
497             uselessVoidCandidates.add(getMethodDescriptor());
498         }
499         handleStatus();
500     }

502     @Override
503     public void sawOpcode(int seen) {
504         if (!allowedFields.isEmpty() && seen == PUTFIELD) {
505             Item objItem = getStack().getStackItem(1);
506             if (objItem.getRegisterNumber() == 0) {
507                 if (allowedFields.contains(getFieldDescriptorOperand())) {
508                     Item valueItem = getStack().getStackItem(0);
509                     if (!isNew(valueItem) && !valueItem.isNull()) {
510                         allowedFields.remove(getFieldDescriptorOperand());
511                     }
512                 }
513             }
514         }
515         if (status == SideEffectStatus.SIDE_EFFECT && allowedFields.isEmpty()) {
516             // Nothing to do: skip the rest of the method
517             throw new EarlyExitException();
518         }
519         if (status == SideEffectStatus.SIDE_EFFECT) {
520             return;
521         }
522         switch (seen) {
523         case ASTORE:
524         case ASTORE_0:
525         case ASTORE_1:
526         case ASTORE_2:
527         case ASTORE_3:
528             if(finallyTargets.contains(getPC())) {
529                 finallyExceptionRegisters.add(getRegisterOperand());
530             }
531             break;
532         case ATHROW: {
533             Item exceptionItem = getStack().getStackItem(0);
534             if(!finallyExceptionRegisters.remove(exceptionItem.getRegisterNumber())) {
535                 uselessVoidCandidate = false;
536                 try {
537                     JavaClass javaClass = exceptionItem.getJavaClass();
538                     if (javaClass != null && ALLOWED_EXCEPTIONS.contains(javaClass.getClassName())) {
539                         break;
540                     }
541                 } catch (ClassNotFoundException e) {
542                 }
543                 status = SideEffectStatus.SIDE_EFFECT;
544             }
545             break;
546         }
547         case PUTSTATIC:
548             if(classInit) {
549                 if(getClassConstantOperand().equals(getClassName())) {
550                     break;
551                 }
552             }
553             status = SideEffectStatus.SIDE_EFFECT;
554             break;
555         case INVOKEDYNAMIC:
556             status = SideEffectStatus.SIDE_EFFECT;
557             break;
558         case PUTFIELD:
559             sawCall(getMethodCall(FIELD_STORE_STUB_METHOD), false);
560             break;
561         case AASTORE:
562         case DASTORE:
563         case CASTORE:
564         case BASTORE:
565         case IASTORE:
566         case LASTORE:
567         case FASTORE:
568         case SASTORE:
569             sawCall(getMethodCall(ARRAY_STORE_STUB_METHOD), false);
570             break;
571         case INVOKESTATIC:
572             if (changesOnlyNewObjects(getMethodDescriptorOperand())) {
573                 break;
574             }
575             sawCall(new MethodCall(getMethodDescriptorOperand(), TARGET_OTHER), false);
576             break;
577         case INVOKESPECIAL:
578         case INVOKEINTERFACE:
579         case INVOKEVIRTUAL: {
580             XMethod xMethodOperand = getXMethodOperand();
581             MethodDescriptor methodDescriptorOperand = xMethodOperand == null ? getMethodDescriptorOperand() : xMethodOperand
582                     .getMethodDescriptor();
583             if (changesOnlyNewObjects(getMethodDescriptorOperand())) {
584                 break;
585             }
586             MethodCall methodCall = getMethodCall(methodDescriptorOperand);
587             sawCall(methodCall, false);
588             break;
589         }
590         default:
591             break;
592         }
593     }

595     private MethodCall getMethodCall(MethodDescriptor methodDescriptorOperand) {
596         Item objItem = getStack().getStackItem(getNumberArguments(methodDescriptorOperand.getSignature()));
597         if (isNew(objItem)) {
598             return new MethodCall(methodDescriptorOperand, TARGET_NEW);
599         }
600         if (objItem.getRegisterNumber() == 0 && !getMethod().isStatic()) {
601             return new MethodCall(methodDescriptorOperand, constructor ? TARGET_NEW : TARGET_THIS);
602         }
603         XField xField = objItem.getXField();
604         if (xField != null) {
605             if (classInit && xField.isStatic() && xField.getClassDescriptor().getClassName().equals(getClassName())) {
606                 return new MethodCall(methodDescriptorOperand, TARGET_NEW);
607             }
608             if (!getMethodDescriptor().isStatic() && objItem.getFieldLoadedFromRegister() == 0
609                     && allowedFields.contains(xField.getFieldDescriptor())) {
610                 fieldsModifyingMethods.add(getMethodDescriptor());
611                 return new MethodCall(methodDescriptorOperand, xField.getFieldDescriptor());
612             }
613         }
614         return new MethodCall(methodDescriptorOperand, TARGET_OTHER);
615     }

617     private void sawCall(MethodCall methodCall, boolean finalPass) {
618         if (status == SideEffectStatus.SIDE_EFFECT) {
619             return;
620         }
621         MethodDescriptor methodDescriptor = methodCall.getMethod();
622         if (hasNoSideEffect(methodDescriptor)) {
623             sawNoSideEffectCall(methodDescriptor);
624             return;
625         }
626         FieldDescriptor target = methodCall.getTarget();
627         SideEffectStatus calledStatus = isObjectOnlyMethod(methodDescriptor) ? SideEffectStatus.OBJECT_ONLY : statusMap
628                 .get(methodDescriptor);
629         if (calledStatus == null) {
630             calledStatus = finalPass ? hasNoSideEffectUnknown(methodDescriptor) ? SideEffectStatus.NO_SIDE_EFFECT : SideEffectStatus.SIDE_EFFECT
631                     : SideEffectStatus.UNSURE;
632         }
633         switch(calledStatus) {
634         case NO_SIDE_EFFECT:
635             sawNoSideEffectCall(methodDescriptor);
636             return;
637         case SIDE_EFFECT:
638             status = SideEffectStatus.SIDE_EFFECT;
639             return;
640         case OBJECT_ONLY:
641             if(target == TARGET_THIS) {
642                 status = status.toObjectOnly();
643             } else if(target == TARGET_OTHER) {
644                 status = SideEffectStatus.SIDE_EFFECT;
645             } else if(target != TARGET_NEW) {
646                 status = status.toObjectOnly();
647                 sawUnsureCall(methodCall);
648             }
649             return;
650         case UNSURE_OBJECT_ONLY:
651             if(target == TARGET_NEW) {
652                 sawUnsureCall(methodCall);
653             } else if(target == TARGET_OTHER) {
654                 status = SideEffectStatus.SIDE_EFFECT;
655             } else {
656                 status = status.toObjectOnly();
657                 sawUnsureCall(methodCall);
658             }
659             return;
660         case UNSURE:
661             sawUnsureCall(methodCall);
662             return;
663         }
664     }

666     /**
667      * @param methodDescriptor
668      */
669     private void sawNoSideEffectCall(MethodDescriptor methodDescriptor) {
670         if(uselessVoidCandidate && Type.getReturnType(methodDescriptor.getSignature()) == Type.VOID
671                 && !methodDescriptor.getName().equals("<init>")) {
672             /* To reduce false-positives we do not mark method as useless void if it calls
673              * another useless void method. If that another method also in the scope of our project
674              * then we will report it instead. If there's a cycle of no-side-effect calls, then
675              * it's probably some delegation pattern and methods can be extended in future/derived
676              * projects to do something useful.
677              */
678             uselessVoidCandidate = false;
679         }
680     }

682     private void sawUnsureCall(MethodCall methodCall) {
683         calledMethods.add(methodCall);
684         status = status.toUnsure();
685     }

687     /**
688      * @param item stack item to check
689      * @return true if this stack item is known to be newly created
690      */
691     private static boolean isNew(OpcodeStack.Item item) {
692         if(item.isNewlyAllocated()) {
693             return true;
694         }
695         XMethod returnValueOf = item.getReturnValueOf();
696         if(returnValueOf == null) {
697             return false;
698         }
699         if("iterator".equals(returnValueOf.getName())
700                 && "()Ljava/util/Iterator;".equals(returnValueOf.getSignature())
701                 && Subtypes2.instanceOf(returnValueOf.getClassName(), "java.lang.Iterable")) {
702             return true;
703         }
704         if(returnValueOf.getClassName().startsWith("[") && returnValueOf.getName().equals("clone")) {
705             return true;
706         }
707         if(NEW_OBJECT_RETURNING_METHODS.contains(returnValueOf.getMethodDescriptor())) {
708             return true;
709         }
710         return false;
711     }

713     private boolean changesOnlyNewObjects(MethodDescriptor methodDescriptor) {
714         int arg = changedArg(methodDescriptor);
715         if(arg == -1) {
716             return false;
717         }
718         int nArgs = getNumberArguments(methodDescriptor.getSignature());
719         if(!isNew(getStack().getStackItem(nArgs-arg-1))) {
720             return false;
721         }
722         return true;
723     }

725     /**
726      * @param m method to check
727      * @return array of argument numbers (0-based) which this method writes into or null if we don't know anything about this method
728      */
729     private static int changedArg(MethodDescriptor m) {
730         if(m.equals(ARRAY_COPY)) {
731             return 2;
732         }
733         if(m.getName().equals("toArray") && m.getSignature().equals("([Ljava/lang/Object;)[Ljava/lang/Object;")
734                 && Subtypes2.instanceOf(m.getClassDescriptor(), "java.util.Collection")) {
735             return 0;
736         }
737         if ((m.getName().equals("sort") || m.getName().equals("fill") || m.getName().equals("reverse") || m.getName().equals(
738                 "shuffle"))
739                 && (m.getSlashedClassName().equals("java/util/Arrays") || m.getSlashedClassName().equals("java/util/Collections"))) {
740             return 0;
741         }
742         return -1;
743     }

745     /**
746      * @param m method to check
747      * @return true if given method is known to have no side effects
748      */
749     private static boolean hasNoSideEffect(MethodDescriptor m) {
750         String className = m.getSlashedClassName();
751         if("java/lang/String".equals(className)) {
752             return !(m.getName().equals("getChars") || (m.getName().equals("getBytes") && m.getSignature().equals("(II[BI)V")));
753         }
754         if("java/lang/Math".equals(className)) {
755             return !m.getName().equals("random");
756         }
757         if("java/lang/Throwable".equals(className)) {
758             return m.getName().startsWith("get");
759         }
760         if("java/lang/Character".equals(className)) {
761             return !m.getName().equals("toChars");
762         }
763         if("java/lang/Class".equals(className) && m.getName().startsWith("is")) {
764             return true;
765         }
766         if("java/awt/Color".equals(className) && m.getName().equals("<init>")) {
767             return true;
768         }
769         if("java/util/regex/Pattern".contains(className)) {
770             // Pattern.compile is often used to check the PatternSyntaxException, thus we consider it as side-effect method
771             return !m.getName().equals("compile") && !m.getName().equals("<init>");
772         }
773         if(className.startsWith("[") && m.getName().equals("clone")) {
774             return true;
775         }
776         if(className.startsWith("org/w3c/dom/") && (m.getName().startsWith("get") || m.getName().startsWith("has") || m.getName().equals("item"))) {
777             return true;
778         }
779         if(className.startsWith("java/util/") &&
780                 (className.endsWith("Set") || className.endsWith("Map") || className.endsWith("Collection")
781                         || className.endsWith("List") || className.endsWith("Queue") || className.endsWith("Deque")
782                         || className.endsWith("Vector")) || className.endsWith("Hashtable") || className.endsWith("Dictionary")) {
783             // LinkedHashSet in accessOrder mode changes internal state during get/getOrDefault
784             if(className.equals("java/util/LinkedHashMap") && m.getName().startsWith("get")) {
785                 return false;
786             }
787             if(NO_SIDE_EFFECT_COLLECTION_METHODS.contains(m.getName()) || (m.getName().equals("toArray") && m.getSignature().equals("()[Ljava/lang/Object;"))) {
788                 return true;
789             }
790         }
791         if(m.getName().equals("binarySearch") && (m.getSlashedClassName().equals("java/util/Arrays") || m.getSlashedClassName().equals("java/util/Collections"))) {
792             return true;
793         }
794         if(m.getName().startsWith("$SWITCH_TABLE$")) {
795             return true;
796         }
797         if(m.getName().equals("<init>") && isObjectOnlyClass(className)) {
798             return true;
799         }
800         if(m.getName().equals("toString") && m.getSignature().equals("()Ljava/lang/String;") && m.getSlashedClassName().startsWith("java/")) {
801             return true;
802         }
803         if(NUMBER_CLASSES.contains(className)) {
804             return !m.getSignature().startsWith("(Ljava/lang/String;");
805         }
806         if(!m.isStatic() && m.getName().equals("equals") &&
807                 m.getSignature().equals("(Ljava/lang/Object;)Z")) {
808             return true;
809         }
810         if(NO_SIDE_EFFECT_METHODS.contains(m)) {
811             return true;
812         }
813         return false;
814     }

816     /**
817      * @param m method to check
818      * @return true if we may assume that given unseen method has no side effect
819      */
820     private static boolean hasNoSideEffectUnknown(MethodDescriptor m) {
821         if(m.isStatic() && m.getName().equals("<clinit>")) {
822             // No side effect for class initializer of unseen class
823             return true;
824         }
825         if(!m.isStatic() && m.getName().equals("toString") && m.getSignature().equals("()Ljava/lang/String;")) {
826             // We assume no side effect for unseen toString methods
827             return true;
828         }
829         if(!m.isStatic() && m.getName().equals("hashCode") && m.getSignature().equals("()I")) {
830             // We assume no side effect for unseen hashCode methods
831             return true;
832         }
833         if(m.isStatic() && m.getName().equals("values") && m.getSignature().startsWith("()")) {
834             // We assume no side effect for unseen enums
835             return Subtypes2.instanceOf(m.getClassDescriptor(), "java.lang.Enum");
836         }
837         return false;
838     }

840     /**
841      * @param m method to check
842      * @return true if given method is known to change its object only
843      */
844     private static boolean isObjectOnlyMethod(MethodDescriptor m) {
845         if (m.isStatic() || m.getName().equals("<init>")) {
846             return false;
847         }
848         String className = m.getSlashedClassName();
849         if(isObjectOnlyClass(className)) {
850             return true;
851         }
852         if(className.startsWith("javax/xml/") && m.getName().startsWith("next")) {
853             return true;
854         }
855         if ((className.startsWith("java/net/") || className.startsWith("javax/servlet"))
856                 && (m.getName().startsWith("remove") || m.getName().startsWith("add") || m.getName().startsWith("set"))) {
857             return true;
858         }
859         if(OBJECT_ONLY_METHODS.contains(m)) {
860             return true;
861         }
862         return false;
863     }

865     /**
866      * @param className class to check
867      * @return true if all methods of this class are known to be object-only or no-side-effect
868      */
869     private static boolean isObjectOnlyClass(String className) {
870         if(OBJECT_ONLY_CLASSES.contains(className)) {
871             return true;
872         }
873         if(className.startsWith("java/lang/") && (className.endsWith("Error") || className.endsWith("Exception"))) {
874             return true;
875         }
876         return className.startsWith("java/util/") &&
877                 (className.endsWith("Set") || className.endsWith("Map") || className.endsWith("Collection")
878                         || className.endsWith("List") || className.endsWith("Queue") || className.endsWith("Deque")
879                         || className.endsWith("Vector"));
880     }

882     @Override
883     public void report() {
884         computeFinalStatus();
885         Set<String> sideEffectClinit = new HashSet<>();
886         for(Entry<MethodDescriptor, SideEffectStatus> entry : statusMap.entrySet()) {
887             if (entry.getValue() == SideEffectStatus.SIDE_EFFECT && entry.getKey().isStatic() && entry.getKey().getName().equals("<clinit>")) {
888                 sideEffectClinit.add(entry.getKey().getSlashedClassName());
889             }
890         }
891         for(Entry<MethodDescriptor, SideEffectStatus> entry : statusMap.entrySet()) {
892             MethodDescriptor m = entry.getKey();
893             if (entry.getValue() == SideEffectStatus.NO_SIDE_EFFECT) {
894                 String returnType = new SignatureParser(m.getSignature()).getReturnTypeSignature();
895                 if (!returnType.equals("V") || m.getName().equals("<init>")) {
896                     if(m.equals(GET_CLASS)) {
897                         /* We do not mark getClass() call as pure, because it can appear in code like this:
898                             public class Outer {
899                               public class Inner {}
900                               public void test(Outer n) { n.new Inner(); }
901                             }
902                             The test method is compiled into (assumably it's done to generate NPE if n is null)
903                                0: new           #16                 // class a/Outer$Inner
904                                3: aload_1
905                                4: dup
906                                5: invokevirtual #18                 // Method java/lang/Object.getClass:()Ljava/lang/Class;
907                                8: pop
908                                9: invokespecial #22                 // Method a/Outer$Inner."<init>":(La/Outer;)V
909                               12: return
910                             So we would have a false-positive here
911                          */
912                         continue;
913                     }
914                     if (m.getName().startsWith("access$") && (!(m instanceof XMethod) || ((XMethod)m).getAccessMethodForMethod() == null)) {
915                         /* We skip field access methods, because they can unnecessarily be used for static calls
916                          * (probably by older javac)
917                          */
918                         continue;
919                     }
920                     if (m.getName().startsWith("jjStopStringLiteral")) {
921                         /* Some old JJTree versions may generate redundant calls to this method
922                          * Skip it as reports in generated code don't help much
923                          */
924                         continue;
925                     }
926                     if (m.isStatic() || m.getName().equals("<init>")) {
927                         if(sideEffectClinit.contains(m.getSlashedClassName())) {
928                             /* Skip static methods and constructors for classes which have
929                              * side-effect class initializer
930                              */
931                             noSideEffectMethods.add(m, MethodSideEffectStatus.SE_CLINIT);
932                             continue;
933                         }
934                     }
935                     if(m.equals(CLASS_GET_NAME) // used sometimes to trigger class loading
936                             || m.equals(HASH_CODE) // found intended hashCode call several times in different projects, need further research
937                             ) {
938                         noSideEffectMethods.add(m, MethodSideEffectStatus.NSE_EX);
939                         continue;
940                     }
941                     if (m.isStatic() && getStaticMethods.contains(m) && !m.getSlashedClassName().startsWith("java/")) {
942                         String returnSlashedClassName = ClassName.fromFieldSignature(returnType);
943                         if(returnSlashedClassName != null) {
944                             String returnClass = ClassName.toDottedClassName(returnSlashedClassName);
945                             if(ClassName.extractPackageName(returnClass).equals(m.getClassDescriptor().getPackageName())) {
946                                 /* Skip methods which only retrieve static field from the same package
947                                  * As they as often used to trigger class initialization
948                                  */
949                                 noSideEffectMethods.add(m, MethodSideEffectStatus.NSE_EX);
950                                 continue;
951                             }
952                         }
953                     }
954                     noSideEffectMethods.add(m, MethodSideEffectStatus.NSE);
955                 } else {    // void methods
956                     if(uselessVoidCandidates.contains(m)) {
957                         if(m.getName().equals("maybeForceBuilderInitialization") && m.getSignature().equals("()V")) {
958                             // Autogenerated by Google protocol buffer compiler
959                             continue;
960                         }
961                         noSideEffectMethods.add(m, MethodSideEffectStatus.USELESS);
962                     }
963                 }
964             } else if(entry.getValue() == SideEffectStatus.OBJECT_ONLY) {
965                 noSideEffectMethods.add(m, MethodSideEffectStatus.OBJ);
966             }
967         }
968     }

970     /**
971      * @param xMethod
972      * @return true if this has other implementations
973      */
974     private static boolean hasOtherImplementations(XMethod xMethod) {
975         Set<XMethod> superMethods = Hierarchy2.findSuperMethods(xMethod);
976         superMethods.add(xMethod);
977         Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
978         Set<ClassDescriptor> subtypes = new HashSet<>();
979         for(XMethod superMethod : superMethods) {
980             try {
981                 subtypes.addAll(subtypes2.getSubtypes(superMethod.getClassDescriptor()));
982             } catch (ClassNotFoundException e) {
983                 // ignore
984             }
985         }
986         subtypes.remove(xMethod.getClassDescriptor());
987         for (ClassDescriptor subtype : subtypes) {
988             try {
989                 XClass xClass = subtype.getXClass();
990                 XMethod subMethod = xClass.findMatchingMethod(xMethod.getMethodDescriptor());
991                 if (subMethod != null) {
992                     if(!subMethod.isAbstract() ) {
993                         return true;
994                     }
995                 }
996             } catch (CheckedAnalysisException e) {
997                 // ignore
998             }
999         }
1000         return false;
1001     }

1003     private void computeFinalStatus() {
1004         boolean changed = true;
1005         while (changed) {
1006             changed = false;
1007             Iterator<Entry<MethodDescriptor, List<MethodCall>>> iterator = callGraph.entrySet().iterator();
1008             while (iterator.hasNext()) {
1009                 Entry<MethodDescriptor, List<MethodCall>> entry = iterator.next();
1010                 MethodDescriptor method = entry.getKey();
1011                 uselessVoidCandidate = uselessVoidCandidates.contains(method);
1012                 SideEffectStatus prevStatus = statusMap.get(method);
1013                 status = prevStatus.toSure();
1014                 calledMethods = new ArrayList<>();
1015                 for(MethodCall methodCall : entry.getValue()) {
1016                     sawCall(methodCall, true);
1017                     if(status == SideEffectStatus.SIDE_EFFECT) {
1018                         break;
1019                     }
1020                 }
1021                 if (!uselessVoidCandidate || (status != SideEffectStatus.UNSURE && status != SideEffectStatus.NO_SIDE_EFFECT)) {
1022                     uselessVoidCandidates.remove(method);
1023                 }
1024                 if (status != prevStatus || !entry.getValue().equals(calledMethods)) {
1025                     statusMap.put(method, status);
1026                     if (status.unsure()) {
1027                         entry.setValue(calledMethods);
1028                     } else {
1029                         iterator.remove();
1030                     }
1031                     changed = true;
1032                 }
1033             }
1034         }
1035         for(Entry<MethodDescriptor, List<MethodCall>> entry : callGraph.entrySet()) {
1036             MethodDescriptor method = entry.getKey();
1037             status = statusMap.get(method);
1038             if(status == SideEffectStatus.UNSURE) {
1039                 boolean safeCycle = true;
1040                 for(MethodCall methodCall : entry.getValue()) {
1041                     SideEffectStatus calledStatus = statusMap.get(methodCall.getMethod());
1042                     if(calledStatus != SideEffectStatus.UNSURE && calledStatus != SideEffectStatus.NO_SIDE_EFFECT) {
1043                         safeCycle = false;
1044                         break;
1045                     }
1046                 }
1047                 if(safeCycle) {
1048                     statusMap.put(method, SideEffectStatus.NO_SIDE_EFFECT);
1049                     uselessVoidCandidate = uselessVoidCandidates.contains(method);
1050                     if(uselessVoidCandidate) {
1051                         for(MethodCall call : entry.getValue()) {
1052                             uselessVoidCandidate = false;
1053                             if((call.getMethod().equals(method) && call.getTarget() == TARGET_THIS) || method.isStatic()) {
1054                                 uselessVoidCandidate = true;
1055                             } else {
1056                                 if(call.getMethod() instanceof XMethod) {
1057                                     XMethod xMethod = (XMethod) call.getMethod();
1058                                     if(xMethod.isFinal() || (!xMethod.isPublic() && !xMethod.isProtected())) {
1059                                         uselessVoidCandidate = true;
1060                                     }
1061                                 }
1062                             }
1063                             if(!uselessVoidCandidate) {
1064                                 break;
1065                             }
1066                         }
1067                         if(!uselessVoidCandidate) {
1068                             uselessVoidCandidates.remove(method);
1069                         }
1070                     }
1071                 }
1072             }
1073         }
1074     }
