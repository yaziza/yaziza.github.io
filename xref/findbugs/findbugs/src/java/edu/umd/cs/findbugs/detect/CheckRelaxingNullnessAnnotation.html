64  /**
65   * Checks that overriding methods do not relax {@link Nonnull} (made
66   * {@link CheckForNull}) on return values or {@link CheckForNull} (made
67   * {@link Nonnull}) on parameters.
68   *
69   * The code accepts also old (deprecated) nullness annotations from
70   * {@link edu.umd.cs.findbugs.annotations} package.
71   *
72   * @author alienisty (Alessandro Nistico)
73   * @author Andrey Loskutov
74   */
75  public class CheckRelaxingNullnessAnnotation extends ClassNodeDetector {
76  
77      XClass xclass;
78  
79      public CheckRelaxingNullnessAnnotation(BugReporter bugReporter) {
80          super(bugReporter);
81      }
82  
83      @Override
84      public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
85          xclass = getClassInfo(classDescriptor);
86          if(xclass != null){
87              super.visitClass(classDescriptor);
88          }
89      }
90  
91      @CheckForNull
92      XClass getClassInfo(ClassDescriptor classDescr){
93          if(classDescr == null){
94              return null;
95          }
96          try {
97              return Global.getAnalysisCache().getClassAnalysis(XClass.class, classDescr);
98          } catch (CheckedAnalysisException e) {
99              bugReporter.reportMissingClass(classDescr);
100             return null;
101         }
102     }
103 
104     @Override
105     public MethodVisitor visitMethod(int methodAccess, String methodName, String desc, String methodSignature, String[] exceptions) {
106         if ((methodAccess & ACC_STATIC) != 0) {
107             // skip static methods
108             return null;
109         }
110         final XMethod xmethod = xclass.findMethod(methodName, desc, false);
111         if (xmethod == null) {
112             // unable to continue the analysis
113             bugReporter.reportSkippedAnalysis(new MethodDescriptor(xclass.getClassDescriptor().getClassName(), methodName, desc,
114                     false));
115             return null;
116         }
117         return new DetectorNode(methodAccess, methodName, desc, methodSignature, exceptions, xmethod);
118     }

120     private final class DetectorNode extends MethodNode {
121 
122         private final XMethod xmethod;
123 
124         private Map<Integer, NullnessAnnotation> nonNullParameter;
125 
126         private boolean relaxedNullReturn;
127 
128         DetectorNode(int access, String name, String desc, String signature, String[] exceptions, XMethod xmethod) {
129             super(FindBugsASM.ASM_VERSION, access, name, desc, signature, exceptions);
130             this.xmethod = xmethod;
131         }
132 
133         @Override
134         public void visitEnd() {
135             super.visitEnd();
136             // 1 test if we have suspicious annotations on method or parameters
137             relaxedNullReturn = containsRelaxedNonNull(visibleAnnotations);
138             if(!relaxedNullReturn){
139                 relaxedNullReturn = containsRelaxedNonNull(invisibleAnnotations);
140             }
141             boolean needsCheck = relaxedNullReturn;
142             if (invisibleParameterAnnotations != null || visibleParameterAnnotations != null) {
143                 nonNullParameter = getNonnullOrNullableParams(visibleParameterAnnotations);
144                 Map<Integer, NullnessAnnotation> nnp = getNonnullOrNullableParams(invisibleParameterAnnotations);
145                 if (nnp != null) {
146                     if (nonNullParameter == null) {
147                         nonNullParameter = nnp;
148                     } else {
149                         nonNullParameter.putAll(nnp);
150                     }
151                 }
152                 needsCheck |= !nonNullParameter.isEmpty();
153             }
154 
155             if (!needsCheck) {
156                 // we can stop, there is no direct violations due annotations applied on the method.
157                 // However it would be nice to flag Bug2672946B violation too (where the entire class
158                 // relaxes the parent contract by applying default annotations)
159                 return;
160                 // If we continue here, we will flag Bug2672946B violation too on method level,
161                 // but it would be nice to do it on class or even package level (if package-info.java
162                 // has default annotations applied which conflicts with parent class contract)
163             }
164 
165             // 2 look in the hierarchy if we have relaxed contract
166             HierarchyIterator hierarchy = new HierarchyIterator(xclass);
167             XClass superClass;
168             boolean done = false;
169             while (!done && (superClass = hierarchy.next()) != null) {
170                 XMethod method = superClass.findMethod(name, desc, false);
171                 if (method != null) {
172                     done = checkMethod(method);
173                 } else {
174                     for (XMethod superMethod : superClass.getXMethods()) {
175                         if (name.equals(superMethod.getName()) && compatibleParameters(desc, superMethod.getSignature())) {
176                             if (checkMethod(superMethod)) {
177                                 done = true;
178                                 break;
179                             }
180                         }
181                     }
182                 }
183             }
184         }
185 
186         private final boolean checkMethod(@Nonnull XMethod method) {
187             boolean foundAny = false;
188             if (relaxedNullReturn && containsNullness(method.getAnnotations(), NONNULL)) {
189                 BugInstance bug = new BugInstance(CheckRelaxingNullnessAnnotation.this, "NP_METHOD_RETURN_RELAXING_ANNOTATION",
190                         HIGH_PRIORITY);
191                 bug.addClassAndMethod(xmethod);
192                 bugReporter.reportBug(bug);
193                 foundAny = true;
194             }
195             if (nonNullParameter != null) {
196                 for(Map.Entry<Integer, NullnessAnnotation> e : nonNullParameter.entrySet()) {
197                     int i = e.getKey();
198                     if (containsNullness(method.getParameterAnnotations(i), CHECK_FOR_NULL)) {
199                         NullnessAnnotation a = e.getValue();
200                         BugInstance bug = new BugInstance(CheckRelaxingNullnessAnnotation.this,
201                                 "NP_METHOD_PARAMETER_TIGHTENS_ANNOTATION", a.equals(NONNULL) ? HIGH_PRIORITY : NORMAL_PRIORITY);
202                         bug.addClassAndMethod(xmethod);
203                         LocalVariableAnnotation lva = null;
204                         if (localVariables != null) {
205                             for(LocalVariableNode lvn : localVariables) {
206                                 if (lvn.index == i+1) {
207                                     lva = new LocalVariableAnnotation(lvn.name, i+1, 0);
208                                     lva.setDescription(LocalVariableAnnotation.PARAMETER_NAMED_ROLE);
209                                     break;
210                                 }
211                             }
212                         }
213                         if (lva==null) {
214                             lva = new LocalVariableAnnotation("?", i+1, 0);
215                             lva.setDescription(LocalVariableAnnotation.PARAMETER_ROLE);
216                         }
217                         bug.add(lva);
218                         bugReporter.reportBug(bug);
219                         foundAny = true;
220                     }
221                 }
222 
223             }
224             return foundAny;
225         }
226     }

228     private class HierarchyIterator {
229         private XClass superclass;
230         private Queue<ClassDescriptor> interfacesToVisit;
231         private final Set<ClassDescriptor> visited;
232 
233         public HierarchyIterator(@Nonnull XClass xclass) {
234             interfacesToVisit = new LinkedList<ClassDescriptor>(Arrays.asList(xclass.getInterfaceDescriptorList()));
235             visited = new HashSet<ClassDescriptor>();
236             superclass = getClassInfo(xclass.getSuperclassDescriptor());
237         }
238 
239         public XClass next() {
240             while (!interfacesToVisit.isEmpty()) {
241                 ClassDescriptor interfaceDescr = interfacesToVisit.poll();
242                 if (visited.add(interfaceDescr)) {
243                     XClass xinterface = getClassInfo(interfaceDescr);
244                     if(xinterface != null){
245                         interfacesToVisit.addAll(Arrays.asList(xinterface.getInterfaceDescriptorList()));
246                         return xinterface;
247                     }
248                 }
249             }
250             // no interfaces => check super classes
251             if (superclass == null) {
252                 return null;
253             }
254             XClass currentSuperclass = superclass;
255             // compute next one
256             superclass = getClassInfo(superclass.getSuperclassDescriptor());
257             if(superclass != null){
258                 interfacesToVisit = new LinkedList<ClassDescriptor>(Arrays.asList(superclass.getInterfaceDescriptorList()));
259             }
260             return currentSuperclass;
261         }
262     }

264     static boolean containsRelaxedNonNull(@CheckForNull List<AnnotationNode> methodAnnotations) {
265         if (methodAnnotations == null) {
266             return false;
267         }
268         for (AnnotationNode annotation : methodAnnotations) {
269             NullnessAnnotation nullness = getNullness(annotation.desc);
270             if (nullness == CHECK_FOR_NULL || nullness == NULLABLE) {
271                 return true;
272             }
273         }
274         return false;
275     }
276 
277     @CheckForNull
278     static Map<Integer, NullnessAnnotation> getNonnullOrNullableParams(@CheckForNull List<AnnotationNode>[] parameterAnnotations) {
279         if (parameterAnnotations == null) {
280             return null;
281         }
282         Map<Integer, NullnessAnnotation> nonNullParameter = new HashMap<Integer, NullnessAnnotation>();
283         for (int i = 0; i < parameterAnnotations.length; i++) {
284             List<AnnotationNode> annotations = parameterAnnotations[i];
285             if (annotations == null) {
286                 continue;
287             }
288             for (AnnotationNode annotation : annotations) {
289                 NullnessAnnotation nullness = getNullness(annotation.desc);
290                 if (nullness == null || nullness == CHECK_FOR_NULL) {
291                     continue;
292                 }
293                 nonNullParameter.put(i, nullness);
294             }
295         }
296         return nonNullParameter;
297     }

299     @CheckForNull
300     static NullnessAnnotation getNullness(@SlashedClassName String annotationDesc) {
301         if (annotationDesc.length() < 2) {
302             return null;
303         }
304         // remove L; from signature
305         String substring = annotationDesc.substring(1, annotationDesc.length() - 1);
306         return Parser.parse(ClassName.toDottedClassName(substring));
307     }
308 
309     static boolean containsNullness(Collection<AnnotationValue> annotations, NullnessAnnotation nullness) {
310         for (AnnotationValue annotation : annotations) {
311             NullnessAnnotation check = Parser.parse(annotation.getAnnotationClass().getDottedClassName());
312             if (check == nullness) {
313                 return true;
314             }
315         }
316         return false;
317     }

319     static boolean compatibleParameters(String signature, String superSignature) {
320         SignatureParser sig = new SignatureParser(signature);
321         SignatureParser superSig = new SignatureParser(superSignature);
322         if (sig.getNumParameters() == superSig.getNumParameters()) {
323             Iterator<String> params = sig.parameterSignatureIterator();
324             Iterator<String> superParams = superSig.parameterSignatureIterator();
325             while (params.hasNext()) {
326                 String param = params.next();
327                 String superParam = superParams.next();
328                 if (areRelated(param, superParam)) {
329                     continue;
330                 }
331                 return false;
332             }
333             String retSig = sig.getReturnTypeSignature();
334             String superRetSig = superSig.getReturnTypeSignature();
335             if (areRelated(retSig, superRetSig)) {
336                 // it is compatible
337                 return true;
338             }
339         }
340         return false;
341     }
342 
343     static boolean areRelated(String sig, String superSig) {
344         try {
345             if (sig.equals(superSig)) {
346                 return true;
347             }
348             if (sig.charAt(0) == 'L' && superSig.charAt(0) == 'L') {
349                 sig = sig.substring(1, sig.length() - 1);
350                 superSig = superSig.substring(1, superSig.length() - 1);
351                 return Hierarchy.isSubtype(sig, superSig);
352             }
353         } catch (ClassNotFoundException e) {
354             AnalysisContext.reportMissingClass(e);
355         }
356         return false;
357     }
