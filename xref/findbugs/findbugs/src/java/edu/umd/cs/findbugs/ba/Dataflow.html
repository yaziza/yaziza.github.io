38  /**
39   * Perform dataflow analysis on a method using a control flow graph. Both
40   * forward and backward analyses can be performed.
41   * <ul>
42   * <li>The "start" point of each block is the entry (forward analyses) or the
43   * exit (backward analyses).
44   * <li>The "result" point of each block is the exit (forward analyses) or the
45   * entry (backward analyses).
46   * </ul>
47   * The analysis's transfer function is applied to transform the meet of the
48   * results of the block's logical predecessors (the block's start facts) into
49   * the block's result facts.
50   *
51   * @author David Hovemeyer
52   * @see CFG
53   * @see DataflowAnalysis
54   */
55  public class Dataflow<Fact, AnalysisType extends DataflowAnalysis<Fact>> {
56      private final CFG cfg;
57  
58      private final AnalysisType analysis;
59  
60      private final BlockOrder blockOrder;
61  
62      private final boolean isForwards;
63  
64      private int numIterations;
65  
66      public static boolean DEBUG = SystemProperties.getBoolean("dataflow.debug");
67  
68      /**
69       * Constructor.
70       *
71       * @param cfg
72       *            the control flow graph
73       * @param analysis
74       *            the DataflowAnalysis to be run
75       */
76      public Dataflow(CFG cfg, AnalysisType analysis) {
77          this.cfg = cfg;
78          this.analysis = analysis;
79          blockOrder = analysis.getBlockOrder(cfg);
80          isForwards = analysis.isForwards();
81          numIterations = 0;
82  
83          // Initialize result facts
84          Iterator<BasicBlock> i = cfg.blockIterator();
85          while (i.hasNext()) {
86              BasicBlock block = i.next();
87  
88              Fact result = analysis.getResultFact(block);
89              if (block == logicalEntryBlock()) {
90                  try {
91                      // Entry block: set to entry fact
92                      analysis.initEntryFact(result);
93                  } catch (DataflowAnalysisException e) {
94                      analysis.makeFactTop(result);
95                  }
96              } else {
97                  // Set to top
98                  analysis.makeFactTop(result);
99              }
100         }
101     }
102 
103     // Maximum number of iterations before we assume there is a bug and give up.
104     private static final int MAX_ITERS = SystemProperties.getInt("dataflow.maxiters", 97);
105 
106     private String getFullyQualifiedMethodName() {
107         String methodName;
108         MethodGen methodGen = cfg.getMethodGen();
109         if (methodGen == null) {
110             methodName = cfg.getMethodName();
111         } else {
112             methodName = SignatureConverter.convertMethodSignature(methodGen);
113         }
114         return methodName;
115     }

117     static class ForwardProgramOrder implements Comparator<BasicBlock>, Serializable {
118 
119         @Override
120         public int compare(BasicBlock o1, BasicBlock o2) {
121             int p1 = o1.getLabel();
122             int p2 = o2.getLabel();
123             return p1 - p2;
124         }
125 
126     }
127 
128     static class BackwardProgramOrder extends ForwardProgramOrder {
129 
130         @Override
131         public int compare(BasicBlock o1, BasicBlock o2) {
132             return super.compare(o2, o1);
133         }
134 
135     }

137     /**
138      * Run the algorithm. Afterwards, caller can use the getStartFact() and
139      * getResultFact() methods to to get dataflow facts at start and result
140      * points of each block.
141      */
142     public void execute() throws DataflowAnalysisException {
143         boolean change;
144         boolean debugWas = DEBUG;
145         if (DEBUG) {
146             reportAnalysis("Executing");
147         }
148 
149         int timestamp = 0;
150         boolean firstTime = true;
151         do {
152             change = false;
153             boolean sawBackEdge = false;
154             ++numIterations;
155             if (numIterations > MAX_ITERS && !DEBUG) {
156                 DEBUG = true;
157                 reportAnalysis("Too many iterations");
158                 System.out.println(this.getClass().getName());
159                 if (this.getClass() == UnconditionalValueDerefDataflow.class || this.getClass() == LiveLocalStoreDataflow.class) {
160                     try {
161                         ClassContext cc = Global.getAnalysisCache().getClassAnalysis(ClassContext.class,
162                                 DescriptorFactory.createClassDescriptorFromDottedClassName(cfg.getMethodGen().getClassName()));
163                         System.out.println("Forwards cfg");
164                         CFGPrinter printer = new CFGPrinter(cfg);
165                         printer.setIsForwards(true);
166                         printer.print(System.out);
167                         System.out.println("Backwards cfg");
168                         printer = new CFGPrinter(cfg);
169                         printer.setIsForwards(false);
170                         printer.print(System.out);
171                         cc.dumpSimpleDataflowInformation(cfg.getMethodGen().getMethod());
172                     } catch (CheckedAnalysisException e) {
173                         e.printStackTrace(System.out);
174                     }
175                 }
176             }
177 
178             if (DEBUG) {
179                 System.out.println("----------------------------------------------------------------------");
180                 System.out.println(this.getClass().getName() + " iteration: " + numIterations + ", timestamp: " + timestamp);
181                 MethodGen mg = cfg.getMethodGen();
182                 System.out.println(mg.getClassName() + "." + mg.getName() + mg.getSignature());
183                 System.out.println("----------------------------------------------------------------------");
184 
185             }
186 
187             if (numIterations >= MAX_ITERS + 9) {
188                 throw new DataflowAnalysisException("Too many iterations (" + numIterations + ") in dataflow when analyzing "
189                         + getFullyQualifiedMethodName());
190             }
191 
192             analysis.startIteration();
193 
194             if (DEBUG && firstTime && blockOrder instanceof ReverseDFSOrder) {
195                 ReverseDFSOrder rBlockOrder = (ReverseDFSOrder) blockOrder;
196                 System.out.println("Entry point is: " + logicalEntryBlock());
197                 System.out.println("Basic block order: ");
198                 Iterator<BasicBlock> i = blockOrder.blockIterator();
199                 while (i.hasNext()) {
200 
201                     BasicBlock block = i.next();
202                     debug(block, "rBlockOrder " + rBlockOrder.rdfs.getDiscoveryTime(block) + "\n");
203                 }
204             }
205             Iterator<BasicBlock> i = blockOrder.blockIterator();
206             if (numIterations > 3 && numIterations % 2 == 0 && blockOrder instanceof ReverseDFSOrder) {
207                 if (DEBUG) {
208                     System.out.println("Trying program order");
209                 }
210                 TreeSet<BasicBlock> bb = new TreeSet<BasicBlock>(new BackwardProgramOrder());
211                 Iterator<BasicBlock> j = blockOrder.blockIterator();
212                 while (j.hasNext()) {
213                     BasicBlock block = j.next();
214                     bb.add(block);
215                 }
216                 if (DEBUG) {
217                     for (BasicBlock block : bb) {
218                         debug(block, "\n");
219                     }
220                 }
221                 i = bb.iterator();
222             }
223             if (DEBUG) {
224                 dumpDataflow(analysis);
225             }
226 
227             // For each block in CFG...
228 
229             while (i.hasNext()) {
230 
231                 BasicBlock block = i.next();
232 
233                 // Get start fact for block.
234                 Fact start = analysis.getStartFact(block);
235                 assert start != null;
236 
237                 boolean needToRecompute = false;
238                 // Get result facts for block,
239                 Fact result = analysis.getResultFact(block);
240                 assert result != null;
241 
242                 int originalResultTimestamp = analysis.getLastUpdateTimestamp(result);
243 
244                 // Meet all of the logical predecessor results into this block's
245                 // start.
246                 // Special case: if the block is the logical entry, then it gets
247                 // the special "entry fact".
248                 if (block == logicalEntryBlock()) {
249                     analysis.makeFactTop(start);
250                     analysis.initEntryFact(start);
251                     if (DEBUG) {
252                         debug(block, "Init entry fact ==> " + analysis.factToString(start) + "\n");
253                     }
254                     needToRecompute = true;
255                 } else {
256                     int lastCalculated = analysis.getLastUpdateTimestamp(start);
257                     Iterator<Edge> predEdgeIter = logicalPredecessorEdgeIterator(block);
258 
259                     int predCount = 0;
260                     int rawPredCount = 0;
261                     while (predEdgeIter.hasNext()) {
262                         Edge edge = predEdgeIter.next();
263                         rawPredCount++;
264                         if (needToRecompute) {
265                             // don't need to check to see if we need to recompute.
266                             if (firstTime && !sawBackEdge) {
267                                 // may need to se sawBackEdge
268                             } else {
269                                 continue;
270                             }
271 
272                         }
273                         BasicBlock logicalPred = isForwards ? edge.getSource() : edge.getTarget();
274 
275                         int direction = blockOrder.compare(block, logicalPred);
276 
277                         if (DEBUG) {
278                             debug(block, "direction " + direction + " for " + blockId(logicalPred) + "\n");
279                         }
280                         if (direction < 0) {
281                             sawBackEdge = true;
282                         }
283 
284                         // Get the predecessor result fact
285                         Fact predFact = analysis.getResultFact(logicalPred);
286                         int predLastUpdated = analysis.getLastUpdateTimestamp(predFact);
287                         if (!analysis.isTop(predFact)) {
288                             predCount++;
289                             if (predLastUpdated >= lastCalculated) {
290 
291                                 needToRecompute = true;
292                                 if (DEBUG) {
293                                     debug(block, "\n Need to recompute. My timestamp = " + lastCalculated + ", pred timestamp = "
294                                             + predLastUpdated + ",\n   pred fact = " + predFact + "\n");
295                                 }
296                                 // break;
297                             }
298                         }
299                     }
300                     if (predCount == 0) {
301                         needToRecompute = true;
302                     }
303 
304                     if (!needToRecompute) {
305                         continue;
306                     }
307 
308                     analysis.makeFactTop(start);
309                     predEdgeIter = logicalPredecessorEdgeIterator(block);
310                     while (predEdgeIter.hasNext()) {
311                         Edge edge = predEdgeIter.next();
312                         BasicBlock logicalPred = isForwards ? edge.getSource() : edge.getTarget();
313 
314                         // Get the predecessor result fact
315                         Fact predFact = analysis.getResultFact(logicalPred);
316 
317                         // Apply the edge transfer function.
318                         Fact edgeFact = analysis.createFact();
319                         analysis.copy(predFact, edgeFact);
320                         analysis.edgeTransfer(edge, edgeFact);
321 
322                         if (DEBUG && !analysis.same(edgeFact, predFact)) {
323                             debug(block, logicalPred, edge, "Edge transfer " + analysis.factToString(predFact) + " ==> "
324                                     + analysis.factToString(edgeFact));
325                         }
326 
327                         // Merge the predecessor fact (possibly transformed
328                         // by the edge transfer function)
329                         // into the block's start fact.
330                         if (DEBUG) {
331                             if (analysis.isTop(start)) {
332                                 debug(block, logicalPred, edge, "\n  First pred is " + analysis.factToString(edgeFact)
333                                         + "\n   last updated at " + analysis.getLastUpdateTimestamp(predFact) + "\n");
334                             } else {
335                                 debug(block, logicalPred, edge, "\n  Meet " + analysis.factToString(start) + "\n   with "
336                                         + analysis.factToString(edgeFact)
337 
338                                         + "\n   pred last updated at " + analysis.getLastUpdateTimestamp(predFact) + "\n");
339                             }
340                         }
341 
342                         if (analysis instanceof UnconditionalValueDerefAnalysis) {
343                             ((UnconditionalValueDerefAnalysis) analysis).meetInto((UnconditionalValueDerefSet) edgeFact,
344                                     edge, (UnconditionalValueDerefSet) start, rawPredCount == 1);
345                         } else {
346                             analysis.meetInto(edgeFact, edge, start);
347                         }
348                         analysis.setLastUpdateTimestamp(start, timestamp);
349 
350                         int pos = -1;
351                         if (block.getFirstInstruction() != null) {
352                             pos = block.getFirstInstruction().getPosition();
353                         }
354                         if (DEBUG) {
355                             System.out.println(" [" + pos + "]==> " + analysis.factToString(start) + " @ " + timestamp
356                                     + " \n");
357                         }
358                     }
359                 }
360                 if (DEBUG) {
361                     debug(block, "start fact is " + analysis.factToString(start) + "\n");
362                 }
363 
364                 // making a copy of result facts (so we can detect if it
365                 // changed).
366                 boolean resultWasTop = analysis.isTop(result);
367                 Fact origResult = null;
368                 if (!resultWasTop) {
369                     origResult = analysis.createFact();
370                     analysis.copy(result, origResult);
371                 }
372 
373                 //                if (true || analysis.isTop(start)) {
374                 // Apply the transfer function.
375 
376                 analysis.transfer(block, null, start, result);
377                 //                } else {
378                 //                    analysis.copy(start, result);
379                 //                }
380 
381                 if (DEBUG && SystemProperties.getBoolean("dataflow.blockdebug")) {
382                     debug(block, "Dumping flow values for block:\n");
383                     Iterator<org.apache.bcel.generic.InstructionHandle> ii = block.instructionIterator();
384                     while (ii.hasNext()) {
385                         org.apache.bcel.generic.InstructionHandle handle = ii.next();
386                         Fact tmpResult = analysis.createFact();
387                         analysis.transfer(block, handle, start, tmpResult);
388                         System.out.println("\t" + handle + " " + analysis.factToString(tmpResult));
389                     }
390                 }
391 
392                 // See if the result changed.
393                 if (DEBUG) {
394                     debug(block, "orig result is " + (origResult == null ? "TOP" : analysis.factToString(origResult)) + "\n");
395                 }
396                 boolean thisResultChanged = false;
397                 if (resultWasTop) {
398                     thisResultChanged = !analysis.isTop(result);
399                 } else {
400                     thisResultChanged = !analysis.same(result, origResult);
401                 }
402                 if (thisResultChanged) {
403                     timestamp++;
404                     if (DEBUG) {
405                         debug(block, "result changed at timestamp " + timestamp + "\n");
406                     }
407                     if (DEBUG && !needToRecompute) {
408                         System.out.println("I thought I didn't need to recompute");
409                     }
410                     change = true;
411                     analysis.setLastUpdateTimestamp(result, timestamp);
412                 } else {
413                     analysis.setLastUpdateTimestamp(result, originalResultTimestamp);
414                 }
415 
416                 if (DEBUG) {
417                     debug(block,
418                             "result is " + analysis.factToString(result) + " @ timestamp "
419                                     + analysis.getLastUpdateTimestamp(result) + "\n");
420                 }
421             }
422 
423             analysis.finishIteration();
424             if (!sawBackEdge) {
425                 break;
426             }
427 
428         } while (change);
429 
430         if (DEBUG) {
431             System.out.println("-- Quiescence achieved-------------------------------------------------");
432             System.out.println(this.getClass().getName() + " iteration: " + numIterations + ", timestamp: " + timestamp);
433             MethodGen mg = cfg.getMethodGen();
434             System.out.println(mg.getClassName() + "." + mg.getName() + mg.getSignature());
435             new RuntimeException("Quiescence achieved----------------------------------------------------------------")
436             .printStackTrace(System.out);
437 
438         }
439         DEBUG = debugWas;
440     }

442     private void reportAnalysis(String msg) {
443         String shortAnalysisName = analysis.getClass().getName();
444         int pkgEnd = shortAnalysisName.lastIndexOf('.');
445         if (pkgEnd >= 0) {
446             shortAnalysisName = shortAnalysisName.substring(pkgEnd + 1);
447         }
448         System.out.println(msg + " " + shortAnalysisName + " on " + getFullyQualifiedMethodName());
449     }

451     private static String blockId(BasicBlock bb) {
452         InstructionHandle handle = bb.getFirstInstruction();
453         if (handle == null) {
454             return "" + bb.getLabel();
455         }
456         return bb.getLabel() + ":" + handle.getPosition() + " " + handle.getInstruction();
457     }

459     private static void debug(BasicBlock bb, String msg) {
460 
461         System.out.print("Dataflow (block " + blockId(bb) + "): " + msg);
462     }
463 
464     private static void debug(BasicBlock bb, BasicBlock pred, Edge edge, String msg) {
465         System.out.print("Dataflow (block " + blockId(bb) + ", predecessor " + blockId(pred) + " ["
466                 + Edge.edgeTypeToString(edge.getType()) + "]): " + msg);
467     }

544     /**
545      * Return an Iterator over edges that connect given block to its logical
546      * predecessors. For forward analyses, this is the incoming edges. For
547      * backward analyses, this is the outgoing edges.
548      */
549     private Iterator<Edge> logicalPredecessorEdgeIterator(BasicBlock block) {
550         return isForwards ? cfg.incomingEdgeIterator(block) : cfg.outgoingEdgeIterator(block);
551     }

553     /**
554      * Get the "logical" entry block of the CFG. For forward analyses, this is
555      * the entry block. For backward analyses, this is the exit block.
556      */
557     private BasicBlock logicalEntryBlock() {
558         return isForwards ? cfg.getEntry() : cfg.getExit();
559     }

561     public void dumpDataflow(AnalysisType analysis) {
562         System.out.println(this.getClass().getName() + " analysis for " + getCFG().getMethodName() + getCFG().getMethodSig()
563                 + " { ");
564         try {
565 
566             for (Location loc : getCFG().orderedLocations()) {
567                 System.out.println("\nBefore: " + analysis.factToString(getFactAtLocation(loc)));
568                 System.out.println("Location: " + loc);
569                 System.out.println("After: " + analysis.factToString(getFactAfterLocation(loc)));
570             }
571         } catch (DataflowAnalysisException e) {
572             AnalysisContext.logError("error dumping dataflow analysis", e);
573             System.out.println(e);
574         }
575         System.out.println("}");
576     }
