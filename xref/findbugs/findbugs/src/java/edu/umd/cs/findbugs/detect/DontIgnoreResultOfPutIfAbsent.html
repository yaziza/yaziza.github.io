99      @Override
100     public void visitClassContext(ClassContext classContext) {
101 
102         JavaClass javaClass = classContext.getJavaClass();
103         ConstantPool pool = javaClass.getConstantPool();
104         boolean found = false;
105         for (Constant constantEntry : pool.getConstantPool()) {
106             if (constantEntry instanceof ConstantNameAndType) {
107                 ConstantNameAndType nt = (ConstantNameAndType) constantEntry;
108                 if ("putIfAbsent".equals(nt.getName(pool))) {
109                     found = true;
110                     break;
111                 }
112             }
113         }
114         if (!found) {
115             return;
116         }
117 
118         Method[] methodList = javaClass.getMethods();
119 
120         for (Method method : methodList) {
121             MethodGen methodGen = classContext.getMethodGen(method);
122             if (methodGen == null) {
123                 continue;
124             }
125 
126             try {
127                 analyzeMethod(classContext, method);
128             } catch (DataflowAnalysisException e) {
129                 bugReporter.logError("Error analyzing " + method.toString(), e);
130             } catch (CFGBuilderException e) {
131                 bugReporter.logError("Error analyzing " + method.toString(), e);
132             }
133         }
134     }

147     private static int getPriorityForBeingMutable(Type type) {
148         if (type instanceof ArrayType) {
149             return HIGH_PRIORITY;
150         } else if (type instanceof ObjectType) {
151             UnreadFieldsData unreadFields = AnalysisContext.currentAnalysisContext().getUnreadFieldsData();
152 
153             ClassDescriptor cd = DescriptorFactory.getClassDescriptor((ObjectType) type);
154             @SlashedClassName
155             String className = cd.getClassName();
156             if (immutableClassNames.contains(className)) {
157                 return Priorities.LOW_PRIORITY;
158             }
159 
160             XClass xClass = AnalysisContext.currentXFactory().getXClass(cd);
161             if (xClass == null) {
162                 return Priorities.IGNORE_PRIORITY;
163             }
164             ClassDescriptor superclassDescriptor = xClass.getSuperclassDescriptor();
165             if (superclassDescriptor != null) {
166                 @SlashedClassName
167                 String superClassName = superclassDescriptor.getClassName();
168                 if ("java/lang/Enum".equals(superClassName)) {
169                     return Priorities.LOW_PRIORITY;
170                 }
171             }
172             boolean hasMutableField = false;
173             boolean hasUpdates = false;
174             for (XField f : xClass.getXFields()) {
175                 if (!f.isStatic()) {
176                     if (!f.isFinal() && !f.isSynthetic()) {
177                         hasMutableField = true;
178                         if (unreadFields.isWrittenOutsideOfInitialization(f)) {
179                             hasUpdates = true;
180                         }
181                     }
182                     String signature = f.getSignature();
183                     if (signature.startsWith("Ljava/util/concurrent") || signature.startsWith("Ljava/lang/StringB")
184                             || signature.charAt(0) == '[' || signature.indexOf("Map") >= 0 || signature.indexOf("List") >= 0
185                             || signature.indexOf("Set") >= 0) {
186                         hasMutableField = hasUpdates = true;
187                     }
188 
189                 }
190             }
191 
192             if (!hasMutableField && !xClass.isInterface() && !xClass.isAbstract()) {
193                 return Priorities.LOW_PRIORITY;
194             }
195             if (hasUpdates || className.startsWith("java/util") || className.indexOf("Map") >= 0
196                     || className.indexOf("List") >= 0) {
197                 return Priorities.HIGH_PRIORITY;
198             }
199             return Priorities.NORMAL_PRIORITY;
200 
201         } else {
202             return Priorities.IGNORE_PRIORITY;
203         }
204     }

206     private void analyzeMethod(ClassContext classContext, Method method) throws DataflowAnalysisException, CFGBuilderException {
207         if (BCELUtil.isSynthetic(method) || (method.getAccessFlags() & Constants.ACC_BRIDGE) == Constants.ACC_BRIDGE) {
208             return;
209         }
210 
211         if (DEBUG) {
212             System.out.println("    Analyzing method " + classContext.getJavaClass().getClassName() + "." + method.getName());
213         }
214 
215         JavaClass javaClass = classContext.getJavaClass();
216         ConstantPoolGen cpg = classContext.getConstantPoolGen();
217         Dataflow<BitSet, LiveLocalStoreAnalysis> llsaDataflow = classContext.getLiveLocalStoreDataflow(method);
218 
219         MethodGen methodGen = classContext.getMethodGen(method);
220         CFG cfg = classContext.getCFG(method);
221         ValueNumberDataflow vnaDataflow = classContext.getValueNumberDataflow(method);
222         TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
223 
224         String sourceFileName = javaClass.getSourceFileName();
225 
226         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
227             Location location = i.next();
228 
229             InstructionHandle handle = location.getHandle();
230             Instruction ins = handle.getInstruction();
231 
232             if (ins instanceof InvokeInstruction) {
233                 InvokeInstruction invoke = (InvokeInstruction) ins;
234                 if ("putIfAbsent".equals(invoke.getMethodName(cpg))) {
235                     String signature = invoke.getSignature(cpg);
236                     if ("(Ljava/lang/Object;Ljava/lang/Object;)Ljava/lang/Object;".equals(signature)
237                             && !(invoke instanceof INVOKESTATIC)) {
238                         TypeFrame typeFrame = typeDataflow.getFactAtLocation(location);
239                         Type objType = typeFrame.getStackValue(2);
240                         if(extendsConcurrentMap(ClassName.toDottedClassName(ClassName.fromFieldSignature(objType.getSignature())))) {
241                             InstructionHandle next = handle.getNext();
242                             boolean isIgnored = next != null && next.getInstruction() instanceof POP;
243                             //                        boolean isImmediateNullTest = next != null
244                             //                                && (next.getInstruction() instanceof IFNULL || next.getInstruction() instanceof IFNONNULL);
245                             if (countOtherCalls || isIgnored) {
246                                 BitSet live = llsaDataflow.getAnalysis().getFactAtLocation(location);
247                                 ValueNumberFrame vna = vnaDataflow.getAnalysis().getFactAtLocation(location);
248                                 ValueNumber vn = vna.getTopValue();
249 
250                                 int locals = vna.getNumLocals();
251                                 //                            boolean isRetained = false;
252                                 for (int pos = 0; pos < locals; pos++) {
253                                     if (vna.getValue(pos).equals(vn) && live.get(pos)) {
254                                         BugAnnotation ba = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, vn,
255                                                 vnaDataflow.getFactAtLocation(location), "VALUE_OF");
256                                         if (ba == null) {
257                                             continue;
258                                         }
259                                         String pattern = "RV_RETURN_VALUE_OF_PUTIFABSENT_IGNORED";
260                                         if (!isIgnored) {
261                                             pattern = "UNKNOWN";
262                                         }
263                                         Type type = typeFrame.getTopValue();
264                                         int priority = getPriorityForBeingMutable(type);
265                                         BugInstance bugInstance = new BugInstance(this, pattern, priority)
266                                         .addClassAndMethod(methodGen, sourceFileName).addCalledMethod(methodGen, invoke)
267                                         .add(new TypeAnnotation(type)).add(ba);
268                                         SourceLineAnnotation where = SourceLineAnnotation.fromVisitedInstruction(classContext,
269                                                 method, location);
270                                         accumulator.accumulateBug(bugInstance, where);
271                                         //                                    isRetained = true;
272                                         break;
273                                     }
274                                 }
275                             }
276                             /*
277                             if (testingEnabled && (countOtherCalls && !isRetained)) {
278                                 int priority = LOW_PRIORITY;
279                                 if (!isImmediateNullTest && !isIgnored) {
280                                     TypeDataflow typeAnalysis = classContext.getTypeDataflow(method);
281                                     Type type = typeAnalysis.getFactAtLocation(location).getTopValue();
282                                     String valueSignature = type.getSignature();
283                                     if (!valueSignature.startsWith("Ljava/util/concurrent/atomic/Atomic")) {
284                                         priority = Priorities.HIGH_PRIORITY;
285                                     }
286                                 }
287                                 BugInstance bugInstance = new BugInstance(this, "TESTING", priority)
288                                 .addClassAndMethod(methodGen, sourceFileName).addString("Counting putIfAbsentCalls")
289                                 .addCalledMethod(methodGen, invoke);
290                                 SourceLineAnnotation where = SourceLineAnnotation.fromVisitedInstruction(classContext, method,
291                                         location);
292                                 accumulator.accumulateBug(bugInstance, where);
293                             }
294                              */
295                         }
296                     } /* else if (testingEnabled && countOtherCalls) {
297                         BugInstance bugInstance = new BugInstance(this, "TESTING2", Priorities.NORMAL_PRIORITY)
298                         .addClassAndMethod(methodGen, sourceFileName).addCalledMethod(methodGen, invoke);
299                         SourceLineAnnotation where = SourceLineAnnotation.fromVisitedInstruction(classContext, method, location);
300                         accumulator.accumulateBug(bugInstance, where);
301 
302                     }
303                      */
304                 }
305 
306             }
307         }
308         accumulator.reportAccumulatedBugs();
309     }

311     private boolean extendsConcurrentMap(@DottedClassName String className) {
312         if ("java.util.concurrent.ConcurrentHashMap".equals(className)
313                 || className.equals(concurrentMapDescriptor.getDottedClassName())) {
314             return true;
315         }
316         ClassDescriptor c = DescriptorFactory.createClassDescriptorFromDottedClassName(className);
317         Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
318 
319         try {
320             if (subtypes2.isSubtype(c, concurrentMapDescriptor)) {
321                 return true;
322             }
323         } catch (ClassNotFoundException e) {
324             AnalysisContext.reportMissingClass(e);
325         }
326 
327         return false;
