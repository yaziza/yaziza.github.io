82      public static @CheckForNull
83      XMethod definedIn(JavaClass clazz, XMethod m) {
84          for (Method m2 : clazz.getMethods()) {
85              if (m.getName().equals(m2.getName()) && m.getSignature().equals(m2.getSignature()) && m.isStatic() == m2.isStatic()) {
86                  return XFactory.createXMethod(clazz, m2);
87              }
88          }
89          return null;
90      }

92      public static boolean confusingMethodNamesWrongCapitalization(XMethod m1, XMethod m2) {
93          if (m1.isStatic() != m2.isStatic()) {
94              return false;
95          }
96          if (m1.getClassName().equals(m2.getClassName())) {
97              return false;
98          }
99          if (m1.getName().equals(m2.getName())) {
100             return false;
101         }
102         if (m1.getName().equalsIgnoreCase(m2.getName())
103                 && removePackageNamesFromSignature(m1.getSignature()).equals(removePackageNamesFromSignature(m2.getSignature()))) {
104             return true;
105         }
106         return false;
107     }

109     public static boolean confusingMethodNamesWrongPackage(XMethod m1, XMethod m2) {
110         if (m1.isStatic() != m2.isStatic()) {
111             return false;
112         }
113         if (m1.getClassName().equals(m2.getClassName())) {
114             return false;
115         }
116 
117         if (!m1.getName().equals(m2.getName())) {
118             return false;
119         }
120         if (m1.getSignature().equals(m2.getSignature())) {
121             return false;
122         }
123         if (removePackageNamesFromSignature(m1.getSignature()).equals(removePackageNamesFromSignature(m2.getSignature()))) {
124             return true;
125         }
126         return false;
127     }

145     private boolean checkSuper(XMethod m, Set<XMethod> others) {
146         if (m.isStatic()) {
147             return false;
148         }
149         if ("<init>".equals(m.getName()) || "<clinit>".equals(m.getName())) {
150             return false;
151         }
152         for (XMethod m2 : others) {
153             try {
154                 if ((confusingMethodNamesWrongCapitalization(m, m2) || confusingMethodNamesWrongPackage(m, m2))
155                         && Repository.instanceOf(m.getClassName(), m2.getClassName())) {
156                     WarningPropertySet<NamingProperty> propertySet = new WarningPropertySet<NamingProperty>();
157 
158                     int priority = HIGH_PRIORITY;
159                     boolean intentional = false;
160                     XMethod m3 = null;
161                     try {
162                         JavaClass clazz = Repository.lookupClass(m.getClassName());
163                         if ((m3 = definedIn(clazz, m2)) != null) {
164                             // the method we don't override is also defined in
165                             // our class
166                             priority = NORMAL_PRIORITY;
167                             intentional = true;
168                         }
169                     } catch (ClassNotFoundException e) {
170                         priority++;
171                         AnalysisContext.reportMissingClass(e);
172                     }
173 
174                     XFactory xFactory = AnalysisContext.currentXFactory();
175                     if (m3 == null && xFactory.isCalled(m)) {
176                         propertySet.addProperty(NamingProperty.METHOD_IS_CALLED);
177                     } else if (m.isDeprecated() || m2.isDeprecated()) {
178                         propertySet.addProperty(NamingProperty.METHOD_IS_DEPRECATED);
179                     }
180 
181                     if (!m.getName().equals(m2.getName()) && m.getName().equalsIgnoreCase(m2.getName())) {
182                         String pattern = intentional ? "NM_VERY_CONFUSING_INTENTIONAL" : "NM_VERY_CONFUSING";
183                         Set<XMethod> overrides = Hierarchy2.findSuperMethods(m);
184                         if (!overrides.isEmpty()) {
185                             if (intentional || allAbstract(overrides)) {
186                                 break;
187                             }
188                             priority++;
189                         }
190                         BugInstance bug = new BugInstance(this, pattern, priority).addClass(m.getClassName()).addMethod(m)
191                                 .addClass(m2.getClassName()).describe(ClassAnnotation.SUPERCLASS_ROLE).addMethod(m2)
192                                 .describe(MethodAnnotation.METHOD_DID_YOU_MEAN_TO_OVERRIDE);
193                         if (m3 != null) {
194                             bug.addMethod(m3).describe(MethodAnnotation.METHOD_OVERRIDDEN);
195                         }
196                         propertySet.decorateBugInstance(bug);
197                         bugReporter.reportBug(bug);
198                     } else if (!m.getSignature().equals(m2.getSignature())
199                             && removePackageNamesFromSignature(m.getSignature()).equals(
200                                     removePackageNamesFromSignature(m2.getSignature()))) {
201                         String pattern = intentional ? "NM_WRONG_PACKAGE_INTENTIONAL" : "NM_WRONG_PACKAGE";
202                         Set<XMethod> overrides = Hierarchy2.findSuperMethods(m);
203                         if (!overrides.isEmpty()) {
204                             if (intentional || allAbstract(overrides)) {
205                                 break;
206                             }
207                             priority++;
208                         }
209                         Iterator<String> s = new SignatureParser(m.getSignature()).parameterSignatureIterator();
210                         Iterator<String> s2 = new SignatureParser(m2.getSignature()).parameterSignatureIterator();
211                         while (s.hasNext()) {
212                             String p = s.next();
213                             String p2 = s2.next();
214                             if (!p.equals(p2)) {
215                                 BugInstance bug = new BugInstance(this, pattern, priority).addClass(m.getClassName())
216                                         .addMethod(m).addClass(m2.getClassName()).describe(ClassAnnotation.SUPERCLASS_ROLE)
217                                         .addMethod(m2).describe(MethodAnnotation.METHOD_DID_YOU_MEAN_TO_OVERRIDE)
218                                         .addFoundAndExpectedType(p, p2);
219                                 if (m3 != null) {
220                                     bug.addMethod(m3).describe(MethodAnnotation.METHOD_OVERRIDDEN);
221                                 }
222                                 propertySet.decorateBugInstance(bug);
223                                 bugReporter.reportBug(bug);
224 
225                             }
226                         }
227 
228                         //
229                     }
230                     return true;
231                 }
232             } catch (ClassNotFoundException e) {
233                 AnalysisContext.reportMissingClass(e);
234             }
235         }
236         return false;
237     }

239     private boolean allAbstract(Set<XMethod> overrides) {
240         boolean allAbstract = true;
241         for (XMethod m4 : overrides) {
242             if (!m4.isAbstract()) {
243                 allAbstract = false;
244             }
245         }
246         return allAbstract;
247     }

249     private boolean checkNonSuper(XMethod m, Set<XMethod> others) {
250         if (m.isStatic()) {
251             return false;
252         }
253         if (m.getName().startsWith("<init>") || m.getName().startsWith("<clinit>")) {
254             return false;
255         }
256         for (XMethod m2 : others) {
257             if (confusingMethodNamesWrongCapitalization(m, m2)) {
258                 XMethod mm1;
259                 XMethod mm2;
260                 if (m.compareTo(m2) < 0) {
261                     mm1 = m;
262                     mm2 = m2;
263                 } else {
264                     mm1 = m2;
265                     mm2 = m;
266                 }
267                 bugReporter.reportBug(new BugInstance(this, "NM_CONFUSING", LOW_PRIORITY).addClass(mm1.getClassName())
268                         .addMethod(mm1).addClass(mm2.getClassName()).addMethod(mm2));
269                 return true;
270             }
271         }
272         return false;
273     }

275     @Override
276     public void report() {
277 
278         for (Map.Entry<String, TreeSet<XMethod>> e : canonicalToXMethod.entrySet()) {
279             TreeSet<XMethod> conflictingMethods = e.getValue();
280             HashSet<String> trueNames = new HashSet<String>();
281 
282             for (XMethod m : conflictingMethods) {
283                 trueNames.add(m.getName() + m.getSignature());
284             }
285             if (trueNames.size() <= 1) {
286                 continue;
287             }
288             for (Iterator<XMethod> j = conflictingMethods.iterator(); j.hasNext();) {
289                 if (checkSuper(j.next(), conflictingMethods)) {
290                     j.remove();
291                 }
292             }
293             for (XMethod conflictingMethod : conflictingMethods) {
294                 if (checkNonSuper(conflictingMethod, conflictingMethods)) {
295                     break;
296                 }
297             }
298         }
299     }

301     public String stripPackageName(String className) {
302         if (className.indexOf('.') >= 0) {
303             return className.substring(className.lastIndexOf('.') + 1);
304         } else if (className.indexOf('/') >= 0) {
305             return className.substring(className.lastIndexOf('/') + 1);
306         } else {
307             return className;
308         }
309     }

311     public boolean sameSimpleName(String class1, String class2) {
312         return class1 != null && class2 != null && stripPackageName(class1).equals(stripPackageName(class2));
313     }

315     @Override
316     public void visitJavaClass(JavaClass obj) {
317         if (BCELUtil.isSynthetic(obj)) {
318             return;
319         }
320         String name = obj.getClassName();
321         if (!visited.add(name)) {
322             return;
323         }
324 
325         String superClassName = obj.getSuperclassName();
326         if (!"java.lang.Object".equals(name)) {
327             if (sameSimpleName(superClassName, name)) {
328                 bugReporter.reportBug(new BugInstance(this, "NM_SAME_SIMPLE_NAME_AS_SUPERCLASS", HIGH_PRIORITY).addClass(name)
329                         .addClass(superClassName));
330             }
331             for (String interfaceName : obj.getInterfaceNames()) {
332                 if (sameSimpleName(interfaceName, name)) {
333                     bugReporter.reportBug(new BugInstance(this, "NM_SAME_SIMPLE_NAME_AS_INTERFACE", NORMAL_PRIORITY).addClass(
334                             name).addClass(interfaceName));
335                 }
336             }
337         }
338         if (obj.isInterface()) {
339             return;
340         }
341 
342         if ("java.lang.Object".equals(superClassName) && !visited.contains(superClassName)) {
343             try {
344                 visitJavaClass(obj.getSuperClass());
345             } catch (ClassNotFoundException e) {
346                 // ignore it
347             }
348         }
349         super.visitJavaClass(obj);
350     }

352     /**
353      * Determine whether the class descriptor ultimately inherits from
354      * java.lang.Exception
355      *
356      * @param d
357      *            class descriptor we want to check
358      * @return true iff the descriptor ultimately inherits from Exception
359      */
360     private static boolean mightInheritFromException(ClassDescriptor d) {
361         while (d != null) {
362             try {
363                 if ("java.lang.Exception".equals(d.getDottedClassName())) {
364                     return true;
365                 }
366                 XClass classNameAndInfo = Global.getAnalysisCache().getClassAnalysis(XClass.class, d);
367                 d = classNameAndInfo.getSuperclassDescriptor();
368             } catch (CheckedAnalysisException e) {
369                 return true; // don't know
370             }
371         }
372         return false;
373     }

379     /**
380      * Eclipse uses reflection to initialize NLS message bundles. Classes which
381      * using this mechanism are usualy extending org.eclipse.osgi.util.NLS class
382      * and contains lots of public static String fields which are used as
383      * message constants. Unfortunately these fields often has bad names which
384      * does not follow Java code convention, so FB reports tons of warnings for
385      * such Eclipse message fields.
386      *
387      * @see edu.umd.cs.findbugs.detect.MutableStaticFields
388      */
389     private boolean isEclipseNLS;
390 
391     @Override
392     public void visit(JavaClass obj) {
393         String name = obj.getClassName();
394         String[] parts = name.split("[$+.]");
395         baseClassName = parts[parts.length - 1];
396         for (String p : name.split("[.]")) {
397             if (p.length() == 1) {
398                 return;
399             }
400         }
401         if (name.indexOf("Proto$") >= 0) {
402             return;
403         }
404         classIsPublicOrProtected = obj.isPublic() || obj.isProtected();
405         if (Character.isLetter(baseClassName.charAt(0)) && !Character.isUpperCase(baseClassName.charAt(0))
406                 && baseClassName.indexOf('_') == -1) {
407             int priority = classIsPublicOrProtected ? NORMAL_PRIORITY : LOW_PRIORITY;
408 
409             bugReporter.reportBug(new BugInstance(this, "NM_CLASS_NAMING_CONVENTION", priority).addClass(this));
410         }
411         if (name.endsWith("Exception")) {
412             // Does it ultimately inherit from Throwable?
413             if (!mightInheritFromException(DescriptorFactory.createClassDescriptor(obj))) {
414                 // It doens't, so the name is misleading
415                 bugReporter.reportBug(new BugInstance(this, "NM_CLASS_NOT_EXCEPTION", NORMAL_PRIORITY).addClass(this));
416             }
417         }
418 
419         int badFieldNames = 0;
420         for (Field f : obj.getFields()) {
421             if (f.getName().length() >= 2 && badFieldName(f)) {
422                 badFieldNames++;
423             }
424         }
425         hasBadFieldNames = badFieldNames > 3 && badFieldNames > obj.getFields().length / 3;
426         int badMethodNames = 0;
427         for (Method m : obj.getMethods()) {
428             if (badMethodName(m.getName())) {
429                 badMethodNames++;
430             }
431         }
432         hasBadMethodNames = badMethodNames > 3 && badMethodNames > obj.getMethods().length / 3;
433         isEclipseNLS = "org.eclipse.osgi.util.NLS".equals(obj.getSuperclassName());
434         super.visit(obj);
435     }

437     @Override
438     public void visit(Field obj) {
439         if (getFieldName().length() == 1) {
440             return;
441         }
442 
443         if (isEclipseNLS) {
444             int flags = obj.getAccessFlags();
445             if ((flags & ACC_STATIC) != 0 && ((flags & ACC_PUBLIC) != 0) && "Ljava/lang/String;".equals(getFieldSig())) {
446                 // ignore "public statis String InstallIUCommandTooltip;"
447                 // messages from Eclipse NLS bundles
448                 return;
449             }
450         }
451         if (badFieldName(obj)) {
452             bugReporter.reportBug(new BugInstance(this, "NM_FIELD_NAMING_CONVENTION", classIsPublicOrProtected
453                     && (obj.isPublic() || obj.isProtected()) && !hasBadFieldNames ? NORMAL_PRIORITY : LOW_PRIORITY)
454             .addClass(this).addVisitedField(this));
455         }
456     }

458     private boolean badFieldName(Field obj) {
459         String fieldName = obj.getName();
460         return !obj.isFinal() && Character.isLetter(fieldName.charAt(0)) && !Character.isLowerCase(fieldName.charAt(0))
461                 && fieldName.indexOf('_') == -1 && Character.isLetter(fieldName.charAt(1))
462                 && Character.isLowerCase(fieldName.charAt(1));
463     }

469     private static @CheckForNull
470     String getSignatureOfOuterClass(JavaClass obj) {
471         for (Field f : obj.getFields()) {
472             if (f.getName().startsWith("this$")) {
473                 return f.getSignature();
474             }
475         }
476         return null;

479     private boolean markedAsNotUsable(Method obj) {
480         for (Attribute a : obj.getAttributes()) {
481             if (a instanceof Deprecated) {
482                 return true;
483             }
484         }
485         Code code = obj.getCode();
486         if (code == null) {
487             return false;
488         }
489         byte[] codeBytes = code.getCode();
490         if (codeBytes.length > 1 && codeBytes.length < 10) {
491             int lastOpcode = codeBytes[codeBytes.length - 1] & 0xff;
492             if (lastOpcode != ATHROW) {
493                 return false;
494             }
495             for (int b : codeBytes) {
496                 if ((b & 0xff) == RETURN) {
497                     return false;
498                 }
499             }
500             return true;
501         }
502         return false;
503     }

505     private static @CheckForNull
506     Method findVoidConstructor(JavaClass clazz) {
507         for (Method m : clazz.getMethods()) {
508             if (isVoidConstructor(clazz, m)) {
509                 return m;
510             }
511         }
512         return null;

516     @Override
517     public void visit(Method obj) {
518         String mName = getMethodName();
519         if (mName.length() == 1) {
520             return;
521         }
522         if ("isRequestedSessionIdFromURL".equals(mName) || "isRequestedSessionIdFromUrl".equals(mName)) {
523             return;
524         }
525         String sig = getMethodSig();
526         if (mName.equals(baseClassName) && "()V".equals(sig)) {
527             Code code = obj.getCode();
528             Method realVoidConstructor = findVoidConstructor(getThisClass());
529             if (code != null && !markedAsNotUsable(obj)) {
530                 int priority = NORMAL_PRIORITY;
531                 if (codeDoesSomething(code)) {
532                     priority--;
533                 } else if (!obj.isPublic() && getThisClass().isPublic()) {
534                     priority--;
535                 }
536                 boolean instanceMembers = false;
537                 for (Method m : this.getThisClass().getMethods()) {
538                     if (!m.isStatic() && m != obj && !isVoidConstructor(getThisClass(), m)) {
539                         instanceMembers = true;
540                     }
541                 }
542                 for (Field f : this.getThisClass().getFields()) {
543                     if (!f.isStatic()) {
544                         instanceMembers = true;
545                     }
546                 }
547                 if (!codeDoesSomething(code) && !instanceMembers && "java/lang/Object".equals(getSuperclassName())) {
548                     priority += 2;
549                 }
550                 if (hasBadMethodNames) {
551                     priority++;
552                 }
553                 if (!getXClass().getAnnotations().isEmpty()) {
554                     priority++;
555                 }
556                 if (realVoidConstructor != null) {
557                     priority = LOW_PRIORITY;
558                 }
559 
560                 bugReporter.reportBug(new BugInstance(this, "NM_METHOD_CONSTRUCTOR_CONFUSION", priority).addClassAndMethod(this)
561                         .lowerPriorityIfDeprecated());
562                 return;
563             }
564         } else if (badMethodName(mName)) {
565             bugReporter.reportBug(new BugInstance(this, "NM_METHOD_NAMING_CONVENTION", classIsPublicOrProtected
566                     && (obj.isPublic() || obj.isProtected()) && !hasBadMethodNames ? NORMAL_PRIORITY : LOW_PRIORITY)
567             .addClassAndMethod(this));
568         }
569 
570         if (obj.isAbstract()) {
571             return;
572         }
573         if (obj.isPrivate()) {
574             return;
575         }
576 
577         if ("equal".equals(mName) && "(Ljava/lang/Object;)Z".equals(sig)) {
578             bugReporter.reportBug(new BugInstance(this, "NM_BAD_EQUAL", HIGH_PRIORITY).addClassAndMethod(this)
579                     .lowerPriorityIfDeprecated());
580             return;
581         }
582         if ("hashcode".equals(mName) && "()I".equals(sig)) {
583             bugReporter.reportBug(new BugInstance(this, "NM_LCASE_HASHCODE", HIGH_PRIORITY).addClassAndMethod(this)
584                     .lowerPriorityIfDeprecated());
585             return;
586         }
587         if ("tostring".equals(mName) && "()Ljava/lang/String;".equals(sig)) {
588             bugReporter.reportBug(new BugInstance(this, "NM_LCASE_TOSTRING", HIGH_PRIORITY).addClassAndMethod(this)
589                     .lowerPriorityIfDeprecated());
590             return;
591         }
592 
593         if (obj.isPrivate() || obj.isStatic() || "<init>".equals(mName)) {
594             return;
595         }
596 
597         String sig2 = removePackageNamesFromSignature(sig);
598         String allSmall = mName.toLowerCase() + sig2;
599 
600         XMethod xm = getXMethod();
601         {
602             TreeSet<XMethod> s = canonicalToXMethod.get(allSmall);
603             if (s == null) {
604                 s = new TreeSet<XMethod>();
605                 canonicalToXMethod.put(allSmall, s);
606             }
607             s.add(xm);
608         }
609 
610     }

612     private static boolean isVoidConstructor(JavaClass clazz, Method m) {
613         String outerClassSignature = getSignatureOfOuterClass(clazz);
614         if (outerClassSignature == null) {
615             outerClassSignature = "";
616         }
617         return "<init>".equals(m.getName()) && m.getSignature().equals("(" + outerClassSignature + ")V");
618     }

620     private boolean badMethodName(String mName) {
621         return mName.length() >= 2 && Character.isLetter(mName.charAt(0)) && !Character.isLowerCase(mName.charAt(0))
622                 && Character.isLetter(mName.charAt(1)) && Character.isLowerCase(mName.charAt(1)) && mName.indexOf('_') == -1;
623     }

625     private boolean codeDoesSomething(Code code) {
626         byte[] codeBytes = code.getCode();
627         return codeBytes.length > 1;
628     }

630     private static String removePackageNamesFromSignature(String sig) {
631         int end = sig.indexOf(')');
632         Matcher m = sigType.matcher(sig.substring(0, end));
633         return m.replaceAll("L$2") + sig.substring(end);
634     }
