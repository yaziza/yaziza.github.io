76          public int score() {
77              switch (this) {
78              case NEW:
79                  return 0;
80  
81              case ACCEPTED:
82              case DUPLICATE:
83              case WILL_NOT_FIX:
84                  return 1;
85  
86              case ASSIGNED:
87              case FIXED:
88              case FIX_LATER:
89              case VERIFIED:
90              case VERIFIER_ASSIGNED:
91                  return 2;
92  
93              default:
94                  throw new IllegalStateException();
95              }
96          }

100             case NEW:
101             case DUPLICATE:
102                 return 0;
103 
104             case ASSIGNED:
105                 return 1;
106 
107             case WILL_NOT_FIX:
108             case ACCEPTED:
109                 return 2;

114                 return 4;
115             case VERIFIED:
116             case VERIFIER_ASSIGNED:

119             default:

121             }

125     static Timestamp bucketByHour(Timestamp t) {
126         Timestamp result = new Timestamp(t.getTime());
127         result.setSeconds(0);
128         result.setMinutes(0);
129         result.setNanos(0);
130         return result;
131     }

133     static class TimeSeries<K, V extends Comparable<? super V>> implements Comparable<TimeSeries<K, V>> {
134         @Override
135         public int hashCode() {
136             final int prime = 31;
137             int result = 1;
138             result = prime * result + ((k == null) ? 0 : k.hashCode());
139             result = prime * result + ((v == null) ? 0 : v.hashCode());
140             return result;
141         }
142 
143         @Override
144         public boolean equals(Object obj) {
145             if (this == obj)
146                 return true;
147             if (!(obj instanceof TimeSeries))
148                 return false;
149             TimeSeries<K, V> other = (TimeSeries) obj;
150             return Util.nullSafeEquals(this.k, other.k) && Util.nullSafeEquals(this.v, other.v);
151         }
152 
153         final K k;
154 
155         final int keyHash;
156 
157         final V v;
158 
159         public TimeSeries(K k, V v) {
160             this.k = k;
161             this.keyHash = System.identityHashCode(k);
162             this.v = v;
163         }
164 
165         @Override
166         public String toString() {
167             return v + " " + k;
168         }
169 
170         public int compareTo(TimeSeries<K, V> o) {
171             if (o == this)
172                 return 0;
173             int result = v.compareTo(o.v);
174             if (result != 0)
175                 return result;
176             if (keyHash < o.keyHash)
177                 return -1;
178             return 1;
179         }
180     }

225             if (pattern != null) {

227                 bugRank.put(id, BugRankCategory.getRank(rank));
228                 detailedBugRank.put(hash, rank);
229                 bugPattern.put(id, pattern.getType());
230             }
231         }

233         ps.close();

238         MergeMap.MinMap<String, Timestamp> firstUse = new MergeMap.MinMap<String, Timestamp>();
239         MergeMap.MinMap<String, Timestamp> reviewers = new MergeMap.MinMap<String, Timestamp>();
240         MergeMap.MinMap<String, Timestamp> uniqueReviews = new MergeMap.MinMap<String, Timestamp>();

489     private static void printPatterns(String filename, String header, FractionalMultiset<String> average,
490             FractionalMultiset<String> variance, Multiset<String> count) throws FileNotFoundException {
491         DetectorFactoryCollection i18n = DetectorFactoryCollection.instance();
492         PrintWriter out = new PrintWriter(filename);
493         out.println(header);
494         for (Map.Entry<String, Double> e : average.entriesInDecreasingOrder()) {
495             String key = e.getKey();
496             BugPattern pattern = i18n.lookupBugPattern(key);
497             if (pattern != null)
498                 out.printf("%1.1f,%1.1f,%d,%d,%s%n", e.getValue(), variance.getValue(key), BugRanker.findRank(pattern, 1),
499                         count.getCount(key), key);
500         }
501         out.close();
502     }

504     /**

506      */
507     private static UserDesignation getDesignationFromScore(int value) {
508         if (value <= -3)
509             return UserDesignation.BAD_ANALYSIS;
510         else
511             switch (value) {
512             case -2:
513                 return UserDesignation.NOT_A_BUG;
514             case -1:
515                 return UserDesignation.MOSTLY_HARMLESS;
516             case 0:
517                 return UserDesignation.NEEDS_STUDY;
518             case 1:
519                 return UserDesignation.SHOULD_FIX;
520             default:
521                 return UserDesignation.MUST_FIX;
522             }
523     }

528      * @return
529      */
530     private static String getDesignationTitle(I18N i18n, edu.umd.cs.findbugs.cloud.Cloud.UserDesignation d) {
531         String designation;
532         switch (d) {
533         case OBSOLETE_CODE:
534             designation = "obsolete code";
535             break;
536         case MUST_FIX:
537             designation = "Must fix";
538             break;
539         case SHOULD_FIX:
540             designation = "Should fix";
541             break;
542         default:
543             designation = i18n.getUserDesignation(d.name());
544         }
545         return designation;
546     }

566     final static Date fixitStart = new Date("May 11, 2009");
567 
568     private static void printTimeSeries(String filename, String title, MergeMap.MinMap<String, Timestamp> firstUse)
569             throws FileNotFoundException {
570         PrintWriter out = new PrintWriter(filename);
571         out.println(title + ",time,full time");
572         TreeSet<TimeSeries<String, Timestamp>> series = new TreeSet<TimeSeries<String, Timestamp>>();
573         for (Map.Entry<String, Timestamp> e : firstUse.entrySet()) {
574             series.add(new TimeSeries<String, Timestamp>(e.getKey(), e.getValue()));
575         }
576 
577         Multiset<Timestamp> counter = new Multiset<Timestamp>(new TreeMap<Timestamp, Integer>());
578         for (TimeSeries<String, Timestamp> t : series) {
579             counter.add(bucketByHour(t.v));
580         }
581         int total = 0;
582         SimpleDateFormat format = new SimpleDateFormat("h a EEE");
583         SimpleDateFormat defaultFormat = new SimpleDateFormat();
584         for (Map.Entry<Timestamp, Integer> e : counter.entrySet()) {
585             Timestamp time = e.getKey();
586             total += e.getValue();
587             if (time.after(fixitStart))
588                 out.printf("%d,%s,%s%n", total, format.format(time), defaultFormat.format(time));
589         }
590         out.close();
591 
592     }
